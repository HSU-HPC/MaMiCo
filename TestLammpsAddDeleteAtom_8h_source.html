12template <unsigned int dim>
13class TestLammpsAddDeleteAtom : public TestLammps<dim> {
14public:
15 TestLammpsAddDeleteAtom(int argc, char **argv, std::string name)
16 : TestLammps<dim>(argc, argv, name) {}
17 virtual ~TestLammpsAddDeleteAtom() {}
18
19 virtual void run() {
20 // LOAD COUPLED SCENARIO
21 // ----------------------------------------------------------------------
22 if (dim == 2) {
23 TestLammps<dim>::loadLammpsTestConfiguration("inputpositionsonly2D.xyz",
24 4);
25 } else {
26 TestLammps<dim>::loadLammpsTestConfiguration("inputpositionsonly3D.xyz",
27 8);
28 }
29 TestLammps<dim>::loadMacroscopicSolverConfiguration();
30 TestLammps<dim>::loadMamicoTestConfiguration();
31 std::cout << "All configs, solvers and coupling loaded and initialised..."
32 << std::endl;
33
34 // DEFINE PROPERTIES OF ATOM TO BE ADDED/DELETED
35 // ---------------------------------------------- helper variables
36 const tarch::la::Vector<dim, unsigned int> linkedCellsPerMacroscopicCell(1);
37 const tarch::la::Vector<dim, unsigned int> linkedCellInMacroscopicCell(0);
38 const coupling::IndexConversion<dim> &indexConversion =
39 TestLammps<dim>::_macroscopicCellService->getIndexConversion();
40 const double tolerance =
41 1.0e-8; // this value should be the same as tolerance defined in
42 // MDSolverInterface of LAMMPS
43 int rank, size;
44 MPI_Comm_size(MPI_COMM_WORLD, &size);
45 MPI_Comm_rank(MPI_COMM_WORLD, &rank);
46
47 // position and dummy values of atom to be deleted
48 tarch::la::Vector<dim, double> posDeleteAtom(0.0);
49 posDeleteAtom[0] = 4.0;
50 posDeleteAtom[1] = 2.0;
51 if (dim != 2)
52 posDeleteAtom[2] = 2.0;
53 const tarch::la::Vector<dim, double> dummyVec(0.0);
54 const double dummyEnergy = 0.0;
55 const coupling::datastructures::Molecule<dim> deleteMolecule(
56 posDeleteAtom, dummyVec, dummyVec, dummyEnergy);
57 tarch::la::Vector<dim, unsigned int> deleteCellIndex(1);
58 deleteCellIndex[0] = 2; // global cell index for deletion
59 // true, if this is the respective rank that should contain this molecule
60 const bool performDeletion = (size == 1) || ((size == 4) && (rank == 0)) ||
61 ((size == 16) && (rank == 1)) ||
62 ((size == 8) && (rank == 0)) ||
63 ((size == 64) && (rank == 1));
64
65 // position, velocity, force of atom to be inserted
66 tarch::la::Vector<dim, double> posAddAtom(3.6);
67 const tarch::la::Vector<dim, double> velAddAtom(5.0);
68 const tarch::la::Vector<dim, double> fAddAtom(10.0);
69 const coupling::datastructures::Molecule<dim> addMolecule(
70 posAddAtom, velAddAtom, fAddAtom, dummyEnergy);
71 tarch::la::Vector<dim, unsigned int> addCellIndex(
72 2); // global cell index for adding atom
73 // true, if this rank holds the respective domain containing the new
74 // molecule's position
75 const bool performAdd = (size == 1) || ((size == 4) && (rank == 0)) ||
76 ((size == 16) && (rank == 5)) ||
77 ((size == 8) && (rank == 0)) ||
78 ((size == 64) && (rank == 21));
79
80 // set a very small time step (so that molecules hardly move and positions
81 // thus remain quasi identical) and run one time step to sort molecules into
82 // mamico cells
83 // TestLammps<dim>::_lammps->input->one("timestep 1e-14");
84 TestLammps<dim>::_lammps->input->one("run 1");
85
86 // for testing purposes, set cut-off radius to 2.5 (=mamico-cell-size); we
87 // can then check whether the newly inserted/deleted atoms are transferred
88 // to the neighbouring processes in the
89 // synchronizeMoleculesAfterMassModification()-method.
90 // TestLammps<dim>::_lammps->input->one("pair_style lj/cut 2.5");
91 // TestLammps<dim>::_lammps->input->one("pair_coeff 1 1 1.0 1.0 2.5");
92
93 // print out all molecules for each rank (synchronised)
94 TestLammps<dim>::printMolecules();
95
96 // PERFORM DELETION TEST
97 // -----------------------------------------------------------------------------------
98 // delete atom
99 if (performDeletion) {
100 const int numberAtoms = TestLammps<dim>::_lammps->atom->nlocal;
101 // convert global to local coordinate
102 deleteCellIndex =
103 indexConversion.convertGlobalToLocalVectorCellIndex(deleteCellIndex);
104 LAMMPS_NS::MamicoCell &deleteCell =
105 coupling::interface::MamicoInterfaceProvider<LAMMPS_NS::MamicoCell,
106 dim>::getInstance()
107 .getMDSolverInterface()
108 ->getLinkedCell(deleteCellIndex, linkedCellInMacroscopicCell,
109 linkedCellsPerMacroscopicCell, indexConversion);
110 coupling::interface::MoleculeIterator<LAMMPS_NS::MamicoCell,
111 dim> *iterator =
112 coupling::interface::MamicoInterfaceProvider<LAMMPS_NS::MamicoCell,
113 dim>::getInstance()
114 .getMDSolverInterface()
115 ->getMoleculeIterator(deleteCell);
116 int cellCounterBeforeDeletion = 0;
117 int cellCounterAfterDeletion = 0;
118
119 // determine number of molecules in this cell before deletion and plot
120 // molecule positions
121 std::cout << "Molecules before deletion:" << std::endl;
122 for (iterator->begin(); iterator->continueIteration(); iterator->next()) {
123 std::cout << iterator->getConst().getPosition() << std::endl;
124 cellCounterBeforeDeletion++;
125 };
126
127 // delete molecule
128 coupling::interface::MamicoInterfaceProvider<LAMMPS_NS::MamicoCell,
129 dim>::getInstance()
130 .getMDSolverInterface()
131 ->deleteMoleculeFromMDSimulation(deleteMolecule, deleteCell);
132
133 // check if number of atoms has been decremented
134 if (numberAtoms - 1 != TestLammps<dim>::_lammps->atom->nlocal) {
135 std::cout << "ERROR TestLammpsAddDeleteAtom::run(): Same number of "
136 "atoms after deletion as before!"
137 << std::endl;
138 exit(EXIT_FAILURE);
139 }
140 // check if deleted atom can be found in mamico cell
141 std::cout << "Molecules after deletion:" << std::endl;
142 for (iterator->begin(); iterator->continueIteration(); iterator->next()) {
143 std::cout << iterator->getConst().getPosition() << std::endl;
144 cellCounterAfterDeletion++;
145 }
146 if (cellCounterAfterDeletion != cellCounterBeforeDeletion - 1) {
147 int rank;
148 MPI_Comm_rank(MPI_COMM_WORLD, &rank);
149 std::cout << "Rank " << rank
150 << ": ERROR TestLammpsAddDeleteAtom::run(): Did not reduce "
151 "number of molecules in the respective mamico cell!"
152 << std::endl;
153 exit(EXIT_FAILURE);
154 }
155
156 // check if deleted atom can be found anywhere in LAMMPS
157 for (int i = 0; i < TestLammps<dim>::_lammps->atom->nlocal; i++) {
158 bool found = true;
159 for (unsigned int d = 0; d < dim; d++) {
160 found = found &&
161 tarch::la::equals(TestLammps<dim>::_lammps->atom->x[i][d],
162 posDeleteAtom[d], tolerance);
163 }
164 if (found) {
165 std::cout << "ERROR TestLammpsAddDeleteAtom::run(): Found atom in "
166 "LAMMPS though it should have been deleted!"
167 << std::endl;
168 exit(EXIT_FAILURE);
169 }
170 }
171
172 delete iterator;
173 } // finish deletion tests
174
175 if (rank == 0) {
176 std::cout << "Print molecules after deletion test..." << std::endl;
177 }
178 TestLammps<dim>::printMolecules();
179
180 // TEST ADDING A MOLECULE
181 // ---------------------------------------------------------
182 if (performAdd) {
183 const int numberAtoms = TestLammps<dim>::_lammps->atom->nlocal;
184 // convert global to local cell index
185 addCellIndex =
186 indexConversion.convertGlobalToLocalVectorCellIndex(addCellIndex);
187 LAMMPS_NS::MamicoCell &addCell =
188 coupling::interface::MamicoInterfaceProvider<LAMMPS_NS::MamicoCell,
189 dim>::getInstance()
190 .getMDSolverInterface()
191 ->getLinkedCell(addCellIndex, linkedCellInMacroscopicCell,
192 linkedCellsPerMacroscopicCell, indexConversion);
193 coupling::interface::MoleculeIterator<LAMMPS_NS::MamicoCell,
194 dim> *iterator =
195 coupling::interface::MamicoInterfaceProvider<LAMMPS_NS::MamicoCell,
196 dim>::getInstance()
197 .getMDSolverInterface()
198 ->getMoleculeIterator(addCell);
199 int cellCounterBeforeAdding = 0;
200 int cellCounterAfterAdding = 0;
201
202 // determine number of molecules in this cell before adding new one and
203 // plot molecule positions
204 std::cout << "Molecules before adding:" << std::endl;
205 for (iterator->begin(); iterator->continueIteration(); iterator->next()) {
206 std::cout << iterator->getConst().getPosition() << ", "
207 << iterator->getConst().getVelocity() << ", "
208 << iterator->getConst().getForce() << std::endl;
209 cellCounterBeforeAdding++;
210 };
211
212 // add molecule
213 coupling::interface::MamicoInterfaceProvider<LAMMPS_NS::MamicoCell,
214 dim>::getInstance()
215 .getMDSolverInterface()
216 ->addMoleculeToMDSimulation(addMolecule);
217
218 // check if number of atoms is now equal again
219 if (numberAtoms + 1 != TestLammps<dim>::_lammps->atom->nlocal) {
220 std::cout << "ERROR TestLammpsAddDeleteAtom::run(): Same number of "
221 "atoms after adding new molecule as before!"
222 << std::endl;
223 exit(EXIT_FAILURE);
224 }
225 // check if deleted atom can be found in mamico cell
226 std::cout << "Molecules after adding new one:" << std::endl;
227 for (iterator->begin(); iterator->continueIteration(); iterator->next()) {
228 std::cout << iterator->getConst().getPosition() << ", "
229 << iterator->getConst().getVelocity() << ", "
230 << iterator->getConst().getForce() << std::endl;
231 cellCounterAfterAdding++;
232 }
233 if (cellCounterAfterAdding != cellCounterBeforeAdding + 1) {
234 int rank;
235 MPI_Comm_rank(MPI_COMM_WORLD, &rank);
236 std::cout << "Rank " << rank
237 << ": ERROR TestLammpsAddDeleteAtom::run(): Did not increase "
238 "number of molecules in the respective mamico cell!"
239 << std::endl;
240 exit(EXIT_FAILURE);
241 }
242
243 // check if new atom can be found in LAMMPS
244 bool foundMolecule = false;
245 for (int i = 0; i < TestLammps<dim>::_lammps->atom->nlocal; i++) {
246 bool found = true;
247 for (unsigned int d = 0; d < dim; d++) {
248 found = found &&
249 tarch::la::equals(TestLammps<dim>::_lammps->atom->x[i][d],
250 posAddAtom[d], tolerance);
251 found = found &&
252 tarch::la::equals(TestLammps<dim>::_lammps->atom->v[i][d],
253 velAddAtom[d], tolerance);
254 found = found &&
255 tarch::la::equals(TestLammps<dim>::_lammps->atom->f[i][d],
256 fAddAtom[d], tolerance);
257 }
258 if (found) {
259 foundMolecule = true;
260 break;
261 }
262 }
263 if (!foundMolecule) {
264 std::cout << "ERROR TestLammpsAddDeleteAtom::run(): Could not find new "
265 "atom in LAMMPS!"
266 << std::endl;
267 exit(EXIT_FAILURE);
268 }
269
270 delete iterator;
271 } // finish adding molecule test
272
273 // test synchronisation of mass across processes
274 {
275 const int totalNumberAtoms = TestLammps<dim>::_lammps->atom->natoms;
276 if (dim == 2) {
277 if (totalNumberAtoms != 4) {
278 std::cout << "ERROR TestLammpsAddDeleteAtom: total number of atoms="
279 << totalNumberAtoms << " instead of 4" << std::endl;
280 exit(EXIT_FAILURE);
281 }
282 }
283 if (dim == 3) {
284 if (totalNumberAtoms != 8) {
285 std::cout << "ERROR TestLammpsAddDeleteAtom: total number of atoms="
286 << totalNumberAtoms << " instead of 8" << std::endl;
287 exit(EXIT_FAILURE);
288 }
289 }
290 coupling::interface::MamicoInterfaceProvider<LAMMPS_NS::MamicoCell,
291 dim>::getInstance()
292 .getMDSolverInterface()
293 ->synchronizeMoleculesAfterMassModification();
294 const int totalNumberAtomsAfterInsertion =
295 TestLammps<dim>::_lammps->atom->natoms;
296 if (totalNumberAtomsAfterInsertion != totalNumberAtoms) {
297 std::cout << "ERROR TestLammpsAddDeleteAtom: total number of atoms has "
298 "changed; new total number of atoms: "
299 << totalNumberAtomsAfterInsertion << "!" << std::endl;
300 exit(EXIT_FAILURE);
301 }
302 }
303
304 if (rank == 0) {
305 std::cout << "Print molecules after insertion test..." << std::endl;
306 }
307 TestLammps<dim>::printMolecules();
308
309 // do one more time step
310 TestLammps<dim>::_lammps->input->one("run 1");
311 }
312};
