12template <unsigned int dim> class TestLammpsAddDeleteAtom : public TestLammps<dim> {
13public:
14 TestLammpsAddDeleteAtom(int argc, char** argv, std::string name) : TestLammps<dim>(argc, argv, name) {}

17 virtual void run() {
18 // LOAD COUPLED SCENARIO
19 // ----------------------------------------------------------------------
20 if (dim == 2) {
21 TestLammps<dim>::loadLammpsTestConfiguration("inputpositionsonly2D.xyz", 4);
22 } else {
23 TestLammps<dim>::loadLammpsTestConfiguration("inputpositionsonly3D.xyz", 8);
24 }
25 TestLammps<dim>::loadMacroscopicSolverConfiguration();
26 TestLammps<dim>::loadMamicoTestConfiguration();
27 std::cout << "All configs, solvers and coupling loaded and initialised..." << std::endl;
28
29 // DEFINE PROPERTIES OF ATOM TO BE ADDED/DELETED
30 // ---------------------------------------------- helper variables
31 const tarch::la::Vector<dim, unsigned int> linkedCellsPerMacroscopicCell(1);
32 const tarch::la::Vector<dim, unsigned int> linkedCellInMacroscopicCell(0);
33 const coupling::IndexConversion<dim>& indexConversion = TestLammps<dim>::_macroscopicCellService->getIndexConversion();
34 const double tolerance = 1.0e-8; // this value should be the same as tolerance defined in
35 // MDSolverInterface of LAMMPS
36 int rank, size;
37 MPI_Comm_size(MPI_COMM_WORLD, &size);
38 MPI_Comm_rank(MPI_COMM_WORLD, &rank);
39
40 // position and dummy values of atom to be deleted
41 tarch::la::Vector<dim, double> posDeleteAtom(0.0);
42 posDeleteAtom[0] = 4.0;
43 posDeleteAtom[1] = 2.0;
44 if (dim != 2)
45 posDeleteAtom[2] = 2.0;
46 const tarch::la::Vector<dim, double> dummyVec(0.0);
47 const double dummyEnergy = 0.0;
48 const coupling::datastructures::Molecule<dim> deleteMolecule(posDeleteAtom, dummyVec, dummyVec, dummyEnergy);
49 tarch::la::Vector<dim, unsigned int> deleteCellIndex(1);
50 deleteCellIndex[0] = 2; // global cell index for deletion
51 // true, if this is the respective rank that should contain this molecule
52 const bool performDeletion =
53 (size == 1) || ((size == 4) && (rank == 0)) || ((size == 16) && (rank == 1)) || ((size == 8) && (rank == 0)) || ((size == 64) && (rank == 1));
54
55 // position, velocity, force of atom to be inserted
56 tarch::la::Vector<dim, double> posAddAtom(3.6);
57 const tarch::la::Vector<dim, double> velAddAtom(5.0);
58 const tarch::la::Vector<dim, double> fAddAtom(10.0);
59 const coupling::datastructures::Molecule<dim> addMolecule(posAddAtom, velAddAtom, fAddAtom, dummyEnergy);
60 tarch::la::Vector<dim, unsigned int> addCellIndex(2); // global cell index for adding atom
61 // true, if this rank holds the respective domain containing the new
62 // molecule's position
63 const bool performAdd =
64 (size == 1) || ((size == 4) && (rank == 0)) || ((size == 16) && (rank == 5)) || ((size == 8) && (rank == 0)) || ((size == 64) && (rank == 21));
65
66 // set a very small time step (so that molecules hardly move and positions
67 // thus remain quasi identical) and run one time step to sort molecules into
68 // mamico cells
69 // TestLammps<dim>::_lammps->input->one("timestep 1e-14");
70 TestLammps<dim>::_lammps->input->one("run 1");
71
72 // for testing purposes, set cut-off radius to 2.5 (=mamico-cell-size); we
73 // can then check whether the newly inserted/deleted atoms are transferred
74 // to the neighbouring processes in the
75 // synchronizeMoleculesAfterMassModification()-method.
76 // TestLammps<dim>::_lammps->input->one("pair_style lj/cut 2.5");
77 // TestLammps<dim>::_lammps->input->one("pair_coeff 1 1 1.0 1.0 2.5");
78
79 // print out all molecules for each rank (synchronised)
80 TestLammps<dim>::printMolecules();
81
82 // PERFORM DELETION TEST
83 // -----------------------------------------------------------------------------------
84 // delete atom
85 if (performDeletion) {
86 const int numberAtoms = TestLammps<dim>::_lammps->atom->nlocal;
87 // convert global to local coordinate
88 deleteCellIndex = indexConversion.convertGlobalToLocalVectorCellIndex(deleteCellIndex);
89 LAMMPS_NS::MamicoCell& deleteCell =
90 coupling::interface::MamicoInterfaceProvider<LAMMPS_NS::MamicoCell, dim>::getInstance().getMDSolverInterface()->getLinkedCell(
91 deleteCellIndex, linkedCellInMacroscopicCell, linkedCellsPerMacroscopicCell, indexConversion);
92 coupling::interface::MoleculeIterator<LAMMPS_NS::MamicoCell, dim>* iterator =
93 coupling::interface::MamicoInterfaceProvider<LAMMPS_NS::MamicoCell, dim>::getInstance().getMDSolverInterface()->getMoleculeIterator(deleteCell);
94 int cellCounterBeforeDeletion = 0;
95 int cellCounterAfterDeletion = 0;
96
97 // determine number of molecules in this cell before deletion and plot
98 // molecule positions
99 std::cout << "Molecules before deletion:" << std::endl;
100 for (iterator->begin(); iterator->continueIteration(); iterator->next()) {
101 std::cout << iterator->getConst().getPosition() << std::endl;
102 cellCounterBeforeDeletion++;
103 };
104
105 // delete molecule
106 coupling::interface::MamicoInterfaceProvider<LAMMPS_NS::MamicoCell, dim>::getInstance().getMDSolverInterface()->deleteMoleculeFromMDSimulation(
107 deleteMolecule, deleteCell);
108
109 // check if number of atoms has been decremented
110 if (numberAtoms - 1 != TestLammps<dim>::_lammps->atom->nlocal) {
111 std::cout << "ERROR TestLammpsAddDeleteAtom::run(): Same number of "
112 "atoms after deletion as before!"
113 << std::endl;
114 exit(EXIT_FAILURE);
115 }
116 // check if deleted atom can be found in mamico cell
117 std::cout << "Molecules after deletion:" << std::endl;
118 for (iterator->begin(); iterator->continueIteration(); iterator->next()) {
119 std::cout << iterator->getConst().getPosition() << std::endl;
120 cellCounterAfterDeletion++;
121 }
122 if (cellCounterAfterDeletion != cellCounterBeforeDeletion - 1) {
123 int rank;
124 MPI_Comm_rank(MPI_COMM_WORLD, &rank);
125 std::cout << "Rank " << rank
126 << ": ERROR TestLammpsAddDeleteAtom::run(): Did not reduce "
127 "number of molecules in the respective mamico cell!"
128 << std::endl;
129 exit(EXIT_FAILURE);
130 }
131
132 // check if deleted atom can be found anywhere in LAMMPS
133 for (int i = 0; i < TestLammps<dim>::_lammps->atom->nlocal; i++) {
134 bool found = true;
135 for (unsigned int d = 0; d < dim; d++) {
136 found = found && tarch::la::equals(TestLammps<dim>::_lammps->atom->x[i][d], posDeleteAtom[d], tolerance);
137 }
138 if (found) {
139 std::cout << "ERROR TestLammpsAddDeleteAtom::run(): Found atom in "
140 "LAMMPS though it should have been deleted!"
141 << std::endl;
142 exit(EXIT_FAILURE);
143 }
144 }
145
146 delete iterator;
147 } // finish deletion tests
148
149 if (rank == 0) {
150 std::cout << "Print molecules after deletion test..." << std::endl;
151 }
152 TestLammps<dim>::printMolecules();
153
154 // TEST ADDING A MOLECULE
155 // ---------------------------------------------------------
156 if (performAdd) {
157 const int numberAtoms = TestLammps<dim>::_lammps->atom->nlocal;
158 // convert global to local cell index
159 addCellIndex = indexConversion.convertGlobalToLocalVectorCellIndex(addCellIndex);
160 LAMMPS_NS::MamicoCell& addCell =
161 coupling::interface::MamicoInterfaceProvider<LAMMPS_NS::MamicoCell, dim>::getInstance().getMDSolverInterface()->getLinkedCell(
162 addCellIndex, linkedCellInMacroscopicCell, linkedCellsPerMacroscopicCell, indexConversion);
163 coupling::interface::MoleculeIterator<LAMMPS_NS::MamicoCell, dim>* iterator =
164 coupling::interface::MamicoInterfaceProvider<LAMMPS_NS::MamicoCell, dim>::getInstance().getMDSolverInterface()->getMoleculeIterator(addCell);
165 int cellCounterBeforeAdding = 0;
166 int cellCounterAfterAdding = 0;
167
168 // determine number of molecules in this cell before adding new one and
169 // plot molecule positions
170 std::cout << "Molecules before adding:" << std::endl;
171 for (iterator->begin(); iterator->continueIteration(); iterator->next()) {
172 std::cout << iterator->getConst().getPosition() << ", " << iterator->getConst().getVelocity() << ", " << iterator->getConst().getForce() << std::endl;
173 cellCounterBeforeAdding++;
174 };
175
176 // add molecule
177 coupling::interface::MamicoInterfaceProvider<LAMMPS_NS::MamicoCell, dim>::getInstance().getMDSolverInterface()->addMoleculeToMDSimulation(addMolecule);
178
179 // check if number of atoms is now equal again
180 if (numberAtoms + 1 != TestLammps<dim>::_lammps->atom->nlocal) {
181 std::cout << "ERROR TestLammpsAddDeleteAtom::run(): Same number of "
182 "atoms after adding new molecule as before!"
183 << std::endl;
184 exit(EXIT_FAILURE);
185 }
186 // check if deleted atom can be found in mamico cell
187 std::cout << "Molecules after adding new one:" << std::endl;
188 for (iterator->begin(); iterator->continueIteration(); iterator->next()) {
189 std::cout << iterator->getConst().getPosition() << ", " << iterator->getConst().getVelocity() << ", " << iterator->getConst().getForce() << std::endl;
190 cellCounterAfterAdding++;
191 }
192 if (cellCounterAfterAdding != cellCounterBeforeAdding + 1) {
193 int rank;
194 MPI_Comm_rank(MPI_COMM_WORLD, &rank);
195 std::cout << "Rank " << rank
196 << ": ERROR TestLammpsAddDeleteAtom::run(): Did not increase "
197 "number of molecules in the respective mamico cell!"
198 << std::endl;
199 exit(EXIT_FAILURE);
200 }
201
202 // check if new atom can be found in LAMMPS
203 bool foundMolecule = false;
204 for (int i = 0; i < TestLammps<dim>::_lammps->atom->nlocal; i++) {
205 bool found = true;
206 for (unsigned int d = 0; d < dim; d++) {
207 found = found && tarch::la::equals(TestLammps<dim>::_lammps->atom->x[i][d], posAddAtom[d], tolerance);
208 found = found && tarch::la::equals(TestLammps<dim>::_lammps->atom->v[i][d], velAddAtom[d], tolerance);
209 found = found && tarch::la::equals(TestLammps<dim>::_lammps->atom->f[i][d], fAddAtom[d], tolerance);
210 }
211 if (found) {
212 foundMolecule = true;
213 break;
214 }
215 }
216 if (!foundMolecule) {
217 std::cout << "ERROR TestLammpsAddDeleteAtom::run(): Could not find new "
218 "atom in LAMMPS!"
219 << std::endl;
220 exit(EXIT_FAILURE);
221 }
222
223 delete iterator;
224 } // finish adding molecule test
225
226 // test synchronisation of mass across processes
227 {
228 const int totalNumberAtoms = TestLammps<dim>::_lammps->atom->natoms;
229 if (dim == 2) {
230 if (totalNumberAtoms != 4) {
231 std::cout << "ERROR TestLammpsAddDeleteAtom: total number of atoms=" << totalNumberAtoms << " instead of 4" << std::endl;
232 exit(EXIT_FAILURE);
233 }
234 }
235 if (dim == 3) {
236 if (totalNumberAtoms != 8) {
237 std::cout << "ERROR TestLammpsAddDeleteAtom: total number of atoms=" << totalNumberAtoms << " instead of 8" << std::endl;
238 exit(EXIT_FAILURE);
239 }
240 }
241 coupling::interface::MamicoInterfaceProvider<LAMMPS_NS::MamicoCell, dim>::getInstance()
242 .getMDSolverInterface()
243 ->synchronizeMoleculesAfterMassModification();
244 const int totalNumberAtomsAfterInsertion = TestLammps<dim>::_lammps->atom->natoms;
245 if (totalNumberAtomsAfterInsertion != totalNumberAtoms) {
246 std::cout << "ERROR TestLammpsAddDeleteAtom: total number of atoms has "
247 "changed; new total number of atoms: "
248 << totalNumberAtomsAfterInsertion << "!" << std::endl;
249 exit(EXIT_FAILURE);
250 }
251 }
252
253 if (rank == 0) {
254 std::cout << "Print molecules after insertion test..." << std::endl;
255 }
256 TestLammps<dim>::printMolecules();
257
258 // do one more time step
259 TestLammps<dim>::_lammps->input->one("run 1");
260 }
261};
