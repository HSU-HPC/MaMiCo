198 template <unsigned int size, class T> static void readVector(tarch::la::Vector<size, T> &result, tinyxml2::XMLElement *node, std::string tag) {
199 const char *myText = node->Attribute(tag.c_str());
200 if (myText == NULL) {
201 std::cout << "Error while reading mandatory argument " << tag << " of XML element " << node->Name() << std::endl;
202 exit(EXIT_FAILURE);
203 }
204 std::string input(myText);
205 for (unsigned int i = 0; i < size; i++) {
206 // search for first non-whitespace entry in this vector entry
207 std::size_t first = input.find_first_not_of(" ");
208 // search for end of this vector component, typically denoted by ";"
209 // -> if this is the last entry, then npos is accepted as well
210 std::size_t last = input.find_first_of(";");
211 // for debugging
212 // std::cout << first << ", " << last << std::endl;
213 if ((i == size - 1) && (last == std::string::npos)) {
214 last = input.size();
215 }
216
217 std::stringstream ss(input.substr(first, last - first));
218 // for debugging
219 // std::cout << ss.str() << std::endl;
220 ss >> result[i];
221 if (i < size - 1) {
222 input = input.substr(last + 1, input.size() - last - 1);
223 }
224 // for debugging
225 // std::cout << input << std::endl;
226 }
227 }
