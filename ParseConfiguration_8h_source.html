129 static void readBoolMandatory(bool &storage, tinyxml2::XMLElement *node,
130 std::string tag) {

217 template <unsigned int size, class T>
218 static void readVector(tarch::la::Vector<size, T> &result,
219 tinyxml2::XMLElement *node, std::string tag) {
220 const char *myText = node->Attribute(tag.c_str());
221 if (myText == NULL) {
222 std::cout << "Error while reading mandatory argument " << tag
223 << " of XML element " << node->Name() << std::endl;
224 exit(EXIT_FAILURE);
225 }
226 std::string input(myText);
227 for (unsigned int i = 0; i < size; i++) {
228 // search for first non-whitespace entry in this vector entry
229 std::size_t first = input.find_first_not_of(" ");
230 // search for end of this vector component, typically denoted by ";"
231 // -> if this is the last entry, then npos is accepted as well
232 std::size_t last = input.find_first_of(";");
233 // for debugging
234 // std::cout << first << ", " << last << std::endl;
235 if ((i == size - 1) && (last == std::string::npos)) {
236 last = input.size();
237 }
238
239 std::stringstream ss(input.substr(first, last - first));
240 // for debugging
241 // std::cout << ss.str() << std::endl;
242 ss >> result[i];
243 if (i < size - 1) {
244 input = input.substr(last + 1, input.size() - last - 1);
245 }
246 // for debugging
247 // std::cout << input << std::endl;
248 }
249 }
