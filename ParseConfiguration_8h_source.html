178template<unsigned int size,class T>
179static void readVector(tarch::la::Vector<size,T>& result, tinyxml2::XMLElement *node, std::string tag){
180 const char *myText = node->Attribute(tag.c_str());
181 if (myText == NULL){
182 std::cout << "Error while reading mandatory argument " << tag << " of XML element " << node->Name() << std::endl; exit(EXIT_FAILURE);
183 }
184 std::string input(myText);
185 for (unsigned int i = 0; i < size; i++){
186 // search for first non-whitespace entry in this vector entry
187 std::size_t first = input.find_first_not_of(" ");
188 // search for end of this vector component, typically denoted by ";"
189 // -> if this is the last entry, then npos is accepted as well
190 std::size_t last = input.find_first_of(";");
191 // for debugging
192 //std::cout << first << ", " << last << std::endl;
193 if ((i==size-1) && (last==std::string::npos)){last = input.size();}
194
195 std::stringstream ss(input.substr(first,last-first));
196 // for debugging
197 //std::cout << ss.str() << std::endl;
198 ss >> result[i];
199 if (i<size-1){ input = input.substr(last+1,input.size()-last-1); }
200 // for debugging
201 //std::cout << input << std::endl;
202 }
203}
