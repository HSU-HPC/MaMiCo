46 void handleCell(LinkedCell& cell,const unsigned int &cellIndex){
47 coupling::interface::MoleculeIterator<LinkedCell,dim> *it = _mdSolverInterface->getMoleculeIterator(cell);
48 it->begin();
49 while(it->continueIteration()){
50 coupling::interface::Molecule<dim> &wrapper(it->get());
51 tarch::la::Vector<dim,double> velocity = wrapper.getVelocity();
52 // scale velocity
53 velocity = _meanVelocity + _factor*(velocity - _meanVelocity);
54 wrapper.setVelocity(velocity);
55 it->next();
56 }
57 delete it;
58 }

61 double getScalingFactor(const double& oldTemperature, const double &newTemperature) const {
62 // only allow re-scaling if the original temperature was not zero (can happen in empty cells that just became populated)
63 if (oldTemperature!= 0.0){
64 return sqrt(newTemperature/oldTemperature);
65 } else {
66 return 1.0;
67 }
68 }
