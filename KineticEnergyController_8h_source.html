74 void setKineticEnergy(
75 coupling::datastructures::MacroscopicCellWithLinkedCells<LinkedCell, dim>
76 &cell,
77 const double &kineticEnergy) const {
78 // determine mass, momentum and old kinetic energy
79 coupling::cellmappings::ComputeMassMapping<LinkedCell, dim>
80 computeMassMapping(_mdSolverInterface);
81 cell.iterateConstCells(computeMassMapping);
82 coupling::cellmappings::ComputeMomentumMapping<LinkedCell, dim>
83 computeMomentumMapping(_mdSolverInterface);
84 cell.iterateConstCells(computeMomentumMapping);
85 coupling::cellmappings::ComputeKineticEnergyMapping<LinkedCell, dim>
86 computeKineticEnergyMapping(_mdSolverInterface);
87 cell.iterateConstCells(computeKineticEnergyMapping);
88 // set new kinetic energy
89 unsigned int numberParticles = computeMassMapping.getNumberOfParticles();
90 tarch::la::Vector<dim, double> meanVelocity =
91 computeMomentumMapping.getMeanVelocity();
92 double oldKineticEnergy = computeKineticEnergyMapping.getKineticEnergy();
93 coupling::cellmappings::SetKineticEnergyMapping<LinkedCell, dim>
94 setKineticEnergyMapping(oldKineticEnergy, kineticEnergy,
95 numberParticles, meanVelocity);
96 cell.iterateCells(setKineticEnergyMapping);
97 }

105 void setTemperature(
106 coupling::datastructures::MacroscopicCellWithLinkedCells<LinkedCell, dim>
107 &cell,
108 const double &temperature) const {
109 coupling::cellmappings::ComputeMomentumMapping<LinkedCell, dim>
110 computeMomentumMapping(_mdSolverInterface);
111 cell.iterateConstCells(computeMomentumMapping);
112 tarch::la::Vector<dim, double> meanVelocity =
113 computeMomentumMapping.getMeanVelocity();
114
115 coupling::cellmappings::ComputeTemperatureMapping<LinkedCell, dim>
116 computeTemperatureMapping(meanVelocity, _mdSolverInterface);
117 cell.iterateConstCells(computeTemperatureMapping);
118 double currentTemperature = computeTemperatureMapping.getTemperature();
119
120 coupling::cellmappings::SetTemperatureMapping<LinkedCell, dim>
121 setTemperatureMapping(currentTemperature, temperature, meanVelocity,
122 _mdSolverInterface);
123 cell.iterateCells(setTemperatureMapping);
124 }

129 *const _mdSolverInterface;
130};
