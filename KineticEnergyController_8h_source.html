62 void setKineticEnergy(coupling::datastructures::MacroscopicCellWithLinkedCells<LinkedCell, dim>& cell, const double& kineticEnergy) const {
63 // determine mass, momentum and old kinetic energy
64 coupling::cellmappings::ComputeMassMapping<LinkedCell, dim> computeMassMapping(_mdSolverInterface);
65 cell.iterateConstCells(computeMassMapping);
66 coupling::cellmappings::ComputeMomentumMapping<LinkedCell, dim> computeMomentumMapping(_mdSolverInterface);
67 cell.iterateConstCells(computeMomentumMapping);
68 coupling::cellmappings::ComputeKineticEnergyMapping<LinkedCell, dim> computeKineticEnergyMapping(_mdSolverInterface);
69 cell.iterateConstCells(computeKineticEnergyMapping);
70 // set new kinetic energy
71 unsigned int numberParticles = computeMassMapping.getNumberOfParticles();
72 tarch::la::Vector<dim, double> meanVelocity = computeMomentumMapping.getMeanVelocity();
73 double oldKineticEnergy = computeKineticEnergyMapping.getKineticEnergy();
74 coupling::cellmappings::SetKineticEnergyMapping<LinkedCell, dim> setKineticEnergyMapping(oldKineticEnergy, kineticEnergy, numberParticles, meanVelocity);
75 cell.iterateCells(setKineticEnergyMapping);
76 }
