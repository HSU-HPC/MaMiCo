102 virtual void writeToCell(const double* const buffer, coupling::datastructures::MacroscopicCell<dim>& cell) {
103 tarch::la::Vector<dim, double> macroscopicMomentum(0.0);
104 for (unsigned int d = 0; d < dim; d++) {
105 macroscopicMomentum[d] = buffer[1 + d];
106 }
107 cell.setMacroscopicMomentum(macroscopicMomentum);
108 cell.setMacroscopicMass(buffer[0]);
109 }
