36 virtual std::vector<unsigned int> getTargetRanks(tarch::la::Vector<dim,unsigned int> globalCellIndex){
37 bool isSource = true;
38 std::vector<unsigned int> vec;
39 // empty target ranks if this is a global cell at the lower or upper end of the diagonal (inside ghost layer)
40 if ( (globalCellIndex==tarch::la::Vector<dim,unsigned int>(0))
41 || (globalCellIndex==_indexConversion.getGlobalNumberMacroscopicCells()+tarch::la::Vector<dim,unsigned int>(1)) ){
42 return vec;
43 }
44 const unsigned int rank = _indexConversion.getUniqueRankForMacroscopicCell(globalCellIndex);
45 int numberProcesses = 1;
46 #if (COUPLING_MD_PARALLEL==COUPLING_MD_YES)
47 MPI_Comm_size(MPI_COMM_WORLD,&numberProcesses);
48 #endif
49
50 for (unsigned int d = 1; d < dim; d++){
51 isSource = isSource && (globalCellIndex[d] == globalCellIndex[0]);
52 }
53
54 // if diagonal AND this is not the last rank, add rank+1 as target
55 if (isSource){
56 vec.push_back( (rank+1)%numberProcesses );
57 }
58 return vec;
59 }

65 virtual std::vector<unsigned int> getSourceRanks(tarch::la::Vector<dim,unsigned int> globalCellIndex){
66 bool isSource = true;
67 std::vector<unsigned int> vec;
68 // empty source ranks if this is a global cell at the lower or upper end of the diagonal (inside ghost layer)
69 if ( (globalCellIndex==tarch::la::Vector<dim,unsigned int>(0))
70 || (globalCellIndex==_indexConversion.getGlobalNumberMacroscopicCells()+tarch::la::Vector<dim,unsigned int>(1)) ){
71 return vec;
72 }
73
74
75 const unsigned int rank = _indexConversion.getUniqueRankForMacroscopicCell(globalCellIndex);
76
77 for (unsigned int d = 1; d < dim; d++){
78 isSource = isSource && (globalCellIndex[d] == globalCellIndex[0]);
79 }

83 vec.push_back(rank);
84 }

115 virtual void getBuffer4MacroscopicSolverCells(unsigned int &numberCells, std::vector<TestCell<dim>* > &cells, unsigned int * &globalCellIndices){
116 // for rank 0: expect data from last rank
117 if (_indexConversion.getThisRank() == 0){
118 // determine number of cells on last rank (this only works for cubic sub-domains) and start coordinate of first
119 // (non-ghost) global cell of last rank
120 const tarch::la::Vector<dim,unsigned int> start =
121 tarch::la::Vector<dim,unsigned int> (1) +
122 (_indexConversion.getNumberProcesses()-tarch::la::Vector<dim,unsigned int>(1))*
123 _indexConversion.getAverageLocalNumberMacroscopicCells()[0];
124
125 numberCells = (_indexConversion.getGlobalNumberMacroscopicCells()
126 - (_indexConversion.getNumberProcesses()-tarch::la::Vector<dim,unsigned int>(1))*
127 _indexConversion.getAverageLocalNumberMacroscopicCells()[0] )[0];
128 for (unsigned int i = 0; i < numberCells; i++){
129 cells.push_back(new TestCell<dim>());
130 if (cells[i]==NULL){std::cout << "ERROR getBuffer4MacroscopicSolverCells: cells[i]==NULL!" << std::endl; exit(EXIT_FAILURE);}
131 }
132 globalCellIndices = new unsigned int [numberCells];
133 if ( globalCellIndices==NULL ){
134 std::cout << "ERROR TestDataExchangeFromMD2Macro::getBuffer4ReceivedCells: NULL ptr returned!" << std::endl;
135 exit(EXIT_FAILURE);
136 }
137
138 for (unsigned int i = 0; i < numberCells; i++){
139 globalCellIndices[i] = _indexConversion.getGlobalCellIndex(start+i*tarch::la::Vector<dim,unsigned int>(1));
140 }
141 // all other cases (rank != 0)
142 } else {
143
144 // determine if we obtain information from the neighbouring rank
145 const unsigned int neighbourRank = _indexConversion.getThisRank() - 1;
146 const tarch::la::Vector<dim,unsigned int> neighbourRankCoords = _indexConversion.getProcessCoordinates(neighbourRank);
147 bool isSource = true;
148 for (unsigned int d = 1; d < dim; d++){
149 isSource = isSource && ( neighbourRankCoords[0] == neighbourRankCoords[d] );
150 }
151
152 if (isSource){
153 // we only plan with cubic sub-domains on the processes...
154 const unsigned int numberLocalCells = _indexConversion.getAverageLocalNumberMacroscopicCells()[0];
155
156 globalCellIndices = new unsigned int [numberLocalCells];
157 for (unsigned int i = 0; i < numberLocalCells; i++){
158 cells.push_back(new TestCell<dim>());
159 if (cells[i] ==NULL){std::cout << "ERROR getBuffer4MacroscopicSolverCells: cells[i]==NULL!" << std::endl; exit(EXIT_FAILURE);}
160 }
161 if ( globalCellIndices==NULL ){
162 std::cout << "ERROR TestDataExchangeFromMD2Macro::getBuffer4ReceivedCells: NULL ptrs returned!" << std::endl;
163 exit(EXIT_FAILURE);
164 }
165 numberCells = numberLocalCells;
166
167 // determine start coordinates of neighbouring local cells
168 tarch::la::Vector<dim,unsigned int> start(1);
169 start = start + neighbourRankCoords*numberLocalCells;
170
171 for (unsigned int i = 0; i< numberLocalCells; i++){
172 globalCellIndices[i] = _indexConversion.getGlobalCellIndex(start+i*tarch::la::Vector<dim,unsigned int>(1));
173 }
174 }
175 }
176 }

178 protected:
179 const coupling::IndexConversion<dim> &_indexConversion;
