37 virtual std::vector<unsigned int>
38 getTargetRanks(tarch::la::Vector<dim, unsigned int> globalCellIndex) {
39 bool isSource = true;
40 std::vector<unsigned int> vec;
41 // empty target ranks if this is a global cell at the lower or upper end of
42 // the diagonal (inside ghost layer)
43 if ((globalCellIndex == tarch::la::Vector<dim, unsigned int>(0)) ||
44 (globalCellIndex == _indexConversion.getGlobalNumberMacroscopicCells() +
45 tarch::la::Vector<dim, unsigned int>(1))) {
46 return vec;
47 }
48 const unsigned int rank =
49 _indexConversion.getUniqueRankForMacroscopicCell(globalCellIndex);
50 int numberProcesses = 1;
51#if (COUPLING_MD_PARALLEL == COUPLING_MD_YES)
52 MPI_Comm_size(MPI_COMM_WORLD, &numberProcesses);
53#endif
54
55 for (unsigned int d = 1; d < dim; d++) {
56 isSource = isSource && (globalCellIndex[d] == globalCellIndex[0]);
57 }
58
59 // if diagonal AND this is not the last rank, add rank+1 as target
60 if (isSource) {
61 vec.push_back((rank + 1) % numberProcesses);
62 }
63 return vec;
64 }

69 virtual std::vector<unsigned int>
70 getSourceRanks(tarch::la::Vector<dim, unsigned int> globalCellIndex) {
71 bool isSource = true;
72 std::vector<unsigned int> vec;
73 // empty source ranks if this is a global cell at the lower or upper end of
74 // the diagonal (inside ghost layer)
75 if ((globalCellIndex == tarch::la::Vector<dim, unsigned int>(0)) ||
76 (globalCellIndex == _indexConversion.getGlobalNumberMacroscopicCells() +
77 tarch::la::Vector<dim, unsigned int>(1))) {
78 return vec;
79 }
80
81 const unsigned int rank =
82 _indexConversion.getUniqueRankForMacroscopicCell(globalCellIndex);
83
84 for (unsigned int d = 1; d < dim; d++) {
85 isSource = isSource && (globalCellIndex[d] == globalCellIndex[0]);
86 }
87
88 // source is rank
89 if (isSource) {
90 vec.push_back(rank);
91 }
92 return vec;
93 }

123 virtual void
124 getBuffer4MacroscopicSolverCells(unsigned int &numberCells,
125 std::vector<TestCell<dim> *> &cells,
126 unsigned int *&globalCellIndices) {
127 // for rank 0: expect data from last rank
128 if (_indexConversion.getThisRank() == 0) {
129 // determine number of cells on last rank (this only works for cubic
130 // sub-domains) and start coordinate of first (non-ghost) global cell of
131 // last rank
132 const tarch::la::Vector<dim, unsigned int> start =
133 tarch::la::Vector<dim, unsigned int>(1) +
134 (_indexConversion.getNumberProcesses() -
135 tarch::la::Vector<dim, unsigned int>(1)) *
136 _indexConversion.getAverageLocalNumberMacroscopicCells()[0];
137
138 numberCells =
139 (_indexConversion.getGlobalNumberMacroscopicCells() -
140 (_indexConversion.getNumberProcesses() -
141 tarch::la::Vector<dim, unsigned int>(1)) *
142 _indexConversion.getAverageLocalNumberMacroscopicCells()[0])[0];
143 for (unsigned int i = 0; i < numberCells; i++) {
144 cells.push_back(new TestCell<dim>());
145 if (cells[i] == NULL) {
146 std::cout << "ERROR getBuffer4MacroscopicSolverCells: cells[i]==NULL!"
147 << std::endl;
148 exit(EXIT_FAILURE);
149 }
150 }
151 globalCellIndices = new unsigned int[numberCells];
152 if (globalCellIndices == NULL) {
153 std::cout
154 << "ERROR TestDataExchangeFromMD2Macro::getBuffer4ReceivedCells: "
155 "NULL ptr returned!"
156 << std::endl;
157 exit(EXIT_FAILURE);
158 }
159
160 for (unsigned int i = 0; i < numberCells; i++) {
161 globalCellIndices[i] = _indexConversion.getGlobalCellIndex(
162 start + i * tarch::la::Vector<dim, unsigned int>(1));
163 }
164 // all other cases (rank != 0)
165 } else {
166
167 // determine if we obtain information from the neighbouring rank
168 const unsigned int neighbourRank = _indexConversion.getThisRank() - 1;
169 const tarch::la::Vector<dim, unsigned int> neighbourRankCoords =
170 _indexConversion.getProcessCoordinates(neighbourRank);
171 bool isSource = true;
172 for (unsigned int d = 1; d < dim; d++) {
173 isSource =
174 isSource && (neighbourRankCoords[0] == neighbourRankCoords[d]);
175 }
176
177 if (isSource) {
178 // we only plan with cubic sub-domains on the processes...
179 const unsigned int numberLocalCells =
180 _indexConversion.getAverageLocalNumberMacroscopicCells()[0];
181
182 globalCellIndices = new unsigned int[numberLocalCells];
183 for (unsigned int i = 0; i < numberLocalCells; i++) {
184 cells.push_back(new TestCell<dim>());
185 if (cells[i] == NULL) {
186 std::cout
187 << "ERROR getBuffer4MacroscopicSolverCells: cells[i]==NULL!"
188 << std::endl;
189 exit(EXIT_FAILURE);
190 }
191 }
192 if (globalCellIndices == NULL) {
193 std::cout
194 << "ERROR TestDataExchangeFromMD2Macro::getBuffer4ReceivedCells: "
195 "NULL ptrs returned!"
196 << std::endl;
197 exit(EXIT_FAILURE);
198 }
199 numberCells = numberLocalCells;
200
201 // determine start coordinates of neighbouring local cells
202 tarch::la::Vector<dim, unsigned int> start(1);
203 start = start + neighbourRankCoords * numberLocalCells;
204
205 for (unsigned int i = 0; i < numberLocalCells; i++) {
206 globalCellIndices[i] = _indexConversion.getGlobalCellIndex(
207 start + i * tarch::la::Vector<dim, unsigned int>(1));
208 }
209 }
210 }
211 }
212
213protected:
214 const coupling::IndexConversion<dim> &_indexConversion;
215};
