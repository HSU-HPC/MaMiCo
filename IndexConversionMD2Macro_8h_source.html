138 //assumes lower boundary to be lower than upper
139 tarch::la::Vector<dim, unsigned int> getGlobalMD2MacroDomainSize() const {
140 //Since both lower and upper boundaries are inclusive, we need to add one. operator+(int) in tarch::la:Vector would be great here...
141 auto plus_one = tarch::la::Vector<dim, unsigned int>(1);
142 return *_upperBoundaryAllRanks - *_lowerBoundaryAllRanks + plus_one;
143 }
144 tarch::la::Vector<dim, unsigned int> getLocalMD2MacroDomainSize() const {
145 auto plus_one = tarch::la::Vector<dim, unsigned int>(1);
146 return *_upperBoundaryThisRank - *_lowerBoundaryThisRank + plus_one;
147 }
148
149 tarch::la::Vector<dim, double> getGlobalMD2MacroDomainOffset() const {
150 tarch::la::Vector<dim, double> offset = _ic->getGlobalMDDomainOffset(); //standard MD offset
151
152 if(_lowerBoundaryAllRanks == nullptr)
153 throw std::runtime_error("ERROR: Calling while _lowerBoundaryAllRanks is uninitialized!");
154
155 //offset of md2macro domain relative to MD domain
156 for(unsigned int d = 0; d < dim; d++) {
157 offset[d] += _ic->getMacroscopicCellSize()[d] * (*_lowerBoundaryAllRanks)[d]; //offset of md2macro domain relative to MD domain
158 }
159
160 //std::cout << offset << std::endl << std::endl;
161 return offset;
162 }

179 unsigned int getLocalCellIndex(tarch::la::Vector<dim,unsigned int> localCellIndex) const{
180 auto numberCells = getLocalMD2MacroDomainSize();
181 unsigned int index = localCellIndex[dim-1];
182 for (int d = dim-2; d >-1; d--)
183 index = numberCells[d]*index + localCellIndex[d];
184 return index;
185 }
