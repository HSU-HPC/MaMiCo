164 tarch::la::Vector<dim, unsigned int> getGlobalMD2MacroDomainSize() const {
165 //Since both lower and upper boundaries are inclusive, we need to add one. operator+(int) in tarch::la:Vector would be great here...
166 auto plus_one = tarch::la::Vector<dim, unsigned int>(1);
167 return *_upperBoundaryAllRanks - *_lowerBoundaryAllRanks + plus_one;
168 }

181 tarch::la::Vector<dim, double> getGlobalMD2MacroDomainOffset() const {
182 tarch::la::Vector<dim, double> offset = _ic->getGlobalMDDomainOffset(); //standard MD offset
183
184 if(_lowerBoundaryAllRanks == nullptr)
185 throw std::runtime_error("ERROR: Calling while _lowerBoundaryAllRanks is uninitialized!");
186
187 //offset of md2macro domain relative to MD domain
188 for(unsigned int d = 0; d < dim; d++) {
189 offset[d] += _ic->getMacroscopicCellSize()[d] * (*_lowerBoundaryAllRanks)[d]; //offset of md2macro domain relative to MD domain
190 }
191
192 //std::cout << offset << std::endl << std::endl;
193 return offset;
194 }

227 unsigned int getLocalCellIndex(tarch::la::Vector<dim,unsigned int> localCellIndex) const{
228 auto numberCells = getLocalMD2MacroDomainSize();
229 unsigned int index = localCellIndex[dim-1];
230 for (int d = dim-2; d >-1; d--)
231 index = numberCells[d]*index + localCellIndex[d];
232 return index;
233 }
