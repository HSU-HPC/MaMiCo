176 tarch::la::Vector<dim, unsigned int> getGlobalMD2MacroDomainSize() const {
177 // Since both lower and upper boundaries are inclusive, we need to add one.
178 // operator+(int) in tarch::la:Vector would be great here...
179 auto plus_one = tarch::la::Vector<dim, unsigned int>(1);
180 return *_upperBoundaryAllRanks - *_lowerBoundaryAllRanks + plus_one;
181 }

196 tarch::la::Vector<dim, double> getGlobalMD2MacroDomainOffset() const {
197 tarch::la::Vector<dim, double> offset = _ic->getGlobalMDDomainOffset(); // standard MD offset
198
199 if (_lowerBoundaryAllRanks == nullptr)
200 throw std::runtime_error("ERROR: Calling while _lowerBoundaryAllRanks is uninitialized!");
201
202 // offset of md2macro domain relative to MD domain
203 for (unsigned int d = 0; d < dim; d++) {
204 offset[d] += _ic->getMacroscopicCellSize()[d] * (*_lowerBoundaryAllRanks)[d]; // offset of md2macro domain
205 // relative to MD domain
206 }
207
208 // std::cout << offset << std::endl << std::endl;
209 return offset;
210 }

246 unsigned int getLocalCellIndex(tarch::la::Vector<dim, unsigned int> localCellIndex) const {
247 auto numberCells = getLocalMD2MacroDomainSize();
248 unsigned int index = localCellIndex[dim - 1];
249 for (int d = dim - 2; d > -1; d--)
250 index = numberCells[d] * index + localCellIndex[d];
251 return index;
252 }
