87template <unsigned int dim> tarch::la::Vector<dim, unsigned int> initDivisionFactor(tarch::la::Vector<dim, unsigned int> numberCells) {
88 tarch::la::Vector<dim, unsigned int> divFactor(1);
89 for (unsigned int d = 1; d < dim; d++) {
90 divFactor[d] = divFactor[d - 1] * (numberCells[d - 1]);
91 }
92 return divFactor;
93}

105template <unsigned int dim>
106tarch::la::Vector<dim, unsigned int> getVectorCellIndex(unsigned int cellIndex, const tarch::la::Vector<dim, unsigned int>& divisionFactor) {
107 tarch::la::Vector<dim, unsigned int> myVector(0);
108 unsigned int help = cellIndex;
109 for (int d = dim - 1; d > 0; d--) {
110 myVector[d] = help / divisionFactor[d];
111 help = help - myVector[d] * divisionFactor[d];
112 }
113 myVector[0] = help;
114
115 return myVector;
116}
