83template <unsigned int dim> tarch::la::Vector<dim, unsigned int> initDivisionFactor(tarch::la::Vector<dim, unsigned int> numberCells) {
84 tarch::la::Vector<dim, unsigned int> divFactor(1);
85 for (unsigned int d = 1; d < dim; d++) {
86 divFactor[d] = divFactor[d - 1] * (numberCells[d - 1]);
87 }
88 return divFactor;
89}

101template <unsigned int dim>
102tarch::la::Vector<dim, unsigned int> getVectorCellIndex(unsigned int cellIndex, const tarch::la::Vector<dim, unsigned int>& divisionFactor) {
103 tarch::la::Vector<dim, unsigned int> myVector(0);
104 unsigned int help = cellIndex;
105 for (int d = dim - 1; d > 0; d--) {
106 myVector[d] = help / divisionFactor[d];
107 help = help - myVector[d] * divisionFactor[d];
108 }
109 myVector[0] = help;
110
111 return myVector;
112}
