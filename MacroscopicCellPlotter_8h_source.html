117 void plotMacroscopicCellFile(unsigned int ID, std::string filename, unsigned int rank, unsigned int t, const coupling::IndexConversion<dim>& indexConversion,
118 coupling::datastructures::MacroscopicCells<LinkedCell, dim>& cells,
119 coupling::interface::MDSolverInterface<LinkedCell, dim>* const mdSolverInterface) {
120 // stringstreams for all cell properties of interest
121 std::stringstream microscopicMasses;
122 microscopicMasses << "SCALARS microscopicMassBuffer float 1" << std::endl;
123 microscopicMasses << "LOOKUP_TABLE DEFAULT" << std::endl;
124 std::stringstream macroscopicMasses;
125 macroscopicMasses << "SCALARS macroscopicMassBuffer float 1" << std::endl;
126 macroscopicMasses << "LOOKUP_TABLE DEFAULT" << std::endl;
127 std::stringstream microscopicMomenta;
128 microscopicMomenta << "VECTORS microscopicMomentumBuffer float" << std::endl;
129 std::stringstream macroscopicMomenta;
130 macroscopicMomenta << "VECTORS macroscopicMomentumBuffer float" << std::endl;
131 std::stringstream meanVelocities;
132 meanVelocities << "VECTORS meanVelocity float" << std::endl;
133 std::stringstream masses;
134 masses << "SCALARS mass float 1" << std::endl;
135 masses << "LOOKUP_TABLE DEFAULT" << std::endl;
136 std::stringstream energies;
137 energies << "SCALARS meanPotentialEnergyBuffer float 1" << std::endl;
138 energies << "LOOKUP_TABLE DEFAULT" << std::endl;
139 std::stringstream temperatures;
140 temperatures << "SCALARS temperature float 1" << std::endl;
141 temperatures << "LOOKUP_TABLE DEFAULT" << std::endl;
142 // ofstream for file
143 std::ofstream ofCell;
144 // zeros appended in vtk file
145 std::string appendFloatZeros("");
146 switch (dim) {
147 case 1:
148 appendFloatZeros = " 0.0 0.0";
149 break;
150 case 2:
151 appendFloatZeros = " 0.0";
152 break;
153 case 3:
154 appendFloatZeros = "";
155 break;
156 default:
157 break;
158 }
159 // number of points
160 const tarch::la::Vector<3, unsigned int> numberCells =
161 coupling::initRange<dim>(indexConversion.getLocalNumberMacroscopicCells() + tarch::la::Vector<dim, unsigned int>(2));
162 tarch::la::Vector<3, double> original(0.0);
163 for (unsigned int d = 0; d < dim; d++) {
164 original[d] =
165 indexConversion.getGlobalMDDomainOffset()[d] - 0.5 * indexConversion.getMacroscopicCellSize()[d] +
166 indexConversion.getAverageLocalNumberMacroscopicCells()[d] * indexConversion.getThisProcess()[d] * indexConversion.getMacroscopicCellSize()[d];
167 }
168 // open file
169 open(ID, filename, "MacroscopicCells", rank, t, ofCell);
170 Writer4Cells writer(microscopicMasses, macroscopicMasses, microscopicMomenta, macroscopicMomenta, meanVelocities, masses, energies, temperatures,
171 appendFloatZeros, indexConversion, mdSolverInterface);
172 cells.applyToAllLocalMacroscopicCellsWithLinkedCells(writer);
173 ofCell << "# vtk DataFile Version 2.0" << std::endl;
174 ofCell << "generated by MaMiCo (Philipp Neumann)" << std::endl;
175 ofCell << "ASCII" << std::endl << std::endl;
176 ofCell << "DATASET STRUCTURED_POINTS" << std::endl;
177 ofCell << "DIMENSIONS " << numberCells[0] << " " << numberCells[1] << " " << numberCells[2] << std::endl;
178 ofCell << "ORIGIN ";
179 for (unsigned int d = 0; d < 3; d++) {
180 ofCell << original[d] << " ";
181 }
182 ofCell << std::endl;
183 ofCell << "SPACING ";
184 for (unsigned int d = 0; d < dim; d++) {
185 ofCell << indexConversion.getMacroscopicCellSize()[d] << " ";
186 }
187 for (unsigned int d = dim; d < 3; d++) {
188 ofCell << "1.0 ";
189 }
190 ofCell << std::endl;
191 ofCell << "POINT_DATA " << numberCells[0] * numberCells[1] * numberCells[2] << std::endl << std::endl;
192 ofCell << microscopicMasses.str() << std::endl << std::endl;
193 microscopicMasses.clear();
194 microscopicMasses.str("");
195 ofCell << macroscopicMasses.str() << std::endl << std::endl;
196 macroscopicMasses.clear();
197 macroscopicMasses.str("");
198 ofCell << masses.str() << std::endl << std::endl;
199 masses.clear();
200 masses.str("");
201 ofCell << energies.str() << std::endl << std::endl;
202 energies.clear();
203 energies.str("");
204 ofCell << temperatures.str() << std::endl << std::endl;
205 temperatures.clear();
206 temperatures.str("");
207 ofCell << microscopicMomenta.str() << std::endl << std::endl;
208 microscopicMomenta.clear();
209 microscopicMomenta.str("");
210 ofCell << macroscopicMomenta.str() << std::endl << std::endl;
211 macroscopicMomenta.clear();
212 macroscopicMomenta.str("");
213 ofCell << meanVelocities.str() << std::endl << std::endl;
214 meanVelocities.clear();
215 meanVelocities.str("");
216 // close file
217 ofCell.close();
218 }

277 void apply(coupling::datastructures::MacroscopicCellWithLinkedCells<LinkedCell, dim>& cell, const unsigned int& index) {
278 // compute local quantities from molecules, if this is an inner cell
279 const tarch::la::Vector<dim, unsigned int> end = _indexConversion.getLocalNumberMacroscopicCells() + tarch::la::Vector<dim, unsigned int>(1);
280 const tarch::la::Vector<dim, unsigned int> localIndex = _indexConversion.getLocalVectorCellIndex(index);
281 bool isInnerCell = true;
282 for (unsigned int d = 0; d < dim; d++) {
283 isInnerCell = isInnerCell && (localIndex[d] > 0) && (localIndex[d] < end[d]);
284 }
285 double mass = 0.0;
286 tarch::la::Vector<dim, double> meanVelocity(0.0);
287 double temperature = 0.0;
288
289 if (isInnerCell) {
290 cell.iterateConstCells(_computeMassMapping);
291 mass = _computeMassMapping.getMass();
292 cell.iterateConstCells(_computeMomentumMapping);
293 meanVelocity = _computeMomentumMapping.getMeanVelocity();
294 temperature = _kineticEnergyController.computeTemperature(cell);
295 }
296
297 // write data to streams
298 _microscopicMasses << cell.getMicroscopicMass() << std::endl;
299 _macroscopicMasses << cell.getMacroscopicMass() << std::endl;
300 _masses << mass << std::endl;
301 _temperatures << temperature << std::endl;
302 _energies << cell.getPotentialEnergy() << std::endl;
303 for (unsigned int d = 0; d < dim; d++) {
304 _microscopicMomenta << cell.getMicroscopicMomentum()[d] << " ";
305 _macroscopicMomenta << cell.getMacroscopicMomentum()[d] << " ";
306 _meanVelocities << meanVelocity[d] << " ";
307 }
308 _microscopicMomenta << _appendFloatZeros << std::endl;
309 _macroscopicMomenta << _appendFloatZeros << std::endl;
310 _meanVelocities << _appendFloatZeros << std::endl;
311 }
312

316 std::stringstream& _microscopicMasses;
319 std::stringstream& _macroscopicMasses;
322 std::stringstream& _microscopicMomenta;
325 std::stringstream& _macroscopicMomenta;
327 std::stringstream& _meanVelocities;
329 std::stringstream& _masses;
331 std::stringstream& _energies;
333 std::stringstream& _temperatures;
335 const std::string _appendFloatZeros;
337 coupling::cellmappings::ComputeMassMapping<LinkedCell, dim> _computeMassMapping;
339 coupling::cellmappings::ComputeMomentumMapping<LinkedCell, dim> _computeMomentumMapping;
341 coupling::KineticEnergyController<LinkedCell, dim> _kineticEnergyController;
343 const coupling::IndexConversion<dim>& _indexConversion;
344 };
