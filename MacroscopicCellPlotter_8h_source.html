101 void plotMacroscopicCellFile(
102 unsigned int ID,
103 std::string filename,
104 unsigned int rank,
105 unsigned int t,
106 const coupling::IndexConversion<dim> &indexConversion,
107 coupling::datastructures::MacroscopicCells<LinkedCell,dim> &cells,
108 coupling::interface::MDSolverInterface<LinkedCell,dim> * const mdSolverInterface
109 ){
110 // stringstreams for all cell properties of interest
111 std::stringstream microscopicMasses;
112 microscopicMasses << "SCALARS microscopicMassBuffer float 1" << std::endl;
113 microscopicMasses << "LOOKUP_TABLE DEFAULT" << std::endl;
114 std::stringstream macroscopicMasses;
115 macroscopicMasses << "SCALARS macroscopicMassBuffer float 1" << std::endl;
116 macroscopicMasses << "LOOKUP_TABLE DEFAULT" << std::endl;
117 std::stringstream microscopicMomenta;
118 microscopicMomenta << "VECTORS microscopicMomentumBuffer float" << std::endl;
119 std::stringstream macroscopicMomenta;
120 macroscopicMomenta << "VECTORS macroscopicMomentumBuffer float" << std::endl;
121 std::stringstream meanVelocities;
122 meanVelocities << "VECTORS meanVelocity float" << std::endl;
123 std::stringstream masses;
124 masses << "SCALARS mass float 1" << std::endl;
125 masses << "LOOKUP_TABLE DEFAULT" << std::endl;
126 std::stringstream energies;
127 energies << "SCALARS meanPotentialEnergyBuffer float 1" << std::endl;
128 energies << "LOOKUP_TABLE DEFAULT" << std::endl;
129 std::stringstream temperatures;
130 temperatures << "SCALARS temperature float 1" << std::endl;
131 temperatures << "LOOKUP_TABLE DEFAULT" << std::endl;
132 // ofstream for file
133 std::ofstream ofCell;
134 // zeros appended in vtk file
135 std::string appendFloatZeros("");
136 switch(dim){
137 case 1:
138 appendFloatZeros=" 0.0 0.0";
139 break;
140 case 2:
141 appendFloatZeros=" 0.0";
142 break;
143 case 3:
144 appendFloatZeros="";
145 break;
146 default:
147 break;
148 }
149 // number of points
150 const tarch::la::Vector<3,unsigned int> numberCells = coupling::initRange<dim>(indexConversion.getLocalNumberMacroscopicCells()+tarch::la::Vector<dim,unsigned int>(2));
151 tarch::la::Vector<3,double> original(0.0);
152 for (unsigned int d = 0; d < dim; d++){
153 original[d] = indexConversion.getGlobalMDDomainOffset()[d] - 0.5*indexConversion.getMacroscopicCellSize()[d]
154 + indexConversion.getAverageLocalNumberMacroscopicCells()[d]*indexConversion.getThisProcess()[d]*indexConversion.getMacroscopicCellSize()[d];
155 }
156
157 // open file
158 open(ID,filename,"MacroscopicCells",rank,t,ofCell);
159
160 Writer4Cells writer( microscopicMasses, macroscopicMasses,
161 microscopicMomenta, macroscopicMomenta,
162 meanVelocities, masses,
163 energies, temperatures,
164 appendFloatZeros, indexConversion, mdSolverInterface);
165 cells.applyToAllLocalMacroscopicCellsWithLinkedCells(writer);
166
167 ofCell << "# vtk DataFile Version 2.0" << std::endl;
168 ofCell << "generated by MaMiCo (Philipp Neumann)" << std::endl;
169 ofCell << "ASCII" << std::endl << std::endl;
170 ofCell << "DATASET STRUCTURED_POINTS" << std::endl;
171 ofCell << "DIMENSIONS " << numberCells[0] << " " << numberCells[1] << " " << numberCells[2] << std::endl;
172 ofCell << "ORIGIN ";
173 for (unsigned int d = 0; d < 3; d++){
174 ofCell << original[d] << " ";
175 }
176 ofCell << std::endl;
177 ofCell << "SPACING ";
178 for (unsigned int d = 0; d < dim; d++){
179 ofCell << indexConversion.getMacroscopicCellSize()[d] << " ";
180 }
181 for (unsigned int d = dim; d < 3; d++){
182 ofCell << "1.0 ";
183 }
184 ofCell << std::endl;
185 ofCell << "POINT_DATA " << numberCells[0]*numberCells[1]*numberCells[2] << std::endl << std::endl;
186 ofCell << microscopicMasses.str() << std::endl << std::endl;
187 microscopicMasses.clear(); microscopicMasses.str("");
188 ofCell << macroscopicMasses.str() << std::endl << std::endl;
189 macroscopicMasses.clear(); macroscopicMasses.str("");
190 ofCell << masses.str() << std::endl << std::endl;
191 masses.clear(); masses.str("");
192 ofCell << energies.str() << std::endl << std::endl;
193 energies.clear(); energies.str("");
194 ofCell << temperatures.str() << std::endl << std::endl;
195 temperatures.clear(); temperatures.str("");
196 ofCell << microscopicMomenta.str() << std::endl << std::endl;
197 microscopicMomenta.clear(); microscopicMomenta.str("");
198 ofCell << macroscopicMomenta.str() << std::endl << std::endl;
199 macroscopicMomenta.clear(); macroscopicMomenta.str("");
200 ofCell << meanVelocities.str() << std::endl << std::endl;
201 meanVelocities.clear(); meanVelocities.str("");
202
203 // close file
204 ofCell.close();
205 }

245 // write cell data to stringstreams
246 void apply(coupling::datastructures::MacroscopicCellWithLinkedCells<LinkedCell,dim> &cell, const unsigned int &index){
247 // compute local quantities from molecules, if this is an inner cell
248 const tarch::la::Vector<dim,unsigned int> end = _indexConversion.getLocalNumberMacroscopicCells()+tarch::la::Vector<dim,unsigned int>(1);
249 const tarch::la::Vector<dim,unsigned int> localIndex = _indexConversion.getLocalVectorCellIndex(index);
250 bool isInnerCell = true;
251 for (unsigned int d = 0; d < dim; d++){
252 isInnerCell = isInnerCell && (localIndex[d]> 0) && (localIndex[d] < end[d]);
253 }
254 double mass = 0.0;
255 tarch::la::Vector<dim,double> meanVelocity(0.0);
256 double temperature = 0.0;
257
258 if (isInnerCell){
259 cell.iterateConstCells(_computeMassMapping);
260 mass = _computeMassMapping.getMass();
261 cell.iterateConstCells(_computeMomentumMapping);
262 meanVelocity = _computeMomentumMapping.getMeanVelocity();
263 temperature = _kineticEnergyController.computeTemperature(cell);
264 }
265
266 // write data to streams
267 _microscopicMasses << cell.getMicroscopicMass() << std::endl;
268 _macroscopicMasses << cell.getMacroscopicMass() << std::endl;
269 _masses << mass << std::endl;
270 _temperatures << temperature << std::endl;
271 _energies << cell.getPotentialEnergy() << std::endl;
272 for (unsigned int d = 0; d < dim; d++){
273 _microscopicMomenta << cell.getMicroscopicMomentum()[d] << " ";
274 _macroscopicMomenta << cell.getMacroscopicMomentum()[d] << " ";
275 _meanVelocities << meanVelocity[d] << " ";
276 }
277 _microscopicMomenta << _appendFloatZeros << std::endl;
278 _macroscopicMomenta << _appendFloatZeros << std::endl;
279 _meanVelocities << _appendFloatZeros << std::endl;
280 }
281
282 private:
283 std::stringstream &_microscopicMasses;
284 std::stringstream &_macroscopicMasses;
285 std::stringstream &_microscopicMomenta;
286 std::stringstream &_macroscopicMomenta;
287 std::stringstream &_meanVelocities;
288 std::stringstream &_masses;
289 std::stringstream &_energies;
290 std::stringstream &_temperatures;
291 const std::string _appendFloatZeros;
292 coupling::cellmappings::ComputeMassMapping<LinkedCell,dim> _computeMassMapping;
293 coupling::cellmappings::ComputeMomentumMapping<LinkedCell,dim> _computeMomentumMapping;
294 coupling::KineticEnergyController<LinkedCell,dim> _kineticEnergyController;
295 const coupling::IndexConversion<dim> &_indexConversion;
296 };
