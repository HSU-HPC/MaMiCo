127 void plotMacroscopicCellFile(
128 unsigned int ID, std::string filename, unsigned int rank, unsigned int t,
129 const coupling::IndexConversion<dim> &indexConversion,
130 coupling::datastructures::MacroscopicCells<LinkedCell, dim> &cells,
131 coupling::interface::MDSolverInterface<LinkedCell, dim>
132 *const mdSolverInterface) {
133 // stringstreams for all cell properties of interest
134 std::stringstream microscopicMasses;
135 microscopicMasses << "SCALARS microscopicMassBuffer float 1" << std::endl;
136 microscopicMasses << "LOOKUP_TABLE DEFAULT" << std::endl;
137 std::stringstream macroscopicMasses;
138 macroscopicMasses << "SCALARS macroscopicMassBuffer float 1" << std::endl;
139 macroscopicMasses << "LOOKUP_TABLE DEFAULT" << std::endl;
140 std::stringstream microscopicMomenta;
141 microscopicMomenta << "VECTORS microscopicMomentumBuffer float"
142 << std::endl;
143 std::stringstream macroscopicMomenta;
144 macroscopicMomenta << "VECTORS macroscopicMomentumBuffer float"
145 << std::endl;
146 std::stringstream meanVelocities;
147 meanVelocities << "VECTORS meanVelocity float" << std::endl;
148 std::stringstream masses;
149 masses << "SCALARS mass float 1" << std::endl;
150 masses << "LOOKUP_TABLE DEFAULT" << std::endl;
151 std::stringstream energies;
152 energies << "SCALARS meanPotentialEnergyBuffer float 1" << std::endl;
153 energies << "LOOKUP_TABLE DEFAULT" << std::endl;
154 std::stringstream temperatures;
155 temperatures << "SCALARS temperature float 1" << std::endl;
156 temperatures << "LOOKUP_TABLE DEFAULT" << std::endl;
157 // ofstream for file
158 std::ofstream ofCell;
159 // zeros appended in vtk file
160 std::string appendFloatZeros("");
161 switch (dim) {
162 case 1:
163 appendFloatZeros = " 0.0 0.0";
164 break;
165 case 2:
166 appendFloatZeros = " 0.0";
167 break;
168 case 3:
169 appendFloatZeros = "";
170 break;
171 default:
172 break;
173 }
174 // number of points
175 const tarch::la::Vector<3, unsigned int> numberCells =
176 coupling::initRange<dim>(
177 indexConversion.getLocalNumberMacroscopicCells() +
178 tarch::la::Vector<dim, unsigned int>(2));
179 tarch::la::Vector<3, double> original(0.0);
180 for (unsigned int d = 0; d < dim; d++) {
181 original[d] = indexConversion.getGlobalMDDomainOffset()[d] -
182 0.5 * indexConversion.getMacroscopicCellSize()[d] +
183 indexConversion.getAverageLocalNumberMacroscopicCells()[d] *
184 indexConversion.getThisProcess()[d] *
185 indexConversion.getMacroscopicCellSize()[d];
186 }
187 // open file
188 open(ID, filename, "MacroscopicCells", rank, t, ofCell);
189 Writer4Cells writer(microscopicMasses, macroscopicMasses,
190 microscopicMomenta, macroscopicMomenta, meanVelocities,
191 masses, energies, temperatures, appendFloatZeros,
192 indexConversion, mdSolverInterface);
193 cells.applyToAllLocalMacroscopicCellsWithLinkedCells(writer);
194 ofCell << "# vtk DataFile Version 2.0" << std::endl;
195 ofCell << "generated by MaMiCo (Philipp Neumann)" << std::endl;
196 ofCell << "ASCII" << std::endl << std::endl;
197 ofCell << "DATASET STRUCTURED_POINTS" << std::endl;
198 ofCell << "DIMENSIONS " << numberCells[0] << " " << numberCells[1] << " "
199 << numberCells[2] << std::endl;
200 ofCell << "ORIGIN ";
201 for (unsigned int d = 0; d < 3; d++) {
202 ofCell << original[d] << " ";
203 }
204 ofCell << std::endl;
205 ofCell << "SPACING ";
206 for (unsigned int d = 0; d < dim; d++) {
207 ofCell << indexConversion.getMacroscopicCellSize()[d] << " ";
208 }
209 for (unsigned int d = dim; d < 3; d++) {
210 ofCell << "1.0 ";
211 }
212 ofCell << std::endl;
213 ofCell << "POINT_DATA " << numberCells[0] * numberCells[1] * numberCells[2]
214 << std::endl
215 << std::endl;
216 ofCell << microscopicMasses.str() << std::endl << std::endl;
217 microscopicMasses.clear();
218 microscopicMasses.str("");
219 ofCell << macroscopicMasses.str() << std::endl << std::endl;
220 macroscopicMasses.clear();
221 macroscopicMasses.str("");
222 ofCell << masses.str() << std::endl << std::endl;
223 masses.clear();
224 masses.str("");
225 ofCell << energies.str() << std::endl << std::endl;
226 energies.clear();
227 energies.str("");
228 ofCell << temperatures.str() << std::endl << std::endl;
229 temperatures.clear();
230 temperatures.str("");
231 ofCell << microscopicMomenta.str() << std::endl << std::endl;
232 microscopicMomenta.clear();
233 microscopicMomenta.str("");
234 ofCell << macroscopicMomenta.str() << std::endl << std::endl;
235 macroscopicMomenta.clear();
236 macroscopicMomenta.str("");
237 ofCell << meanVelocities.str() << std::endl << std::endl;
238 meanVelocities.clear();
239 meanVelocities.str("");
240 // close file
241 ofCell.close();
242 }

317 void
318 apply(coupling::datastructures::MacroscopicCellWithLinkedCells<LinkedCell,
319 dim> &cell,
320 const unsigned int &index) {
321 // compute local quantities from molecules, if this is an inner cell
322 const tarch::la::Vector<dim, unsigned int> end =
323 _indexConversion.getLocalNumberMacroscopicCells() +
324 tarch::la::Vector<dim, unsigned int>(1);
325 const tarch::la::Vector<dim, unsigned int> localIndex =
326 _indexConversion.getLocalVectorCellIndex(index);
327 bool isInnerCell = true;
328 for (unsigned int d = 0; d < dim; d++) {
329 isInnerCell =
330 isInnerCell && (localIndex[d] > 0) && (localIndex[d] < end[d]);
331 }
332 double mass = 0.0;
333 tarch::la::Vector<dim, double> meanVelocity(0.0);
334 double temperature = 0.0;
335
336 if (isInnerCell) {
337 cell.iterateConstCells(_computeMassMapping);
338 mass = _computeMassMapping.getMass();
339 cell.iterateConstCells(_computeMomentumMapping);
340 meanVelocity = _computeMomentumMapping.getMeanVelocity();
341 temperature = _kineticEnergyController.computeTemperature(cell);
342 }
343
344 // write data to streams
345 _microscopicMasses << cell.getMicroscopicMass() << std::endl;
346 _macroscopicMasses << cell.getMacroscopicMass() << std::endl;
347 _masses << mass << std::endl;
348 _temperatures << temperature << std::endl;
349 _energies << cell.getPotentialEnergy() << std::endl;
350 for (unsigned int d = 0; d < dim; d++) {
351 _microscopicMomenta << cell.getMicroscopicMomentum()[d] << " ";
352 _macroscopicMomenta << cell.getMacroscopicMomentum()[d] << " ";
353 _meanVelocities << meanVelocity[d] << " ";
354 }
355 _microscopicMomenta << _appendFloatZeros << std::endl;
356 _macroscopicMomenta << _appendFloatZeros << std::endl;
357 _meanVelocities << _appendFloatZeros << std::endl;
358 }
359
360 private:
363 std::stringstream &_microscopicMasses;
366 std::stringstream &_macroscopicMasses;
369 std::stringstream &_microscopicMomenta;
372 std::stringstream &_macroscopicMomenta;
374 std::stringstream &_meanVelocities;
376 std::stringstream &_masses;
378 std::stringstream &_energies;
380 std::stringstream &_temperatures;
382 const std::string _appendFloatZeros;
384 coupling::cellmappings::ComputeMassMapping<LinkedCell, dim>
385 _computeMassMapping;
387 coupling::cellmappings::ComputeMomentumMapping<LinkedCell, dim>
388 _computeMomentumMapping;
390 coupling::KineticEnergyController<LinkedCell, dim> _kineticEnergyController;
392 const coupling::IndexConversion<dim> &_indexConversion;
393 };
