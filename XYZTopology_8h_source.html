35 tarch::la::Vector<dim,unsigned int> getProcessCoordinates(unsigned int rank) const {
36 #if (COUPLING_MD_DEBUG==COUPLING_MD_YES)
37 unsigned int intNumberProcesses = _numberProcesses[0]; for (unsigned int d = 1 ; d < dim; d++){ intNumberProcesses = intNumberProcesses*_numberProcesses[d]; }
38 if ( (rank<_topologyOffset) || (rank>_topologyOffset+intNumberProcesses-1)){
39 // TODO will be thrown on macroOnly services
40 std::cout << "Warning coupling::paralleltopology::XYZTopology::getProcessCoordinates(): rank out of range!" << std::endl;
41 std::cout << "Offset=" << _topologyOffset << ", rank=" << rank << std::endl;
42 }
43 std::cout << "Rank=" << rank << " corresponds to process coordinates=" << coupling::getVectorCellIndex<dim>(rank-_topologyOffset,_divisionFactor4NumberProcesses) << std::endl;
44 #endif
45 return coupling::getVectorCellIndex<dim>(rank-_topologyOffset,_divisionFactor4NumberProcesses);
46 }

49 unsigned int getRank(tarch::la::Vector<dim,unsigned int> processCoordinates) const {
50 unsigned int index = processCoordinates[dim-1];
51 for (int d = dim-2; d >-1; d--){
52 index = _numberProcesses[d]*index + processCoordinates[d];
53 }
54 #if (COUPLING_MD_DEBUG==COUPLING_MD_YES)
55 std::cout << "Process coordinates=" << processCoordinates << " correspond to rank=" << index+_topologyOffset << std::endl;
56 #endif
57 return index+_topologyOffset;
58 }

60 private:
61 /* number of processes */
62 const tarch::la::Vector<dim,unsigned int> _numberProcesses;
63 /* division factor for number of processes */
64 const tarch::la::Vector<dim,unsigned int> _divisionFactor4NumberProcesses;
65 /* offset in ranks for linear shift of xyz-topology */
66 const unsigned int _topologyOffset;
67};
