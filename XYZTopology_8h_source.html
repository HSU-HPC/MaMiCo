42 tarch::la::Vector<dim, unsigned int> getProcessCoordinates(unsigned int rank) const {
43#if (COUPLING_MD_DEBUG == COUPLING_MD_YES)
44 unsigned int intNumberProcesses = _numberProcesses[0];
45 for (unsigned int d = 1; d < dim; d++) {
46 intNumberProcesses = intNumberProcesses * _numberProcesses[d];
47 }
48 if ((rank < _topologyOffset) || (rank > _topologyOffset + intNumberProcesses - 1)) {
49 // TODO will be thrown on macroOnly services
50 std::cout << "Warning "
51 "coupling::paralleltopology::XYZTopology::"
52 "getProcessCoordinates(): rank out of range!"
53 << std::endl;
54 std::cout << "Offset=" << _topologyOffset << ", rank=" << rank << std::endl;
55 }
56 std::cout << "Rank=" << rank
57 << " corresponds to process coordinates=" << coupling::getVectorCellIndex<dim>(rank - _topologyOffset, _divisionFactor4NumberProcesses)
58 << std::endl;
59#endif
60 return coupling::getVectorCellIndex<dim>(rank - _topologyOffset, _divisionFactor4NumberProcesses);
61 }
62
63 unsigned int getRank(tarch::la::Vector<dim, unsigned int> processCoordinates) const {
64 unsigned int index = processCoordinates[dim - 1];
65 for (int d = dim - 2; d > -1; d--) {
66 index = _numberProcesses[d] * index + processCoordinates[d];
67 }
68#if (COUPLING_MD_DEBUG == COUPLING_MD_YES)
69 std::cout << "Process coordinates=" << processCoordinates << " correspond to rank=" << index + _topologyOffset << std::endl;
70#endif
71 return index + _topologyOffset;
72 }

74private:

77 /* division factor for number of processes */
78 const tarch::la::Vector<dim, unsigned int> _divisionFactor4NumberProcesses;
79 /* offset in ranks for linear shift of xyz-topology */
80 const unsigned int _topologyOffset;
