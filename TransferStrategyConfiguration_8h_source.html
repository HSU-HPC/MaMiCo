31class coupling::configurations::TransferStrategyConfiguration:
32public tarch::configuration::Configuration {
33 public:

35 DirectTransferStrategy=0,DifferenceTransferStrategy=1,TransferStrategy4FluxCoupling=2,TransferStrategy4SchwarzCoupling=3,AveragingTransferStrategy=4,
36 TransferStrategy4NieCoupling=5

41 virtual ~TransferStrategyConfiguration(){}

43 void parseSubtag( tinyxml2::XMLElement* node ){
44 std::string value;
45 tarch::configuration::ParseConfiguration::readStringMandatory(value,node,"type");
46 if (value=="direct-transfer"){
47 _type = DirectTransferStrategy;
48 } else if (value=="difference-transfer"){
49 _type = DifferenceTransferStrategy;
50 } else if (value=="schwarz-transfer"){
51 _type = TransferStrategy4SchwarzCoupling;
52 } else if (value=="nie-transfer"){
53 _type = TransferStrategy4NieCoupling;
54 } else if (value=="averaging"){
55 _type = AveragingTransferStrategy;
56 } else {
57 std::cout << "ERROR coupling::TransferStrategyConfiguration: Wrong insertion type!" << std::endl;
58 _isValid = false;
59 exit(EXIT_FAILURE);
60 }
61
62
63 if (_type==TransferStrategy4NieCoupling){
64 const std::string boundaries[6] = {"mass-flux-west","mass-flux-east","mass-flux-south","mass-flux-north","mass-flux-bottom","mass-flux-top"};
65 for (unsigned int d = 0; d < 2*dim; d++){
66 tarch::configuration::ParseConfiguration::readBoolMandatory(_massFluxBoundary[d],node,boundaries[d]);
67 }
68 // by default: no shifting of time interval
69 _shiftTimestep=0.0;
70 tarch::configuration::ParseConfiguration::readDoubleOptional(_shiftTimestep,node,"shift-by-timesteps");
71 if (_shiftTimestep<0.0 || _shiftTimestep>1.0){
72 std::cout << "Warning coupling::configurations::TransferStrategyConfiguration: shift-by-timesteps=" << _shiftTimestep << "; typical values range between 0 and 1!" << std::endl;
73 }
74 }
75 }

96 template<class LinkedCell>
97 coupling::transferstrategies::TransferStrategy<LinkedCell,dim>* interpreteConfiguration(
98 coupling::interface::MDSolverInterface<LinkedCell,dim> * const mdSolverInterface,
99 const coupling::IndexConversion<dim> &indexConversion,
100 unsigned int numberOfMDTimesteps
101 ) const {
102 if (_type == DirectTransferStrategy){
103 return new coupling::transferstrategies::DirectTransferStrategy<LinkedCell,dim>(mdSolverInterface,indexConversion);
104 } else if (_type == DifferenceTransferStrategy){
105 return new coupling::transferstrategies::DifferenceTransferStrategy<LinkedCell,dim>(mdSolverInterface,indexConversion,numberOfMDTimesteps);
106 } else if (_type == TransferStrategy4SchwarzCoupling){
107 return new coupling::transferstrategies::TransferStrategy4SchwarzCoupling<LinkedCell,dim>(mdSolverInterface,indexConversion,numberOfMDTimesteps);
108 } else if (_type == TransferStrategy4NieCoupling){
109 return new coupling::transferstrategies::TransferStrategy4NieCoupling<LinkedCell,dim>(mdSolverInterface,indexConversion,numberOfMDTimesteps,_shiftTimestep,_massFluxBoundary);
110 } else if (_type == AveragingTransferStrategy){
111 return new coupling::transferstrategies::AveragingTransferStrategy<LinkedCell,dim>(mdSolverInterface,indexConversion);
112 } else {
113 return NULL;
114 }
115 }

122 double _shiftTimestep; // used for Nie coupling: time interval by which the evaluation of the continuum flow field should be shifted. See also TransferStrategy4NieCoupling for more details.
123
124 bool _isValid;
125};
