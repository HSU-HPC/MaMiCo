270 void shutdown() {

272 coupling::interface::MamicoInterfaceProvider<LinkedCell, dim>::getInstance().setMDSolverInterface(_mdSolverInterface[i]);
273 if (_mdSimulations[i] != nullptr) {

278 _mdSolverInterface[i] = coupling::interface::MamicoInterfaceProvider<LinkedCell, dim>::getInstance().getMDSolverInterface();

281 for (auto& solverInterface : _mdSolverInterface) {
282 if (solverInterface != nullptr) {

284 solverInterface = nullptr;
285 }
