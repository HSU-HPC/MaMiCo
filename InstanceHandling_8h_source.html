67 ~InstanceHandling() {
68 for(unsigned int i=0;i<_simpleMD.size();++i) {
69 coupling::interface::MamicoInterfaceProvider<LinkedCell,dim>::getInstance().setMDSolverInterface(_mdSolverInterface[i]);
70 if(_simpleMD[i] != nullptr) {
71 _simpleMD[i]->shutdown();
72 delete _simpleMD[i];
73 _simpleMD[i] = nullptr;
74 }
75 _mdSolverInterface[i] = coupling::interface::MamicoInterfaceProvider<LinkedCell, dim>::getInstance().getMDSolverInterface();
76 }
77 _simpleMD.clear();
78 for(auto & solverInterface : _mdSolverInterface) {
79 if(solverInterface != nullptr) {
80 delete solverInterface;
81 solverInterface = nullptr;
82 }
83 }
84 _mdSolverInterface.clear();
