43 InstanceHandling(
44 simplemd::configurations::MolecularDynamicsConfiguration &mdConfig,
45 coupling::configurations::MaMiCoConfiguration<dim> &mamicoConfig,
46 tarch::utils::MultiMDService<dim> &multiMDService)
47 : _simpleMD(), _mdSolverInterface(), _mdConfig(mdConfig),
48 _mamicoConfig(mamicoConfig), _multiMDService(multiMDService) {
49 for (unsigned int i = 0; i < multiMDService.getLocalNumberOfMDSimulations();
50 i++) {
51 _simpleMD.push_back(
52 coupling::interface::SimulationAndInterfaceFactory::getInstance()
53 .getMDSimulation(_mdConfig, _mamicoConfig
54#if (COUPLING_MD_PARALLEL == COUPLING_MD_YES)
55 ,
56 _multiMDService.getLocalCommunicator()
57#endif
58 ));
59
60 if (_simpleMD[i] == nullptr) {
61 std::cout << "ERROR InstanceHandling : _simpleMD [" << i << "] == NULL!"
62 << std::endl;
63 std::exit(EXIT_FAILURE);
64 }
65
66 _simpleMD[i]->init(_multiMDService,
67 _multiMDService.getGlobalNumberOfLocalMDSimulation(i));
68 }
69 }

73 ~InstanceHandling() {
74 for (unsigned int i = 0; i < _simpleMD.size(); ++i) {
75 coupling::interface::MamicoInterfaceProvider<LinkedCell,
76 dim>::getInstance()
77 .setMDSolverInterface(_mdSolverInterface[i]);
78 if (_simpleMD[i] != nullptr) {
79 _simpleMD[i]->shutdown();
80 delete _simpleMD[i];
81 _simpleMD[i] = nullptr;
82 }
83 _mdSolverInterface[i] =
84 coupling::interface::MamicoInterfaceProvider<LinkedCell,
85 dim>::getInstance()
86 .getMDSolverInterface();
87 }
88 _simpleMD.clear();
89 for (auto &solverInterface : _mdSolverInterface) {
90 if (solverInterface != nullptr) {
91 delete solverInterface;
92 solverInterface = nullptr;
93 }
94 }
95 _mdSolverInterface.clear();
96 }
