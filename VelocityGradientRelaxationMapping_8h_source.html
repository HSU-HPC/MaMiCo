83 void handleCell(LinkedCell& cell, const unsigned int& cellIndex) {
84 // if this macroscopic cell is not interesting, skip it immediately
85 if (_ignoreThisCell) {
86 return;
87 }
88
89 // std::cout << "Handle VelocityGradientRelaxation in cell " <<
90 // _localVectorCellIndex << "=" << _currentLocalMacroscopicCellIndex <<
91 // std::endl;
92 coupling::interface::MoleculeIterator<LinkedCell, dim>* it = _mdSolverInterface->getMoleculeIterator(cell);
93 it->begin();
94 while (it->continueIteration()) {
95 coupling::interface::Molecule<dim>& wrapper(it->get());
96 tarch::la::Vector<dim, double> velocity = wrapper.getVelocity();
97 const tarch::la::Vector<dim, double> position(wrapper.getPosition());
98#if (COUPLING_MD_DEBUG == COUPLING_MD_YES)
99 std::cout << "VelocityGradientRelaxationMapping: Process cell " << _globalVectorCellIndex << ", molecule " << position << std::endl;
100#endif
101 if (relaxMolecule(position)) {
102 tarch::la::Vector<dim, double> normalisedPosition(0.0);
103 bool secondCake = false;
104 // index of macroscopic cell values that are involved in interpolation
105 // process
106 unsigned int macroscopicCellIndex[10];
107 // determine cell indices and the "correct cake" for interpolation
108 createMacroscopicCellIndex4SecondOrderInterpolation(position, normalisedPosition, secondCake, macroscopicCellIndex);
109
110 tarch::la::Vector<dim, double> newVelocity(0.0);
111 tarch::la::Vector<dim, double> oldVelocity(0.0);
112
113 // get newVelocity (microscopicMomentum) with second-order interpolation
114 interpolateVelocitySecondOrder(macroscopicCellIndex, normalisedPosition, secondCake, newVelocity);
115 // get current velocity directly from macroscopic cell
116 oldVelocity = _macroscopicCells[_currentLocalMacroscopicCellIndex].getCurrentVelocity();
117
118 velocity += _velocityRelaxationFactor * (newVelocity - oldVelocity);
119 wrapper.setVelocity(velocity);
120
121#if (COUPLING_MD_DEBUG == COUPLING_MD_YES)
122 for (unsigned int d = 0; d < dim; d++) {
123 if (normalisedPosition[d] > 2.0) {
124 std::cout << "ERROR cellmappings::VelocityGradientRelaxationMapping: "
125 "normalisedPosition>2.0! "
126 << "Local vector index=" << _localVectorCellIndex << " , global vector index=" << _globalVectorCellIndex
127 << ", currentLocalCellIndex=" << _currentLocalMacroscopicCellIndex << ", Norm. position=" << normalisedPosition
128 << ", Actual position=" << position << std::endl;
129 for (unsigned int i = 0; i < 10; i++) {
130 std::cout << macroscopicCellIndex[i] << " ";
131 }
132 std::cout << std::endl;
133 exit(EXIT_FAILURE);
134 }
135 if (normalisedPosition[d] < 0.0) {
136 std::cout << "ERROR cellmappings::VelocityGradientRelaxationMapping: "
137 "normalisedPosition<0.0! "
138 << _localVectorCellIndex << " ," << normalisedPosition << std::endl;
139 for (unsigned int i = 0; i < 10; i++) {
140 std::cout << macroscopicCellIndex[i] << " ";
141 }
142 std::cout << std::endl;
143 exit(EXIT_FAILURE);
144 }
145 }
146 if ((tarch::la::dot(newVelocity, newVelocity) > 1000.0) || (tarch::la::dot(oldVelocity, oldVelocity) > 1000.0)) {
147 for (unsigned int i = 0; i < 10; i++) {
148 std::cout << macroscopicCellIndex[i] << " ";
149 }
150 std::cout << std::endl;
151 std::cout << "ERROR: velocity to high! " << oldVelocity << " " << newVelocity << " " << _localVectorCellIndex << std::endl;
152 exit(EXIT_FAILURE);
153 }
154#endif
155 }
156 it->next();
157 }
158 delete it;
159 }

168 virtual bool relaxMolecule(const tarch::la::Vector<dim, double>& position) const {
169 // check if molecule is in the respective boundary strip
170 bool isInsideOuter = true;
171 bool isOutsideInner = false;
172 for (unsigned int d = 0; d < dim; d++) {
173 isInsideOuter = isInsideOuter && (position[d] > _outerLowerLeft[d]) && (position[d] < _outerUpperRight[d]);
174 isOutsideInner = isOutsideInner || (position[d] < _innerLowerLeft[d]) || (position[d] > _innerUpperRight[d]);
175 }
176 return isInsideOuter && isOutsideInner;
177 }

256 void createMacroscopicCellIndex4SecondOrderInterpolation(const tarch::la::Vector<dim, double>& position, tarch::la::Vector<dim, double>& normalisedPosition,
257 bool& secondCake, unsigned int* macroscopicCellIndex) const {
258 // compute lower left cell index and normalised position;
259 // the normalised position is chosen such that the origin (0,0...,0)
260 // coincides with the lower left... macroscopic cell's midpoint
261 tarch::la::Vector<dim, unsigned int> globalCellIndex = _globalVectorCellIndex;
262 tarch::la::Vector<dim, double> cellMidpoint = _domainOffset - 0.5 * _cellSize;
263 tarch::la::Vector<dim, unsigned int> lowerLeftCellIndex = _localVectorCellIndex;
264 unsigned int lowerLeftIndex = 0;
265
266 // determine cell mid point and normalised position in this loop
267 for (unsigned int d = 0; d < dim; d++) {
268 // determine mid point of current macroscopic cell
269 cellMidpoint[d] = cellMidpoint[d] + globalCellIndex[d] * _cellSize[d];
270#if (COUPLING_MD_DEBUG == COUPLING_MD_YES)
271 if ((position[d] < cellMidpoint[d] - 0.5 * _cellSize[d]) || (position[d] > cellMidpoint[d] + 0.5 * _cellSize[d])) {
272 std::cout << "ERROR "
273 "VelocityGradientRelaxationMapping::"
274 "createMacroscopicCellIndex4SecondOrderInterpolation: "
275 "Input position "
276 << position;
277 std::cout << " out of range of cell " << globalCellIndex << "!" << std::endl;
278 exit(EXIT_FAILURE);
279 }
280#endif
281
282 // put lowerLeftCellIndex such that it's right below the position
283 if (position[d] < cellMidpoint[d]) {
284 lowerLeftCellIndex[d]--;
285 globalCellIndex[d]--;
286 }
287 // if this is on the left/ lower boundary, but still to far up, reduce
288 // height
289 if (lowerLeftCellIndex[d] == 1) {
290 lowerLeftCellIndex[d]--;
291 globalCellIndex[d]--;
292 } else if (lowerLeftCellIndex[d] == _localNumberCellsWithGhostLayer[d] - 2) {
293 lowerLeftCellIndex[d]--;
294 globalCellIndex[d]--;
295 }
296
297 // determine normalised position
298 normalisedPosition[d] = (position[d] - (_domainOffset[d] - 0.5 * _cellSize[d]) - globalCellIndex[d] * _cellSize[d]) / _cellSize[d];
299 }
300 lowerLeftIndex = _indexConversion.getLocalCellIndex(lowerLeftCellIndex);
301
302 if (dim == 2) {
303 std::cout << "Not implemented correctly yet!" << std::endl;
304 exit(EXIT_FAILURE);
305 } else if (dim == 3) {
306 const unsigned int localNumberCells01 = _localNumberCellsWithGhostLayer[0] * _localNumberCellsWithGhostLayer[1];
307 const unsigned int localNumber2Cells0 = 2 * _localNumberCellsWithGhostLayer[0];
308 const unsigned int localNumber2Cells01Plus2Cells0 = 2 * localNumberCells01 + 2 * _localNumberCellsWithGhostLayer[0];
309
310 tarch::la::Vector<dim, double> normal;
311 normal[0] = 1.0;
312 normal[1] = 1.0;
313 normal[2] = 0.0;
314 tarch::la::Vector<dim, double> relPos(normalisedPosition);
315 relPos[0] -= 2.0;
316
317 // first cake:
318 if (tarch::la::dot(relPos, normal) < 0.0) {
319 secondCake = false;
320 macroscopicCellIndex[0] = lowerLeftIndex;
321 macroscopicCellIndex[1] = lowerLeftIndex + 1;
322 macroscopicCellIndex[2] = lowerLeftIndex + 2;
323 macroscopicCellIndex[3] = lowerLeftIndex + _localNumberCellsWithGhostLayer[0];
324 macroscopicCellIndex[4] = lowerLeftIndex + localNumber2Cells0;
325 macroscopicCellIndex[5] = lowerLeftIndex + localNumberCells01;
326 macroscopicCellIndex[6] = lowerLeftIndex + localNumberCells01 + _localNumberCellsWithGhostLayer[0] + 1;
327 macroscopicCellIndex[7] = lowerLeftIndex + 2 * localNumberCells01;
328 macroscopicCellIndex[8] = lowerLeftIndex + 2 * localNumberCells01 + 2;
329 macroscopicCellIndex[9] = lowerLeftIndex + localNumber2Cells01Plus2Cells0;
330 // second cake
331 } else {
332 secondCake = true;
333 macroscopicCellIndex[0] = lowerLeftIndex + 2;
334 macroscopicCellIndex[1] = lowerLeftIndex + _localNumberCellsWithGhostLayer[0] + 2;
335 macroscopicCellIndex[2] = lowerLeftIndex + localNumber2Cells0;
336 macroscopicCellIndex[3] = lowerLeftIndex + localNumber2Cells0 + 1;
337 macroscopicCellIndex[4] = lowerLeftIndex + localNumber2Cells0 + 2;
338 macroscopicCellIndex[5] = lowerLeftIndex + localNumberCells01 + _localNumberCellsWithGhostLayer[0] + 1;
339 macroscopicCellIndex[6] = lowerLeftIndex + localNumberCells01 + 2 * _localNumberCellsWithGhostLayer[0] + 2;
340 macroscopicCellIndex[7] = lowerLeftIndex + 2 * localNumberCells01 + 2;
341 macroscopicCellIndex[8] = lowerLeftIndex + localNumber2Cells01Plus2Cells0;
342 macroscopicCellIndex[9] = lowerLeftIndex + localNumber2Cells01Plus2Cells0 + 2;
343 }
344 } else {
345 std::cout << "ERROR createMacroscopicCellIndex4Interpolation(position): "
346 "only 2D/3D supported!"
347 << std::endl;
348 exit(EXIT_FAILURE);
349 }
350 }

359 void interpolateVelocitySecondOrder(unsigned int* cellIndex, const tarch::la::Vector<dim, double>& normalisedPosition, const bool& secondCake,
360 tarch::la::Vector<dim, double>& newVelocity) const {
361 if (dim == 2) {
362 // TODO
363 } else if (dim == 3) {
364 // compute interpolation coefficients
365 tarch::la::Vector<dim, double> coefficients[10];
366 tarch::la::Vector<dim, double> velocity[10];
367
368 // extract velocities from cells
369 for (int i = 0; i < 10; i++) {
370 velocity[i] = _macroscopicCells[cellIndex[i]].getMicroscopicMomentum();
371 }
372
373 if (secondCake) {
374 const tarch::la::Vector<dim, double> v78 = velocity[7] + velocity[8];
375 coefficients[0] =
376 2.0 * (velocity[0] + velocity[2] + velocity[9]) + 4.0 * (velocity[5] - velocity[1] - velocity[6] - velocity[3]) + 5.0 * velocity[4] - v78;
377 coefficients[1] =
378 0.5 * (v78 - velocity[0]) + 2.0 * (velocity[1] - velocity[2] - velocity[5] + velocity[6]) + 4.0 * velocity[3] - 3.5 * velocity[4] - velocity[9];
379 coefficients[2] =
380 2.0 * (velocity[3] - velocity[0] - velocity[5] + velocity[6]) + 4.0 * velocity[1] - 3.5 * velocity[4] + 0.5 * (v78 - velocity[2]) - velocity[9];
381 coefficients[3] = 0.5 * (v78 - velocity[0] - velocity[2] - velocity[4]) + 2.0 * velocity[6] - 1.5 * velocity[9];
382 coefficients[4] = 0.5 * (velocity[2] + velocity[4]) - velocity[3];
383 coefficients[5] = 0.5 * (velocity[0] + velocity[4]) - velocity[1];
384 coefficients[6] = 0.5 * (velocity[4] + velocity[9]) - velocity[6];
385 coefficients[7] =
386 0.25 * (velocity[0] + velocity[2] - v78) - velocity[1] - velocity[3] + 1.5 * velocity[4] + velocity[5] - velocity[6] + 0.5 * velocity[9];
387 coefficients[8] = 0.25 * (velocity[2] - velocity[4] - velocity[8] + velocity[9]);
388 coefficients[9] = 0.25 * (velocity[0] - velocity[4] - velocity[7] + velocity[9]);
389 } else {
390 const tarch::la::Vector<dim, double> minus15V0 = -1.5 * velocity[0];
391 coefficients[0] = velocity[0];
392 coefficients[1] = minus15V0 + 2.0 * velocity[1] - 0.5 * velocity[2];
393 coefficients[2] = minus15V0 + 2.0 * velocity[3] - 0.5 * velocity[4];
394 coefficients[3] = minus15V0 + 2.0 * velocity[5] - 0.5 * velocity[7];
395 coefficients[4] = 0.5 * (velocity[0] + velocity[2]) - velocity[1];
396 coefficients[5] = 0.5 * (velocity[0] + velocity[4]) - velocity[3];
397 coefficients[6] = 0.5 * (velocity[0] + velocity[7]) - velocity[5];
398 coefficients[7] = 1.5 * velocity[0] + velocity[6] - velocity[1] - velocity[3] - velocity[5] + 0.5 * velocity[7] +
399 0.25 * (velocity[2] + velocity[4] - velocity[8] - velocity[9]);
400 coefficients[8] = 0.25 * (velocity[0] - velocity[2] - velocity[7] + velocity[8]);
401 coefficients[9] = 0.25 * (velocity[0] - velocity[4] - velocity[7] + velocity[9]);
402 }
403
404 // evaluate polynomial
405 const tarch::la::Vector<dim, double> contribution0 =
406 normalisedPosition[0] *
407 (coefficients[1] + coefficients[4] * normalisedPosition[0] + coefficients[7] * normalisedPosition[1] + coefficients[8] * normalisedPosition[2]);
408 const tarch::la::Vector<dim, double> contribution1 =
409 normalisedPosition[1] * (coefficients[2] + coefficients[5] * normalisedPosition[1] + coefficients[9] * normalisedPosition[2]);
410 const tarch::la::Vector<dim, double> contribution2 = normalisedPosition[2] * (coefficients[3] + coefficients[6] * normalisedPosition[2]);
411
412 newVelocity = coefficients[0] + contribution0 + contribution1 + contribution2;
413 }
414 }
