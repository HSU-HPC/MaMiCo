106 void handleCell(LinkedCell &cell, const unsigned int &cellIndex) {
107 // if this macroscopic cell is not interesting, skip it immediately
108 if (_ignoreThisCell) {
109 return;
110 }
111
112 // std::cout << "Handle VelocityGradientRelaxation in cell " <<
113 // _localVectorCellIndex << "=" << _currentLocalMacroscopicCellIndex <<
114 // std::endl;
115 coupling::interface::MoleculeIterator<LinkedCell, dim> *it =
116 _mdSolverInterface->getMoleculeIterator(cell);
117 it->begin();
118 while (it->continueIteration()) {
119 coupling::interface::Molecule<dim> &wrapper(it->get());
120 tarch::la::Vector<dim, double> velocity = wrapper.getVelocity();
121 const tarch::la::Vector<dim, double> position(wrapper.getPosition());
122#if (COUPLING_MD_DEBUG == COUPLING_MD_YES)
123 std::cout << "VelocityGradientRelaxationMapping: Process cell "
124 << _globalVectorCellIndex << ", molecule " << position
125 << std::endl;
126#endif
127 if (relaxMolecule(position)) {
128 tarch::la::Vector<dim, double> normalisedPosition(0.0);
129 bool secondCake = false;
130 // index of macroscopic cell values that are involved in interpolation
131 // process
132 unsigned int macroscopicCellIndex[10];
133 // determine cell indices and the "correct cake" for interpolation
134 createMacroscopicCellIndex4SecondOrderInterpolation(
135 position, normalisedPosition, secondCake, macroscopicCellIndex);
136
137 tarch::la::Vector<dim, double> newVelocity(0.0);
138 tarch::la::Vector<dim, double> oldVelocity(0.0);
139
140 // get newVelocity (microscopicMomentum) with second-order interpolation
141 interpolateVelocitySecondOrder(macroscopicCellIndex, normalisedPosition,
142 secondCake, newVelocity);
143 // get current velocity directly from macroscopic cell
144 oldVelocity = _macroscopicCells[_currentLocalMacroscopicCellIndex]
145 .getCurrentVelocity();
146
147 velocity += _velocityRelaxationFactor * (newVelocity - oldVelocity);
148 wrapper.setVelocity(velocity);
149
150#if (COUPLING_MD_DEBUG == COUPLING_MD_YES)
151 for (unsigned int d = 0; d < dim; d++) {
152 if (normalisedPosition[d] > 2.0) {
153 std::cout
154 << "ERROR cellmappings::VelocityGradientRelaxationMapping: "
155 "normalisedPosition>2.0! "
156 << "Local vector index=" << _localVectorCellIndex
157 << " , global vector index=" << _globalVectorCellIndex
158 << ", currentLocalCellIndex="
159 << _currentLocalMacroscopicCellIndex
160 << ", Norm. position=" << normalisedPosition
161 << ", Actual position=" << position << std::endl;
162 for (unsigned int i = 0; i < 10; i++) {
163 std::cout << macroscopicCellIndex[i] << " ";
164 }
165 std::cout << std::endl;
166 exit(EXIT_FAILURE);
167 }
168 if (normalisedPosition[d] < 0.0) {
169 std::cout
170 << "ERROR cellmappings::VelocityGradientRelaxationMapping: "
171 "normalisedPosition<0.0! "
172 << _localVectorCellIndex << " ," << normalisedPosition
173 << std::endl;
174 for (unsigned int i = 0; i < 10; i++) {
175 std::cout << macroscopicCellIndex[i] << " ";
176 }
177 std::cout << std::endl;
178 exit(EXIT_FAILURE);
179 }
180 }
181 if ((tarch::la::dot(newVelocity, newVelocity) > 1000.0) ||
182 (tarch::la::dot(oldVelocity, oldVelocity) > 1000.0)) {
183 for (unsigned int i = 0; i < 10; i++) {
184 std::cout << macroscopicCellIndex[i] << " ";
185 }
186 std::cout << std::endl;
187 std::cout << "ERROR: velocity to high! " << oldVelocity << " "
188 << newVelocity << " " << _localVectorCellIndex
189 << std::endl;
190 exit(EXIT_FAILURE);
191 }
192#endif
193 }
194 it->next();
195 }
196 delete it;
197 }

206 virtual bool
207 relaxMolecule(const tarch::la::Vector<dim, double> &position) const {
208 // check if molecule is in the respective boundary strip
209 bool isInsideOuter = true;
210 bool isOutsideInner = false;
211 for (unsigned int d = 0; d < dim; d++) {
212 isInsideOuter = isInsideOuter && (position[d] > _outerLowerLeft[d]) &&
213 (position[d] < _outerUpperRight[d]);
214 isOutsideInner = isOutsideInner || (position[d] < _innerLowerLeft[d]) ||
215 (position[d] > _innerUpperRight[d]);
216 }
217 return isInsideOuter && isOutsideInner;
218 }

310 void createMacroscopicCellIndex4SecondOrderInterpolation(
311 const tarch::la::Vector<dim, double> &position,
312 tarch::la::Vector<dim, double> &normalisedPosition, bool &secondCake,
313 unsigned int *macroscopicCellIndex) const {
314 // compute lower left cell index and normalised position;
315 // the normalised position is chosen such that the origin (0,0...,0)
316 // coincides with the lower left... macroscopic cell's midpoint
317 tarch::la::Vector<dim, unsigned int> globalCellIndex =
318 _globalVectorCellIndex;
319 tarch::la::Vector<dim, double> cellMidpoint =
320 _domainOffset - 0.5 * _cellSize;
321 tarch::la::Vector<dim, unsigned int> lowerLeftCellIndex =
322 _localVectorCellIndex;
323 unsigned int lowerLeftIndex = 0;
324
325 // determine cell mid point and normalised position in this loop
326 for (unsigned int d = 0; d < dim; d++) {
327 // determine mid point of current macroscopic cell
328 cellMidpoint[d] = cellMidpoint[d] + globalCellIndex[d] * _cellSize[d];
329#if (COUPLING_MD_DEBUG == COUPLING_MD_YES)
330 if ((position[d] < cellMidpoint[d] - 0.5 * _cellSize[d]) ||
331 (position[d] > cellMidpoint[d] + 0.5 * _cellSize[d])) {
332 std::cout << "ERROR "
333 "VelocityGradientRelaxationMapping::"
334 "createMacroscopicCellIndex4SecondOrderInterpolation: "
335 "Input position "
336 << position;
337 std::cout << " out of range of cell " << globalCellIndex << "!"
338 << std::endl;
339 exit(EXIT_FAILURE);
340 }
341#endif
342
343 // put lowerLeftCellIndex such that it's right below the position
344 if (position[d] < cellMidpoint[d]) {
345 lowerLeftCellIndex[d]--;
346 globalCellIndex[d]--;
347 }
348 // if this is on the left/ lower boundary, but still to far up, reduce
349 // height
350 if (lowerLeftCellIndex[d] == 1) {
351 lowerLeftCellIndex[d]--;
352 globalCellIndex[d]--;
353 } else if (lowerLeftCellIndex[d] ==
354 _localNumberCellsWithGhostLayer[d] - 2) {
355 lowerLeftCellIndex[d]--;
356 globalCellIndex[d]--;
357 }
358
359 // determine normalised position
360 normalisedPosition[d] =
361 (position[d] - (_domainOffset[d] - 0.5 * _cellSize[d]) -
362 globalCellIndex[d] * _cellSize[d]) /
363 _cellSize[d];
364 }
365 lowerLeftIndex = _indexConversion.getLocalCellIndex(lowerLeftCellIndex);
366
367 if (dim == 2) {
368 std::cout << "Not implemented correctly yet!" << std::endl;
369 exit(EXIT_FAILURE);
370 } else if (dim == 3) {
371 const unsigned int localNumberCells01 =
372 _localNumberCellsWithGhostLayer[0] *
373 _localNumberCellsWithGhostLayer[1];
374 const unsigned int localNumber2Cells0 =
375 2 * _localNumberCellsWithGhostLayer[0];
376 const unsigned int localNumber2Cells01Plus2Cells0 =
377 2 * localNumberCells01 + 2 * _localNumberCellsWithGhostLayer[0];
378
379 tarch::la::Vector<dim, double> normal;
380 normal[0] = 1.0;
381 normal[1] = 1.0;
382 normal[2] = 0.0;
383 tarch::la::Vector<dim, double> relPos(normalisedPosition);
384 relPos[0] -= 2.0;
385
386 // first cake:
387 if (tarch::la::dot(relPos, normal) < 0.0) {
388 secondCake = false;
389 macroscopicCellIndex[0] = lowerLeftIndex;
390 macroscopicCellIndex[1] = lowerLeftIndex + 1;
391 macroscopicCellIndex[2] = lowerLeftIndex + 2;
392 macroscopicCellIndex[3] =
393 lowerLeftIndex + _localNumberCellsWithGhostLayer[0];
394 macroscopicCellIndex[4] = lowerLeftIndex + localNumber2Cells0;
395 macroscopicCellIndex[5] = lowerLeftIndex + localNumberCells01;
396 macroscopicCellIndex[6] = lowerLeftIndex + localNumberCells01 +
397 _localNumberCellsWithGhostLayer[0] + 1;
398 macroscopicCellIndex[7] = lowerLeftIndex + 2 * localNumberCells01;
399 macroscopicCellIndex[8] = lowerLeftIndex + 2 * localNumberCells01 + 2;
400 macroscopicCellIndex[9] =
401 lowerLeftIndex + localNumber2Cells01Plus2Cells0;
402 // second cake
403 } else {
404 secondCake = true;
405 macroscopicCellIndex[0] = lowerLeftIndex + 2;
406 macroscopicCellIndex[1] =
407 lowerLeftIndex + _localNumberCellsWithGhostLayer[0] + 2;
408 macroscopicCellIndex[2] = lowerLeftIndex + localNumber2Cells0;
409 macroscopicCellIndex[3] = lowerLeftIndex + localNumber2Cells0 + 1;
410 macroscopicCellIndex[4] = lowerLeftIndex + localNumber2Cells0 + 2;
411 macroscopicCellIndex[5] = lowerLeftIndex + localNumberCells01 +
412 _localNumberCellsWithGhostLayer[0] + 1;
413 macroscopicCellIndex[6] = lowerLeftIndex + localNumberCells01 +
414 2 * _localNumberCellsWithGhostLayer[0] + 2;
415 macroscopicCellIndex[7] = lowerLeftIndex + 2 * localNumberCells01 + 2;
416 macroscopicCellIndex[8] =
417 lowerLeftIndex + localNumber2Cells01Plus2Cells0;
418 macroscopicCellIndex[9] =
419 lowerLeftIndex + localNumber2Cells01Plus2Cells0 + 2;
420 }
421 } else {
422 std::cout << "ERROR createMacroscopicCellIndex4Interpolation(position): "
423 "only 2D/3D supported!"
424 << std::endl;
425 exit(EXIT_FAILURE);
426 }
427 }

436 void interpolateVelocitySecondOrder(
437 unsigned int *cellIndex,
438 const tarch::la::Vector<dim, double> &normalisedPosition,
439 const bool &secondCake,
440 tarch::la::Vector<dim, double> &newVelocity) const {
441 if (dim == 2) {
442 // TODO
443 } else if (dim == 3) {
444 // compute interpolation coefficients
445 tarch::la::Vector<dim, double> coefficients[10];
446 tarch::la::Vector<dim, double> velocity[10];
447
448 // extract velocities from cells
449 for (int i = 0; i < 10; i++) {
450 velocity[i] = _macroscopicCells[cellIndex[i]].getMicroscopicMomentum();
451 }
452
453 if (secondCake) {
454 const tarch::la::Vector<dim, double> v78 = velocity[7] + velocity[8];
455 coefficients[0] =
456 2.0 * (velocity[0] + velocity[2] + velocity[9]) +
457 4.0 * (velocity[5] - velocity[1] - velocity[6] - velocity[3]) +
458 5.0 * velocity[4] - v78;
459 coefficients[1] =
460 0.5 * (v78 - velocity[0]) +
461 2.0 * (velocity[1] - velocity[2] - velocity[5] + velocity[6]) +
462 4.0 * velocity[3] - 3.5 * velocity[4] - velocity[9];
463 coefficients[2] =
464 2.0 * (velocity[3] - velocity[0] - velocity[5] + velocity[6]) +
465 4.0 * velocity[1] - 3.5 * velocity[4] + 0.5 * (v78 - velocity[2]) -
466 velocity[9];
467 coefficients[3] =
468 0.5 * (v78 - velocity[0] - velocity[2] - velocity[4]) +
469 2.0 * velocity[6] - 1.5 * velocity[9];
470 coefficients[4] = 0.5 * (velocity[2] + velocity[4]) - velocity[3];
471 coefficients[5] = 0.5 * (velocity[0] + velocity[4]) - velocity[1];
472 coefficients[6] = 0.5 * (velocity[4] + velocity[9]) - velocity[6];
473 coefficients[7] = 0.25 * (velocity[0] + velocity[2] - v78) -
474 velocity[1] - velocity[3] + 1.5 * velocity[4] +
475 velocity[5] - velocity[6] + 0.5 * velocity[9];
476 coefficients[8] =
477 0.25 * (velocity[2] - velocity[4] - velocity[8] + velocity[9]);
478 coefficients[9] =
479 0.25 * (velocity[0] - velocity[4] - velocity[7] + velocity[9]);
480 } else {
481 const tarch::la::Vector<dim, double> minus15V0 = -1.5 * velocity[0];
482 coefficients[0] = velocity[0];
483 coefficients[1] = minus15V0 + 2.0 * velocity[1] - 0.5 * velocity[2];
484 coefficients[2] = minus15V0 + 2.0 * velocity[3] - 0.5 * velocity[4];
485 coefficients[3] = minus15V0 + 2.0 * velocity[5] - 0.5 * velocity[7];
486 coefficients[4] = 0.5 * (velocity[0] + velocity[2]) - velocity[1];
487 coefficients[5] = 0.5 * (velocity[0] + velocity[4]) - velocity[3];
488 coefficients[6] = 0.5 * (velocity[0] + velocity[7]) - velocity[5];
489 coefficients[7] =
490 1.5 * velocity[0] + velocity[6] - velocity[1] - velocity[3] -
491 velocity[5] + 0.5 * velocity[7] +
492 0.25 * (velocity[2] + velocity[4] - velocity[8] - velocity[9]);
493 coefficients[8] =
494 0.25 * (velocity[0] - velocity[2] - velocity[7] + velocity[8]);
495 coefficients[9] =
496 0.25 * (velocity[0] - velocity[4] - velocity[7] + velocity[9]);
497 }
498
499 // evaluate polynomial
500 const tarch::la::Vector<dim, double> contribution0 =
501 normalisedPosition[0] *
502 (coefficients[1] + coefficients[4] * normalisedPosition[0] +
503 coefficients[7] * normalisedPosition[1] +
504 coefficients[8] * normalisedPosition[2]);
505 const tarch::la::Vector<dim, double> contribution1 =
506 normalisedPosition[1] *
507 (coefficients[2] + coefficients[5] * normalisedPosition[1] +
508 coefficients[9] * normalisedPosition[2]);
509 const tarch::la::Vector<dim, double> contribution2 =
510 normalisedPosition[2] *
511 (coefficients[3] + coefficients[6] * normalisedPosition[2]);
512
513 newVelocity =
514 coefficients[0] + contribution0 + contribution1 + contribution2;
515 }
516 }

533class coupling::cellmappings::VelocityGradientRelaxationTopOnlyMapping

567 // the following snippet is basically a replacement of the method
568 // ignoreThisCell(). Since this function is called in the constructor of the
569 // based class, we cannot overwrite it; hence, we solve it by overwriting
570 // the respective variable from within this constructor (called after base
571 // object is constructed)
572 const tarch::la::Vector<dim, unsigned int> globalIndex =
573 coupling::cellmappings::VelocityGradientRelaxationMapping<
574 LinkedCell, dim>::_indexConversion
575 .convertLocalToGlobalVectorCellIndex(
576 coupling::cellmappings::VelocityGradientRelaxationMapping<
577 LinkedCell, dim>::_indexConversion
578 .getLocalVectorCellIndex(currentLocalMacroscopicCellIndex));
579 coupling::cellmappings::VelocityGradientRelaxationMapping<
580 LinkedCell, dim>::_ignoreThisCell =
581 (globalIndex[dim - 1] <
582 coupling::cellmappings::VelocityGradientRelaxationMapping<
583 LinkedCell, dim>::_globalNumberCellsWithGhostLayer[dim - 1] -
584 3);
585 }
586
587protected:
594 virtual bool
595 relaxMolecule(const tarch::la::Vector<dim, double> &position) const {
596 // check if molecule is in the respective boundary strip (only upper part is
597 // considered)
598 return (position[dim - 1] >
599 coupling::cellmappings::VelocityGradientRelaxationMapping<
600 LinkedCell, dim>::_innerUpperRight[dim - 1]) &&
601 (position[dim - 1] <
602 coupling::cellmappings::VelocityGradientRelaxationMapping<
603 LinkedCell, dim>::_outerUpperRight[dim - 1]);
604 }
605};
