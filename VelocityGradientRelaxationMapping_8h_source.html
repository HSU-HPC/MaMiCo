90 void handleCell(LinkedCell& cell,const unsigned int &cellIndex){
91 // if this macroscopic cell is not interesting, skip it immediately
92 if (_ignoreThisCell){return;}
93
94 //std::cout << "Handle VelocityGradientRelaxation in cell " << _localVectorCellIndex << "=" << _currentLocalMacroscopicCellIndex << std::endl;
95 coupling::interface::MoleculeIterator<LinkedCell,dim> *it = _mdSolverInterface->getMoleculeIterator(cell);
96 it->begin();
97 while(it->continueIteration()){
98 coupling::interface::Molecule<dim> &wrapper(it->get());
99 tarch::la::Vector<dim,double> velocity = wrapper.getVelocity();
100 const tarch::la::Vector<dim,double> position(wrapper.getPosition());
101 #if (COUPLING_MD_DEBUG==COUPLING_MD_YES)
102 std::cout << "VelocityGradientRelaxationMapping: Process cell " << _globalVectorCellIndex << ", molecule " << position << std::endl;
103 #endif
104 if (relaxMolecule(position)){
105 tarch::la::Vector<dim,double> normalisedPosition(0.0);
106 bool secondCake = false;
107 // index of macroscopic cell values that are involved in interpolation process
108 unsigned int macroscopicCellIndex[10];
109 //determine cell indices and the "correct cake" for interpolation
110 createMacroscopicCellIndex4SecondOrderInterpolation(position,normalisedPosition,secondCake, macroscopicCellIndex);
111
112 tarch::la::Vector<dim,double> newVelocity(0.0);
113 tarch::la::Vector<dim,double> oldVelocity(0.0);
114
115 // get newVelocity (microscopicMomentum) with second-order interpolation
116 interpolateVelocitySecondOrder(macroscopicCellIndex,normalisedPosition,secondCake,newVelocity);
117 // get current velocity directly from macroscopic cell
118 oldVelocity = _macroscopicCells[_currentLocalMacroscopicCellIndex].getCurrentVelocity();
119
120 velocity += _velocityRelaxationFactor*(newVelocity-oldVelocity);
121 wrapper.setVelocity(velocity);
122
123 #if (COUPLING_MD_DEBUG==COUPLING_MD_YES)
124 for (unsigned int d = 0; d < dim; d++){
125 if (normalisedPosition[d]>2.0){
126 std::cout << "ERROR cellmappings::VelocityGradientRelaxationMapping: normalisedPosition>2.0! " << "Local vector index=" <<_localVectorCellIndex << " , global vector index=" << _globalVectorCellIndex << ", currentLocalCellIndex=" << _currentLocalMacroscopicCellIndex << ", Norm. position=" << normalisedPosition << ", Actual position=" << position << std::endl;
127 for (unsigned int i = 0; i < 10; i++){ std::cout << macroscopicCellIndex[i] << " ";} std::cout << std::endl;
128 exit(EXIT_FAILURE);
129 }
130 if (normalisedPosition[d]<0.0){
131 std::cout << "ERROR cellmappings::VelocityGradientRelaxationMapping: normalisedPosition<0.0! " << _localVectorCellIndex << " ," << normalisedPosition << std::endl;
132 for (unsigned int i = 0; i < 10; i++){ std::cout << macroscopicCellIndex[i] << " ";} std::cout << std::endl;
133 exit(EXIT_FAILURE);
134 }
135 }
136 if ( (tarch::la::dot(newVelocity,newVelocity) > 1000.0) || (tarch::la::dot(oldVelocity,oldVelocity) > 1000.0) ){
137 for (unsigned int i = 0; i < 10; i++){ std::cout << macroscopicCellIndex[i] << " ";} std::cout << std::endl;
138 std::cout << "ERROR: velocity to high! " << oldVelocity << " " << newVelocity << " " << _localVectorCellIndex << std::endl;
139 exit(EXIT_FAILURE);
140 }
141 #endif
142 }
143 it->next();
144 }
145 delete it;
146 }

153 virtual bool relaxMolecule(const tarch::la::Vector<dim,double> &position) const {
154 // check if molecule is in the respective boundary strip
155 bool isInsideOuter = true;
156 bool isOutsideInner = false;
157 for (unsigned int d = 0; d < dim; d++){
158 isInsideOuter = isInsideOuter && (position[d]>_outerLowerLeft[d]) && (position[d]<_outerUpperRight[d]);
159 isOutsideInner = isOutsideInner || (position[d]<_innerLowerLeft[d]) || (position[d]>_innerUpperRight[d]);
160 }
161 return isInsideOuter && isOutsideInner;
162 }

238 void createMacroscopicCellIndex4SecondOrderInterpolation(
239 const tarch::la::Vector<dim,double> &position, tarch::la::Vector<dim,double> &normalisedPosition, bool &secondCake,
240 unsigned int *macroscopicCellIndex
241 ) const {
242 // compute lower left cell index and normalised position;
243 // the normalised position is chosen such that the origin (0,0...,0) coincides with the lower left... macroscopic cell's midpoint
244 tarch::la::Vector<dim,unsigned int> globalCellIndex = _globalVectorCellIndex;
245 tarch::la::Vector<dim,double> cellMidpoint = _domainOffset - 0.5*_cellSize;
246 tarch::la::Vector<dim,unsigned int> lowerLeftCellIndex = _localVectorCellIndex;
247 unsigned int lowerLeftIndex = 0;
248
249 // determine cell mid point and normalised position in this loop
250 for (unsigned int d = 0; d < dim; d++){
251 // determine mid point of current macroscopic cell
252 cellMidpoint[d] = cellMidpoint[d] + globalCellIndex[d]*_cellSize[d];
253 #if (COUPLING_MD_DEBUG==COUPLING_MD_YES)
254 if ( (position[d]<cellMidpoint[d]-0.5*_cellSize[d]) || (position[d] > cellMidpoint[d]+0.5*_cellSize[d])){
255 std::cout << "ERROR VelocityGradientRelaxationMapping::createMacroscopicCellIndex4SecondOrderInterpolation: Input position " << position;
256 std::cout << " out of range of cell " << globalCellIndex << "!" << std::endl; exit(EXIT_FAILURE);
257 }
258 #endif
259
260 // put lowerLeftCellIndex such that it's right below the position
261 if ( position[d] < cellMidpoint[d] ){ lowerLeftCellIndex[d]--; globalCellIndex[d]--; }
262 // if this is on the left/ lower boundary, but still to far up, reduce height
263 if (lowerLeftCellIndex[d]==1 ){ lowerLeftCellIndex[d]--; globalCellIndex[d]--; }
264 else if (lowerLeftCellIndex[d]==_localNumberCellsWithGhostLayer[d]-2){ lowerLeftCellIndex[d]--; globalCellIndex[d]--; }
265
266 // determine normalised position
267 normalisedPosition[d] = (position[d] - (_domainOffset[d]-0.5*_cellSize[d]) - globalCellIndex[d]*_cellSize[d]) / _cellSize[d];
268 }
269 lowerLeftIndex = _indexConversion.getLocalCellIndex(lowerLeftCellIndex);
270
271
272 if (dim==2){
273 std::cout << "Not implemented correctly yet!" << std::endl; exit(EXIT_FAILURE);
274 } else if (dim==3){
275 const unsigned int localNumberCells01 = _localNumberCellsWithGhostLayer[0]*_localNumberCellsWithGhostLayer[1];
276 const unsigned int localNumber2Cells0 = 2*_localNumberCellsWithGhostLayer[0];
277 const unsigned int localNumber2Cells01Plus2Cells0 = 2*localNumberCells01+2*_localNumberCellsWithGhostLayer[0];
278
279 tarch::la::Vector<dim,double> normal; normal[0] = 1.0; normal[1] = 1.0; normal[2] = 0.0;
280 tarch::la::Vector<dim,double> relPos(normalisedPosition); relPos[0] -= 2.0;
281
282 // first cake:
283 if (tarch::la::dot(relPos,normal) < 0.0){
284 secondCake = false;
285 macroscopicCellIndex[0] = lowerLeftIndex;
286 macroscopicCellIndex[1] = lowerLeftIndex+1;
287 macroscopicCellIndex[2] = lowerLeftIndex+2;
288 macroscopicCellIndex[3] = lowerLeftIndex+_localNumberCellsWithGhostLayer[0];
289 macroscopicCellIndex[4] = lowerLeftIndex+localNumber2Cells0;
290 macroscopicCellIndex[5] = lowerLeftIndex+localNumberCells01;
291 macroscopicCellIndex[6] = lowerLeftIndex+localNumberCells01+_localNumberCellsWithGhostLayer[0]+1;
292 macroscopicCellIndex[7] = lowerLeftIndex+2*localNumberCells01;
293 macroscopicCellIndex[8] = lowerLeftIndex+2*localNumberCells01+2;
294 macroscopicCellIndex[9] = lowerLeftIndex+localNumber2Cells01Plus2Cells0;
295 // second cake
296 } else {
297 secondCake = true;
298 macroscopicCellIndex[0] = lowerLeftIndex+2;
299 macroscopicCellIndex[1] = lowerLeftIndex+_localNumberCellsWithGhostLayer[0]+2;
300 macroscopicCellIndex[2] = lowerLeftIndex+localNumber2Cells0;
301 macroscopicCellIndex[3] = lowerLeftIndex+localNumber2Cells0+1;
302 macroscopicCellIndex[4] = lowerLeftIndex+localNumber2Cells0+2;
303 macroscopicCellIndex[5] = lowerLeftIndex+localNumberCells01+_localNumberCellsWithGhostLayer[0]+1;
304 macroscopicCellIndex[6] = lowerLeftIndex+localNumberCells01+2*_localNumberCellsWithGhostLayer[0]+2;
305 macroscopicCellIndex[7] = lowerLeftIndex+2*localNumberCells01+2;
306 macroscopicCellIndex[8] = lowerLeftIndex+localNumber2Cells01Plus2Cells0;
307 macroscopicCellIndex[9] = lowerLeftIndex+localNumber2Cells01Plus2Cells0+2;
308 }
309 } else {
310 std::cout << "ERROR createMacroscopicCellIndex4Interpolation(position): only 2D/3D supported!" << std::endl; exit(EXIT_FAILURE);
311 }
312
313 }

322 void interpolateVelocitySecondOrder(
323 unsigned int* cellIndex,const tarch::la::Vector<dim,double> &normalisedPosition,const bool &secondCake,
324 tarch::la::Vector<dim,double>& newVelocity
325 ) const {
326 if (dim==2){
327 // TODO
328 } else if (dim==3){
329 // compute interpolation coefficients
330 tarch::la::Vector<dim,double> coefficients[10];
331 tarch::la::Vector<dim,double> velocity[10];
332
333 // extract velocities from cells
334 for (int i = 0; i < 10; i++){
335 velocity[i] = _macroscopicCells[cellIndex[i]].getMicroscopicMomentum();
336 }
337
338 if (secondCake){
339 const tarch::la::Vector<dim,double> v78 = velocity[7]+velocity[8];
340 coefficients[0] = 2.0*(velocity[0] + velocity[2] + velocity[9]) + 4.0*(velocity[5] - velocity[1] - velocity[6] - velocity[3]) + 5.0*velocity[4] - v78;
341 coefficients[1] = 0.5*(v78 - velocity[0]) + 2.0*(velocity[1] - velocity[2] - velocity[5] + velocity[6]) + 4.0*velocity[3] - 3.5*velocity[4] - velocity[9];
342 coefficients[2] = 2.0*(velocity[3] - velocity[0] - velocity[5] + velocity[6]) + 4.0*velocity[1] - 3.5*velocity[4] + 0.5*(v78 - velocity[2]) - velocity[9];
343 coefficients[3] = 0.5*(v78 - velocity[0] - velocity[2] - velocity[4]) + 2.0*velocity[6] - 1.5*velocity[9];
344 coefficients[4] = 0.5*(velocity[2] + velocity[4]) - velocity[3];
345 coefficients[5] = 0.5*(velocity[0] + velocity[4]) - velocity[1];
346 coefficients[6] = 0.5*(velocity[4] + velocity[9]) - velocity[6];
347 coefficients[7] = 0.25*(velocity[0] + velocity[2] - v78) - velocity[1] - velocity[3] + 1.5*velocity[4] + velocity[5] - velocity[6] + 0.5*velocity[9];
348 coefficients[8] = 0.25*(velocity[2] - velocity[4] - velocity[8] + velocity[9]);
349 coefficients[9] = 0.25*(velocity[0] - velocity[4] - velocity[7] + velocity[9]);
350 } else {
351 const tarch::la::Vector<dim,double> minus15V0 = -1.5*velocity[0];
352 coefficients[0] = velocity[0];
353 coefficients[1] = minus15V0 + 2.0*velocity[1] - 0.5*velocity[2];
354 coefficients[2] = minus15V0 + 2.0*velocity[3] - 0.5*velocity[4];
355 coefficients[3] = minus15V0 + 2.0*velocity[5] - 0.5*velocity[7];
356 coefficients[4] = 0.5*(velocity[0] + velocity[2]) - velocity[1];
357 coefficients[5] = 0.5*(velocity[0] + velocity[4]) - velocity[3];
358 coefficients[6] = 0.5*(velocity[0] + velocity[7]) - velocity[5];
359 coefficients[7] = 1.5*velocity[0] + velocity[6] - velocity[1] - velocity[3] - velocity[5]
360 + 0.5*velocity[7] + 0.25*(velocity[2] + velocity[4] - velocity[8] - velocity[9]);
361 coefficients[8] = 0.25*(velocity[0] - velocity[2] - velocity[7] + velocity[8]);
362 coefficients[9] = 0.25*(velocity[0] - velocity[4] - velocity[7] + velocity[9]);
363 }
364
365 // evaluate polynomial
366 const tarch::la::Vector<dim,double> contribution0 = normalisedPosition[0]*(
367 coefficients[1] + coefficients[4]*normalisedPosition[0]
368 + coefficients[7]*normalisedPosition[1] + coefficients[8]*normalisedPosition[2] );
369 const tarch::la::Vector<dim,double> contribution1 = normalisedPosition[1]*(
370 coefficients[2] + coefficients[5]*normalisedPosition[1]
371 + coefficients[9]*normalisedPosition[2] );
372 const tarch::la::Vector<dim,double> contribution2 = normalisedPosition[2]*(
373 coefficients[3] + coefficients[6]*normalisedPosition[2] );
374
375 newVelocity = coefficients[0]+contribution0+contribution1+contribution2;
376
377 }
