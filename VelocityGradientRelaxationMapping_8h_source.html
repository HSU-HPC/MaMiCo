71 void handleCell(LinkedCell& cell,const unsigned int &cellIndex){
72 // if this macroscopic cell is not interesting, skip it immediately
73 if (_ignoreThisCell){return;}
74
75 //std::cout << "Handle VelocityGradientRelaxation in cell " << _localVectorCellIndex << "=" << _currentLocalMacroscopicCellIndex << std::endl;
76 coupling::interface::MoleculeIterator<LinkedCell,dim> *it = _mdSolverInterface->getMoleculeIterator(cell);
77 it->begin();
78 while(it->continueIteration()){
79 coupling::interface::Molecule<dim> &wrapper(it->get());
80 tarch::la::Vector<dim,double> velocity = wrapper.getVelocity();
81 const tarch::la::Vector<dim,double> position(wrapper.getPosition());
82 #if (COUPLING_MD_DEBUG==COUPLING_MD_YES)
83 std::cout << "VelocityGradientRelaxationMapping: Process cell " << _globalVectorCellIndex << ", molecule " << position << std::endl;
84 #endif
85 if (relaxMolecule(position)){
86 tarch::la::Vector<dim,double> normalisedPosition(0.0);
87 bool secondCake = false;
88 // index of macroscopic cell values that are involved in interpolation process
89 unsigned int macroscopicCellIndex[10];
90 //determine cell indices and the "correct cake" for interpolation
91 createMacroscopicCellIndex4SecondOrderInterpolation(position,normalisedPosition,secondCake, macroscopicCellIndex);
92
93 tarch::la::Vector<dim,double> newVelocity(0.0);
94 tarch::la::Vector<dim,double> oldVelocity(0.0);
95
96 // get newVelocity (microscopicMomentum) with second-order interpolation
97 interpolateVelocitySecondOrder(macroscopicCellIndex,normalisedPosition,secondCake,newVelocity);
98 // get current velocity directly from macroscopic cell
99 oldVelocity = _macroscopicCells[_currentLocalMacroscopicCellIndex].getCurrentVelocity();
100
101 velocity += _velocityRelaxationFactor*(newVelocity-oldVelocity);
102 wrapper.setVelocity(velocity);
103
104 #if (COUPLING_MD_DEBUG==COUPLING_MD_YES)
105 for (unsigned int d = 0; d < dim; d++){
106 if (normalisedPosition[d]>2.0){
107 std::cout << "ERROR cellmappings::VelocityGradientRelaxationMapping: normalisedPosition>2.0! " << "Local vector index=" <<_localVectorCellIndex << " , global vector index=" << _globalVectorCellIndex << ", currentLocalCellIndex=" << _currentLocalMacroscopicCellIndex << ", Norm. position=" << normalisedPosition << ", Actual position=" << position << std::endl;
108 for (unsigned int i = 0; i < 10; i++){ std::cout << macroscopicCellIndex[i] << " ";} std::cout << std::endl;
109 exit(EXIT_FAILURE);
110 }
111 if (normalisedPosition[d]<0.0){
112 std::cout << "ERROR cellmappings::VelocityGradientRelaxationMapping: normalisedPosition<0.0! " << _localVectorCellIndex << " ," << normalisedPosition << std::endl;
113 for (unsigned int i = 0; i < 10; i++){ std::cout << macroscopicCellIndex[i] << " ";} std::cout << std::endl;
114 exit(EXIT_FAILURE);
115 }
116 }
117 if ( (tarch::la::dot(newVelocity,newVelocity) > 1000.0) || (tarch::la::dot(oldVelocity,oldVelocity) > 1000.0) ){
118 for (unsigned int i = 0; i < 10; i++){ std::cout << macroscopicCellIndex[i] << " ";} std::cout << std::endl;
119 std::cout << "ERROR: velocity to high! " << oldVelocity << " " << newVelocity << " " << _localVectorCellIndex << std::endl;
120 exit(EXIT_FAILURE);
121 }
122 #endif
123 }
124 it->next();
125 }
126 delete it;
127 }

130 // returns true, if the position 'position' is within the respective boundary strip that is spanned by the two outermost macroscopic cell midpoints
131 virtual bool relaxMolecule(const tarch::la::Vector<dim,double> &position) const {
132 // check if molecule is in the respective boundary strip
133 bool isInsideOuter = true;
134 bool isOutsideInner = false;
135 for (unsigned int d = 0; d < dim; d++){
136 isInsideOuter = isInsideOuter && (position[d]>_outerLowerLeft[d]) && (position[d]<_outerUpperRight[d]);
137 isOutsideInner = isOutsideInner || (position[d]<_innerLowerLeft[d]) || (position[d]>_innerUpperRight[d]);
138 }
139 return isInsideOuter && isOutsideInner;
140 }

143 bool ignoreThisCell(const unsigned int& currentLocalMacroscopicCellIndex) const {
144 const tarch::la::Vector<dim,unsigned int> globalIndex = _indexConversion.convertLocalToGlobalVectorCellIndex(_indexConversion.getLocalVectorCellIndex(currentLocalMacroscopicCellIndex));

146 for (unsigned int d = 0; d < dim; d++){
147 innerCell = innerCell && ( (globalIndex[d]>2) && (globalIndex[d]<_globalNumberCellsWithGhostLayer[d]-3) );
148 }
149
150 return innerCell;
151 }

195 void createMacroscopicCellIndex4SecondOrderInterpolation(
196 const tarch::la::Vector<dim,double> &position, tarch::la::Vector<dim,double> &normalisedPosition, bool &secondCake,
197 unsigned int *macroscopicCellIndex
198 ) const {
199 // compute lower left cell index and normalised position;
200 // the normalised position is chosen such that the origin (0,0...,0) coincides with the lower left... macroscopic cell's midpoint
201 tarch::la::Vector<dim,unsigned int> globalCellIndex = _globalVectorCellIndex;
202 tarch::la::Vector<dim,double> cellMidpoint = _domainOffset - 0.5*_cellSize;
203 tarch::la::Vector<dim,unsigned int> lowerLeftCellIndex = _localVectorCellIndex;
204 unsigned int lowerLeftIndex = 0;
205
206 // determine cell mid point and normalised position in this loop
207 for (unsigned int d = 0; d < dim; d++){
208 // determine mid point of current macroscopic cell
209 cellMidpoint[d] = cellMidpoint[d] + globalCellIndex[d]*_cellSize[d];
210 #if (COUPLING_MD_DEBUG==COUPLING_MD_YES)
211 if ( (position[d]<cellMidpoint[d]-0.5*_cellSize[d]) || (position[d] > cellMidpoint[d]+0.5*_cellSize[d])){
212 std::cout << "ERROR VelocityGradientRelaxationMapping::createMacroscopicCellIndex4SecondOrderInterpolation: Input position " << position;
213 std::cout << " out of range of cell " << globalCellIndex << "!" << std::endl; exit(EXIT_FAILURE);
214 }
215 #endif
216
217 // put lowerLeftCellIndex such that it's right below the position
218 if ( position[d] < cellMidpoint[d] ){ lowerLeftCellIndex[d]--; globalCellIndex[d]--; }
219 // if this is on the left/ lower boundary, but still to far up, reduce height
220 if (lowerLeftCellIndex[d]==1 ){ lowerLeftCellIndex[d]--; globalCellIndex[d]--; }
221 else if (lowerLeftCellIndex[d]==_localNumberCellsWithGhostLayer[d]-2){ lowerLeftCellIndex[d]--; globalCellIndex[d]--; }
222
223 // determine normalised position
224 normalisedPosition[d] = (position[d] - (_domainOffset[d]-0.5*_cellSize[d]) - globalCellIndex[d]*_cellSize[d]) / _cellSize[d];
225 }
226 lowerLeftIndex = _indexConversion.getLocalCellIndex(lowerLeftCellIndex);
227
228
229 if (dim==2){
230 std::cout << "Not implemented correctly yet!" << std::endl; exit(EXIT_FAILURE);
231 } else if (dim==3){
232 const unsigned int localNumberCells01 = _localNumberCellsWithGhostLayer[0]*_localNumberCellsWithGhostLayer[1];
233 const unsigned int localNumber2Cells0 = 2*_localNumberCellsWithGhostLayer[0];
234 const unsigned int localNumber2Cells01Plus2Cells0 = 2*localNumberCells01+2*_localNumberCellsWithGhostLayer[0];
235
236 tarch::la::Vector<dim,double> normal; normal[0] = 1.0; normal[1] = 1.0; normal[2] = 0.0;
237 tarch::la::Vector<dim,double> relPos(normalisedPosition); relPos[0] -= 2.0;
238
239 // first cake:
240 if (tarch::la::dot(relPos,normal) < 0.0){
241 secondCake = false;
242 macroscopicCellIndex[0] = lowerLeftIndex;
243 macroscopicCellIndex[1] = lowerLeftIndex+1;
244 macroscopicCellIndex[2] = lowerLeftIndex+2;
245 macroscopicCellIndex[3] = lowerLeftIndex+_localNumberCellsWithGhostLayer[0];
246 macroscopicCellIndex[4] = lowerLeftIndex+localNumber2Cells0;
247 macroscopicCellIndex[5] = lowerLeftIndex+localNumberCells01;
248 macroscopicCellIndex[6] = lowerLeftIndex+localNumberCells01+_localNumberCellsWithGhostLayer[0]+1;
249 macroscopicCellIndex[7] = lowerLeftIndex+2*localNumberCells01;
250 macroscopicCellIndex[8] = lowerLeftIndex+2*localNumberCells01+2;
251 macroscopicCellIndex[9] = lowerLeftIndex+localNumber2Cells01Plus2Cells0;
252 // second cake
253 } else {
254 secondCake = true;
255 macroscopicCellIndex[0] = lowerLeftIndex+2;
256 macroscopicCellIndex[1] = lowerLeftIndex+_localNumberCellsWithGhostLayer[0]+2;
257 macroscopicCellIndex[2] = lowerLeftIndex+localNumber2Cells0;
258 macroscopicCellIndex[3] = lowerLeftIndex+localNumber2Cells0+1;
259 macroscopicCellIndex[4] = lowerLeftIndex+localNumber2Cells0+2;
260 macroscopicCellIndex[5] = lowerLeftIndex+localNumberCells01+_localNumberCellsWithGhostLayer[0]+1;
261 macroscopicCellIndex[6] = lowerLeftIndex+localNumberCells01+2*_localNumberCellsWithGhostLayer[0]+2;
262 macroscopicCellIndex[7] = lowerLeftIndex+2*localNumberCells01+2;
263 macroscopicCellIndex[8] = lowerLeftIndex+localNumber2Cells01Plus2Cells0;
264 macroscopicCellIndex[9] = lowerLeftIndex+localNumber2Cells01Plus2Cells0+2;
265 }
266 } else {
267 std::cout << "ERROR createMacroscopicCellIndex4Interpolation(position): only 2D/3D supported!" << std::endl; exit(EXIT_FAILURE);
268 }
269
270 }

274 void interpolateVelocitySecondOrder(
275 unsigned int* cellIndex,const tarch::la::Vector<dim,double> &normalisedPosition,const bool &secondCake,
276 tarch::la::Vector<dim,double>& newVelocity
277 ) const {
278 if (dim==2){
279 // TODO
280 } else if (dim==3){
281 // compute interpolation coefficients
282 tarch::la::Vector<dim,double> coefficients[10];
283 tarch::la::Vector<dim,double> velocity[10];
284
285 // extract velocities from cells
286 for (int i = 0; i < 10; i++){
287 velocity[i] = _macroscopicCells[cellIndex[i]].getMicroscopicMomentum();
288 }
289
290 if (secondCake){
291 const tarch::la::Vector<dim,double> v78 = velocity[7]+velocity[8];
292 coefficients[0] = 2.0*(velocity[0] + velocity[2] + velocity[9]) + 4.0*(velocity[5] - velocity[1] - velocity[6] - velocity[3]) + 5.0*velocity[4] - v78;
293 coefficients[1] = 0.5*(v78 - velocity[0]) + 2.0*(velocity[1] - velocity[2] - velocity[5] + velocity[6]) + 4.0*velocity[3] - 3.5*velocity[4] - velocity[9];
294 coefficients[2] = 2.0*(velocity[3] - velocity[0] - velocity[5] + velocity[6]) + 4.0*velocity[1] - 3.5*velocity[4] + 0.5*(v78 - velocity[2]) - velocity[9];
295 coefficients[3] = 0.5*(v78 - velocity[0] - velocity[2] - velocity[4]) + 2.0*velocity[6] - 1.5*velocity[9];
296 coefficients[4] = 0.5*(velocity[2] + velocity[4]) - velocity[3];
297 coefficients[5] = 0.5*(velocity[0] + velocity[4]) - velocity[1];
298 coefficients[6] = 0.5*(velocity[4] + velocity[9]) - velocity[6];
299 coefficients[7] = 0.25*(velocity[0] + velocity[2] - v78) - velocity[1] - velocity[3] + 1.5*velocity[4] + velocity[5] - velocity[6] + 0.5*velocity[9];
300 coefficients[8] = 0.25*(velocity[2] - velocity[4] - velocity[8] + velocity[9]);
301 coefficients[9] = 0.25*(velocity[0] - velocity[4] - velocity[7] + velocity[9]);
302 } else {
303 const tarch::la::Vector<dim,double> minus15V0 = -1.5*velocity[0];
304 coefficients[0] = velocity[0];
305 coefficients[1] = minus15V0 + 2.0*velocity[1] - 0.5*velocity[2];
306 coefficients[2] = minus15V0 + 2.0*velocity[3] - 0.5*velocity[4];
307 coefficients[3] = minus15V0 + 2.0*velocity[5] - 0.5*velocity[7];
308 coefficients[4] = 0.5*(velocity[0] + velocity[2]) - velocity[1];
309 coefficients[5] = 0.5*(velocity[0] + velocity[4]) - velocity[3];
310 coefficients[6] = 0.5*(velocity[0] + velocity[7]) - velocity[5];
311 coefficients[7] = 1.5*velocity[0] + velocity[6] - velocity[1] - velocity[3] - velocity[5]
312 + 0.5*velocity[7] + 0.25*(velocity[2] + velocity[4] - velocity[8] - velocity[9]);
313 coefficients[8] = 0.25*(velocity[0] - velocity[2] - velocity[7] + velocity[8]);
314 coefficients[9] = 0.25*(velocity[0] - velocity[4] - velocity[7] + velocity[9]);
315 }
316
317 // evaluate polynomial
318 const tarch::la::Vector<dim,double> contribution0 = normalisedPosition[0]*(
319 coefficients[1] + coefficients[4]*normalisedPosition[0]
320 + coefficients[7]*normalisedPosition[1] + coefficients[8]*normalisedPosition[2] );
321 const tarch::la::Vector<dim,double> contribution1 = normalisedPosition[1]*(
322 coefficients[2] + coefficients[5]*normalisedPosition[1]
323 + coefficients[9]*normalisedPosition[2] );
324 const tarch::la::Vector<dim,double> contribution2 = normalisedPosition[2]*(
325 coefficients[3] + coefficients[6]*normalisedPosition[2] );
326
327 newVelocity = coefficients[0]+contribution0+contribution1+contribution2;
328
329 }
