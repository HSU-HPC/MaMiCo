47 // loop over all global number cell-configurations

53 // loop over all number of process configurations
54 for (unsigned int znp = 1; znp < testNumberProcesses[2]; znp++) {

58 const unsigned int numberRanks = xnp * ynp * znp;

60 // loop over all local process configurations

64 conversion.getLocalNumberMacroscopicCells() + tarch::la::Vector<3, unsigned int>(2);
65
66 // loop over all cells in the local domain

70

195 template <unsigned int dim>
196 void testUniqueRanks(const tarch::la::Vector<dim, unsigned int>& testGlobalNumberCells, const tarch::la::Vector<dim, unsigned int>& testNumberProcesses,
197 const tarch::la::Vector<dim, double>& size, const tarch::la::Vector<dim, double>& offset) {
198 const coupling::IndexConversion<dim> indexConversion(testGlobalNumberCells, testNumberProcesses, 0, size, offset, coupling::paralleltopology::XYZ);
199 const tarch::la::Vector<3, unsigned int> end = coupling::initRange<dim>(testGlobalNumberCells + tarch::la::Vector<dim, unsigned int>(2));
200 tarch::la::Vector<3, unsigned int> loop(0);
201 for (loop[2] = 0; loop[2] < end[2]; loop[2]++) {
202 for (loop[1] = 0; loop[1] < end[1]; loop[1]++) {
203 for (loop[0] = 0; loop[0] < end[0]; loop[0]++) {
204 std::cout << "Unique rank for cell " << coupling::initDimVector<dim>(loop) << ": "
205 << indexConversion.getUniqueRankForMacroscopicCell(coupling::initDimVector<dim>(loop)) << std::endl;
206 }
207 }
