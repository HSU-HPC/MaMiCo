94 coupling::interface::EspressoMDSolverInterface _test;

96 Particle newpart;
97 newpart.p.identity = max_seen_particle + 1;
98 tarch::la::Vector<3, double> pos(0.05);

101 newpart.r.p[i] = pos[i];
102 newpart.m.v[i] = vel[i];
103 newpart.f.f[i] = vel[i];

105 newpart.p.type = 0;
106 coupling::interface::EspressoMDMolecule new_part(&newpart);
107 new_part.setPosition(pos);
108 new_part.setVelocity(vel);
109 new_part.setForce(vel);

111 // Call the method addMoleculeToMDSimulation provided in
112 // EspressoMDSolverInterface
113 _test.addMoleculeToMDSimulation(new_part);

115 // Iterate over all the particles in the local domain (excluding ghost
116 // cells) in Espresso
117 Cell *cell;
118 int c, i, np, cnt = 0;
119 for (c = 0; c < local_cells.n; c++) {
