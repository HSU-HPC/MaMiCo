11template <unsigned int dim>
12class TestLammpsInitialVelocity : public TestLammps<dim> {
13public:
14 TestLammpsInitialVelocity(int argc, char **argv, std::string name)
15 : TestLammps<dim>(argc, argv, name) {}
16 virtual ~TestLammpsInitialVelocity() {}
17
18 virtual void run() {
19 // initialise all interfaces and simulation parts
20 if (dim == 2) {
21 TestLammps<dim>::loadLammpsTestConfiguration(
22 "inputpositions2D_moleculeiterator.xyz", 24);
23 } else {
24 TestLammps<dim>::loadLammpsTestConfiguration(
25 "inputpositions3D_moleculeiterator.xyz", 96);
26 }
27 TestLammps<dim>::loadMacroscopicSolverConfiguration();
28 TestLammps<dim>::loadMamicoTestConfiguration();
29 coupling::interface::MDSolverInterface<LAMMPS_NS::MamicoCell, dim>
30 *mdInterface =
31 coupling::interface::MamicoInterfaceProvider<LAMMPS_NS::MamicoCell,
32 dim>::getInstance()
33 .getMDSolverInterface();
34
35 const double kB = mdInterface->getKB(); // value for kB
36 const double temperature = 3.5; // value for temperature
37 const tarch::la::Vector<dim, double> meanVelocity(
38 1.0); // value for avg velocity
39 const double stdDev =
40 std::sqrt(dim * kB * temperature / mdInterface->getMoleculeMass());
41 const int numberExperiments = 100000;
42 const double tol = 1.0e-2;
43
44 std::vector<tarch::la::Vector<dim, double>> velocities;
45 tarch::la::Vector<dim, double> expAvg(0.0);
46 double expStdDev = 0.0;
47
48 // sample velocities
49 for (int i = 0; i < numberExperiments; i++) {
50 tarch::la::Vector<dim, double> initVel(0.0);
51 mdInterface->getInitialVelocity(meanVelocity, kB, temperature, initVel);
52 velocities.push_back(initVel);
53 }
54
55 // compute avg
56 for (int i = 0; i < numberExperiments; i++) {
57 expAvg = expAvg + velocities[i];
58 }
59 expAvg = (1.0 / numberExperiments) * expAvg;
60 // compute std dev
61 for (int i = 0; i < numberExperiments; i++) {
62 expStdDev = expStdDev + tarch::la::dot(velocities[i] - expAvg,
63 velocities[i] - expAvg);
64 }
65 expStdDev = std::sqrt(expStdDev / numberExperiments);
66
67 std::cout << "Average: " << meanVelocity << " measured: " << expAvg
68 << ", ||. - .||_2: " << tarch::la::norm2(meanVelocity - expAvg)
69 << std::endl;
70 std::cout << "Std-dev: " << stdDev << " measured: " << expStdDev
71 << ", ||. - .||_2: " << fabs(stdDev - expStdDev) << std::endl;
72 if (tarch::la::norm2(meanVelocity - expAvg) > tol) {
73 std::cout
74 << "ERROR TestLammpsInitialVelocity: Avg. velocity deviation exceeds "
75 << tol << "!" << std::endl;
76 exit(EXIT_FAILURE);
77 }
78 if (fabs(stdDev - expStdDev) > tol) {
79 std::cout
80 << "ERROR TestLammpsInitialVelocity: Std. dev. deviation exceeds "
81 << tol << "!" << std::endl;
82 exit(EXIT_FAILURE);
83 }
84 }
85};
