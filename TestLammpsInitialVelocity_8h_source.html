11template <unsigned int dim> class TestLammpsInitialVelocity : public TestLammps<dim> {
12public:
13 TestLammpsInitialVelocity(int argc, char** argv, std::string name) : TestLammps<dim>(argc, argv, name) {}
14 virtual ~TestLammpsInitialVelocity() {}
15
16 virtual void run() {
17 // initialise all interfaces and simulation parts
18 if (dim == 2) {
19 TestLammps<dim>::loadLammpsTestConfiguration("inputpositions2D_moleculeiterator.xyz", 24);
20 } else {
21 TestLammps<dim>::loadLammpsTestConfiguration("inputpositions3D_moleculeiterator.xyz", 96);
22 }
23 TestLammps<dim>::loadMacroscopicSolverConfiguration();
24 TestLammps<dim>::loadMamicoTestConfiguration();
25 coupling::interface::MDSolverInterface<LAMMPS_NS::MamicoCell, dim>* mdInterface =
26 coupling::interface::MamicoInterfaceProvider<LAMMPS_NS::MamicoCell, dim>::getInstance().getMDSolverInterface();
27
28 const double kB = mdInterface->getKB(); // value for kB
29 const double temperature = 3.5; // value for temperature
30 const tarch::la::Vector<dim, double> meanVelocity(1.0); // value for avg velocity
31 const double stdDev = std::sqrt(dim * kB * temperature / mdInterface->getMoleculeMass());
32 const int numberExperiments = 100000;
33 const double tol = 1.0e-2;
34
35 std::vector<tarch::la::Vector<dim, double>> velocities;
36 tarch::la::Vector<dim, double> expAvg(0.0);
37 double expStdDev = 0.0;
38
39 // sample velocities
40 for (int i = 0; i < numberExperiments; i++) {
41 tarch::la::Vector<dim, double> initVel(0.0);
42 mdInterface->getInitialVelocity(meanVelocity, kB, temperature, initVel);
43 velocities.push_back(initVel);
44 }
45
46 // compute avg
47 for (int i = 0; i < numberExperiments; i++) {
48 expAvg = expAvg + velocities[i];
49 }
50 expAvg = (1.0 / numberExperiments) * expAvg;
51 // compute std dev
52 for (int i = 0; i < numberExperiments; i++) {
53 expStdDev = expStdDev + tarch::la::dot(velocities[i] - expAvg, velocities[i] - expAvg);
54 }
55 expStdDev = std::sqrt(expStdDev / numberExperiments);
56
57 std::cout << "Average: " << meanVelocity << " measured: " << expAvg << ", ||. - .||_2: " << tarch::la::norm2(meanVelocity - expAvg) << std::endl;
58 std::cout << "Std-dev: " << stdDev << " measured: " << expStdDev << ", ||. - .||_2: " << fabs(stdDev - expStdDev) << std::endl;
59 if (tarch::la::norm2(meanVelocity - expAvg) > tol) {
60 std::cout << "ERROR TestLammpsInitialVelocity: Avg. velocity deviation exceeds " << tol << "!" << std::endl;
61 exit(EXIT_FAILURE);
62 }
63 if (fabs(stdDev - expStdDev) > tol) {
64 std::cout << "ERROR TestLammpsInitialVelocity: Std. dev. deviation exceeds " << tol << "!" << std::endl;
65 exit(EXIT_FAILURE);
66 }
67 }
68};
