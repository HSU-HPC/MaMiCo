66 tarch::la::Vector<3,unsigned int> numberProcesses(1);
67 unsigned int rank=0;

70 tarch::la::Vector<3,double> macroscopicCellSize(0.0);

72 for(unsigned int d=0; d<3; d++){
73 globalMDDomainSize[d]=30.0;
74 globalMDDomainOffset[d]=0.0;
75 macroscopicCellSize[d]=1.25*linkedCellsPerMacroscopicCell[d];

82 unsigned int sendCounter=0;
83 unsigned int recvCounter=0;

85 // Update the sendCounter or recvCounter based on whether the cell sends data to or receives data from MD Solver
86 for (loop[2]=2; loop[2] < 16; loop[2]++){
87 for (loop[1]=2; loop[1] < 16; loop[1]++){
88 for (loop[0]=2; loop[0] < 16; loop[0]++){
89 const tarch::la::Vector<3,unsigned int> index=coupling::initDimVector<3>(loop);
90 const double mass=_dummySolver.getDensity(index[0],index[1],index[2]);
91 const tarch::la::Vector<3,double> momentum=_dummySolver.getVelocity(index[0],index[1],index[2]);
92 bool flagsend=DummySolverInterfaceService::getInstance().addToSendBuffer(mass, momentum, index);
93 if(flagsend==true) {
94 sendCounter++;
95 }
96 double massr = 1.0;
97 tarch::la::Vector<3,double> momentumr(1.0);
98 bool flagrecv=DummySolverInterfaceService::getInstance().getFromReceiveBuffer(massr, momentumr, index);
99 if(flagrecv==true) {
100 recvCounter++;
101 }
102 }
103 }
104 }
105
106 if(sendCounter == 2232 && recvCounter == 512) {
107 std::cout << "The test was successful" << std::endl;
