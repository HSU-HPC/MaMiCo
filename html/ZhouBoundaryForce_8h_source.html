85 void handleCell(LinkedCell& cell,const unsigned int &cellIndex){
86 coupling::interface::MoleculeIterator<LinkedCell,dim> *it = _mdSolverInterface->getMoleculeIterator(cell);
87 it->begin();
88 while(it->continueIteration()){
89 coupling::interface::Molecule<dim> &wrapper(it->get());
90 //std::cout << "apply force to Molecule " << wrapper.getPosition() << std::endl;
91
92 // extract position and force of each molecule
93 const tarch::la::Vector<dim,double> position(wrapper.getPosition());
94 tarch::la::Vector<dim,double> force(wrapper.getForce());
95 // add boundary force
96 force = force + getBoundaryForces(position);
97 wrapper.setForce(force);
98
99 it->next();
100 }
101 delete it;

104 double getPotentialEnergy(const tarch::la::Vector<dim,double>& position) const {
105 double energy(0);
106 const double distance = 2.5*_sigma;
107 for (unsigned int d = 0; d < dim; d++){
108 if (_boundary[2*d] && (position[d]-_domainLowerLeft[d]<distance)) { energy += getPotentialEnergy((position[d]-_domainLowerLeft[d])/_sigma);}
109 if (_boundary[2*d+1] && (_domainUpperRight[d]-position[d]<distance)){ energy += getPotentialEnergy((_domainUpperRight[d]-position[d])/_sigma);}
110 }
111 return energy;
112 }

117 tarch::la::Vector<dim,double> getBoundaryForces(const tarch::la::Vector<dim,double>& position) const {
118 tarch::la::Vector<dim,double> force(0.0);
119 const double distance = 2.5*_sigma;
120
121 for (unsigned int d = 0; d < dim; d++){
122 // for left/lower/back boundary: add force; for right/upper/front boundary: subtract force value
123 if (_boundary[2*d] && (position[d]-_domainLowerLeft[d]<distance)) { force[d] += _forceFactor*getScalarBoundaryForce( (position[d]-_domainLowerLeft[d])/_sigma); }
124 if (_boundary[2*d+1] && (_domainUpperRight[d]-position[d]<distance)){ force[d] -= _forceFactor*getScalarBoundaryForce((_domainUpperRight[d]-position[d])/_sigma); }
125 }
126 return force;
127 }

132 double getPotentialEnergy(double rw) const {
133 double dx(2.5 * _sigma / (_energyResolution-1));
134 int upper = (int)((2.5-rw)/2.5*(_energyResolution-1)-.5);
135 int lower = (int)((2.5-rw)/2.5*(_energyResolution-1)+.5);
136 return _energyTable[lower] * ((2.5 - upper * dx - rw) / dx)
137 + _energyTable[upper] * ((rw - 2.5 + lower * dx) / dx);
138 }

141 double getScalarBoundaryForce(double rw) const {
142 if (rw < 1.04){ return _p1 + _p2*exp(pow((rw+0.25),3.4))*cos(_p3*rw) ; }
143 else { return -1.0/(_q1+_q2*(2.5-rw)*(2.5-rw) + _q3/((2.5-rw)*(2.5-rw))); }
144 }

147 double getP1(const double& density, const double& temperature) const {
148 const double logrho = log(density);
149 const double T2 = temperature*temperature;
150 const double T3 = T2*temperature;
151
152 // an additional factor 10**-5 is in the expression 4.599/(...) -> private communication with Wenjing Zhou (typo in paper)
153 const double p1 = (-18.953 + 53.369*temperature - 1.253*T2 + 4.599/100000.0*T3
154 +59.871*logrho + 19.737*logrho*logrho)/
155 (1.0 + 2.592*temperature - 0.557*T2 + 0.049*T3
156 -13.912*logrho + 18.657*logrho*logrho );
157 return p1;
158 }
159 double getP2(const double& density,const double &temperature) const {
160 const double logrho = log(density);
161 const double T2 = temperature*temperature;
162 const double T3 = T2*temperature;
163
164 const double p2 = (-0.094 + 2.808*temperature - 0.019*T2 - 0.001*T3
165 +2.823*logrho + 2.071*logrho*logrho)/ (1.0 + 0.168*temperature - 0.013*T2
166 -4.323*logrho + 2.557*logrho*logrho - 2.155*logrho*logrho*logrho);
167 return p2;
168 }
169 double getP3(const double& density,const double &temperature) const {
170 const double T394 = pow(temperature,0.394);
171 const double rho17437 = pow(density,17.437);
172 const double p3 = 3.934 + 0.099*T394 - 0.097*rho17437 + 0.075*T394 * rho17437;
173 return p3;
174 }
175 double getQ1(const double& density) const {
176 const double rho2 = density*density;
177 const double rho3 = rho2*density;
178 const double rho4 = rho2*rho2;
179 return -30.471 + 113.879*density - 207.205*rho2 + 184.242*rho3 - 62.879*rho4;
180 }
181 double getQ2(const double& density) const {
182 const double rho2 = density*density;
183 const double rho3 = rho2*density;
184 const double rho4 = rho2*rho2;
185 return 6.938 - 25.788*density + 46.773*rho2 - 41.768*rho3 + 14.394*rho4;
186 }
187 double getQ3(const double& density) const {
188 const double rho2 = density*density;
189 const double rho3 = rho2*density;
190 const double rho4 = rho2*rho2;
191 return 39.634 - 147.821*density + 269.519*rho2 - 239.066*rho3 + 81.439*rho4;
192 }
