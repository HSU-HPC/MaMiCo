50 virtual tarch::la::Vector<dim,double> getVelocity(tarch::la::Vector<dim,double> pos) const {
51 tarch::la::Vector<dim,double> v(0.0);
52 v[0] = _wallVelocity*(1.0-pos[dim-1]/_channelheight);
53 const double pi = 3.141592653589793238;
54 double sum = 0.0;
55 for (int k = 1; k < 30; k++){
56 sum += 1.0/k * sin(k*pi*pos[dim-1]/_channelheight) * exp(-k*k * pi*pi/(_channelheight*_channelheight) * _kinVisc * _time);
57 }
58 v[0] = v[0] - 2.0*_wallVelocity/pi * sum;
59 return v;
60 }

62 virtual void setWallVelocity(const tarch::la::Vector<dim,double> wallVelocity){
63 _wallVelocity = wallVelocity[0];
64 }

66 private:
67 const double _channelheight; // height of couette channel
68 double _wallVelocity; // velocity of moving wall
69 const double _kinVisc; // kinematic viscosity
70 double _time; // current time
71};
