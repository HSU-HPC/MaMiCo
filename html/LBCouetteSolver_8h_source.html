53 std::cout << "tau=" << 1.0/_omega << std::endl;

105 void setMDBoundaryValues(
106 std::vector<coupling::datastructures::MacroscopicCell<3>* >& recvBuffer,const unsigned int * const recvIndices,
107 const coupling::IndexConversion<3>& indexConversion)override{
108 if (skipRank()){return ;}
109
110 // loop over all received cells
111 const unsigned int size = (unsigned int) recvBuffer.size();
112 for (unsigned int i = 0; i < size; i++){
113 // determine cell index of this cell in LB domain
114 tarch::la::Vector<3,unsigned int> globalCellCoords = indexConversion.getGlobalVectorCellIndex(recvIndices[i]);
115 globalCellCoords[0] = (globalCellCoords[0]+_offset[0]) - _coords[0]*_avgDomainSizeX;
116 globalCellCoords[1] = (globalCellCoords[1]+_offset[1]) - _coords[1]*_avgDomainSizeY;
117 globalCellCoords[2] = (globalCellCoords[2]+_offset[2]) - _coords[2]*_avgDomainSizeZ;
118 #if (COUPLING_MD_DEBUG==COUPLING_MD_YES)
119 std::cout << "Process coords: " << _coords << ": GlobalCellCoords for index " << indexConversion.getGlobalVectorCellIndex(recvIndices[i]) << ": " << globalCellCoords << std::endl;
120 #endif
121 const int index = get(globalCellCoords[0],globalCellCoords[1],globalCellCoords[2]);
122 #if (COUPLING_MD_DEBUG==COUPLING_MD_YES)
123 if (_flag[index]!=MD_BOUNDARY){std::cout << "ERROR LBCouetteSolver::setMDBoundaryValues(): Cell " << index << " is no MD boundary cell!" << std::endl; exit(EXIT_FAILURE);}
124 #endif
125 // set velocity value and pdfs in MD boundary cell (before streaming); the boundary velocities are interpolated between the neighbouring and this cell. This interpolation is valid for
126 // FLUID-MD_BOUNDARY neighbouring relations only.
127 // determine local velocity received from MaMiCo and convert it to LB units; store the velocity in _vel
128 tarch::la::Vector<3,double> localVel( (1.0/recvBuffer[i]->getMacroscopicMass())*(_dt/_dx)*recvBuffer[i]->getMacroscopicMomentum());
129 for (unsigned int d = 0; d<3; d++){ _vel[3*index+d] = localVel[d]; }
130 // loop over all pdfs and set them according to interpolated moving-wall conditions
131 for (unsigned int q = 0; q < 19; q++){
132 // index of neighbour cell; only if cell is located inside local domain
133 if ( ((int)globalCellCoords[0]+_C[q][0] > 0) && ((int)globalCellCoords[0]+_C[q][0]<_domainSizeX+1)
134 && ((int)globalCellCoords[1]+_C[q][1] > 0) && ((int)globalCellCoords[1]+_C[q][1]<_domainSizeY+1)
135 && ((int)globalCellCoords[2]+_C[q][2] > 0) && ((int)globalCellCoords[2]+_C[q][2]<_domainSizeZ+1)){
136 const int nbIndex = get((_C[q][0]+globalCellCoords[0]),(_C[q][1]+globalCellCoords[1]),(_C[q][2]+globalCellCoords[2]));
137 const tarch::la::Vector<3,double> interpolVel(0.5*(_vel[3*index]+_vel[3*nbIndex]),0.5*(_vel[3*index+1]+_vel[3*nbIndex+1]),0.5*(_vel[3*index+2]+_vel[3*nbIndex+2]));
138 _pdf1[19*index+q] = _pdf1[19*nbIndex+18-q] - 6.0*_W[q]*_density[nbIndex]*(_C[18-q][0]*interpolVel[0]+_C[18-q][1]*interpolVel[1]+_C[18-q][2]*interpolVel[2]);
139 }
140 }
141 }

144 /*returns velocity at a certain position.*/
145 tarch::la::Vector<3,double> getVelocity(tarch::la::Vector<3,double> pos) const override {
146 tarch::la::Vector<3,unsigned int> coords;
147 const tarch::la::Vector<3,double> domainOffset(_coords[0]*_dx*_avgDomainSizeX,_coords[1]*_dx*_avgDomainSizeY,_coords[2]*_dx*_avgDomainSizeZ);
148
149 // check pos-data for process locality (todo: put this in debug mode in future releases)
150 if ( (pos[0]<domainOffset[0]) || (pos[0]>domainOffset[0]+_domainSizeX*_dx)
151 || (pos[1]<domainOffset[1]) || (pos[1]>domainOffset[1]+_domainSizeY*_dx)
152 || (pos[2]<domainOffset[2]) || (pos[2]>domainOffset[2]+_domainSizeZ*_dx) ){
153 std::cout << "ERROR LBCouetteSolver::getVelocity(): Position " << pos << " out of range!" << std::endl; exit(EXIT_FAILURE);
154 }
155 // compute index for respective cell (_dx+... for ghost cells); use coords to store local cell coordinates
156 for (unsigned int d = 0; d < 3; d++){ coords[d] = (unsigned int) ((_dx+pos[d]-domainOffset[d])/_dx);}
157 const int index = get(coords[0],coords[1],coords[2]);
158 tarch::la::Vector<3,double> vel(0.0);
159 // extract and scale velocity to "real"=MD units
160 for (int d = 0; d < 3; d++){ vel[d] = _dx/_dt*_vel[3*index+d]; }
161 #if (COUPLING_MD_DEBUG==COUPLING_MD_YES)
162 std::cout << "Position " << pos << " corresponds to cell: " << coords << "; vel=" << vel << std::endl;
163 #endif
164 return vel;
165 }

210 // stream from pdf1 to pdf2
211 void stream(int index){
212 const int pI = 19*index;
213 for (int q = 0; q < 9; q++){
214 const int nb= 19*(_C[q][0]+_C[q][1]*_xO+_C[q][2]*_yO);
215 _pdf2[pI+q] = _pdf1[pI+q-nb];
216 _pdf2[pI+18-q] = _pdf1[pI+18-q+nb];
217 }
218 _pdf2[pI+9] = _pdf1[pI+9];
219 }

221 // collide within pdf2
222 void collide(int index,int x, int y, int z){
223 // index of start of cell-local pdfs in AoS
224 const int pI = 19*index;
225
226 // compute and store density, velocity
227 double *vel = &_vel[3*index];
228 computeDensityAndVelocity(vel,_density[index],&_pdf2[pI]);
229
230 // collide (BGK); always handle pdfs no. q and inv(q)=18-q in one step
231 const double u2 = 1.0 - 1.5*(vel[0]*vel[0] + vel[1]*vel[1] + vel[2]*vel[2]);
232 // pdf 0,18
233 double cu = -vel[1]-vel[2];
234 int nb = -_xO-_yO;
235 double feq = _W[0]*_density[index]*(u2 + 3.0*cu + 4.5*cu*cu);
236 _pdf2[pI] -= _omega*(_pdf2[pI] - feq);
237 boundary(_pdf2,pI,x,y,z,0,_flag[index+nb],pI+19*nb);
238 feq -= 6.0*_W[0]*_density[index]*cu;
239 _pdf2[pI+18] -= _omega*(_pdf2[pI+18] - feq);
240 boundary(_pdf2,pI,x,y,z,18,_flag[index-nb],pI-19*nb);
241 // pdf 1,17
242 cu = -vel[0]-vel[2];
243 nb = -1-_yO;
244 feq = _W[1]*_density[index]*(u2 + 3.0*cu + 4.5*cu*cu);
245 _pdf2[pI+1] -= _omega*(_pdf2[pI+1] - feq);
246 boundary(_pdf2,pI,x,y,z,1,_flag[index+nb],pI+19*nb);
247 feq -= 6.0*_W[1]*_density[index]*cu;
248 _pdf2[pI+17] -= _omega*(_pdf2[pI+17] - feq);
249 boundary(_pdf2,pI,x,y,z,17,_flag[index-nb],pI-19*nb);
250 // pdf 2,16
251 cu = -vel[2];
252 nb = -_yO;
253 feq = _W[2]*_density[index]*(u2 + 3.0*cu + 4.5*cu*cu);
254 _pdf2[pI+2] -= _omega*(_pdf2[pI+2] - feq);
255 boundary(_pdf2,pI,x,y,z,2,_flag[index+nb],pI+19*nb);
256 feq -= 6.0*_W[2]*_density[index]*cu;
257 _pdf2[pI+16] -= _omega*(_pdf2[pI+16] - feq);
258 boundary(_pdf2,pI,x,y,z,16,_flag[index-nb],pI-19*nb);
259 // pdf 3,15
260 cu = vel[0]-vel[2];
261 nb = 1-_yO;
262 feq = _W[3]*_density[index]*(u2 + 3.0*cu + 4.5*cu*cu);
263 _pdf2[pI+3] -= _omega*(_pdf2[pI+3] - feq);
264 boundary(_pdf2,pI,x,y,z,3,_flag[index+nb],pI+19*nb);
265 feq -= 6.0*_W[3]*_density[index]*cu;
266 _pdf2[pI+15] -= _omega*(_pdf2[pI+15] - feq);
267 boundary(_pdf2,pI,x,y,z,15,_flag[index-nb],pI-19*nb);
268 // pdf 4,14
269 cu = vel[1]-vel[2];
270 nb = _xO-_yO;
271 feq = _W[4]*_density[index]*(u2 + 3.0*cu + 4.5*cu*cu);
272 _pdf2[pI+4] -= _omega*(_pdf2[pI+4] - feq);
273 boundary(_pdf2,pI,x,y,z,4,_flag[index+nb],pI+19*nb);
274 feq -= 6.0*_W[4]*_density[index]*cu;
275 _pdf2[pI+14] -= _omega*(_pdf2[pI+14] - feq);
276 boundary(_pdf2,pI,x,y,z,14,_flag[index-nb],pI-19*nb);
277 // pdf 5,13
278 cu = -vel[0]-vel[1];
279 nb = -1-_xO;
280 feq = _W[5]*_density[index]*(u2 + 3.0*cu + 4.5*cu*cu);
281 _pdf2[pI+5] -= _omega*(_pdf2[pI+5] - feq);
282 boundary(_pdf2,pI,x,y,z,5,_flag[index+nb],pI+19*nb);
283 feq -= 6.0*_W[5]*_density[index]*cu;
284 _pdf2[pI+13] -= _omega*(_pdf2[pI+13] - feq);
285 boundary(_pdf2,pI,x,y,z,13,_flag[index-nb],pI-19*nb);
286 // pdf 6,12
287 cu = -vel[1];
288 nb = -_xO;
289 feq = _W[6]*_density[index]*(u2 + 3.0*cu + 4.5*cu*cu);
290 _pdf2[pI+6] -= _omega*(_pdf2[pI+6] - feq);
291 boundary(_pdf2,pI,x,y,z,6,_flag[index+nb],pI+19*nb);
292 feq -= 6.0*_W[6]*_density[index]*cu;
293 _pdf2[pI+12] -= _omega*(_pdf2[pI+12] - feq);
294 boundary(_pdf2,pI,x,y,z,12,_flag[index-nb],pI-19*nb);
295 // pdf 7,11
296 cu = vel[0]-vel[1];
297 nb = 1-_xO;
298 feq = _W[7]*_density[index]*(u2 + 3.0*cu + 4.5*cu*cu);
299 _pdf2[pI+7] -= _omega*(_pdf2[pI+7] - feq);
300 boundary(_pdf2,pI,x,y,z,7,_flag[index+nb],pI+19*nb);
301 feq -= 6.0*_W[7]*_density[index]*cu;
302 _pdf2[pI+11] -= _omega*(_pdf2[pI+11] - feq);
303 boundary(_pdf2,pI,x,y,z,11,_flag[index-nb],pI-19*nb);
304 // pdf 8,10
305 cu = -vel[0];
306 nb = -1;
307 feq = _W[8]*_density[index]*(u2 + 3.0*cu + 4.5*cu*cu);
308 _pdf2[pI+8] -= _omega*(_pdf2[pI+8] - feq);
309 boundary(_pdf2,pI,x,y,z,8,_flag[index+nb],pI+19*nb);
310 feq -= 6.0*_W[8]*_density[index]*cu;
311 _pdf2[pI+10] -= _omega*(_pdf2[pI+10] - feq);
312 boundary(_pdf2,pI,x,y,z,10,_flag[index-nb],pI-19*nb);
313 // pdf 9
314 _pdf2[pI+9] -= _omega*(_pdf2[pI+9] - _W[9]*_density[index]*u2);
315 }

317 // boundary treatment; pdf - distributions (typically _pdf2), index - start index for current cell in pdf-array, x/y/z - cell coordinates of current cell,
318 // q - distribution function no., flag - boundary flag of neighbouring cell, nbIndex - index of neighbouring cell
319 void boundary(double * const pdf, int index, int x, int y, int z, int q, const Flag &flag,int nbIndex){
320 if (flag!=FLUID){
321 if (flag==NO_SLIP){
322 // half-way bounce back
323 pdf[nbIndex+18-q] = pdf[index+q];
324 }
325 else if (flag==MOVING_WALL){
326 // half-way bounce back + moving wall acceleration (only x-direction for wall supported at the moment)
327 pdf[nbIndex+18-q] = pdf[index+q] - 6.0*_W[q]*_density[index/19]*(_C[q][0]*_wallVelocity[0]+_C[q][1]*_wallVelocity[1]+_C[q][2]*_wallVelocity[2]);
328 }
329 else if (flag==PERIODIC){
330 // periodic treatment
331 int target[3] = {x,y,z};
332 if (target[0]+_C[q][0]==0){target[0] = _domainSizeX+1;} else if (target[0]+_C[q][0]==_domainSizeX+1){target[0] = 0;}
333 if (target[1]+_C[q][1]==0){target[1] = _domainSizeY+1;} else if (target[1]+_C[q][1]==_domainSizeY+1){target[1] = 0;}
334 if (target[2]+_C[q][2]==0){target[2] = _domainSizeZ+1;} else if (target[2]+_C[q][2]==_domainSizeZ+1){target[2] = 0;}
335 const int periodicNb = target[0] + (_domainSizeX+2)*(target[1] + (_domainSizeY+2)*target[2]);
336 pdf[19*periodicNb+q] = pdf[index+q];
337 }
338 }
339 }

341 //compute density and velocity on pdf
342 void computeDensityAndVelocity(double * const vel, double &density, const double * const pdf){
343 vel[0] = -(pdf[1]+pdf[5]+pdf[8]+pdf[11]+pdf[15]);
344 density= pdf[3]+pdf[7]+pdf[10]+pdf[13]+pdf[17];
345 vel[1] = (pdf[4]+pdf[11]+pdf[12]+pdf[13]+pdf[18]) - (pdf[0]+pdf[5]+pdf[6]+pdf[7]+pdf[14]);
346 vel[0] = density + vel[0];
347 density= density + pdf[0]+pdf[1]+pdf[2] + pdf[4]+pdf[5]+pdf[6] + pdf[8]+pdf[9] + pdf[11]+pdf[12] + pdf[14]+pdf[15]+pdf[16] + pdf[18];
348 vel[2] = (pdf[14]+pdf[15]+pdf[16]+pdf[17]+pdf[18]) - (pdf[0]+pdf[1]+pdf[2]+pdf[3]+pdf[4]);
349 vel[0] = vel[0]/density;
350 vel[1] = vel[1]/density;
351 vel[2] = vel[2]/density;
352 }

362 void communicatePart(double *pdf, double *sendBuffer, double *recvBuffer, NbFlag nbFlagTo, NbFlag nbFlagFrom,
363 tarch::la::Vector<3,int> startSend, tarch::la::Vector<3,int> endSend,
364 tarch::la::Vector<3,int> startRecv, tarch::la::Vector<3,int> endRecv){
365 #if (COUPLING_MD_PARALLEL==COUPLING_MD_YES)
366 // directions that point to LEFT/RIGHT,... -> same ordering as enums!
367 const int directions[6][5] = { { 1, 5, 8,11,15}, { 3, 7,10,13,17},
368 { 4,11,12,13,18}, { 0, 5, 6, 7,14},
369 { 0, 1, 2, 3, 4}, {14,15,16,17,18} };
370 MPI_Request requests[2];
371 MPI_Status status[2];
372 tarch::la::Vector<2,int> plane;
373 tarch::la::Vector<2,int> domainSize;
374 // find out plane coordinates
375 if (nbFlagTo==LEFT || nbFlagTo==RIGHT){plane[0]=1; plane[1]=2; domainSize[0] = _domainSizeY; domainSize[1] = _domainSizeZ;
376 } else if (nbFlagTo==FRONT|| nbFlagTo==BACK) {plane[0]=0; plane[1]=2; domainSize[0] = _domainSizeX; domainSize[1] = _domainSizeZ;
377 } else if (nbFlagTo==TOP || nbFlagTo==BOTTOM){plane[0]=0; plane[1]=1; domainSize[0] = _domainSizeX; domainSize[1] = _domainSizeY; }
378 else { std::cout << "ERROR LBCouetteSolver::communicatePart: d >2 or d < 0!" << std::endl; exit(EXIT_FAILURE);}
379 // extract data and write to send buffer
380 tarch::la::Vector<3,int> coords(0);
381 for (coords[2] = startSend[2]; coords[2] < endSend[2]; coords[2]++){
382 for (coords[1] = startSend[1]; coords[1] < endSend[1]; coords[1]++){
383 for (coords[0] = startSend[0]; coords[0] < endSend[0]; coords[0]++){
384 for (int q = 0; q < 5; q++){
385 sendBuffer[q+5*getParBuf(coords[plane[0]],coords[plane[1]],domainSize[0],domainSize[1])] = pdf[directions[nbFlagTo][q]+19*get(coords[0],coords[1],coords[2])];
386 }
387 }}}
388 // send and receive data
389 MPI_Irecv(recvBuffer,(domainSize[0]+2)*(domainSize[1]+2)*5,MPI_DOUBLE,_parallelNeighbours[nbFlagFrom],1000,MPI_COMM_WORLD,&requests[0]);
390 MPI_Isend(sendBuffer,(domainSize[0]+2)*(domainSize[1]+2)*5,MPI_DOUBLE,_parallelNeighbours[nbFlagTo], 1000,MPI_COMM_WORLD,&requests[1]);
391 MPI_Waitall(2,requests,status);
392 // write data back to pdf field
393 if (_parallelNeighbours[nbFlagFrom]!=MPI_PROC_NULL){
394 for (coords[2] = startRecv[2]; coords[2] < endRecv[2]; coords[2]++){
395 for (coords[1] = startRecv[1]; coords[1] < endRecv[1]; coords[1]++){
396 for (coords[0] = startRecv[0]; coords[0] < endRecv[0]; coords[0]++){
397 for (int q = 0; q < 5; q++){
