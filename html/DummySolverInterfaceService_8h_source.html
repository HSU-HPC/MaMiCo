66 bool isInsideTransferRegion(const tarch::la::Vector<3,unsigned int> &index) const {
67 bool isInside = true;
68 tarch::la::Vector<3,unsigned int> _transferDomainOffsetIndex(2);
69 tarch::la::Vector<3,unsigned int> _transferDomainNumCells(0);
70 for (unsigned int d=0; d<3; d++) {
71 _transferDomainNumCells[d] = _globalNumberMacroscopicCells[d] + 2;
72 }
73 for (unsigned int d=0; d<3; d++) {
74 isInside = isInside && (index[d] >= _transferDomainOffsetIndex[d]) && (index[d] < _transferDomainOffsetIndex[d]+_transferDomainNumCells[d]);
75 }
76 return isInside;
77 }

80 tarch::la::Vector<3,unsigned int> globalCellIndexfromLocalIndex(const tarch::la::Vector<3,unsigned int> _localIndex) const {
81 tarch::la::Vector<3,unsigned int> _globalIndex(0);
82 bool flag=isInsideTransferRegion(_localIndex);
83 tarch::la::Vector<3,unsigned int> _transferDomainOffsetIndex(2);

85 for (unsigned int d = 0; d < 3; d++) {
86 _globalIndex[d]=_localIndex[d]-_transferDomainOffsetIndex[d];

91 std::cout << "ERROR this cell does not lie in the transfer region" << std::endl;
