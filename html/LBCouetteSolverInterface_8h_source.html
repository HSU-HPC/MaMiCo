55 virtual std::vector<unsigned int> getRanks(tarch::la::Vector<3,unsigned int> globalCellIndex){
56 std::vector<unsigned int> ranks;
57
58 // determine global index of cell in LB simulation
59 tarch::la::Vector<3,unsigned int> globalLBCellIndex(globalCellIndex + _offsetMDDomain);
60 // modify global LB cell index due to ghost layer
61 for (int d = 0; d < 3; d++){
62 #if (COUPLING_MD_DEBUG==COUPLING_MD_YES)
63 std::cout << "LB cell index for global cell index " << globalCellIndex << ": " << globalLBCellIndex << std::endl;
64 #endif
65 if (globalLBCellIndex[d]>0){ globalLBCellIndex[d]--;}
66 }
67
68 // loop over all neighbouring cells within a one-cell surrounding and detect the respective ranks. IMPROVE: This currently only allows for simulations with MD located inside the domain
69 // (no simulation across boundary)
70 for (int z = -1; z < 2; z++){ for (int y = -1; y < 2; y++){ for (int x = -1; x < 2; x++){
71 // neighbour cell index
72 const tarch::la::Vector<3,unsigned int> nbIndex(globalLBCellIndex[0]+x,globalLBCellIndex[1]+y,globalLBCellIndex[2]+z);
73 // coordinates of process of neighbour celll
74 const tarch::la::Vector<3,unsigned int> processCoordinates((nbIndex[0]/_avgNumberLBCells[0]),(nbIndex[1]/_avgNumberLBCells[1]),(nbIndex[2]/_avgNumberLBCells[2]));
75 // corresponding rank
76 const unsigned int rank = processCoordinates[0] + _numberProcesses[0]*(processCoordinates[1]+processCoordinates[2]*_numberProcesses[1]);
77
78 // if this rank is not part of the vector, push it back
79 bool found=false;
80 for (unsigned int i = 0; i<ranks.size();i++){found= found||(rank==ranks[i]);}
81 if (!found){ ranks.push_back(rank); }
82 }}}
83 #if (COUPLING_MD_DEBUG==COUPLING_MD_YES)
84 std::cout << "Ranks for cell " << globalCellIndex << ":"; for (unsigned int i = 0; i < ranks.size(); i++){ std::cout << " " << ranks[i];} std::cout << std::endl;
85 #endif
86 return ranks;
87 }
