137 void testMolecules(
138 coupling::datastructures::MacroscopicCells<LAMMPS_NS::MamicoCell,dim>& macroscopicCells,
139 coupling::cellmappings::MoleculeExtractor<LAMMPS_NS::MamicoCell,dim>& moleculeExtractor,
140 const coupling::IndexConversion<dim>& indexConversion,
141 const std::vector<tarch::la::Vector<2,unsigned int> > &numberMoleculesPerMacroscopicCell
142 ){
143 const tarch::la::Vector<dim,unsigned int> localCells = indexConversion.getLocalNumberMacroscopicCells();
144 tarch::la::Vector<dim,unsigned int> loop(0);
145 if (dim==2){
146 for (loop[1] = 1; loop[1] < localCells[1]+1; loop[1]++){
147 for (loop[0] = 1; loop[0] < localCells[0]+1; loop[0]++){
148 // determine linearised indices
149 const unsigned int localIndex = indexConversion.getLocalCellIndex(loop);
150 const unsigned int globalIndex = indexConversion.convertLocalToGlobalCellIndex(localIndex);
151 // determine number of molecules in this cell, based on cell structure
152 (macroscopicCells.getMacroscopicCellsWithLinkedCells())[localIndex].iterateConstCells(moleculeExtractor);
153 const unsigned int numberMoleculesFound = (unsigned int) moleculeExtractor.getExtractedMolecules().size();
154 // determine the number of molecules as it was expected
155 const unsigned int numberMoleculesExpected = findNumberMolecules(globalIndex,numberMoleculesPerMacroscopicCell);
156 if (numberMoleculesFound != numberMoleculesExpected){
157 std::cout << "ERROR TestLammpsMoleculeIterator: Found molecules=" << numberMoleculesFound << ", expected=" << numberMoleculesExpected << std::endl;
158 std::cout << "This process: " << indexConversion.getThisProcess() << "; considered (global cell): " << globalIndex << std::endl;
159 exit(EXIT_FAILURE);
160 }
161 std::cout << "Global cell index=" << globalIndex << ", molecule coordinates:" << std::endl;
162 for (unsigned int i = 0; i < numberMoleculesFound; i++){
163 std::cout << (moleculeExtractor.getExtractedMolecules())[i] << std::endl;
164 }
165 }
166 }
167 } else {
168 for (loop[2] = 1; loop[2] < localCells[2]+1; loop[2]++){
169 for (loop[1] = 1; loop[1] < localCells[1]+1; loop[1]++){
170 for (loop[0] = 1; loop[0] < localCells[0]+1; loop[0]++){
171 // determine linearised indices
172 const unsigned int localIndex = indexConversion.getLocalCellIndex(loop);
173 const unsigned int globalIndex = indexConversion.convertLocalToGlobalCellIndex(localIndex);
174 // determine number of molecules in this cell, based on cell structure
175 (macroscopicCells.getMacroscopicCellsWithLinkedCells())[localIndex].iterateConstCells(moleculeExtractor);
176 const unsigned int numberMoleculesFound = (unsigned int) moleculeExtractor.getExtractedMolecules().size();
177 // determine the number of molecules as it was expected
178 const unsigned int numberMoleculesExpected = findNumberMolecules(globalIndex,numberMoleculesPerMacroscopicCell);
179 if (numberMoleculesFound != numberMoleculesExpected){
180 std::cout << "ERROR TestLammpsMoleculeIterator: Found molecules=" << numberMoleculesFound << ", expected=" << numberMoleculesExpected << std::endl;
181 std::cout << "This process: " << indexConversion.getThisProcess() << "; considered (global cell): " << globalIndex << std::endl;

198 unsigned int findNumberMolecules(const unsigned int &globalIndex, const std::vector<tarch::la::Vector<2,unsigned int> > &numberMoleculesPerMacroscopicCell) const {
199 const unsigned int size = (unsigned int) numberMoleculesPerMacroscopicCell.size();
200 for (unsigned int i = 0; i < size; i++){
201 if (globalIndex==numberMoleculesPerMacroscopicCell[i][0]){ return numberMoleculesPerMacroscopicCell[i][1]; }
202 }

204 std::cout << "ERROR TestLammpsMoleculeIterator::findNumberMolecules(): Could not find molecule number for global cell index " << globalIndex << std::endl;
