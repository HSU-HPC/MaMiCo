18 _domain = new double[_Nx*_Ny*_Nz*4];
19 for(unsigned int x=0; x<_Nx; x++){
20 for(unsigned int y=0; y<_Ny; y++){
21 for(unsigned int z=0; z<_Nz; z++){
22 _domain[x+_Nx*y+_Nx*_Ny*z+_Nx*_Ny*_Nz*0]=0.6; // density=1.0
23 _domain[x+_Nx*y+_Nx*_Ny*z+_Nx*_Ny*_Nz*1]=(4.0*_Vmax*((double)(y)/(_Ny-1))*(1-((double)(y)/(_Ny-1)))); // Vx=Vmax*y*(Ny-y)
24 _domain[x+_Nx*y+_Nx*_Ny*z+_Nx*_Ny*_Nz*2]=0.0; // Vy=0.0
