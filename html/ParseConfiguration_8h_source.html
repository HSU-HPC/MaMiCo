20 const std::string filename,
21 const std::string topleveltag,

23){
24 tinyxml2::XMLDocument conffile;

32
33 config.parseSubtag(node);

126template<unsigned int size,class T>
127static void readVector(tarch::la::Vector<size,T>& result, tinyxml2::XMLElement *node, std::string tag){
128 const char *myText = node->Attribute(tag.c_str());
129 if (myText == NULL){
130 std::cout << "Error while reading mandatory argument " << tag << " of XML element " << node->Name() << std::endl; exit(EXIT_FAILURE);
131 }
132 std::string input(myText);
133 for (unsigned int i = 0; i < size; i++){
134 // search for first non-whitespace entry in this vector entry
135 std::size_t first = input.find_first_not_of(" ");
136 // search for end of this vector component, typically denoted by ";"
137 // -> if this is the last entry, then npos is accepted as well
138 std::size_t last = input.find_first_of(";");
139 // for debugging
140 //std::cout << first << ", " << last << std::endl;
141 if ((i==size-1) && (last==std::string::npos)){last = input.size();}
142
143 std::stringstream ss(input.substr(first,last-first));
144 // for debugging
145 //std::cout << ss.str() << std::endl;
146 ss >> result[i];
147 if (i<size-1){ input = input.substr(last+1,input.size()-last-1); }
148 // for debugging
149 //std::cout << input << std::endl;
150 }
151}
152};
