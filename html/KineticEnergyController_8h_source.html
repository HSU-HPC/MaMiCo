55 void setKineticEnergy(coupling::datastructures::MacroscopicCellWithLinkedCells<LinkedCell,dim>& cell,const double& kineticEnergy) const{
56 // determine mass, momentum and old kinetic energy
57 coupling::cellmappings::ComputeMassMapping<LinkedCell,dim> computeMassMapping(_mdSolverInterface);
58 cell.iterateConstCells(computeMassMapping);
59 coupling::cellmappings::ComputeMomentumMapping<LinkedCell,dim> computeMomentumMapping(_mdSolverInterface);
60 cell.iterateConstCells(computeMomentumMapping);
61 coupling::cellmappings::ComputeKineticEnergyMapping<LinkedCell,dim> computeKineticEnergyMapping(_mdSolverInterface);
62 cell.iterateConstCells(computeKineticEnergyMapping);
63
64 // set new kinetic energy
65 unsigned int numberParticles = computeMassMapping.getNumberOfParticles();
66 tarch::la::Vector<dim,double> meanVelocity = computeMomentumMapping.getMeanVelocity();
67 double oldKineticEnergy = computeKineticEnergyMapping.getKineticEnergy();
68 coupling::cellmappings::SetKineticEnergyMapping<LinkedCell,dim> setKineticEnergyMapping(
69 oldKineticEnergy,kineticEnergy,numberParticles,meanVelocity
70 );
71 cell.iterateCells(setKineticEnergyMapping);
72 }

86 coupling::cellmappings::SetTemperatureMapping<LinkedCell,dim> setTemperatureMapping(currentTemperature,temperature,meanVelocity,_mdSolverInterface);
87 cell.iterateCells(setTemperatureMapping);
