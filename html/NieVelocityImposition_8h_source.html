23template<class LinkedCell,unsigned int dim>
24class coupling::NieVelocityImposition:
25public coupling::MomentumInsertion<LinkedCell,dim> {
26 public:
27 NieVelocityImposition(
28 coupling::interface::MDSolverInterface<LinkedCell,dim> * const mdSolverInterface,
29 const coupling::IndexConversion<dim>& indexConversion,
30 const unsigned int &outermostLayer,
31 const unsigned int &innermostLayer
32 ):
33 coupling::MomentumInsertion<LinkedCell,dim>(mdSolverInterface),
34 _indexConversion(indexConversion),

40 virtual unsigned int getTimeIntervalPerMomentumInsertion() const { return 1;}
41
42 virtual void insertMomentum(
43 coupling::datastructures::MacroscopicCellWithLinkedCells<LinkedCell,dim>& cell,
44 const unsigned int& currentLocalMacroscopicCellIndex
45 ) const {
46 // nop if this is not an imposition cell
47 if (!isInsideImpositionLayer(currentLocalMacroscopicCellIndex)){ return;}
48 // set continuum velocity
49 tarch::la::Vector<dim,double> continuumVelocity(cell.getMicroscopicMomentum());
50
51 coupling::cellmappings::ComputeAvgForceAndVelocity<LinkedCell,dim> computeForceAndVelocity(coupling::MomentumInsertion<LinkedCell,dim>::_mdSolverInterface);
52 cell.iterateConstCells(computeForceAndVelocity);
53 const tarch::la::Vector<dim,double> avgVel(computeForceAndVelocity.getAvgVelocity());
54 const tarch::la::Vector<dim,double> avgF(computeForceAndVelocity.getAvgForce());
55
56 coupling::cellmappings::NieVelocityImpositionMapping<LinkedCell,dim> velocityImposition(continuumVelocity,avgVel,avgF,coupling::MomentumInsertion<LinkedCell,dim>::_mdSolverInterface);
57 cell.iterateCells(velocityImposition);
58 }
59
60 private:
65 bool isInsideImpositionLayer(const unsigned int &currentLocalMacroscopicCellIndex) const{
66 const tarch::la::Vector<dim,unsigned int> globalNumberMacroscopicCells(_indexConversion.getGlobalNumberMacroscopicCells());
67 const tarch::la::Vector<dim,unsigned int> globalCellIndex(_indexConversion.getGlobalVectorCellIndex(_indexConversion.convertLocalToGlobalCellIndex(currentLocalMacroscopicCellIndex)));
68 bool inner = true;
69 for (unsigned int d = 0; d < dim; d++)
70 inner = inner && (globalCellIndex[d]>_innermostLayer && globalCellIndex[d]< 1+globalNumberMacroscopicCells[d]-_innermostLayer);
71 bool outer = false;
72 for (unsigned int d = 0; d < dim; d++)
73 outer = outer || (globalCellIndex[d]< _outermostLayer || globalCellIndex[d] > 1+globalNumberMacroscopicCells[d]-_outermostLayer);
74 return !inner && !outer;
75 }
76
77 const coupling::IndexConversion<dim>& _indexConversion;
78 const unsigned int _outermostLayer;
79 const unsigned int _innermostLayer;
80};
