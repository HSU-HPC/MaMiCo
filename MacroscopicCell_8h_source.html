162template <class LinkedCell, unsigned int dim>
163class coupling::datastructures::MacroscopicCellWithLinkedCells
164 : public coupling::datastructures::MacroscopicCell<dim> {
165public:
171 MacroscopicCellWithLinkedCells(tarch::la::Vector<dim, unsigned int> blockSize)
172 : coupling::datastructures::MacroscopicCell<dim>(),
173 _numberCells(getNumberCells(blockSize)), _linkedCells(NULL) {
174
175 _linkedCells = new LinkedCell *[_numberCells];
176 if (_linkedCells == NULL) {
177 std::cout
178 << "ERROR coupling::datastructures::MacroscopicCellWithLinkedCells: "
179 "_linkedCells == NULL"
180 << std::endl;
181 exit(EXIT_FAILURE);
182 }
183 // set each pointer to a NULL pointer
184 for (unsigned int i = 0; i < _numberCells; i++) {
185 _linkedCells[i] = NULL;
186 }
187 }
189 virtual ~MacroscopicCellWithLinkedCells() {
190 if (_linkedCells != NULL) {
191 for (unsigned int i = 0; i < _numberCells; i++) {
192 _linkedCells[i] = NULL;
193 }
194 delete[] _linkedCells;
195 _linkedCells = NULL;
196 }
197 }
198
205 void addLinkedCell(LinkedCell &cell, const unsigned int &index) {
206 _linkedCells[index] = &cell;
207 }
208
216 template <class A> void iterateCells(A &a) {
217 a.beginCellIteration();
218 for (unsigned int i = 0; i < _numberCells; i++) {
219 a.handleCell(*(_linkedCells[i]), i);
220 }
221 a.endCellIteration();
222 }
223
232 template <class A> void iterateConstCells(A &a) const {
233 a.beginCellIteration();
234 for (unsigned int i = 0; i < _numberCells; i++) {
235 a.handleCell(*(_linkedCells[i]), i);
236 }
237 a.endCellIteration();
238 }
239
240private:
246 unsigned int
247 getNumberCells(tarch::la::Vector<dim, unsigned int> blockSize) const {
248 unsigned int num = 1;
249 for (unsigned int d = 0; d < dim; d++) {
250 num = num * blockSize[d];
251 }
252 return num;
253 }
254
256 const unsigned int _numberCells;
257
260 LinkedCell **_linkedCells;
261};
