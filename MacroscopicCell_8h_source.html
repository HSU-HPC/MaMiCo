142template <class LinkedCell, unsigned int dim>
143class coupling::datastructures::MacroscopicCellWithLinkedCells : public coupling::datastructures::MacroscopicCell<dim> {
144public:
150 MacroscopicCellWithLinkedCells(tarch::la::Vector<dim, unsigned int> blockSize)
151 : coupling::datastructures::MacroscopicCell<dim>(), _numberCells(getNumberCells(blockSize)), _linkedCells(NULL) {
152
153 _linkedCells = new LinkedCell*[_numberCells];
154 if (_linkedCells == NULL) {
155 std::cout << "ERROR coupling::datastructures::MacroscopicCellWithLinkedCells: "
156 "_linkedCells == NULL"
157 << std::endl;
158 exit(EXIT_FAILURE);
159 }
160 // set each pointer to a NULL pointer
161 for (unsigned int i = 0; i < _numberCells; i++) {
162 _linkedCells[i] = NULL;
163 }
164 }
166 virtual ~MacroscopicCellWithLinkedCells() {
167 if (_linkedCells != NULL) {
168 for (unsigned int i = 0; i < _numberCells; i++) {
169 _linkedCells[i] = NULL;
170 }
171 delete[] _linkedCells;
172 _linkedCells = NULL;
173 }
174 }
175
182 void addLinkedCell(LinkedCell& cell, const unsigned int& index) { _linkedCells[index] = &cell; }
183
191 template <class A> void iterateCells(A& a) {
192 a.beginCellIteration();
193 for (unsigned int i = 0; i < _numberCells; i++) {
194 a.handleCell(*(_linkedCells[i]), i);
195 }
196 a.endCellIteration();
197 }
198
207 template <class A> void iterateConstCells(A& a) const {
208 a.beginCellIteration();
209 for (unsigned int i = 0; i < _numberCells; i++) {
210 a.handleCell(*(_linkedCells[i]), i);
211 }
212 a.endCellIteration();
213 }
214
215private:
221 unsigned int getNumberCells(tarch::la::Vector<dim, unsigned int> blockSize) const {
222 unsigned int num = 1;
223 for (unsigned int d = 0; d < dim; d++) {
224 num = num * blockSize[d];
225 }
226 return num;
227 }

230 const unsigned int _numberCells;

234 LinkedCell** _linkedCells;
