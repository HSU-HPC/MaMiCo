57 {
58 // Initialize flowfield
59 for(int t = 0; t < tws; ++t)
60 for(const tarch::la::Vector<dim, unsigned int>& idx : cellIndices) {
61 coupling::filtering::Quantities<dim>& q = _flowfield(idx, t);
62 q[0] = 1;
63 for(unsigned int d = 1; d <= dim; ++d) q[d] = 0;
64 coupling::filtering::Quantities<dim>& qp = _flowfield_prefiltered(idx, t);
65 qp[0] = 1;
66 for(unsigned int d = 1; d <= dim; ++d) qp[d] = 0;
67 }
68
69 // Initialize innerCellIndices
70 auto domainSize = _ic->getLocalMD2MacroDomainSize();
71 for(auto idx : cellIndices){
72 for (unsigned int d = 0; d < dim; d++)
73 if(idx[d] == 0 || idx[d] == domainSize[d] - 1)
74 goto continue_loop;
75 _innerCellIndices.push_back(idx);
76 continue_loop:;
77 }
78

80 std::cout << " NLM: Created NLM instance." << std::endl;
81 std::cout << " WARNING: Regardless of configuration, NLM always filters macroscopic mass and momentum." << std::endl;
82 #endif
83 }
