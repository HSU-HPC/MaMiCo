64 for(unsigned int d = 1; d <= dim; ++d) q[d] = 0;
65 coupling::filtering::Quantities<dim>& qp = _flowfield_prefiltered(idx, t);
66 qp[0] = 1;
67 for(unsigned int d = 1; d <= dim; ++d) qp[d] = 0;
68 }
69
70 // Initialize innerCellIndices
71 auto domainSize = _ic->getLocalMD2MacroDomainSize();
72 for(auto idx : cellIndices){
73 for (unsigned int d = 0; d < dim; d++)
74 if(idx[d] == 0 || idx[d] == domainSize[d] - 1)
75 goto continue_loop;
76 _innerCellIndices.push_back(idx);
77 continue_loop:;
78 }
