44 : coupling::filtering::JunctorInterface<dim, 2, 1>(
45 {inputCellVector_unfiltered, inputCellVector_prefiltered},
46 {outputCellVector}, filteredValues, "NLM"),
47 _timeWindowSize(tws), _M(M), _d(d), _sigsq(sigsq),
48 _sigsq_rel(sigsq_rel), _hsq(hsq), _hsq_rel(hsq_rel), _cycleCounter(0),
49 _t(0), _flowfield(CellIndex_T::numberCellsInDomain, tws),
50 _flowfield_prefiltered(CellIndex_T::numberCellsInDomain, tws),
51 _patchfield(CellIndex_T::numberCellsInDomain -
52 tarch::la::Vector<dim, unsigned int>(2),
53 tws),
54 _innerCellIndices() {
55 // Initialize flowfield
56 for (int t = 0; t < tws; ++t)
57 for (auto idx : CellIndex_T()) {
58 tarch::la::Vector<dim, unsigned int> idxv(idx.get());
59 coupling::filtering::Quantities<dim> &q = _flowfield(idxv, t);
60 q[0] = 1;
61 for (unsigned int d = 1; d <= dim; ++d)
62 q[d] = 0;
63 coupling::filtering::Quantities<dim> &qp =
64 _flowfield_prefiltered(idxv, t);
65 qp[0] = 1;
66 for (unsigned int d = 1; d <= dim; ++d)
67 qp[d] = 0;
68 }
69
70 // Initialize innerCellIndices
71 auto domainSize = CellIndex_T::numberCellsInDomain;
72 for (auto idx : CellIndex_T()) {
73 tarch::la::Vector<dim, unsigned int> idxv(idx.get());
74 for (unsigned int d = 0; d < dim; d++)
75 if (idxv[d] == 0 || idxv[d] == domainSize[d] - 1)
76 goto continue_loop;
77 _innerCellIndices.push_back(idx);
78 continue_loop:;
79 }

83 std::cout << " WARNING: Regardless of configuration, NLM always filters "
84 "macroscopic mass and momentum."
85 << std::endl;
