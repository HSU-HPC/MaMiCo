66 tarch::la::Vector<dim,double> momentum(fraction*cell.getMicroscopicMomentum());
67 tarch::la::Vector<dim,double> zeroMomentum(0.0);

69 // if there is some momentum to be transferred, do so
70 if (tarch::la::dot(momentum,momentum) != 0.0){
