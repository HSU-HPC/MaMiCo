61 virtual void insertMomentum(
62 coupling::datastructures::MacroscopicCellWithLinkedCells<LinkedCell, dim>
63 &cell,
64 const unsigned int &currentMacroscopicCellIndex) const {
65 const unsigned int timeIntervalMomentumInsertion =
66 getTimeIntervalPerMomentumInsertion();
67 // determine fraction of momentum that is to be inserted in this frame
68 double fraction =
69 1.0 /
70 ((_numberMDTimestepsPerCouplingCycle / timeIntervalMomentumInsertion) +
71 (_numberMDTimestepsPerCouplingCycle % timeIntervalMomentumInsertion !=
72 0));
73 tarch::la::Vector<dim, double> momentum(fraction *
74 cell.getMicroscopicMomentum());
75 tarch::la::Vector<dim, double> zeroMomentum(0.0);
76
77 // if there is some momentum to be transferred, do so
78 if (tarch::la::dot(momentum, momentum) != 0.0) {
79 coupling::cellmappings::ComputeMassMapping<LinkedCell, dim>
80 computeMassMapping(
81 coupling::MomentumInsertion<LinkedCell, dim>::_mdSolverInterface);
82 cell.iterateConstCells(computeMassMapping);
83 unsigned int numberParticles = computeMassMapping.getNumberOfParticles();
84 coupling::cellmappings::SetMomentumMapping<LinkedCell, dim>
85 setMomentumMapping(
86 zeroMomentum, momentum, numberParticles,
87 coupling::MomentumInsertion<LinkedCell, dim>::_mdSolverInterface);
88 cell.iterateCells(setMomentumMapping);
89 }
90 }
