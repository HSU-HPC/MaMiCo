124 void testMolecules() {
125 unsigned int flag = 1;
126 for (unsigned int i = 0; i < local_cells.n; i++) {
127 unsigned int numberOfMoleculesFound = 0;
128 ParticleList* list = local_cells.cell[i];
129 coupling::interface::EspressoMDMoleculeIterator it(*list);
130 Particle* part;
131 part = list->part;
132 unsigned int numberOfMoleculesInCurrentCell = list->n;
133 for (it.begin(); it.continueIteration(); it.next()) {
134 numberOfMoleculesFound++;
135 }
136 if (numberOfMoleculesFound != numberOfMoleculesInCurrentCell) {
137 flag = 0;
138 std::cout << "ERROR: The number of molecules in the current cell " << i << " are expected to be " << numberOfMoleculesInCurrentCell
139 << " while our implementation of molecule iterator returns " << numberOfMoleculesFound << std::endl;
140 }
141 }
142 if (flag == 1) {
143 std::cout << "The test was successful " << std::endl;
144 } else {
145 std::cout << "The test was not successful " << std::endl;
146 }
147 }
