51 // number of processes used for a single MD simulation. Currently, the total number of MPI processes needs to
52 // be a multiple of the number of processes per MD simulation (=product of the vector components)
53 const tarch::la::Vector<dim,unsigned int> _numberProcessesPerMDSimulation;
54 // number of local communicators
55 unsigned int _numberLocalComms;
56 // total number of MD simulations
57 unsigned int _totalNumberMDSimulations;
58 // average number of MD simulations that is processed per local communicator.
59 // If we have 4 processes per MD simulation and 12 processes available and want to run 26 MD simulations,
60 // then this value is given by 26/(12/4) = 26/3 = 8
61 unsigned int _avgNumberMDSimulationsPerLocalComm;
62 // number of MD simulations for this current local communicator. Except for the "last communicator", this value
63 // equals _avgNumberMDSimulationsPerLocalComm. The last comunicator group fills up the missing MD simulations.
64 // If we have 4 processes per MD simulation and 12 processes available and want to run 26 MD simulations,
65 // then we have 12/4=3 communicator groups, of which group 0 and 1 handle 26/3=8 MD simulations. The last group 2
66 // handles 26-2*8 = 10 MD simulations.
67 unsigned int _thisNumberMDSimulations;
68 int _globalSize; // global number of available MPI processes
69 int _globalRank; // rank in global communicator MPI_COMM_WORLD
70
71 int _localSize; // size of communicator _localComm
72 int _localRank; // local rank in communicator _localComm
73 #if (TARCH_PARALLEL==TARCH_YES)
74 MPI_Comm _localComm; // communicator of "local" MD simulation
75 #endif
76};
