65 void handleCell(LinkedCell& cell,const unsigned int &cellIndex){
66 coupling::interface::MoleculeIterator<LinkedCell,dim> *it = _mdSolverInterface->getMoleculeIterator(cell);
67 it->begin();
68 while(it->continueIteration()){
69 coupling::interface::Molecule<dim>& wrapper(it->get());
70 tarch::la::Vector<dim,double> velocity = wrapper.getVelocity();
71
72 // set new velocity: still with same mean, but re-scale the deviation for correct thermal energy
73 velocity = _meanVelocity + _correctionFactor*(velocity - _meanVelocity);
74 wrapper.setVelocity(velocity);
75
76 it->next();
77 }
78 delete it;
79 }
80
81 private:
88 double getCorrectionFactor(
89 const double& oldKineticEnergy, const double& newKineticEnergy,
90 const unsigned int& numberParticles,
91 const tarch::la::Vector<dim,double>& meanVelocity
92 ) const {
93 const double mass = _mdSolverInterface->getMoleculeMass();
94
95 // no correction possible if the correction factor would tend to infinity; I just hard-coded 1e-7 for this case
96 if (oldKineticEnergy - 0.5*mass*numberParticles*tarch::la::dot(meanVelocity,meanVelocity)<1e-7){
97 return 1.0;
98 }
99
100 double correctionFactor = newKineticEnergy - 0.5*mass*numberParticles*tarch::la::dot(meanVelocity,meanVelocity);
101 correctionFactor = correctionFactor /(
102 oldKineticEnergy - 0.5*mass*numberParticles*tarch::la::dot(meanVelocity,meanVelocity) );
103 correctionFactor = sqrt(correctionFactor);
104 return correctionFactor;
105 }
106
107 coupling::interface::MDSolverInterface<LinkedCell,dim> * const _mdSolverInterface;
108 const tarch::la::Vector<dim,double> _meanVelocity;
109 const double _correctionFactor;
110};
