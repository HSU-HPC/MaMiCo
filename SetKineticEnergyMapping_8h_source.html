56 void handleCell(LinkedCell& cell, const unsigned int& cellIndex) {
57 coupling::interface::MoleculeIterator<LinkedCell, dim>* it = _mdSolverInterface->getMoleculeIterator(cell);
58 it->begin();
59 while (it->continueIteration()) {
60 coupling::interface::Molecule<dim>& wrapper(it->get());
61 tarch::la::Vector<dim, double> velocity = wrapper.getVelocity();
62
63 // set new velocity: still with same mean, but re-scale the deviation for
64 // correct thermal energy
65 velocity = _meanVelocity + _correctionFactor * (velocity - _meanVelocity);
66 wrapper.setVelocity(velocity);
67
68 it->next();
69 }
70 delete it;
71 }

73private:
81 double getCorrectionFactor(const double& oldKineticEnergy, const double& newKineticEnergy, const unsigned int& numberParticles,
82 const tarch::la::Vector<dim, double>& meanVelocity) const {
83 const double mass = _mdSolverInterface->getMoleculeMass();
84
85 // no correction possible if the correction factor would tend to infinity; I
86 // just hard-coded 1e-7 for this case
87 if (oldKineticEnergy - 0.5 * mass * numberParticles * tarch::la::dot(meanVelocity, meanVelocity) < 1e-7) {
88 return 1.0;
89 }
90
91 double correctionFactor = newKineticEnergy - 0.5 * mass * numberParticles * tarch::la::dot(meanVelocity, meanVelocity);
92 correctionFactor = correctionFactor / (oldKineticEnergy - 0.5 * mass * numberParticles * tarch::la::dot(meanVelocity, meanVelocity));
93 correctionFactor = sqrt(correctionFactor);
94 return correctionFactor;
95 }

97 coupling::interface::MDSolverInterface<LinkedCell, dim>* const _mdSolverInterface;
98 const tarch::la::Vector<dim, double> _meanVelocity;
99 const double _correctionFactor;
