135private:
136 unsigned int
137 computeScalarSize(const tarch::la::Vector<dim, unsigned int> &spatialSize,
138 const unsigned int &temporalSize) const {
139 unsigned int res = spatialSize[0];
140 for (unsigned int d = 1; d < dim; d++) {
141 res *= (spatialSize[d]);
142 }
143 res *= temporalSize;
144 return res;
145 }
146
147 unsigned int idx(const tarch::la::Vector<dim, unsigned int> &pos,
148 const unsigned int &t) const {
149#ifdef NLM_DEBUG
150 for (unsigned int d = 0; d < dim; d++) {
151 if (pos[d] < 0 || pos[d] > _spatialSize[d] - 1) {
152 std::cout << "ERROR Field idx(): pos out of range!" << std::endl;
153 std::cout << "pos=" << pos << ", _spatialSize=" << _spatialSize
154 << std::endl;
155 exit(EXIT_FAILURE);
156 }
157 }
158 if (t < 0 || t > _temporalSize - 1) {
159 std::cout << "ERROR Field idx(): t out of range!" << std::endl;
160 std::cout << "t=" << t << ", _temporalSize=" << _temporalSize
161 << std::endl;
162 exit(EXIT_FAILURE);
163 }
164#endif
165 unsigned int idx = 0, step = 1;
166 for (unsigned int d = 0; d < dim; d++) {
167 idx += pos[d] * step;
168 step *= _spatialSize[d];
169 }
170 idx += t * step;
171 return idx;
172 }

175 const tarch::la::Vector<dim, unsigned int> _spatialSize;
176 const unsigned int _temporalSize;
177 const unsigned int _scalarSize;
178 T *const _data;

180 friend class coupling::filtering::Patch<dim>; // patches need direct access to
181 // their field's _data for
182 // memory efficiency

188/***
189 * includes local flowfield, and mean and standard deviation of quantities
190 */
191template <unsigned int dim> class coupling::filtering::Patch {
192public:
193 Patch(const tarch::la::Vector<dim, unsigned int> &spatialSize,
194 const unsigned int &temporalSize, const Flowfield<dim> &basefield,
195 const tarch::la::Vector<dim, unsigned int> &pos, const unsigned int &t)
196 : _flowfield(spatialSize, temporalSize), _localMean(0.0),
197 _localStandardDeviation(0.0) {
198 fillFromBasefield(basefield, pos, t);
199 computeLocalMean();
200 computeLocalStandardDeviation();
201 }
202
203 double distance(const coupling::filtering::Patch<dim> &other) {
204 unsigned int size = _flowfield.getScalarSize() * (dim + 1);
205
206 double *const my_data = reinterpret_cast<double *const>(_flowfield._data);
207 double *const other_data =
208 reinterpret_cast<double *const>(other._flowfield._data);
209
210 double res = 0;
211 for (unsigned int i = 0; i < size; i += 1) {
212 double diff(my_data[i] - other_data[i]);
213 res += diff * diff;
214 }
215 return res;
216 }
217
218 ~Patch() {}
219
220 void print() { _flowfield.print(); }
221
222private:
223 inline unsigned int posmod(int i, int n) { return (i % n + n) % n; }
224
225 void fillFromBasefield(const Flowfield<dim> &basefield,
226 const tarch::la::Vector<dim, unsigned int> &pos,
227 const unsigned int &t) {
228 static_assert(dim == 2 || dim == 3,
229 "ERROR filtering::Patch::fillFromBasefield only implemented "
230 "for 2D and 3D");
231
232 tarch::la::Vector<dim, unsigned int> center;
233 for (unsigned int d = 0; d < dim; d++) {
234 center[d] = _flowfield.getSpatialSize()[d] / 2;
235 }
236
237 tarch::la::Vector<dim, unsigned int> local_pos(0);
238 unsigned int local_t(0);
239 tarch::la::Vector<dim, unsigned int> base_pos(0);
240 unsigned int base_t(0);
241
242 for (int offset_t = 0; offset_t > -(int)_flowfield.getTemporalSize();
243 offset_t--) {
244 local_t = posmod(offset_t, _flowfield.getTemporalSize());
245 base_t = posmod(t + offset_t, basefield.getTemporalSize());
246 for (int offset_x = -(int)(_flowfield.getSpatialSize()[0]) / 2;
247 offset_x <= (int)(_flowfield.getSpatialSize()[0]) / 2; offset_x++) {
248 local_pos[0] = center[0] + offset_x;
249 base_pos[0] = pos[0] + offset_x;
250 for (int offset_y = -(int)(_flowfield.getSpatialSize()[1]) / 2;
251 offset_y <= (int)(_flowfield.getSpatialSize()[1]) / 2;
252 offset_y++) {
253 local_pos[1] = center[1] + offset_y;
254 base_pos[1] = pos[1] + offset_y;
255 if constexpr (dim == 3) {
256 for (int offset_z = -(int)(_flowfield.getSpatialSize()[2]) / 2;
257 offset_z <= (int)(_flowfield.getSpatialSize()[2]) / 2;
258 offset_z++) {
259 local_pos[2] = center[2] + offset_z;
260 base_pos[2] = pos[2] + offset_z;
261 _flowfield(local_pos, local_t) = basefield(base_pos, base_t);
262 }
263 } else {
264 _flowfield(local_pos, local_t) = basefield(base_pos, base_t);
265 }
266 }
267 }
268 }
269 }
270
271 void computeLocalMean() {
272 for (unsigned int i = 0; i < _flowfield.getScalarSize(); i++)
273 _localMean += _flowfield[i];
274 _localMean = _localMean * (1.0 / _flowfield.getScalarSize());
275 }
276
277 void computeLocalStandardDeviation() {
278 for (unsigned int i = 0; i < _flowfield.getScalarSize(); i++) {
279 Quantities<dim> diff = _localMean - _flowfield[i];
280 _localStandardDeviation += tarch::la::dot(diff, diff);
281 }
282 _localStandardDeviation =
283 sqrt(_localStandardDeviation / _flowfield.getScalarSize());
284 }
285
286 Flowfield<dim> _flowfield;
287 Quantities<dim> _localMean;
288 double _localStandardDeviation;
289};

295template <unsigned int dim> class coupling::filtering::PatchView {
296public:
297 PatchView(const tarch::la::Vector<dim, unsigned int> &spatialSize,
298 const unsigned int &temporalSize, const Flowfield<dim> &basefield,
299 const tarch::la::Vector<dim, unsigned int> &pos,
300 const unsigned int &t)
301 : _spatialSize(spatialSize), _temporalSize(temporalSize),
302 _basefield(basefield), _pos(pos), _t(t) {}
303
304 double distance(const coupling::filtering::PatchView<dim> &other) const {
305 double res = 0;
306
307 tarch::la::Vector<dim, unsigned int> base_pos(0);
308 unsigned int base_t(0);
309
310 for (int offset_t = 0; offset_t > -(int)_temporalSize; offset_t--) {
311 base_t = posmod(_t + offset_t, _basefield.getTemporalSize());
312 for (int offset_x = -(int)(_spatialSize[0]) / 2;
313 offset_x <= (int)(_spatialSize[0]) / 2; offset_x++) {
314 base_pos[0] = _pos[0] + offset_x;
315 for (int offset_y = -(int)(_spatialSize[1]) / 2;
316 offset_y <= (int)(_spatialSize[1]) / 2; offset_y++) {
317 base_pos[1] = _pos[1] + offset_y;
318 if constexpr (dim == 3) {
319 for (int offset_z = -(int)(_spatialSize[2]) / 2;
320 offset_z <= (int)(_spatialSize[2]) / 2; offset_z++) {
321 base_pos[2] = _pos[2] + offset_z;
322 // TODO: this is not correct. fix me
323 auto diff(_basefield(base_pos, base_t) -
324 _basefield(base_pos + other._pos - _pos,
325 posmod(base_t + other._t - _t,
326 _basefield.getTemporalSize())));
327 res += diff[0] * diff[0];
328 }
329 } else {
330 auto diff(_basefield(base_pos, base_t) -
331 other._basefield(base_pos, base_t));
332 res += tarch::la::dot(diff, diff);
333 }
334 }
335 }
336 }
337 return res;
338 }
