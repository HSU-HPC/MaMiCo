80 virtual void processInnerMacroscopicCellAfterMDTimestep(coupling::datastructures::MacroscopicCellWithLinkedCells<LinkedCell, dim>& cell,
81 const unsigned int& index) {
82 // compute total mass/momentum from previous samples
83 const double oldMass = (_sampleCounter - 1) * cell.getMacroscopicMass();
84 const tarch::la::Vector<dim, double> oldMomentum = ((double)(_sampleCounter - 1)) * cell.getMacroscopicMomentum();
85
86 // compute new averaged mass and momentum
87 cell.iterateConstCells(_massMapping);
88 cell.iterateConstCells(_momentumMapping);
89 const double mass = (1.0 / _sampleCounter) * (oldMass + _massMapping.getMass());
90 const tarch::la::Vector<dim, double> momentum = (1.0 / _sampleCounter) * (oldMomentum + _momentumMapping.getMomentum());
91 _avgMass += mass;
92 _avgMomentum = _avgMomentum + momentum;
93 // set mass and momentum in buffers
94 cell.setMacroscopicMass(mass);
95 cell.setMacroscopicMomentum(momentum);
96 }

109 tarch::la::Vector<dim, double> _avgMomentum;
112 double _avgMass;
