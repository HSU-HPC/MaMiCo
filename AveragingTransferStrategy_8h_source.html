83 virtual void processInnerMacroscopicCellAfterMDTimestep(
84 coupling::datastructures::MacroscopicCellWithLinkedCells<LinkedCell,dim> &cell, const unsigned int &index
85 ){
86 // compute total mass/momentum from previous samples
87 const double oldMass = (_sampleCounter-1)*cell.getMacroscopicMass();
88 const tarch::la::Vector<dim,double> oldMomentum = ((double)(_sampleCounter-1))*cell.getMacroscopicMomentum();
89
90 // compute new averaged mass and momentum
91 cell.iterateConstCells(_massMapping);
92 cell.iterateConstCells(_momentumMapping);
93 const double mass = (1.0/_sampleCounter)*(oldMass+_massMapping.getMass());
94 const tarch::la::Vector<dim,double> momentum = (1.0/_sampleCounter)*(oldMomentum+_momentumMapping.getMomentum());
95 _avgMass += mass;
96 _avgMomentum = _avgMomentum + momentum;
97 // set mass and momentum in buffers
98 cell.setMacroscopicMass(mass);
99 cell.setMacroscopicMomentum(momentum);
100 }
