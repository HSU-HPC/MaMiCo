70 MarDynMDSolverInterface* mdsi = (MarDynMDSolverInterface*)coupling::interface::MamicoInterfaceProvider<MarDynCell, 3>::getInstance().getMDSolverInterface();
71 double potentialEnergy = 0.0;

73 // get particle container from simulation
74 LinkedCells* cells = (LinkedCells*)mdsi->getSimulation()->getMolecules();
75
76 Molecule* curMolecule;
77 int counter = 0;
78 for (curMolecule = cells->begin(); curMolecule != cells->end(); curMolecule = cells->next()) {
79 counter++;
80
81 // std::cout << "cellIndex is: " << cells->getCellIndexOfMolecule(curMolecule) << std::endl;
82 // std::cout << "pos : ";
83 // for(int d=0; d<3; d++) std::cout << curMolecule->r(d) << ", ";

87 tarch::la::Vector<3, double> position(0.0);
88 tarch::la::Vector<3, double> velocity(0.0);
89 double force[3] = {0.0, 0.0, 0.0};
90 for (int d = 0; d < 3; d++) {
91 position[d] = curMolecule->r(d);
92 velocity[d] = curMolecule->v(d);

95

106 // compute force and potential energy

108 potentialEnergy = molecule.getPotentialEnergy();
109
110 if (detail)

115 if (potentialEnergy == 0.0)

117 if (detail)

234 void compareMoleculeList(bool detail, int& currentCellIndex, int& errors) {
235 // get particle container from simulation
236 LinkedCells* cells = (LinkedCells*)this->_marDyn->getMolecules();
237
238 // for testing get first cell and create MarDynCell
239 ParticleCell pc;
240 bool cellFound = false;
241 for (int i = currentCellIndex; i < 2744; i++) {
242 pc = cells->getCell(i);
243 if (pc.getMoleculeCount() > 0) {
244 std::cout << "Test 2 " << i << " " << pc.getMoleculeCount() << std::endl;
245 currentCellIndex = i + 1;
246 cellFound = true;
247 break;
248 }
249 }
250 if (!cellFound) {
251 currentCellIndex = 9999;
252 return;
253 }
254 MarDynCell cell = MarDynCell(&pc, cells->getCutoff());
255
256 // vector containing all molecules of the particle cell
257 std::vector<Molecule*> moleculesInParticleCell = pc.getParticlePointers();
258
259 // set up molecule iterator for testing
260 MarDynMoleculeIterator mdmi(cell);
261
262 // helper variables
263 int moleculeCount = pc.getMoleculeCount();
264 std::vector<tarch::la::Vector<3, double>> listMarDyn;
265 std::vector<tarch::la::Vector<3, double>> listMarDynCouplingInterface;
266 tarch::la::Vector<3, double> position(0.0);
267 tarch::la::Vector<3, double> velocity(0.0);
268 tarch::la::Vector<3, double> force(0.0);
269 int counter = 0;
270
271 // loop through all molecules in the cell
272 for (mdmi.begin(); mdmi.continueIteration(); mdmi.next()) {
273 if (detail)
274 std::cout << "Molecule #" << counter << ":" << std::endl;
275 // get position and velocity through interface
276 position = mdmi.get().getPosition();
277 velocity = mdmi.get().getVelocity();
278 force = mdmi.get().getForce();
279 if (detail)
280 std::cout << "interface: " << position << " | " << velocity << " | " << force << std::endl;
281 listMarDynCouplingInterface.push_back(position);
282 listMarDynCouplingInterface.push_back(velocity);
283 listMarDynCouplingInterface.push_back(force);
284
285 // get position and velocity from the ParticleCell for comparison
286 for (int d = 0; d < 3; d++) {
287 position[d] = moleculesInParticleCell[counter]->r(d);
288 velocity[d] = moleculesInParticleCell[counter]->v(d);
289 force[d] = moleculesInParticleCell[counter]->F(d);
290 }
291 if (detail)
292 std::cout << "cell_____: " << position << " | " << velocity << " | " << force << std::endl;
293 listMarDyn.push_back(position);
294 listMarDyn.push_back(velocity);
295 listMarDyn.push_back(force);
296
297 counter++;
298 }
299
300 if (listMarDyn != listMarDynCouplingInterface) {
301 errors = 1;
302 std::cout << "ERROR: the molecule values obtained from the MarDyn simulation and through the Coupling interface do not match! (Cell #" << currentCellIndex
303 << ")" << std::endl;
304 }
305 }
