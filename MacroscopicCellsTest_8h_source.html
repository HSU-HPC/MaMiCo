123 template<unsigned int dim>
124 void testMacroscopicCells(){
125 TestMDSolverInterface<dim> *testInterface = new TestMDSolverInterface<dim>();
126 if (testInterface==NULL){std::cout << "ERROR MacroscopicCellsTest::testMacroscopicCells(): testInterface==NULL!" << std::endl; exit(EXIT_FAILURE);}
127 const tarch::la::Vector<dim,unsigned int> globalNumberCells(3);
128 const tarch::la::Vector<dim,unsigned int> numberProcesses(1);
129 const unsigned int rank = 0;
130 const coupling::IndexConversion<dim> indexConversion(
131 globalNumberCells,numberProcesses,rank,testInterface->getGlobalMDDomainSize(),testInterface->getGlobalMDDomainOffset(),coupling::paralleltopology::XYZ
132 );
133 const tarch::la::Vector<dim,unsigned int> numberLinkedCellsPerMacroscopicCell(2);
134
135 coupling::datastructures::MacroscopicCells<simplemd::LinkedCell,dim> macroscopicCells(
136 numberLinkedCellsPerMacroscopicCell,indexConversion, testInterface
137 );
138 WriteIndex<dim> writeIndex(indexConversion);
139 PrintIndex<dim> printIndex;
140
141 // write indices to inner macroscopic cells
142 macroscopicCells.applyToLocalNonGhostMacroscopicCellsWithLinkedCells(writeIndex);
143 macroscopicCells.applyToLocalNonGhostMacroscopicCellsWithLinkedCells(printIndex);
144
145 // write indices to outer macroscopic cells
146 macroscopicCells.applyToLocalGhostMacroscopicCellsWithLinkedCells(writeIndex);
147 macroscopicCells.applyToLocalGhostMacroscopicCellsWithLinkedCells(printIndex);
148
149 delete testInterface; testInterface = NULL;
150 }
