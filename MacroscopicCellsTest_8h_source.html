121 template <unsigned int dim> void testMacroscopicCells() {
122 TestMDSolverInterface<dim>* testInterface = new TestMDSolverInterface<dim>();
123 if (testInterface == NULL) {
124 std::cout << "ERROR MacroscopicCellsTest::testMacroscopicCells(): testInterface==NULL!" << std::endl;
125 exit(EXIT_FAILURE);
126 }
127 const tarch::la::Vector<dim, unsigned int> globalNumberCells(3);
128 const tarch::la::Vector<dim, unsigned int> numberProcesses(1);
129 const unsigned int rank = 0;
130 const coupling::IndexConversion<dim> indexConversion(globalNumberCells, numberProcesses, rank, testInterface->getGlobalMDDomainSize(),
131 testInterface->getGlobalMDDomainOffset(), coupling::paralleltopology::XYZ);
132 const tarch::la::Vector<dim, unsigned int> numberLinkedCellsPerMacroscopicCell(2);
133
134 coupling::datastructures::MacroscopicCells<simplemd::LinkedCell, dim> macroscopicCells(numberLinkedCellsPerMacroscopicCell, indexConversion, testInterface);
135 WriteIndex<dim> writeIndex(indexConversion);
136 PrintIndex<dim> printIndex;
137
138 // write indices to inner macroscopic cells
139 macroscopicCells.applyToLocalNonGhostMacroscopicCellsWithLinkedCells(writeIndex);
140 macroscopicCells.applyToLocalNonGhostMacroscopicCellsWithLinkedCells(printIndex);
141
142 // write indices to outer macroscopic cells
143 macroscopicCells.applyToLocalGhostMacroscopicCellsWithLinkedCells(writeIndex);
144 macroscopicCells.applyToLocalGhostMacroscopicCellsWithLinkedCells(printIndex);
145
146 delete testInterface;
147 testInterface = NULL;
148 }
149};
