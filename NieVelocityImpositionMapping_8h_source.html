53 void handleCell(LinkedCell& cell,const unsigned int &cellIndex){
54 coupling::interface::MoleculeIterator<LinkedCell,dim> *it = _mdSolverInterface->getMoleculeIterator(cell);
55
56 // compute average force contribution inside this linked cell
57 it->begin();
58 while(it->continueIteration()){
59 coupling::interface::Molecule<dim> &wrapper(it->get());
60 tarch::la::Vector<dim,double> force (wrapper.getForce());
61 force = force - _avgForce - (_mdSolverInterface->getMoleculeMass()/_mdSolverInterface->getDt()) * (_avgMDVelocity - _continuumVelocity);
62 wrapper.setForce(force);
63 it->next();
64 }
65 delete it;
66 }
67
68 private:
69 // MD solver interface, required for molecule iterator and molecule mass
70 coupling::interface::MDSolverInterface<LinkedCell,dim> * const _mdSolverInterface;
71 // current velocity in this macroscopic cell (=velocity from continuum solver)
72 const tarch::la::Vector<dim,double> _continuumVelocity;
73 // current avg. velocity of molecules
74 const tarch::la::Vector<dim,double> _avgMDVelocity;
75 // average force within this macroscopic cell
76 const tarch::la::Vector<dim,double> _avgForce;
77};
