180/*
181 A dynamic array of Plain Old Data. Doesn't support constructors, etc.
182 Has a small initial memory pool, so that low or no usage will not
183 cause a call to new/delete
184*/
185template <class T, int INIT> class DynArray {
186public:
187 DynArray<T, INIT>() {
188 _mem = _pool;
189 _allocated = INIT;
190 _size = 0;
191 }
192
193 ~DynArray() {
194 if (_mem != _pool) {
195 delete[] _mem;
196 }
197 }
198
199 void Push(T t) {
200 EnsureCapacity(_size + 1);
201 _mem[_size++] = t;
202 }
203
204 T *PushArr(int count) {
205 EnsureCapacity(_size + count);
206 T *ret = &_mem[_size];
207 _size += count;
208 return ret;
209 }
210
211 T Pop() { return _mem[--_size]; }
212
213 void PopArr(int count) {
214 TIXMLASSERT(_size >= count);
215 _size -= count;
216 }
217
218 bool Empty() const { return _size == 0; }
219
220 T &operator[](int i) {
221 TIXMLASSERT(i >= 0 && i < _size);
222 return _mem[i];
223 }
224
225 const T &operator[](int i) const {
226 TIXMLASSERT(i >= 0 && i < _size);
227 return _mem[i];
228 }
229
230 int Size() const { return _size; }
231
232 int Capacity() const { return _allocated; }
233
234 const T *Mem() const { return _mem; }
235
236 T *Mem() { return _mem; }
237
238private:
239 void EnsureCapacity(int cap) {
240 if (cap > _allocated) {
241 int newAllocated = cap * 2;
242 T *newMem = new T[newAllocated];
243 memcpy(newMem, _mem,
244 sizeof(T) *
245 _size); // warning: not using constructors, only works for PODs
246 if (_mem != _pool) {
247 delete[] _mem;
248 }
249 _mem = newMem;
250 _allocated = newAllocated;
251 }
252 }
253
254 T *_mem;
255 T _pool[INIT];
256 int _allocated; // objects allocated
257 int _size; // number objects in use
258};

275/*
276 Template child class to create pools of the correct type.
277*/
278template <int SIZE> class MemPoolT : public MemPool {
279public:
280 MemPoolT()
281 : _root(0), _currentAllocs(0), _nAllocs(0), _maxAllocs(0),
282 _nUntracked(0) {}
283 ~MemPoolT() {
284 // Delete the blocks.
285 for (int i = 0; i < _blockPtrs.Size(); ++i) {
286 delete _blockPtrs[i];
287 }
288 }
289
290 virtual int ItemSize() const { return SIZE; }
291 int CurrentAllocs() const { return _currentAllocs; }
292
293 virtual void *Alloc() {
294 if (!_root) {
295 // Need a new block.
296 Block *block = new Block();
297 _blockPtrs.Push(block);
298
299 for (int i = 0; i < COUNT - 1; ++i) {
300 block->chunk[i].next = &block->chunk[i + 1];
301 }
302 block->chunk[COUNT - 1].next = 0;
303 _root = block->chunk;
304 }
305 void *result = _root;
306 _root = _root->next;
307
308 ++_currentAllocs;
309 if (_currentAllocs > _maxAllocs) {
310 _maxAllocs = _currentAllocs;
311 }
312 _nAllocs++;
313 _nUntracked++;
314 return result;
315 }
316 virtual void Free(void *mem) {
317 if (!mem) {
318 return;
319 }
320 --_currentAllocs;
321 Chunk *chunk = (Chunk *)mem;
322#ifdef DEBUG
323 memset(chunk, 0xfe, sizeof(Chunk));
324#endif
325 chunk->next = _root;
326 _root = chunk;
327 }
328 void Trace(const char *name) {
329 printf("Mempool %s watermark=%d [%dk] current=%d size=%d nAlloc=%d "
330 "blocks=%d\n",
331 name, _maxAllocs, _maxAllocs * SIZE / 1024, _currentAllocs, SIZE,
332 _nAllocs, _blockPtrs.Size());
333 }
334
335 void SetTracked() { _nUntracked--; }
336
337 int Untracked() const { return _nUntracked; }
338
339 // This number is perf sensitive. 4k seems like a good tradeoff on my machine.
340 // The test file is large, 170k.
341 // Release: VS2010 gcc(no opt)
342 // 1k: 4000
343 // 2k: 4000
344 // 4k: 3900 21000
345 // 16k: 5200
346 // 32k: 4300
347 // 64k: 4000 21000
348 enum {
349 COUNT = (4 * 1024) / SIZE
350 }; // Some compilers do not accept to use COUNT in private part if COUNT is
351 // private
352
353private:
354 union Chunk {
355 Chunk *next;
356 char mem[SIZE];
357 };
358 struct Block {
359 Chunk chunk[COUNT];
360 };
361 DynArray<Block *, 10> _blockPtrs;
362 Chunk *_root;
363
364 int _currentAllocs;
365 int _nAllocs;
366 int _maxAllocs;
367 int _nUntracked;
368};

454 int n = 0;
