179/*
180 A dynamic array of Plain Old Data. Doesn't support constructors, etc.
181 Has a small initial memory pool, so that low or no usage will not
182 cause a call to new/delete
183*/
184template <class T, int INIT> class DynArray {
185public:
186 DynArray<T, INIT>() {
187 _mem = _pool;
188 _allocated = INIT;
189 _size = 0;
190 }
191
192 ~DynArray() {
193 if (_mem != _pool) {
194 delete[] _mem;
195 }
196 }
197
198 void Push(T t) {
199 EnsureCapacity(_size + 1);
200 _mem[_size++] = t;
201 }
202
203 T *PushArr(int count) {
204 EnsureCapacity(_size + count);
205 T *ret = &_mem[_size];
206 _size += count;
207 return ret;
208 }
209
210 T Pop() { return _mem[--_size]; }
211
212 void PopArr(int count) {
213 TIXMLASSERT(_size >= count);
214 _size -= count;
215 }
216
217 bool Empty() const { return _size == 0; }
218
219 T &operator[](int i) {
220 TIXMLASSERT(i >= 0 && i < _size);
221 return _mem[i];
222 }
223
224 const T &operator[](int i) const {
225 TIXMLASSERT(i >= 0 && i < _size);
226 return _mem[i];
227 }
228
229 int Size() const { return _size; }
230
231 int Capacity() const { return _allocated; }
232
233 const T *Mem() const { return _mem; }
234
235 T *Mem() { return _mem; }
236
237private:
238 void EnsureCapacity(int cap) {
239 if (cap > _allocated) {
240 int newAllocated = cap * 2;
241 T *newMem = new T[newAllocated];
242 memcpy(newMem, _mem,
243 sizeof(T) * _size); // warning: not using constructors, only works for PODs
244 if (_mem != _pool) {
245 delete[] _mem;
246 }
247 _mem = newMem;
248 _allocated = newAllocated;
249 }
250 }
251
252 T *_mem;
253 T _pool[INIT];
254 int _allocated; // objects allocated
255 int _size; // number objects in use
256};

273/*
274 Template child class to create pools of the correct type.
275*/
276template <int SIZE> class MemPoolT : public MemPool {
277public:
278 MemPoolT() : _root(0), _currentAllocs(0), _nAllocs(0), _maxAllocs(0), _nUntracked(0) {}
279 ~MemPoolT() {
280 // Delete the blocks.
281 for (int i = 0; i < _blockPtrs.Size(); ++i) {
282 delete _blockPtrs[i];
283 }
284 }
285
286 virtual int ItemSize() const { return SIZE; }
287 int CurrentAllocs() const { return _currentAllocs; }
288
289 virtual void *Alloc() {
290 if (!_root) {
291 // Need a new block.
292 Block *block = new Block();
293 _blockPtrs.Push(block);
294
295 for (int i = 0; i < COUNT - 1; ++i) {
296 block->chunk[i].next = &block->chunk[i + 1];
297 }
298 block->chunk[COUNT - 1].next = 0;
299 _root = block->chunk;
300 }
301 void *result = _root;
302 _root = _root->next;
303
304 ++_currentAllocs;
305 if (_currentAllocs > _maxAllocs) {
306 _maxAllocs = _currentAllocs;
307 }
308 _nAllocs++;
309 _nUntracked++;
310 return result;
311 }
312 virtual void Free(void *mem) {
313 if (!mem) {
314 return;
315 }
316 --_currentAllocs;
317 Chunk *chunk = (Chunk *)mem;
318#ifdef DEBUG
319 memset(chunk, 0xfe, sizeof(Chunk));
320#endif
321 chunk->next = _root;
322 _root = chunk;
323 }
324 void Trace(const char *name) {
325 printf("Mempool %s watermark=%d [%dk] current=%d size=%d nAlloc=%d "
326 "blocks=%d\n",
327 name, _maxAllocs, _maxAllocs * SIZE / 1024, _currentAllocs, SIZE, _nAllocs, _blockPtrs.Size());
328 }
329
330 void SetTracked() { _nUntracked--; }
331
332 int Untracked() const { return _nUntracked; }
333
334 // This number is perf sensitive. 4k seems like a good tradeoff on my machine.
335 // The test file is large, 170k.
336 // Release: VS2010 gcc(no opt)
337 // 1k: 4000
338 // 2k: 4000
339 // 4k: 3900 21000
340 // 16k: 5200
341 // 32k: 4300
342 // 64k: 4000 21000
343 enum {
344 COUNT = (4 * 1024) / SIZE
345 }; // Some compilers do not accept to use COUNT in private part if COUNT is
346 // private
347
348private:
349 union Chunk {
350 Chunk *next;
351 char mem[SIZE];
352 };
353 struct Block {
354 Chunk chunk[COUNT];
355 };
356 DynArray<Block *, 10> _blockPtrs;
357 Chunk *_root;
358
359 int _currentAllocs;
360 int _nAllocs;
361 int _maxAllocs;
362 int _nUntracked;
363};

436 inline static bool StringEqual(const char *p, const char *q, int nChar = INT_MAX) {
437 int n = 0;
438 if (p == q) {

443 ++q;
444 ++n;
