68 for (int d = 0; d < 3; d++){ _vel[i*3+d] = (double)0.0;}
69 _density[i] = 1.0;

79 // front - periodic

81 // back - periodic
82 for (int z = 1; z < _domainSizeZ+1; z++){for (int x = 0; x < _domainSizeX+2; x++){ _flag[get(x,_domainSizeY+1,z)] = PERIODIC; } }
83 // left - periodic
84 for (int z = 1; z < _domainSizeZ+1; z++){for (int y = 1; y < _domainSizeY+1; y++){ _flag[get(0,y,z)] = PERIODIC; }}
85 for (int z = 1; z < _domainSizeZ+1; z++){for (int y = 1; y < _domainSizeY+1; y++){ _flag[get(_domainSizeX+1,y,z)] = PERIODIC; }}
86 // correct boundary flags in case of MPI-parallel simulations (Couette scenario)
87 setParallelBoundaryFlags();

183 void determineParallelNeighbours(){
184 #if (COUPLING_MD_PARALLEL==COUPLING_MD_YES)
185 int rank;
186 int size;
187 MPI_Comm_rank(MPI_COMM_WORLD,&rank);
188 MPI_Comm_size(MPI_COMM_WORLD,&size);
189 // check if enough ranks are available
190 if (_processes[0]*_processes[1]*_processes[2]>(unsigned int)size){
191 std::cout << "ERROR NumericalSolver::determineParallelNeighbours(): Not enough ranks available!" << std::endl; exit(EXIT_FAILURE);
192 }
193 // neighbour dependencies based on Couette problem
194 // left,right: periodic
195 _parallelNeighbours[LEFT] = ((_coords[0]+_processes[0]-1) % _processes[0]) + _processes[0]*(_coords[1]+_processes[1]*_coords[2]);
196 _parallelNeighbours[RIGHT]= ((_coords[0]+_processes[0]+1) % _processes[0]) + _processes[0]*(_coords[1]+_processes[1]*_coords[2]);
197 // back,front: periodic
198 _parallelNeighbours[FRONT]= _coords[0] + _processes[0]*( ((_coords[1]+_processes[1]-1) % _processes[1]) + _processes[1]*_coords[2]);
199 _parallelNeighbours[BACK] = _coords[0] + _processes[0]*( ((_coords[1]+_processes[1]+1) % _processes[1]) + _processes[1]*_coords[2]);
200 // top: either neighbour or MPI_PROC_NULL
201 if (_coords[2]==_processes[2]-1){ _parallelNeighbours[TOP] = MPI_PROC_NULL; } else { _parallelNeighbours[TOP] = _coords[0]+_processes[0]*(_coords[1]+_processes[1]*(_coords[2]+1)); }
202 // bottom either neighbour or MPI_PROC_NULL
203 if (_coords[2]==0) { _parallelNeighbours[BOTTOM]=MPI_PROC_NULL;} else { _parallelNeighbours[BOTTOM]=_coords[0]+_processes[0]*(_coords[1]+_processes[1]*(_coords[2]-1));}
204 // std::cout << "Parallel neighbours for rank " << rank << ": " << _parallelNeighbours << std::endl;
205 #endif
206 }

209 void setParallelBoundaryFlags(){
210 #if (COUPLING_MD_PARALLEL==COUPLING_MD_YES)
211 // bottom - moving wall
212 if (_coords[2]!=0){
213 for (int i = 0; i < (_domainSizeX+2)*(_domainSizeY+2); i++){ _flag[i] = PARALLEL_BOUNDARY; }
214 }
215 // top - noslip
216 if (_coords[2]!=_processes[2]-1){
217 for (int i = (_domainSizeX+2)*(_domainSizeY+2)*(_domainSizeZ+1); i < (_domainSizeX+2)*(_domainSizeY+2)*(_domainSizeZ+2); i++){ _flag[get(i)] = PARALLEL_BOUNDARY;}
218 }
219 // for all other boundaries front,back,left,right, we use parallel boundaries. If we send from one processes to itself, this is still the same as periodic conditions
220 // front - periodic
221 for (int z = 1; z < _domainSizeZ+1; z++){for (int x = 0; x < _domainSizeX+2; x++){ _flag[get(x,0,z)] = PARALLEL_BOUNDARY; } }
222 // back - periodic
223 for (int z = 1; z < _domainSizeZ+1; z++){for (int x = 0; x < _domainSizeX+2; x++){ _flag[get(x,_domainSizeY+1,z)] = PARALLEL_BOUNDARY; } }
224 // left - periodic
225 for (int z = 1; z < _domainSizeZ+1; z++){for (int y = 1; y < _domainSizeY+1; y++){ _flag[get(0,y,z)] = PARALLEL_BOUNDARY; }}
226 for (int z = 1; z < _domainSizeZ+1; z++){for (int y = 1; y < _domainSizeY+1; y++){ _flag[get(_domainSizeX+1,y,z)] = PARALLEL_BOUNDARY; }}
227 #endif // COUPLING_MD_PARALLEL

232 int getDomainSize(double channelheight, double dx, tarch::la::Vector<3,unsigned int> processes, int d) const {
233 int globalDomainSize = floor( (channelheight+0.5)/dx );
234 tarch::la::Vector<3,unsigned int> coords(0);
235 #if (COUPLING_MD_PARALLEL==COUPLING_MD_YES)
236 int rank;
237 MPI_Comm_rank(MPI_COMM_WORLD,&rank);
238 // if this rank is outside the range given by processes: return 0
239 // -> cannot use method skipRank() here, since _processes may not be initialized yet!
240 if ((unsigned int)rank > processes[0]*processes[1]*processes[2]-1){return 0;}
241 // determine rank coordinates
242 coords[2] = ((unsigned int)rank)/(processes[0]*processes[1]);
243 coords[1] =(((unsigned int)rank)-coords[2]*processes[0]*processes[1])/processes[0];
244 coords[0] =((unsigned int)rank)-coords[2]*processes[0]*processes[1]-coords[1]*processes[0];
245 #endif
246 // if this is not the last process along this direction: just return avg. number of cells
247 if (coords[d]<processes[d]-1){
248 return globalDomainSize/processes[d];
249 // otherwise: add the cells that have not been distributed yet
250 } else {return globalDomainSize/processes[d] + globalDomainSize%processes[d];}
251 }

254 int getAvgDomainSize(double channelheight,double dx, tarch::la::Vector<3,unsigned int> processes, int d) const {
255 int globalDomainSize = floor( (channelheight+0.5)/dx );
256 return globalDomainSize/processes[d];
257 }
