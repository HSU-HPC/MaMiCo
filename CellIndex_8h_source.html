221 static void setDomainParameters() {
222 auto cellnum = upperBoundary.get() - lowerBoundary.get() + tarch::la::Vector<dim, int>{1};
223 for (unsigned int d = 0; d < dim; d++)
224 cellnum[d] = std::max(0, cellnum[d]);
225 numberCellsInDomain = tarch::la::Vector<dim, unsigned int>{cellnum};
226
227 linearNumberCellsInDomain = 1;
228 for (unsigned int d = 0; d < dim; d++)
229 linearNumberCellsInDomain *= numberCellsInDomain[d];
230
231 tarch::la::Vector<dim, unsigned int> divFactor{1};
232 for (unsigned int d = 1; d < dim; d++)
233 divFactor[d] = divFactor[d - 1] * (numberCellsInDomain[d - 1]);
234 divisionFactor = divFactor;
235 }
