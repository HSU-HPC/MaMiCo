247 static void setDomainParameters() {
248 auto cellnum = upperBoundary.get() - lowerBoundary.get() +
249 tarch::la::Vector<dim, int>{1};
250 for (unsigned int d = 0; d < dim; d++)
251 cellnum[d] = std::max(0, cellnum[d]);
252 numberCellsInDomain = tarch::la::Vector<dim, unsigned int>{cellnum};
253
254 linearNumberCellsInDomain = 1;
255 for (unsigned int d = 0; d < dim; d++)
256 linearNumberCellsInDomain *= numberCellsInDomain[d];
257
258 tarch::la::Vector<dim, unsigned int> divFactor{1};
259 for (unsigned int d = 1; d < dim; d++)
260 divFactor[d] = divFactor[d - 1] * (numberCellsInDomain[d - 1]);
261 divisionFactor = divFactor;
262 }
