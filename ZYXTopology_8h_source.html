36 tarch::la::Vector<dim,unsigned int> getProcessCoordinates(unsigned int rank) const {
37 #if (COUPLING_MD_DEBUG==COUPLING_MD_YES)
38 unsigned int intNumberProcesses = _numberProcesses[0]; for (unsigned int d = 1 ; d < dim; d++){ intNumberProcesses = intNumberProcesses*_numberProcesses[d]; }
39 if ( (rank<_topologyOffset) || (rank>_topologyOffset+intNumberProcesses-1)){
40 std::cout << "Warning coupling::paralleltopology::ZYXTopology::getProcessCoordinates(): rank out of range!" << std::endl;
41 std::cout << "Offset=" << _topologyOffset << ", rank=" << rank << std::endl;
42 }
43 #endif
44 tarch::la::Vector<dim,unsigned int> processCoordinates(0);
45 unsigned int help = rank-_topologyOffset;
46 for (unsigned int d = 0; d < dim; d++){
47 processCoordinates[d] = help/_divisionFactor4NumberProcesses[d];
48 help = help - processCoordinates[d]*_divisionFactor4NumberProcesses[d];
49 }
50 return processCoordinates;
51 }

54 unsigned int getRank(tarch::la::Vector<dim,unsigned int> processCoordinates) const {
55 unsigned int rank = processCoordinates[0];
56 for (unsigned int d = 1; d < dim; d++) {
57 rank = rank*_numberProcesses[d] + processCoordinates[d];
58 }
59 return rank+_topologyOffset;
60 }
61
62 private:
64 tarch::la::Vector<dim,unsigned int> initDivisionFactor(tarch::la::Vector<dim,unsigned int> numberProcesses) const {
65 tarch::la::Vector<dim,unsigned int> div(1);
66 for (int d = dim-2; d > -1; d--){
67 div[d] = div[d+1]*numberProcesses[d+1];
68 }
69 return div;
70 }
71
72 /* number of processes */
73 const tarch::la::Vector<dim,unsigned int> _numberProcesses;
74 /* division factor */
75 const tarch::la::Vector<dim,unsigned int> _divisionFactor4NumberProcesses;
76 const unsigned int _topologyOffset;
77};
