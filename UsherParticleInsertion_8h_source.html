187 double getStepRef(double numberDensity, double sigma) const {
188 // if the coeff. was not parsed and thus set to -1.0, we choose the
189 // strategy described in the usher paper...
190 if (_stepRefCoeff == -1.0) {
191 return 0.1 * pow((1.0 / numberDensity), 3.0 / 2.0) * sigma;
192 // ... otherwise we fix the step size in units of sigma
193 } else {
194 return _stepRefCoeff * sigma;
195 }
196 }
