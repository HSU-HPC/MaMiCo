215 double getStepRef(double numberDensity, double sigma) const {
216 // if the coeff. was not parsed and thus set to -1.0, we choose the
217 // strategy described in the usher paper...
218 if (_stepRefCoeff == -1.0) {
219 return 0.1 * pow((1.0 / numberDensity), 3.0 / 2.0) * sigma;
220 // ... otherwise we fix the step size in units of sigma
221 } else {
222 return _stepRefCoeff * sigma;
223 }
224 }
