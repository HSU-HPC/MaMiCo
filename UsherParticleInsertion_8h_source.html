180 double getStepRef(double numberDensity, double sigma) const {
181 // if the coeff. was not parsed and thus set to -1.0, we choose the strategy
182 // described in the usher paper...
183 if (_stepRefCoeff==-1.0){
184 return 0.1*pow((1.0/numberDensity),3.0/2.0)*sigma;
185 // ... otherwise we fix the step size in units of sigma
186 } else {
187 return _stepRefCoeff*sigma;
188 }
189 }
