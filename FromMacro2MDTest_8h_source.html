24class FromMacro2MDTest : public Test {
25private:
26 template <unsigned int dim> void test() {
27 // define current rank, total size (only allowed to be cubic)
28 int rank = 0;
29 int size = 1;
30#if (COUPLING_MD_PARALLEL == COUPLING_MD_YES)
31 MPI_Comm_size(MPI_COMM_WORLD, &size);
32 MPI_Comm_rank(MPI_COMM_WORLD, &rank);
33#endif
34
35 const unsigned int oneDir = (unsigned int)floor(pow(((double)size), (1.0 / ((double)dim))) + 0.5);
36 if (pow(oneDir, dim) != size) {
37 std::cout << "FromMD2MacroTest::test: this is not a cubic domain decomposition!" << std::endl;
38 std::cout << "Size=" << size << ", in each direction: " << oneDir << std::endl;
39 std::cout << dim << "D case is not carried out." << std::endl;
40 return;
41 }
42
43 // define domain sizes for cells and MD
44 const tarch::la::Vector<dim, unsigned int> globalNumberMacroscopicCells(5 * oneDir + 2);
45 unsigned int numberCellsInclBoundary = globalNumberMacroscopicCells[0] + 2;
46 for (unsigned int d = 1; d < dim; d++) {
47 numberCellsInclBoundary = numberCellsInclBoundary * (globalNumberMacroscopicCells[d] + 2);
48 }
49 const tarch::la::Vector<dim, unsigned int> numberProcesses(oneDir);
50 const tarch::la::Vector<dim, double> mdDomainSize(1.0);
51 const tarch::la::Vector<dim, double> mdDomainOffset(0.0);
52 // output information
53 if (rank == 0) {
54 std::cout << "Global number macroscopic cells: " << globalNumberMacroscopicCells << std::endl;
55 std::cout << "Number processes: " << numberProcesses << std::endl;
56 }
57
58 // define functional objects
59 coupling::IndexConversion<dim> indexConversion(globalNumberMacroscopicCells, numberProcesses, rank, mdDomainSize, mdDomainOffset,
60 coupling::paralleltopology::XYZ);
61 coupling::sendrecv::FromMacro2MD<TestCell<dim>, dim> fromMacro2MD;
62 TestDataExchangeFromMacro2MD<dim> testDataExchangeFromMacro2MD(10, indexConversion);
63
64 // initialise the test cells of MaMiCo and set all cell values to 0.0
65 std::vector<TestCell<dim>*> cellsMamico;
66 for (unsigned int i = 0; i < numberCellsInclBoundary; i++) {
67 cellsMamico.push_back(new TestCell<dim>());
68 if (cellsMamico[i] == NULL) {
69 std::cout << "ERROR FromMD2MacroTest::testSequential3D: cellsMamico[i] "
70 "==NULL!"
71 << std::endl;
72 exit(EXIT_FAILURE);
73 }
74 cellsMamico[i]->setBuffer1(tarch::la::Vector<dim, double>(0.0));
75 cellsMamico[i]->setBuffer2(0.0);
76 }
77
78 // get macroscopic solver buffers and initialise cells before sending
79 std::vector<TestCell<dim>*> sentCells;
80 unsigned int* sentGlobalIndices = NULL;
81 unsigned int numberSentCells = 0;
82 testDataExchangeFromMacro2MD.getBuffer4MacroscopicSolverCells(numberSentCells, sentCells, sentGlobalIndices);
83 for (unsigned int i = 0; i < numberSentCells; i++) {
84 sentCells[i]->setBuffer1(tarch::la::Vector<dim, double>(sentGlobalIndices[i]));
85 sentCells[i]->setBuffer2(sentGlobalIndices[i]);
86 }
87 // INCLUDE THE FOLLOWING LINES FOR BETTER OUTPUT
88 sleep(indexConversion.getThisRank());
89 std::cout << "Send cells from rank " << indexConversion.getThisRank() << ": " << std::endl;
90 for (unsigned int i = 0; i < numberSentCells; i++) {
91 std::cout << "Cell " << sentGlobalIndices[i] << "(" << indexConversion.getGlobalVectorCellIndex(sentGlobalIndices[i])
92 << "): " << sentCells[i]->getBuffer1() << " , " << sentCells[i]->getBuffer2() << std::endl;
93 }
94#if (COUPLING_MD_PARALLEL == COUPLING_MD_YES)
95 MPI_Barrier(MPI_COMM_WORLD);
96#endif
97
98 // exchange quantities from MD to macro
99 fromMacro2MD.sendFromMacro2MD(indexConversion, testDataExchangeFromMacro2MD, cellsMamico, sentCells, sentGlobalIndices);
100
101 // INCLUDE THE FOLLOWING LINES FOR BETTER OUTPUT
102 sleep(indexConversion.getThisRank());
103 std::cout << "Received information on rank " << indexConversion.getThisRank() << ": " << std::endl;
104 for (unsigned int i = 0; i < numberCellsInclBoundary; i++) {
105 if (cellsMamico[i]->getBuffer2() > 0.0) {
106 std::cout << "Cell " << indexConversion.convertLocalToGlobalCellIndex(i) << "("
107 << indexConversion.getGlobalVectorCellIndex(indexConversion.convertLocalToGlobalCellIndex(i)) << "): " << cellsMamico[i]->getBuffer1()
108 << " , " << cellsMamico[i]->getBuffer2() << std::endl;
109 }
110 }
111
112 // free memory
113 for (unsigned int i = 0; i < (unsigned int)sentCells.size(); i++) {
114 if (sentCells[i] != NULL) {
115 delete sentCells[i];
116 sentCells[i] = NULL;
117 }
118 }
119 if (sentGlobalIndices != NULL) {
120 delete[] sentGlobalIndices;
121 sentGlobalIndices = NULL;
122 }
123 for (unsigned int i = 0; i < (unsigned int)cellsMamico.size(); i++) {
124 if (cellsMamico[i] != NULL) {
125 delete cellsMamico[i];
126 cellsMamico[i] = NULL;
127 }
128 }
129 }

132 FromMacro2MDTest() : Test("FromMacro2MDTest") {}

135 virtual void run() {
136 std::cout << "FromMacro2MDTest: Test 2D..." << std::endl;
137 test<2>();
138 std::cout << "FromMacro2MDTest: Test 3D..." << std::endl;
139 test<3>();
140 }
141};
