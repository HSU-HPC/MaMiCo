322 tarch::la::Vector<3,unsigned int> getGlobalNumberMacroscopicCells(
323 const simplemd::configurations::MolecularDynamicsConfiguration &simpleMDConfig, const coupling::configurations::MaMiCoConfiguration<3> &mamicoConfig) const {
324 tarch::la::Vector<3,double> domainSize(simpleMDConfig.getDomainConfiguration().getGlobalDomainSize());
325 tarch::la::Vector<3,double> dx(mamicoConfig.getMacroscopicCellConfiguration().getMacroscopicCellSize());
326 tarch::la::Vector<3,unsigned int> globalNumberMacroscopicCells(0);
327 for (unsigned int d = 0; d < 3; d++){
328 int buf = floor(domainSize[d]/dx[d]+0.5);
329 globalNumberMacroscopicCells[d] = (unsigned int) buf;
330 }
331 return globalNumberMacroscopicCells;
332 }

335 unsigned int* allocateSendBuffer(std::vector<coupling::datastructures::MacroscopicCell<3>* >& sendBuffer, const coupling::IndexConversion<3>& indexConversion,int rank,
336 coupling::interface::MacroscopicSolverInterface<3> &couetteSolverInterface) const {
337 // determine global number of cells
338 const tarch::la::Vector<3,unsigned int> cells(indexConversion.getGlobalNumberMacroscopicCells()+tarch::la::Vector<3,unsigned int>(2));
339 const unsigned int num = cells[0]*cells[1]*cells[2];
340
341 // delete all potential entries of sendBuffer
342 deleteBuffer(sendBuffer);
343
344 // count number of cells to be sent from this process; therefore, loop over all global macroscopic cells...
345 unsigned int numCellsSent=0;
346 for (unsigned int i =0; i < num; i++){
347 // ... and find out, if the current cell should be send to MD from this couette solver process
348 if(couetteSolverInterface.sendMacroscopicQuantityToMDSolver(indexConversion.getGlobalVectorCellIndex(i))){
349 std::vector<unsigned int> ranks = couetteSolverInterface.getSourceRanks(indexConversion.getGlobalVectorCellIndex(i));
350 bool containsThisRank=false;
351 for (unsigned int k = 0; k < ranks.size(); k++){
352 containsThisRank = containsThisRank || (ranks[k]==(unsigned int)rank);
353 }
354 if (containsThisRank){ numCellsSent++; }
355 }
356 }
357
358 // allocate array for cell indices
359 unsigned int* indices = new unsigned int [numCellsSent];
360 if (indices==NULL){std::cout << "ERROR NieTest::allocateSendBuffer(): indices==NULL!" << std::endl; exit(EXIT_FAILURE); }
361
362 // allocate sendBuffer and initialise all entries, incl. indices
363 for (unsigned int i = 0; i < num; i++){
364 if (couetteSolverInterface.sendMacroscopicQuantityToMDSolver(indexConversion.getGlobalVectorCellIndex(i))){
365 std::vector<unsigned int> ranks = couetteSolverInterface.getSourceRanks(indexConversion.getGlobalVectorCellIndex(i));
366 bool containsThisRank=false;
367 for (unsigned int k = 0; k < ranks.size(); k++){
368 containsThisRank = containsThisRank || (ranks[k]==(unsigned int)rank);
369 }
370 if (containsThisRank){
371 sendBuffer.push_back(new coupling::datastructures::MacroscopicCell<3>());
372 if (sendBuffer[sendBuffer.size()-1]==NULL){std::cout << "ERROR NieTest::allocateSendBuffer: sendBuffer[" << sendBuffer.size()-1 << "]==NULL!" << std::endl; exit(EXIT_FAILURE);}
373 indices[sendBuffer.size()-1] = i;
374 }
375 }
376 }
377
378 #if (COUPLING_MD_DEBUG==COUPLING_MD_YES)
379 for (unsigned int i = 0; i < numCellsSent; i++){
380 std::vector<unsigned int> ranks = couetteSolverInterface.getSourceRanks(indexConversion.getGlobalVectorCellIndex(indices[i]));
381 std::cout << "Current rank= " << rank << ", Send cell " << indexConversion.getGlobalVectorCellIndex(indices[i]) << ", ranks=";
382 for (unsigned int j = 0; j < ranks.size(); j++){ std::cout << " " << ranks[j];}
383 std::cout << std::endl;
384 }
385 #endif
386 return indices;
387 }

449 const coupling::IndexConversion<3>& indexConversion,int rank, int couplingCycle
450 ) const {
451 // form file name and open file
452 std::stringstream ss;
453 ss << "CouetteAvgMultiMDCells_" << rank << "_" << couplingCycle << ".csv";
454 std::ofstream file(ss.str().c_str());
455 if (!file.is_open()){std::cout << "ERROR NieTest::write2CSV(): Could not open file " << ss.str() << "!" << std::endl; exit(EXIT_FAILURE);}
456
457 // loop over received cells; read macroscopic mass+momentum buffers and write cell index, mass and velocity to one line in the csv-file
458 const unsigned int numCellsRecv = recvBuffer.size();
459 for (unsigned int i = 0; i < numCellsRecv; i++){
