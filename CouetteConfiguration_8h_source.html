51 static CouetteConfig parseCouetteConfiguration(const std::string& filename) {
52 CouetteConfig cfg;
53
54 ParseConfiguration::XMLConfiguration xmlConfig = ParseConfiguration::XMLConfiguration::load(filename);
55 tinyxml2::XMLElement* node = xmlConfig.root->FirstChildElement("couette-test");
56
57 if (node == NULL) {
58 std::cout << "Could not read input file " << filename
59 << ": missing element "
60 "<couette-test>"
61 << std::endl;
62 exit(EXIT_FAILURE);
63 }
64
65 tinyxml2::XMLElement* n_mamico = node->NextSiblingElement();
66 if (n_mamico == NULL) {
67 std::cout << "Could not read input file " << filename << ": missing element <mamico>" << std::endl;
68 exit(EXIT_FAILURE);
69 }
70 tinyxml2::XMLElement* n_md = n_mamico->NextSiblingElement();
71 if (n_md == NULL) {
72 std::cout << "Could not read input file " << filename
73 << ": missing element "
74 "<molecular-dynamics>"
75 << std::endl;
76 exit(EXIT_FAILURE);
77 }
78 tinyxml2::XMLElement* n_fp = n_md->NextSiblingElement();
79 if (n_fp == NULL) {
80 std::cout << "Could not read input file " << filename
81 << ": missing element "
82 "<filter-pipeline>"
83 << std::endl;
84 exit(EXIT_FAILURE);
85 }
86 tinyxml2::XMLElement* n_unexpected = n_fp->NextSiblingElement();
87 if (n_unexpected != NULL) {
88 std::cout << "Could not read input file " << filename << ": unknown element " << n_unexpected->Name() << std::endl;
89 exit(EXIT_FAILURE);
90 }
91
92 tinyxml2::XMLElement* subtag = node->FirstChildElement("domain");
93 if (subtag == NULL) {
94 std::cout << "Could not read input file " << filename << ": Missing subtag: domain" << std::endl;
95 exit(EXIT_FAILURE);
96 }
97 tarch::configuration::ParseConfiguration::readDoubleMandatory(cfg.channelheight, subtag, "channelheight");
98 tarch::configuration::ParseConfiguration::readVectorMandatory<3, double>(cfg.wallVelocity, subtag, "wall-velocity");
99 cfg.wallInitCycles = 0;
100 tarch::configuration::ParseConfiguration::readIntOptional(cfg.wallInitCycles, subtag, "wall-init-cycles");
101 if (cfg.wallInitCycles > 0)
102 tarch::configuration::ParseConfiguration::readVectorMandatory<3, double>(cfg.wallInitVelocity, subtag, "wall-init-velocity");
103 cfg.wallOscillations = 0;
104 tarch::configuration::ParseConfiguration::readDoubleOptional(cfg.wallOscillations, subtag, "wall-oscillations");
105
106 subtag = node->FirstChildElement("coupling");
107 if (subtag == NULL) {
108 std::cout << "Could not read input file couette.xml: Missing subtag: coupling" << std::endl;
109 exit(EXIT_FAILURE);
110 }
111 tarch::configuration::ParseConfiguration::readIntMandatory(cfg.couplingCycles, subtag, "coupling-cycles");
112 tarch::configuration::ParseConfiguration::readBoolMandatory(cfg.twoWayCoupling, subtag, "two-way-coupling");
113 tarch::configuration::ParseConfiguration::readBoolMandatory(cfg.md2Macro, subtag, "send-from-md-to-macro");
114 tarch::configuration::ParseConfiguration::readBoolMandatory(cfg.macro2Md, subtag, "send-from-macro-to-md");
115 tarch::configuration::ParseConfiguration::readIntMandatory(cfg.filterInitCycles, subtag, "filter-init-cycles");
116 tarch::configuration::ParseConfiguration::readIntMandatory(cfg.csvEveryTimestep, subtag, "write-csv-every-timestep");
117 tarch::configuration::ParseConfiguration::readBoolMandatory(cfg.computeSNR, subtag, "compute-snr");
118
119 subtag = node->FirstChildElement("microscopic-solver");
120 if (subtag == NULL) {
121 std::cout << "Could not read input file " << filename
122 << ": Missing subtag: "
123 "microscopic-solver"
124 << std::endl;
125 exit(EXIT_FAILURE);
126 }
127 std::string type;
128 tarch::configuration::ParseConfiguration::readStringMandatory(type, subtag, "type");
129 if (type == "md") {
130 cfg.miSolverType = SIMPLEMD;
131 tarch::configuration::ParseConfiguration::readDoubleMandatory(cfg.temp, subtag, "temperature");
132 tarch::configuration::ParseConfiguration::readIntMandatory(cfg.equSteps, subtag, "equilibration-steps");
133
134 std::string num_MD;
135 tarch::configuration::ParseConfiguration::readStringMandatory(num_MD, subtag, "number-md-simulations");
136 if (num_MD == "dynamic") {
137 cfg.totalNumberMDSimulations = -1;
138 cfg.lowerBoundNumberMDSimulations = 1;
139 tarch::configuration::ParseConfiguration::readIntOptional(cfg.lowerBoundNumberMDSimulations, subtag, "min-number-md");
140 tarch::configuration::ParseConfiguration::readDoubleMandatory(cfg.absVelErrStart, subtag, "error-start");
141 tarch::configuration::ParseConfiguration::readDoubleMandatory(cfg.absVelErrEnd, subtag, "error-end");
142 } else {
143 tarch::configuration::ParseConfiguration::readIntMandatory(cfg.totalNumberMDSimulations, subtag, "number-md-simulations");
144 if (cfg.totalNumberMDSimulations < 1) {
145 std::cout << "Could not read input file " << filename
146 << ": "
147 "number-md-simulations < 1"
148 << std::endl;
149 exit(EXIT_FAILURE);
150 }
151 }
152 } else if (type == "synthetic") {
153 cfg.miSolverType = SYNTHETIC;
154 tarch::configuration::ParseConfiguration::readDoubleMandatory(cfg.noiseSigma, subtag, "noise-sigma");
155 cfg.totalNumberMDSimulations = 1;
156 tarch::configuration::ParseConfiguration::readIntOptional(cfg.totalNumberMDSimulations, subtag, "number-md-simulations");
157 } else if (type == "ls1") {
158 cfg.miSolverType = LS1;
159
160 cfg.totalNumberMDSimulations = 1;
161 tarch::configuration::ParseConfiguration::readDoubleMandatory(cfg.temp, subtag, "temperature");
162 tarch::configuration::ParseConfiguration::readIntMandatory(cfg.equSteps, subtag, "equilibration-steps");
163 tarch::configuration::ParseConfiguration::readIntOptional(cfg.totalNumberMDSimulations, subtag, "number-md-simulations");
164 } else {
165 std::cout << "Could not read input file " << filename
166 << ": Unknown microscopic "
167 "solver type!"
168 << std::endl;
169 exit(EXIT_FAILURE);
170 }
171 tarch::configuration::ParseConfiguration::readDoubleMandatory(cfg.density, subtag, "density");
172
173 subtag = node->FirstChildElement("macroscopic-solver");
174 if (subtag == NULL) {
175 std::cout << "Could not read input file " << filename
176 << ": Missing subtag: "
177 "macroscopic-solver"
178 << std::endl;
179 exit(EXIT_FAILURE);
180 }
181 cfg.lbNumberProcesses = tarch::la::Vector<3, unsigned int>(1);
182 tarch::configuration::ParseConfiguration::readStringMandatory(type, subtag, "type");
183 if (type == "lb") {
184 cfg.maSolverType = COUETTE_LB;
185 tarch::configuration::ParseConfiguration::readVectorMandatory<3, unsigned int>(cfg.lbNumberProcesses, subtag, "number-of-processes");
186 tarch::configuration::ParseConfiguration::readIntMandatory(cfg.plotEveryTimestep, subtag, "plot-every-timestep");
187 } else if (type == "fd") {
188 cfg.maSolverType = COUETTE_FD;
189 tarch::configuration::ParseConfiguration::readVectorMandatory<3, unsigned int>(cfg.lbNumberProcesses, subtag, "number-of-processes");
190 tarch::configuration::ParseConfiguration::readIntMandatory(cfg.plotEveryTimestep, subtag, "plot-every-timestep");
191 }
192#if (BUILD_WITH_OPENFOAM)
193 else if (type == "foam") {
194 _cfg.maSolverType = COUETTE_FOAM;
195 tarch::configuration::ParseConfiguration::readIntMandatory(_cfg.plotEveryTimestep, subtag, "plot-every-timestep");
196 tarch::configuration::ParseConfiguration::readStringMandatory(_cfg.foam.directory, subtag, "foam-setup-directory");
197 tarch::configuration::ParseConfiguration::readStringMandatory(_cfg.foam.folder, subtag, "foam-setup-folder");
198 tarch::configuration::ParseConfiguration::readVectorMandatory<12, unsigned int>(_cfg.foam.boundariesWithMD, subtag, "boundaries-with-MD");
199 if (!_cfg.twoWayCoupling && _cfg.foam.boundariesWithMD != tarch::la::Vector<12, unsigned int>{0}) {
200 std::cout << "ERROR: Two-way coupling is disabled, but boundaries with "
201 "MD for openfoam were defined"
202 << std::endl;
203 exit(EXIT_FAILURE);
204 }
205 }
206#endif
207 else if (type == "analytical") {
208 cfg.maSolverType = COUETTE_ANALYTICAL;
209 if (!(cfg.wallVelocity[1] == 0.0 && cfg.wallVelocity[2] == 0.0)) {
210 std::cout << "analytic solver only supports flow in x-direction" << std::endl;
211 exit(EXIT_FAILURE);
212 }
213 } else {
214 std::cout << "Could not read input file " << filename
215 << ": Unknown macroscopic "
216 "solver type!"
217 << std::endl;
218 exit(EXIT_FAILURE);
219 }
220 double vis;
221 tarch::configuration::ParseConfiguration::readDoubleMandatory(vis, subtag, "viscosity");
222 cfg.kinVisc = vis / cfg.density;
223 tarch::configuration::ParseConfiguration::readIntMandatory(cfg.initAdvanceCycles, subtag, "init-advance-cycles");
224
225 subtag = node->FirstChildElement("tws-loop");
226 if (subtag == NULL)
227 cfg.twsLoop = false;
228 else {
229 cfg.twsLoop = true;
230 tarch::configuration::ParseConfiguration::readIntMandatory(cfg.twsLoopMin, subtag, "min");
231 tarch::configuration::ParseConfiguration::readIntMandatory(cfg.twsLoopMax, subtag, "max");
232 cfg.twsLoopStep = 1;
233 tarch::configuration::ParseConfiguration::readIntOptional(cfg.twsLoopStep, subtag, "step");
234 }
235
236 if (cfg.miSolverType == SYNTHETIC) {
237 if (cfg.macro2Md || cfg.totalNumberMDSimulations > 1 || cfg.lbNumberProcesses[0] != 1 || cfg.lbNumberProcesses[1] != 1 ||
238 cfg.lbNumberProcesses[2] != 1) {
239 std::cout << "Invalid configuration: Synthetic MD runs sequentially on "
240 "rank 0 only. "
241 << "It does neither support parallel communication nor "
242 "multi-instance sampling"
243 << std::endl;
244 exit(EXIT_FAILURE);
245 }
246 }
247 if (cfg.maSolverType == COUETTE_ANALYTICAL) {
248 if (cfg.twoWayCoupling) {
249 std::cout << "Invalid configuration: COUETTE_ANALYTICAL does not "
250 "support twoWayCoupling"
251 << std::endl;
252 exit(EXIT_FAILURE);
253 }
254 }
255
256 return cfg;
257 }
