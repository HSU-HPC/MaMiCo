86 static CouetteConfig parseCouetteConfiguration(const std::string &filename) {
87 CouetteConfig _cfg;
88
89 tinyxml2::XMLDocument conffile;
90 tinyxml2::XMLElement *node = NULL;
91 conffile.LoadFile("couette.xml");
92 node = conffile.FirstChildElement("couette-test");
93 if (node == NULL) {
94 std::cout << "Could not read input file couette.xml: missing element "
95 "<couette-test>"
96 << std::endl;
97 exit(EXIT_FAILURE);
98 }
99
100 tinyxml2::XMLElement *n_mamico = node->NextSiblingElement();
101 if (n_mamico == NULL) {
102 std::cout << "Could not read input file couette.xml: missing element <mamico>" << std::endl;
103 exit(EXIT_FAILURE);
104 }
105 tinyxml2::XMLElement *n_md = n_mamico->NextSiblingElement();
106 if (n_md == NULL) {
107 std::cout << "Could not read input file couette.xml: missing element "
108 "<molecular-dynamics>"
109 << std::endl;
110 exit(EXIT_FAILURE);
111 }
112 tinyxml2::XMLElement *n_fp = n_md->NextSiblingElement();
113 if (n_fp == NULL) {
114 std::cout << "Could not read input file couette.xml: missing element "
115 "<filter-pipeline>"
116 << std::endl;
117 exit(EXIT_FAILURE);
118 }
119 tinyxml2::XMLElement *n_unexpected = n_fp->NextSiblingElement();
120 if (n_unexpected == NULL) {
121 std::cout << "Could not read input file couette.xml: unknown element " << n_unexpected->Name() << std::endl;
122 exit(EXIT_FAILURE);
123 }
124
125 tinyxml2::XMLElement *subtag = node->FirstChildElement("domain");
126 if (subtag == NULL) {
127 std::cout << "Could not read input file couette.xml: Missing subtag: domain" << std::endl;
128 exit(EXIT_FAILURE);
129 }
130 tarch::configuration::ParseConfiguration::readDoubleMandatory(_cfg.channelheight, subtag, "channelheight");
131 tarch::configuration::ParseConfiguration::readVector<3, double>(_cfg.wallVelocity, subtag, "wall-velocity");
132 _cfg.wallInitCycles = 0;
133 tarch::configuration::ParseConfiguration::readIntOptional(_cfg.wallInitCycles, subtag, "wall-init-cycles");
134 if (_cfg.wallInitCycles > 0)
135 tarch::configuration::ParseConfiguration::readVector<3, double>(_cfg.wallInitVelocity, subtag, "wall-init-velocity");
136 _cfg.wallOscillations = 0;
137 tarch::configuration::ParseConfiguration::readDoubleOptional(_cfg.wallOscillations, subtag, "wall-oscillations");
138
139 subtag = node->FirstChildElement("coupling");
140 if (subtag == NULL) {
141 std::cout << "Could not read input file couette.xml: Missing subtag: coupling" << std::endl;
142 exit(EXIT_FAILURE);
143 }
144 tarch::configuration::ParseConfiguration::readIntMandatory(_cfg.couplingCycles, subtag, "coupling-cycles");
145 tarch::configuration::ParseConfiguration::readBoolMandatory(_cfg.twoWayCoupling, subtag, "two-way-coupling");
146 tarch::configuration::ParseConfiguration::readBoolMandatory(_cfg.md2Macro, subtag, "send-from-md-to-macro");
147 tarch::configuration::ParseConfiguration::readBoolMandatory(_cfg.macro2Md, subtag, "send-from-macro-to-md");
148 tarch::configuration::ParseConfiguration::readIntMandatory(_cfg.filterInitCycles, subtag, "filter-init-cycles");
149 tarch::configuration::ParseConfiguration::readIntMandatory(_cfg.csvEveryTimestep, subtag, "write-csv-every-timestep");
150 tarch::configuration::ParseConfiguration::readBoolMandatory(_cfg.computeSNR, subtag, "compute-snr");
151
152 subtag = node->FirstChildElement("microscopic-solver");
153 if (subtag == NULL) {
154 std::cout << "Could not read input file couette.xml: Missing subtag: "
155 "microscopic-solver"
156 << std::endl;
157 exit(EXIT_FAILURE);
158 }
159 std::string type;
160 tarch::configuration::ParseConfiguration::readStringMandatory(type, subtag, "type");
161 if (type == "md") {
162 _cfg.miSolverType = SIMPLEMD;
163 tarch::configuration::ParseConfiguration::readDoubleMandatory(_cfg.temp, subtag, "temperature");
164 tarch::configuration::ParseConfiguration::readIntMandatory(_cfg.equSteps, subtag, "equilibration-steps");
165 tarch::configuration::ParseConfiguration::readIntMandatory(_cfg.totalNumberMDSimulations, subtag, "number-md-simulations");
166 if (_cfg.totalNumberMDSimulations < 1) {
167 std::cout << "Could not read input file couette.xml: "
168 "number-md-simulations < 1"
169 << std::endl;
170 exit(EXIT_FAILURE);
171 }
172 } else if (type == "synthetic") {
173 _cfg.miSolverType = SYNTHETIC;
174 tarch::configuration::ParseConfiguration::readDoubleMandatory(_cfg.noiseSigma, subtag, "noise-sigma");
175 _cfg.totalNumberMDSimulations = 1;
176 tarch::configuration::ParseConfiguration::readIntOptional(_cfg.totalNumberMDSimulations, subtag, "number-md-simulations");
177 } else {
178 std::cout << "Could not read input file couette.xml: Unknown microscopic "
179 "solver type!"
180 << std::endl;
181 exit(EXIT_FAILURE);
182 }
183 tarch::configuration::ParseConfiguration::readDoubleMandatory(_cfg.density, subtag, "density");
184
185 subtag = node->FirstChildElement("macroscopic-solver");
186 if (subtag == NULL) {
187 std::cout << "Could not read input file couette.xml: Missing subtag: "
188 "macroscopic-solver"
189 << std::endl;
190 exit(EXIT_FAILURE);
191 }
192 _cfg.lbNumberProcesses = tarch::la::Vector<3, unsigned int>(1);
193 tarch::configuration::ParseConfiguration::readStringMandatory(type, subtag, "type");
194 if (type == "lb") {
195 _cfg.maSolverType = COUETTE_LB;
196 tarch::configuration::ParseConfiguration::readVector<3, unsigned int>(_cfg.lbNumberProcesses, subtag, "number-of-processes");
197 tarch::configuration::ParseConfiguration::readIntMandatory(_cfg.plotEveryTimestep, subtag, "plot-every-timestep");
198 } else if (type == "fd") {
199 _cfg.maSolverType = COUETTE_FD;
200 tarch::configuration::ParseConfiguration::readVector<3, unsigned int>(_cfg.lbNumberProcesses, subtag, "number-of-processes");
201 tarch::configuration::ParseConfiguration::readIntMandatory(_cfg.plotEveryTimestep, subtag, "plot-every-timestep");
202 } else if (type == "analytical") {
203 _cfg.maSolverType = COUETTE_ANALYTICAL;
204 if (!(_cfg.wallVelocity[1] == 0.0 && _cfg.wallVelocity[2] == 0.0)) {
205 std::cout << "analytic solver only supports flow in x-direction" << std::endl;
206 exit(EXIT_FAILURE);
207 }
208 } else {
209 std::cout << "Could not read input file couette.xml: Unknown macroscopic "
210 "solver type!"
211 << std::endl;
212 exit(EXIT_FAILURE);
213 }
214 double vis;
215 tarch::configuration::ParseConfiguration::readDoubleMandatory(vis, subtag, "viscosity");
216 _cfg.kinVisc = vis / _cfg.density;
217 tarch::configuration::ParseConfiguration::readIntMandatory(_cfg.initAdvanceCycles, subtag, "init-advance-cycles");
218
219 subtag = node->FirstChildElement("tws-loop");
220 if (subtag == NULL)
221 _cfg.twsLoop = false;
222 else {
223 _cfg.twsLoop = true;
224 tarch::configuration::ParseConfiguration::readIntMandatory(_cfg.twsLoopMin, subtag, "min");
225 tarch::configuration::ParseConfiguration::readIntMandatory(_cfg.twsLoopMax, subtag, "max");
226 _cfg.twsLoopStep = 1;
227 tarch::configuration::ParseConfiguration::readIntOptional(_cfg.twsLoopStep, subtag, "step");
228 }
229
230 if (_cfg.miSolverType == SYNTHETIC) {
231 if (_cfg.md2Macro || _cfg.macro2Md || _cfg.totalNumberMDSimulations > 1 || _cfg.lbNumberProcesses[0] != 1 || _cfg.lbNumberProcesses[1] != 1 ||
232 _cfg.lbNumberProcesses[2] != 1) {
233 std::cout << "Invalid configuration: Synthetic MD runs sequentially on "
234 "rank 0 only. "
235 << "It does neither support parallel communication nor "
236 "multi-instance sampling"
237 << std::endl;
238 exit(EXIT_FAILURE);
239 }
240 }
241 if (_cfg.maSolverType == COUETTE_ANALYTICAL) {
242 if (_cfg.twoWayCoupling) {
243 std::cout << "Invalid configuration: COUETTE_ANALYTICAL does not "
244 "support twoWayCoupling"
245 << std::endl;
246 exit(EXIT_FAILURE);
247 }
248 }
249
250 return _cfg;
251 }
