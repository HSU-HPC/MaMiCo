70 // advance the solver dt in time, this may include several timesteps,
71 // the sequence is based on the OpenFOAM IcoFoam solver
72 void advance(double dt) override {
73 if (skipRank()) {
74 return;
75 }
76 size_t number = floor(dt / runTime.deltaTValue() + 0.5);
77 for (size_t i = 0; i < number; i++) {
78 using namespace Foam;
79 ++runTime;
80 Info << "Time = " << runTime.timeName() << nl << endl;
81
82 // Momentum predictor
83 fvVectorMatrix UEqn(fvm::ddt(U) + fvm::div(phi, U) - fvm::laplacian(nu, U));
84 if (piso.momentumPredictor()) {
85 solve(UEqn == -fvc::grad(p));
86 }
87 // --- PISO loop
88 while (piso.correct()) {
89 volScalarField rAU(1.0 / UEqn.A());
90 volVectorField HbyA(constrainHbyA(rAU * UEqn.H(), U, p));
91 surfaceScalarField phiHbyA("phiHbyA", fvc::flux(HbyA) + fvc::interpolate(rAU) * fvc::ddtCorr(U, phi));
92 adjustPhi(phiHbyA, U, p);
93
94 // Update the pressure BCs to ensure flux consistency
95 constrainPressure(p, U, phiHbyA, rAU);
96
97 // Non-orthogonal pressure corrector loop
98 while (piso.correctNonOrthogonal()) {
99 // Pressure corrector
100 fvScalarMatrix pEqn(fvm::laplacian(rAU, p) == fvc::div(phiHbyA));
101 pEqn.setReference(pRefCell, pRefValue);
102 pEqn.solve();
103 if (piso.finalNonOrthogonalIter()) {
104 phi = phiHbyA - pEqn.flux();
105 }
106 }
107 U = HbyA - rAU * fvc::grad(p);
108 U.correctBoundaryConditions();
109 }
110 // runTime.write(); // writes the original OpenFOAM output
111 // plot();
112 _timestepCounter++;
113 plottxt();
114 }
115 }

117 // Get the velocity at position pos for the current time step
118 tarch::la::Vector<3, double> getVelocity(tarch::la::Vector<3, double> pos) const override {
119 const Foam::vector foamPosition(pos[0], pos[1], pos[2]);
120 const int foamIndice = U.mesh().findCell(foamPosition);
121 if (foamIndice > 0) {
122 return tarch::la::Vector<3, double>(U[foamIndice][0], U[foamIndice][1], U[foamIndice][2]);
123 } else {
124 return tarch::la::Vector<3, double>(0, 0, 0);
125 }
126 };

138 // Applies the MD data (just velocities) as boundary condition, the mapping between the conntinuum and the MD is provided by the setMDBoundary()
139 void setMDBoundaryValues(std::vector<coupling::datastructures::MacroscopicCell<3>*>& recvBuffer, const unsigned int* const recvIndices,
140 const coupling::IndexConversion<3>& indexConversion) {
141 if (skipRank()) {
142 return;
143 }
144 for (size_t i = 0; i < _numberBoundaryPoints; i++) {
145 size_t outer = _boundary2RecvBufferIndicesOuter[i];
146 size_t inner = _boundary2RecvBufferIndicesInner[i];
147 tarch::la::Vector<3, double> localOuterVel((1.0 / recvBuffer[outer]->getMacroscopicMass()) * recvBuffer[outer]->getMacroscopicMomentum());
148 tarch::la::Vector<3, double> localInnerVel((1.0 / recvBuffer[inner]->getMacroscopicMass()) * recvBuffer[inner]->getMacroscopicMomentum());
149 _boundaryIndices[i]->x() = (localOuterVel[0] + localInnerVel[0]) * 0.5;
150 _boundaryIndices[i]->y() = (localOuterVel[1] + localInnerVel[1]) * 0.5;
151 _boundaryIndices[i]->z() = (localOuterVel[2] + localInnerVel[2]) * 0.5;
152 }
153 }

155 // Settup for the mapping from MD to continuum data. Velocity values are necessary directly on the boundary. Therefore the MD values from the two cells beside
156 // the boundary are interpolated. The function looks for the index of every cell that data is necessary from. This indices will be stored in two arrays.
157 void setMDBoundary(tarch::la::Vector<3, double> mdDomainOffset, tarch::la::Vector<3, double> mdDomainSize, unsigned int overlapStrip,
158 const coupling::IndexConversion<3>& indexConversion, const unsigned int* const recvIndice, unsigned int size) {
159 if (skipRank()) {
160 return;
161 }
162 size_t innerMDBoundaryIndex = 0;
163 while (_boundariesWithMD[innerMDBoundaryIndex] == 0) {
164 innerMDBoundaryIndex++;
165 }
166 _numberBoundaryPoints = 6 * U.boundaryFieldRef()[innerMDBoundaryIndex].size();
167 _boundary2RecvBufferIndicesOuter = new size_t[_numberBoundaryPoints];
168 _boundary2RecvBufferIndicesInner = new size_t[_numberBoundaryPoints];
169 _boundaryIndices = new Foam::vector*[_numberBoundaryPoints];
170 size_t counter = 0;
171 for (size_t boundary = 0; boundary < 12; boundary++) {
172 if (_boundariesWithMD[boundary] == 1) {
173 size_t MDPointsPerBoundary = _numberBoundaryPoints / 6;
174 for (size_t j = 0; j < MDPointsPerBoundary; j++) {
175 _boundaryIndices[counter] = &(U.boundaryFieldRef()[boundary][j]);
176 const size_t globalIndexOuter = indexConversion.getGlobalCellIndex(indexConversion.getGlobalVectorCellIndex(getOuterPointFromBoundary(boundary, j)));
177 const size_t globalIndexInner = indexConversion.getGlobalCellIndex(indexConversion.getGlobalVectorCellIndex(getInnerPointFromBoundary(boundary, j)));
178 // const size_t globalIndexInner = indexConversion.getGlobalCellIndex(indexConversion.getGlobalVectorCellIndex(getInnerPointFromBoundary(boundary,
179 // j)));
180 for (size_t k = 0; k < size; k++) {
181 if (globalIndexOuter == recvIndice[k]) {
182 _boundary2RecvBufferIndicesOuter[counter] = k;
183 goto endloop;
184 }
185 }
186 std::cout << "IcoFoamInterface: Within the mapping of the FoamBoundary and the SimpleMD cells there was an error" << std::endl;
187 endloop:
188 for (size_t k = 0; k < size; k++) {
189 if (globalIndexInner == recvIndice[k]) {
190 _boundary2RecvBufferIndicesInner[counter] = k;
191 goto endloop2;
192 }
193 }
194 std::cout << "IcoFoamInterface: Within the mapping of the FoamBoundary and the SimpleMD cells there was an error" << std::endl;
195 endloop2:
196 counter++;
197 }
198 }
199 }

224 // Calculates the analytical solution for U for z=pos for the current time step
225 const double getAnalyticalCouetteU(const double& pos) const {
226 const double pi = 3.141592653589793238;
227 double u_analytical = 0.0;
228 for (int k = 1; k < 30; k++) {
229 u_analytical += 1.0 / k * std::sin(k * pi * pos / _channelheight) *
230 std::exp(-k * k * pi * pi / (_channelheight * _channelheight) * nu.value() * runTime.timeOutputValue());
231 }
232 return _uWall[0] * (1.0 - pos / _channelheight - 2.0 / pi * u_analytical);
233 }
