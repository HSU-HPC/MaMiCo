39 void parseSubtag( tinyxml2::XMLElement* node ){
40 std::string value;
41 tarch::configuration::ParseConfiguration::readStringMandatory(value,node,"type");
42 const std::string boundaries[6] = {"west","east","south","north","bottom","top"};
43
44 if (value=="zhou-boundary-force"){
45 _insertionType = ZHOU;
46 } else if (value=="none"){
47 _insertionType = NO_BOUNDARYFORCE;
48 } else {
49 std::cout << "ERROR coupling::BoundaryForceConfiguration: Wrong insertion type!" << std::endl;
50 _isValid = false;
51 exit(EXIT_FAILURE);
52 }
53
54 // for all boundary force types: read the sides at which boundary force shall be applied
55 // this may not be useful for NO_BOUNDARYFORCE; however, once read, the user can still use the flags,
56 // even in the NO_BOUNDARYFORCE case, to e.g. determine a setup with boundary forcing using external implementations
57 for (unsigned int d = 0; d < 2*dim; d++){
58 tarch::configuration::ParseConfiguration::readBoolMandatory(_boundary[d],node,boundaries[d]);
59 }
60
61 if (_insertionType==ZHOU){
62 tarch::configuration::ParseConfiguration::readDoubleMandatory(_density,node,"density");
63 if (_density<=0.0){
64 std::cout << "ERROR coupling::BoundaryForceConfiguration: density is smaller than zero! Density=" << _density;
65 _isValid = false;
66 exit(EXIT_FAILURE);
67 }
68
69 tarch::configuration::ParseConfiguration::readDoubleMandatory(_temperature,node,"temperature");
70 if (_temperature<=0.0){
71 std::cout << "ERROR coupling::BoundaryForceConfiguration: temperature is smaller than zero! Temperature=" << _temperature;
72 _isValid = false;
73 exit(EXIT_FAILURE);
74 }
75 }
76 }

101 template<class LinkedCell>
102 coupling::BoundaryForceController<LinkedCell,dim>* interpreteConfiguration(
103 coupling::interface::MDSolverInterface<LinkedCell,dim> * const mdSolverInterface
104 ) const {
105 if (_insertionType == ZHOU){
106 if (dim!=3){
107 std::cout << "ERROR coupling::configurations::BoundaryForceConfiguration::interpreteConfiguration(): Zhou boundary force only valid in 3D!" << std::endl;
108 exit(EXIT_FAILURE);
109 }
110 return new coupling::ZhouBoundaryForceController<LinkedCell,dim>(_density,_temperature,_boundary,mdSolverInterface);
111 } else if (_insertionType == NO_BOUNDARYFORCE){
112 return new coupling::NoBoundaryForce<LinkedCell,dim>(mdSolverInterface);
113 }
114 return NULL;
115 }
