48 void parseSubtag( tinyxml2::XMLElement* node ){
49 std::string value;
50 tarch::configuration::ParseConfiguration::readStringMandatory(value,node,"type");
51 const std::string boundaries[6] = {"west","east","south","north","bottom","top"};
52
53 if (value=="zhou-boundary-force"){
54 _insertionType = ZHOU;
55 } else if (value=="none"){
56 _insertionType = NO_BOUNDARYFORCE;
57 } else {
58 std::cout << "ERROR coupling::BoundaryForceConfiguration: Wrong insertion type!" << std::endl;
59 _isValid = false;
60 exit(EXIT_FAILURE);
61 }
62
63 // for all boundary force types: read the sides at which boundary force shall be applied
64 // this may not be useful for NO_BOUNDARYFORCE; however, once read, the user can still use the flags,
65 // even in the NO_BOUNDARYFORCE case, to e.g. determine a setup with boundary forcing using external implementations
66 for (unsigned int d = 0; d < 2*dim; d++){
67 tarch::configuration::ParseConfiguration::readBoolMandatory(_boundary[d],node,boundaries[d]);
68 }
69
70 if (_insertionType==ZHOU){
71 tarch::configuration::ParseConfiguration::readDoubleMandatory(_density,node,"density");
72 if (_density<=0.0){
73 std::cout << "ERROR coupling::BoundaryForceConfiguration: density is smaller than zero! Density=" << _density;
74 _isValid = false;
75 exit(EXIT_FAILURE);
76 }
77
78 tarch::configuration::ParseConfiguration::readDoubleMandatory(_temperature,node,"temperature");
79 if (_temperature<=0.0){
80 std::cout << "ERROR coupling::BoundaryForceConfiguration: temperature is smaller than zero! Temperature=" << _temperature;
81 _isValid = false;
82 exit(EXIT_FAILURE);
83 }
84 }
85 }
