391

395 // MultiMDCellService, because SYNTHETIC should fill the same cells which

398

400 _simpleMDConfig.getMPIConfiguration().getNumberOfProcesses(),

408#if (COUPLING_MD_PARALLEL == COUPLING_MD_YES)

411#endif
412 ));

414 std::cout << "ERROR CouetteTest: _simpleMD[" << i << "]==NULL!"

416 exit(EXIT_FAILURE);
417 }
418 }
419
420 _mdStepCounter = 0;
421 if (_rank == 0) {
422 gettimeofday(&_tv.start, NULL);
423 }
424 for (unsigned int i = 0; i < _localMDInstances; i++) {
425 _simpleMD[i]->init(
426 *_multiMDService,
427 _multiMDService->getGlobalNumberOfLocalMDSimulation(i));
428 }

459 _mamicoConfig.getMomentumInsertionConfiguration()
460 .getInnerOverlap());
461
462 if (_cfg.twsLoop) {
463 // initialise macroscopic cell service for multi-MD case and set single
464 // cell services in each MD simulation
465 _multiMDCellService =
466 new coupling::services::MultiMDCellService<MY_LINKEDCELL, 3>(

472 _mamicoConfig.getBoundaryForceConfiguration(),
473 _mamicoConfig.getTransferStrategyConfiguration(),
474 _mamicoConfig.getParallelTopologyConfiguration(),
475 _mamicoConfig.getThermostatConfiguration(),
476 _simpleMDConfig.getSimulationConfiguration()
477 .getNumberOfTimesteps(),
478 _mamicoConfig.getMacroscopicCellConfiguration(), "couette.xml",

480 } else {
481 // initialise macroscopic cell service for multi-MD case and set single
482 // cell services in each MD simulation
483 _multiMDCellService =
484 new coupling::services::MultiMDCellService<MY_LINKEDCELL, 3>(
485 _mdSolverInterface, couetteSolverInterface,
486 _simpleMDConfig.getMPIConfiguration().getNumberOfProcesses(),

500 // init indexing
501 coupling::indexing::IndexingService<3>::getInstance().init(

504
505 // init filtering for all md instances
506 _multiMDCellService->constructFilterPipelines();

524 /*
525 * A synthethic solver is modeled using a dynamically linked filter,
526 * i.e. a lambda function producing artifical data in every filter step.
527 *
528 * This is how to properly instanciate and use a synthethic solver:
529 * - Create a sequence named like the macro SYNTHETICMD_SEQUENCE in xml. Use
530 * whatever input, but make sure the filter system's output is set to this
531 * sequence.
532 * - Set filtered-values = "macro-mass macro-momentum" for that sequence.
533 * - Use that sequence as input for all sequences that want (unfiltered) MD
534 * input.
535 *
536 * Note that this synthethic solver is designed to be used on sequential
537 * mode only and with only one MD instance.
538 *
539 * TODO
540 * - major bug when there is ONLY a FFF in a sequence
541 * - reduce capture: most variables in lambda can be defined beforehand as
542 * they are const (e.g. everything coming from cfg)
543 * - totalNumberMDSimulations > 1 is theoretically possible with this
544 * redesign. test it and remove the restriction for it to be 1
545 */
546
547 else if (_cfg.miSolverType == SYNTHETIC) {
548 // Synthetic MD runs sequentially only, as described above.
549 if (_cfg.macro2Md || _cfg.totalNumberMDSimulations > 1 ||
550 _cfg.lbNumberProcesses[0] != 1 || _cfg.lbNumberProcesses[1] != 1 ||
551 _cfg.lbNumberProcesses[2] != 1) {
552 throw std::runtime_error(
553 "ERROR: Syntethic MD is only available in sequential mode!");
554 }
555
556 // set couette solver interface in MamicoInterfaceProvider
557 coupling::interface::MamicoInterfaceProvider<MY_LINKEDCELL,
558 3>::getInstance()
559 .setMacroscopicSolverInterface(couetteSolverInterface);
560
561 // Create new FilterFromFunction instance and insert it into Filtering
562 // System.
563 try {
564 for (unsigned int i = 0; i < _localMDInstances; i++)
565 _multiMDCellService->getMacroscopicCellService(i)
566 .getFilterPipeline()
567 ->getSequence(SYNTHETICMD_SEQUENCE)
568 ->addFilter(
569 new std::function<std::vector<double>(
570 std::vector<double>,
571 std::vector<std::array<unsigned int, 3>>)>{
572 // applyScalar
573 [this](std::vector<double>
574 inputScalars, // not actually used as input:
575 // matching MCS's
576 // addFilterToSequence(...)
577 // signature
578 std::vector<std::array<unsigned int, 3>>
579 cellIndices // not used either
580 ) {
581 if (_rank == 0) {
582 gettimeofday(&_tv.start, NULL);
583 }
584
585 // std::cout << "Entering synthetic MD scalar..." <<
586 // std::endl;
587 // TODO: replace usage of deprecated IndexConversion
588 const coupling::IndexConversion<3> &indexConversion =
589 _multiMDCellService->getMacroscopicCellService(0)
590 .getIndexConversion();
591 const unsigned int size = inputScalars.size();
592 const tarch::la::Vector<3, double> macroscopicCellSize(
593 indexConversion.getMacroscopicCellSize());
594 const double mass =
595 (_cfg.density) * macroscopicCellSize[0] *
596 macroscopicCellSize[1] * macroscopicCellSize[2];
597
598 std::vector<double> syntheticMasses;
599 for (unsigned int i = 0; i < size; i++) {
600 syntheticMasses.push_back(mass);
601 }
602 // std::cout << "Generated masses!" << std::endl;
603
604 if (_rank == 0) {
605 gettimeofday(&_tv.end, NULL);
606 _tv.micro +=
607 (_tv.end.tv_sec - _tv.start.tv_sec) * 1000000 +
608 (_tv.end.tv_usec - _tv.start.tv_usec);
609 }
610
611 return syntheticMasses;
612 }},
613 new std::function<std::vector<std::array<double, 3>>(
614 std::vector<std::array<double, 3>>,
615 std::vector<std::array<unsigned int, 3>>)>{
616 // applyVector
617 [this](std::vector<std::array<double, 3>>
618 inputVectors, // same as above
619 std::vector<std::array<unsigned int, 3>>
620 cellIndices // same as above
621 ) {
622 if (_rank == 0) {
623 gettimeofday(&_tv.start, NULL);
624 }
625
626 // std::cout << "Entering synthetic MD vector." <<
627 // std::endl;
628
629 const coupling::IndexConversion<3> &indexConversion =
630 _multiMDCellService->getMacroscopicCellService(0)
631 .getIndexConversion();
632 const unsigned int size = inputVectors.size();
633 const tarch::la::Vector<3, double> macroscopicCellSize(
634 indexConversion.getMacroscopicCellSize());
635 const double mass =
636 (_cfg.density) * macroscopicCellSize[0] *
637 macroscopicCellSize[1] * macroscopicCellSize[2];
638
639 using coupling::indexing::IndexTrait;
640 using coupling::indexing::CellIndex;
641
642 const tarch::la::Vector<3, double>
643 md2MacroDomainOffset = {
644 CellIndex<3, IndexTrait::local,
645 IndexTrait::md2macro,
646 IndexTrait::noGhost>::lowerBoundary
647 .get()[0] *
648 macroscopicCellSize[0],
649 CellIndex<3, IndexTrait::local,
650 IndexTrait::md2macro,
651 IndexTrait::noGhost>::lowerBoundary
652 .get()[1] *
653 macroscopicCellSize[1],
654 CellIndex<3, IndexTrait::local,
655 IndexTrait::md2macro,
656 IndexTrait::noGhost>::lowerBoundary
657 .get()[2] *
658 macroscopicCellSize[2],
659 };
660
661 std::normal_distribution<double> distribution(
662 0.0, _cfg.noiseSigma);
663 std::vector<std::array<double, 3>> syntheticMomenta;
664 for (unsigned int i = 0; i < size; i++) {
665 // determine cell midpoint
666 CellIndex<3, IndexTrait::vector> globalIndex(
667 CellIndex<3, IndexTrait::local,
668 IndexTrait::md2macro,
669 IndexTrait::noGhost>{
670 i}); // construct global CellIndex from buffer
671 // and convert it to vector

673 md2MacroDomainOffset - 0.5 * macroscopicCellSize);

1256 void write2CSV(

1258 const unsigned int *const recvIndices,
1259 const coupling::IndexConversion<3> &indexConversion,
1260 int couplingCycle) const {

1267 ss << "CouetteAvgMultiMDCells_" << _rank << "_" << couplingCycle << ".csv";
1268 std::ofstream file(ss.str().c_str());
1269 if (!file.is_open()) {

1282 }
1283 const tarch::la::Vector<3, unsigned int> counter(
1284 indexConversion.getGlobalVectorCellIndex(recvIndices[i]));

1289 }
