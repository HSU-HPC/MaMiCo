361 /*
362 * A synthethic solver is modeled using a dynamically linked filter,
363 * i.e. a lambda function producing artifical data in every filter step.
364 *
365 * This is how to properly instanciate and use a synthethic solver:
366 * - Create a sequence named like the macro SYNTHETICMD_SEQUENCE in xml. Use whatever input, but make sure the filter system's output is set to this sequence.
367 * - Set filtered-values = "macro-mass macro-momentum" for that sequence.
368 * - Use that sequence as input for all sequences that want (unfiltered) MD input.
369 *
370 * Note that this synthethic solver is designed to be used on sequential mode only and with only one MD instance.
371 *
372 * TODO
373 * - major bug when there is ONLY a FFF in a sequence
374 * - reduce capture: most variables in lambda can be defined beforehand as they are const (e.g. everything coming from cfg)
375 * - totalNumberMDSimulations > 1 is theoretically possible with this redesign. test it and remove the restriction for it to be 1
376 */
377
378 else if(_cfg.miSolverType == SYNTHETIC) {
379
380 /*
381 * Synthetic MD runs sequentially only, as described above.
382 */
383 if(/*_cfg.md2Macro ||*/ _cfg.macro2Md || _cfg.totalNumberMDSimulations > 1 ||
384 _cfg.lbNumberProcesses[0] != 1 || _cfg.lbNumberProcesses[1] != 1 || _cfg.lbNumberProcesses[2] != 1) {
385 throw std::runtime_error("ERROR: Syntethic MD is only available in sequential mode!");
386 }
387
388 /*
389 * Create new FilterFromFunction instance and insert it into Filtering System.
390 */
391 try {
392 for(unsigned int i = 0; i < _localMDInstances; i++)
393 _multiMDCellService->getMacroscopicCellService(i).getFilterPipeline().getSequence(SYNTHETICMD_SEQUENCE)->addFilter(
394 new std::function<std::vector<double> (std::vector<double>, std::vector<std::array<unsigned int, 3>>)>{ //applyScalar
395 [this] (
396 std::vector<double> inputScalars, //doesnt get used: matching MCS's addFilterToSequence(...) signature
397 std::vector<std::array<unsigned int, 3>> cellIndices //only gets used to determine "size" (see below)
398 ) {
399 if (_rank==0){ gettimeofday(&_tv.start,NULL); }
400
401 //sanity check
402 if(inputScalars.size() != cellIndices.size())
403 throw std::runtime_error("ERROR: Cell data and indexing of non-matching shapes!");
404
405 //std::cout << "Entering synthetic MD scalar..." << std::endl;
406 const coupling::IndexConversion<3>& indexConversion = _multiMDCellService->getMacroscopicCellService(0).getIndexConversion();
407 const unsigned int size = cellIndices.size();
408 const tarch::la::Vector<3,double> macroscopicCellSize(indexConversion.getMacroscopicCellSize());
409 const double mass = (_cfg.density)*macroscopicCellSize[0]*macroscopicCellSize[1]*macroscopicCellSize[2];
410
411 std::vector<double> syntheticMasses;
412 for (unsigned int i = 0; i < size; i++){
413 syntheticMasses.push_back(mass);
414 }
415 //std::cout << "Generated masses!" << std::endl;
416
417 if (_rank==0){
418 gettimeofday(&_tv.end,NULL);
419 _tv.micro += (_tv.end.tv_sec - _tv.start.tv_sec)*1000000 + (_tv.end.tv_usec - _tv.start.tv_usec);
420 }
421
422 return syntheticMasses;
423
424 }},
425 new std::function<std::vector<std::array<double, 3>> (std::vector<std::array<double,3>>, std::vector<std::array<unsigned int, 3>>)> { //applyVector
426 [this] (
427 std::vector<std::array<double, 3>> inputVectors, //same for these 2
428 std::vector<std::array<unsigned int, 3>> cellIndices
429 ) {
430 if (_rank==0){ gettimeofday(&_tv.start,NULL); }
431
432 //sanity check
433 if(inputVectors.size() != cellIndices.size())
434 throw std::runtime_error("ERROR: Cell data and indexing of non-matching shapes!");
435
436 //std::cout << "Entering synthetic MD vector." << std::endl;
437
438 //unlike for the scalar case, we need the MD2Macro version of IC to calculate correct offsets
439 const coupling::IndexConversionMD2Macro<3>* indexConversionMD2Macro =
440 _multiMDCellService->getMacroscopicCellService(0).getFilterPipeline().getICM2M();
441
442 const unsigned int size = cellIndices.size();
443 const tarch::la::Vector<3,double> md2MacroDomainOffset = indexConversionMD2Macro->getGlobalMD2MacroDomainOffset();
444
445 const tarch::la::Vector<3,double> macroscopicCellSize(indexConversionMD2Macro->getBaseIC()->getMacroscopicCellSize());
446 const double mass = (_cfg.density)*macroscopicCellSize[0]*macroscopicCellSize[1]*macroscopicCellSize[2];
447
448 std::normal_distribution<double> distribution (0.0,_cfg.noiseSigma);
449 std::vector<std::array<double, 3>> syntheticMomenta;
450 for (unsigned int i = 0; i < size; i++){
451 // determine cell midpoint
452 const tarch::la::Vector<3,unsigned int> globalIndex(indexConversionMD2Macro->getBaseIC()->getGlobalVectorCellIndex(_buf.globalCellIndices4RecvBuffer[i]));
453 tarch::la::Vector<3,double> cellMidPoint(md2MacroDomainOffset-0.5*macroscopicCellSize);
454 for (unsigned int d = 0; d < 3; d++){ cellMidPoint[d] = cellMidPoint[d] + ((double)globalIndex[d])*macroscopicCellSize[d]; }
455
456 //compute momentum
457 const tarch::la::Vector<3,double> noise(distribution(_generator),distribution(_generator),distribution(_generator));
458 const tarch::la::Vector<3,double> momentum(mass*((*_couetteSolver).getVelocity(cellMidPoint)+noise));
459
460 //conversion from tarch::la::Vector to std::array
461 syntheticMomenta.push_back({momentum[0], momentum[1], momentum[2]});
462 }
463 //std::cout << "Generated momenta!" << std::endl;
464
465 if (_rank==0){
466 gettimeofday(&_tv.end,NULL);
467 _tv.micro += (_tv.end.tv_sec - _tv.start.tv_sec)*1000000 + (_tv.end.tv_usec - _tv.start.tv_usec);
468 }
469
470 return syntheticMomenta;
471 }},
472 0 //filterIndex
473 );
474 }
475 catch(std::runtime_error& e) {
476 auto expectedError = std::string("ERROR: Could not find Filter Sequence named ").append(SYNTHETICMD_SEQUENCE);
477 if(expectedError.compare(e.what()) == 0) {
478 std::cout << "ERROR: Synthetic MD solver selected without providing filter sequence '" << SYNTHETICMD_SEQUENCE <<"' in config." << std::endl;
479 exit(EXIT_FAILURE);
480 }
481 else throw e;
482 }
483 }
484
485 // allocate buffers for send/recv operations
486 allocateSendBuffer(_multiMDCellService->getMacroscopicCellService(0).getIndexConversion(),*couetteSolverInterface);
487 allocateRecvBuffer(_multiMDCellService->getMacroscopicCellService(0).getIndexConversion(),*couetteSolverInterface);
488
489 if(_cfg.initAdvanceCycles > 0 && _couetteSolver != NULL)
490 _couetteSolver->advance(_cfg.initAdvanceCycles * _simpleMDConfig.getSimulationConfiguration().getDt()
491 * _simpleMDConfig.getSimulationConfiguration().getNumberOfTimesteps());
492
493 // finish time measurement for initialisation
494 if(_rank == 0){
495 gettimeofday(&_tv.end,NULL);
496 double runtime = (_tv.end.tv_sec - _tv.start.tv_sec)*1000000 + (_tv.end.tv_usec - _tv.start.tv_usec);
497 std::cout << "Initialization: " << (int)(runtime/1000) << "ms" << std::endl;
498 }
499
500 if(_cfg.computeSNR){
501 std::cout << "Output for every coupling cycle, for the cell 87 in recvBuffer:" << std::endl;
502 std::cout << "cycle number (after filter-init-cycles), vel_x macroscopic solver, vel_x filter output" << std::endl;
503 _sum_signal = 0;
504 _sum_noise = 0;
505 }
506
507 if (_rank == 0){ gettimeofday(&_tv.start_total,NULL); }
508 std::cout << "Finish CouetteTest::initSolvers() " << std::endl;
