30 std::vector<coupling::interface::MDSolverInterface<LinkedCell, dim> *>

33 coupling::interface::MacroscopicSolverInterface<dim>
34 *macroscopicSolverInterface,
35 tarch::la::Vector<dim, unsigned int> numberProcesses, unsigned int rank,
36 unsigned int totalNumberMDSimulations,
37 const coupling::configurations::ParticleInsertionConfiguration
38 &particleInsertionConfiguration,
39 const coupling::configurations::MomentumInsertionConfiguration
40 &momentumInsertionConfiguration,
41 const coupling::configurations::BoundaryForceConfiguration<dim>
42 &boundaryForceConfiguration,
43 const coupling::configurations::TransferStrategyConfiguration<dim>
44 &transferStrategyConfiguration,
45 const coupling::configurations::ParallelTopologyConfiguration
46 &parallelTopologyConfiguration,
47 const coupling::configurations::ThermostatConfiguration
48 &thermostatConfiguration,
49 unsigned int numberMDTimestepsPerCouplingCycle,
50 const coupling::configurations::MacroscopicCellConfiguration<dim>
51 &macroscopicCellConfiguration,
52 const char
53 *filterPipelineConfiguration, // location of the .xml file containing
54 // the static filter pipeline config
55 const tarch::utils::MultiMDService<dim> &multiMDService, int tws = 0)
56 : _localNumberMDSimulations((unsigned int)mdSolverInterfaces.size()),
57 _totalNumberMDSimulations(totalNumberMDSimulations),
58 _macroscopicCellServices(NULL),
59 _topologyOffset(computeTopologyOffset(numberProcesses, rank)),
60 _tws(tws),
61 _intNumberProcesses(computeScalarNumberProcesses(numberProcesses)),
62 _macroscopicSolverInterface(macroscopicSolverInterface),
63 _filterPipelineConfiguration(filterPipelineConfiguration),
64 _multiMDService(multiMDService), _rank(rank),
65 _postMultiInstanceFilterPipeline(nullptr) {
66
67 // If we allow for zero MD instances on ranks, this initializion process
68 // would segfault...
69 if (mdSolverInterfaces.size() == 0) {
70 std::cout << "ERROR: Zero MD instances on rank " << rank
71 << ". Maybe you forgot to adjust number-md-simulations in "
72 "couette.xml?"
73 << std::endl;
74 exit(EXIT_FAILURE);
75 }
76
77 const tarch::la::Vector<dim, double> mdDomainSize(
78 mdSolverInterfaces[0]->getGlobalMDDomainSize());
79 const tarch::la::Vector<dim, double> mdDomainOffset(
80 mdSolverInterfaces[0]->getGlobalMDDomainOffset());
81
82 // determine globally unique IDs for each macroscopic cell service.
83 // Assumptions:
84 // - the global parallel topology is subdivided into equally sized blocks of
85 // ranks
86 // - on each block of ranks, the SAME number of MD simulations is executed.
87 // This yields that the variable _topologyOffset together with the local ID
88 // of an MD simulation yields a unique global ID
89 // - an MD simulation executes on one full block of ranks
90 _macroscopicCellServices =
91 new coupling::services::MacroscopicCellService<dim>
92 *[_totalNumberMDSimulations];
93 if (_macroscopicCellServices == NULL) {
94 std::cout << "ERROR "
95 "coupling::services::MultiMDCellService::MultiMDCellService("
96 "...): _macroscopicCellServices==NULL!"
97 << std::endl;
98 exit(EXIT_FAILURE);
99 }
100
101 // allocate all macroscopic cell services for macro-only-solver BEFORE
102 // topology offset
103 // -> we have localNumberMDSimulations that run per block of ranks on
104 // intNumberProcesses
105 // -> this yields
106 // localNumberMDSimulations*topologyOffset/intNumberProcesses MD
107 // simulations before the actual block of ranks
108 for (unsigned int i = 0;
109 i < _localNumberMDSimulations * _topologyOffset / _intNumberProcesses;
110 i++) {
111 _macroscopicCellServices[i] =
112 new coupling::services::MacroscopicCellServiceMacroOnly<dim>(
113 i, macroscopicSolverInterface, numberProcesses, rank,
114 mdDomainSize, mdDomainOffset, parallelTopologyConfiguration,
115 macroscopicCellConfiguration,
116 (i / _localNumberMDSimulations) * _intNumberProcesses);
117 if (_macroscopicCellServices[i] == NULL) {
118 std::cout << "ERROR "
119 "coupling::services::MultiMDCellService::"
120 "MultiMDCellService(...): _macroscopicCellServices["
121 << i << "]==NULL!" << std::endl;
122 exit(EXIT_FAILURE);
123 }
124 }
125
126 // allocate all macroscopic cell services for macro- and micro-interactions
127 for (unsigned int i =
128 _localNumberMDSimulations * _topologyOffset / _intNumberProcesses;
129 i < _localNumberMDSimulations *
130 (_topologyOffset + _intNumberProcesses) / _intNumberProcesses;
131 i++) {
132 _macroscopicCellServices[i] =
133 new coupling::services::MacroscopicCellServiceImpl<LinkedCell, dim>(
134 i,
135 mdSolverInterfaces[i - _localNumberMDSimulations *
136 _topologyOffset / _intNumberProcesses],
137 macroscopicSolverInterface, numberProcesses, rank,
138 particleInsertionConfiguration, momentumInsertionConfiguration,
139 boundaryForceConfiguration, transferStrategyConfiguration,
140 parallelTopologyConfiguration, thermostatConfiguration,
141 numberMDTimestepsPerCouplingCycle, macroscopicCellConfiguration,
142 filterPipelineConfiguration, multiMDService, _topologyOffset,
143 _tws);
144 if (_macroscopicCellServices[i] == NULL) {
145 std::cout << "ERROR "
146 "coupling::services::MultiMDCellService::"
147 "MultiMDCellService(...): _macroscopicCellServices["
148 << i << "]==NULL!" << std::endl;
149 exit(EXIT_FAILURE);
150 }
151 }
152
153 // allocate all macroscopic cell services for macro-only-solver AFTER
154 // topology offset
155 for (unsigned int i = _localNumberMDSimulations *
156 (_topologyOffset + _intNumberProcesses) /
157 _intNumberProcesses;
158 i < _totalNumberMDSimulations; i++) {
159 _macroscopicCellServices[i] =
160 new coupling::services::MacroscopicCellServiceMacroOnly<dim>(
161 i, macroscopicSolverInterface, numberProcesses, rank,
162 mdDomainSize, mdDomainOffset, parallelTopologyConfiguration,
163 macroscopicCellConfiguration,
164 (i / _localNumberMDSimulations) * _intNumberProcesses);
165 if (_macroscopicCellServices[i] == NULL) {
166 std::cout << "ERROR "
167 "coupling::services::MultiMDCellService::"
168 "MultiMDCellService(...): _macroscopicCellServices["
169 << i << "]==NULL!" << std::endl;
170 exit(EXIT_FAILURE);
171 }
172 }
173 }

237 double sendFromMD2Macro(
238 const std::vector<coupling::datastructures::MacroscopicCell<dim> *>
239 &macroscopicCellsFromMacroscopicSolver,
240 const unsigned int *const globalCellIndicesFromMacroscopicSolver) {
241 double res = 0;
242 auto size = macroscopicCellsFromMacroscopicSolver
243 .size(); // we assume globalCellIndicesFromMacroscopicSolver
244 // to be of identical size
245
246 /*
247 * If this is first coupling step, we must allocate space for the
248 * macroscopic cells we filter and determine averages with.
249 */
250 if (_macroscopicCells.empty()) {
251 // Allocate & init _macroscopicCells
252 for (unsigned int c = _macroscopicCells.size(); c < size; c++)
253 _macroscopicCells.push_back(
254 new coupling::datastructures::MacroscopicCell<dim>());
255 }
256
257 /*
258 * If this is the first coupling step, copy cells to be in format compatible
259 * with filtering system. Can be optimized if indices don't change
260 * dynamically.
261 */
262 if (_cellIndices.empty()) {
263 for (unsigned int i = 0; i < size; i++)
264 _cellIndices.push_back(globalCellIndicesFromMacroscopicSolver[i]);
265 }
266
267/*
268 * If this is the first coupling step, we must init the post multi instance
269 * filter pipeline operating on averaged cell data. The
270 * ENABLE_POST_MULTI_INSTANCE_FILTERING flag is used for debugging purposes and
271 * shall be removed later. If you wish to not use post multi-instance filtering
272 * in deployment, you can simply leave the corresponding XML-Subtag empty.
273 */
274#ifdef ENABLE_POST_MULTI_INSTANCE_FILTERING
275 if (_postMultiInstanceFilterPipeline == nullptr) {
276 // Init filter pipeline
277 _postMultiInstanceFilterPipeline =
278 new coupling::filtering::FilterPipeline<dim>(
279 _macroscopicCells, coupling::filtering::Scope::postMultiInstance,
280 _multiMDService, _filterPipelineConfiguration.c_str());
281 }
282#endif
283
284 // reset macroscopic data (only those should be used by macroscopic solver
285 // anyway) in cells
286 for (unsigned int i = 0; i < size; i++) {
287 _macroscopicCells[i]->setMacroscopicMass(0.0);
288 _macroscopicCells[i]->setMacroscopicMomentum(
289 tarch::la::Vector<dim, double>(0.0));
290 }
291
292 // receive data from each MD simulation and accumulate information in cells
293 for (unsigned int l = 0; l < _totalNumberMDSimulations; l++) {
294 // std::cout << "Rank " <<
295 // _macroscopicCellServices[l]->getIndexConversion().getThisRank() << ":
296 // Send from MD to Macro for Simulation no. " << l << std::endl;
297 res += _macroscopicCellServices[l]->sendFromMD2Macro(
298 macroscopicCellsFromMacroscopicSolver,
299 globalCellIndicesFromMacroscopicSolver);
300 for (unsigned int i = 0; i < size; i++) {
301 _macroscopicCells[i]->addMacroscopicMass(
302 macroscopicCellsFromMacroscopicSolver[i]->getMacroscopicMass());
303 _macroscopicCells[i]->addMacroscopicMomentum(
304 macroscopicCellsFromMacroscopicSolver[i]->getMacroscopicMomentum());
305 }
306 }
307
308 // average data
309 for (unsigned int i = 0; i < size; i++) {
310 _macroscopicCells[i]->setMacroscopicMass(
311 _macroscopicCells[i]->getMacroscopicMass() /
312 _totalNumberMDSimulations);
313 _macroscopicCells[i]->setMacroscopicMomentum(
314 1.0 / _totalNumberMDSimulations *
315 _macroscopicCells[i]->getMacroscopicMomentum());
316 }
317
318// apply post multi instance FilterPipeline on cell data
319#ifdef ENABLE_POST_MULTI_INSTANCE_FILTERING
320
321#ifdef DEBUG_FILTER_PIPELINE
322 std::cout << "FP: Now applying post-multi-instance filter pipeline"
323 << std::endl;
324#endif
325
326 (*_postMultiInstanceFilterPipeline)();
327#endif
328
329 // store data in macroscopicCellsFromMacroscopicSolver
330 for (unsigned int i = 0; i < size; i++) {
331 macroscopicCellsFromMacroscopicSolver[i]->setMacroscopicMass(
332 _macroscopicCells[i]->getMacroscopicMass());
333 macroscopicCellsFromMacroscopicSolver[i]->setMacroscopicMomentum(
334 _macroscopicCells[i]->getMacroscopicMomentum());
335 }
336
337 return res;
338 }

340 // Must be called after construction of MultiMDCellService, but before the
341 // first coupling step.
342 void constructFilterPipelines() {
343 for (unsigned int md = 0; md < _totalNumberMDSimulations; md++) {
344 // get macroscopic cell service of instance
345 auto &mcs = _macroscopicCellServices[md];
346
347 // only Impl instances of MCS contain filtering
348 if (dynamic_cast<coupling::services::MacroscopicCellServiceImpl<
349 LinkedCell, dim> *>(mcs) == nullptr)
350 continue;
351
352 if (mcs->getFilterPipeline() == nullptr) {
353 mcs->initFiltering();
354 }
355 }

359 unsigned int
360 computeTopologyOffset(tarch::la::Vector<dim, unsigned int> numberProcesses,
361 unsigned int rank) const {
362 // determine topology offset of this rank
363 const unsigned int intNumberProcesses =
364 computeScalarNumberProcesses(numberProcesses);
365 const unsigned int topologyOffset =
366 (rank / intNumberProcesses) * intNumberProcesses;
367 return topologyOffset;
368 }
369
370 unsigned int computeScalarNumberProcesses(
371 tarch::la::Vector<dim, unsigned int> numberProcesses) const {
372 unsigned int np = numberProcesses[0];
373 for (unsigned int d = 1; d < dim; d++) {
374 np = np * numberProcesses[d];
375 }
376 return np;
377 }
