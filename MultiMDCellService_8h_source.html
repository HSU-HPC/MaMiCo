26 public:
27 MultiMDCellService(
28 std::vector<coupling::interface::MDSolverInterface<LinkedCell,dim>* > mdSolverInterfaces, // MD solver interfaces for each MD simulation (which uses this rank)
29 coupling::interface::MacroscopicSolverInterface<dim> *macroscopicSolverInterface,
30 tarch::la::Vector<dim,unsigned int> numberProcesses,
31 unsigned int rank,
32 unsigned int totalNumberMDSimulations,
33 const coupling::configurations::ParticleInsertionConfiguration &particleInsertionConfiguration,
34 const coupling::configurations::MomentumInsertionConfiguration &momentumInsertionConfiguration,
35 const coupling::configurations::BoundaryForceConfiguration<dim> &boundaryForceConfiguration,
36 const coupling::configurations::TransferStrategyConfiguration<dim>& transferStrategyConfiguration,
37 const coupling::configurations::ParallelTopologyConfiguration& parallelTopologyConfiguration,
38 const coupling::configurations::ThermostatConfiguration& thermostatConfiguration,
39 unsigned int numberMDTimestepsPerCouplingCycle,
40 const coupling::configurations::MacroscopicCellConfiguration<dim> &macroscopicCellConfiguration,
41 const char* filterPipelineConfiguration, //location of the .xml file containing the static filter pipeline config
42 const tarch::utils::MultiMDService<dim>& multiMDService, int tws = 0
43 ):
44 _localNumberMDSimulations((unsigned int)mdSolverInterfaces.size()), _totalNumberMDSimulations(totalNumberMDSimulations),
45 _macroscopicCellServices(NULL), _topologyOffset(computeTopologyOffset(numberProcesses,rank)), _tws(tws), _intNumberProcesses(computeScalarNumberProcesses(numberProcesses)),
46 _macroscopicSolverInterface(macroscopicSolverInterface),
47 _filterPipelineConfiguration(filterPipelineConfiguration),
48 _multiMDService(multiMDService),
49 _rank(rank),
50 _postMultiInstanceFilterPipeline(nullptr) {
51
52 //If we allow for zero MD instances on ranks, this initializion process would segfault...
53 if(mdSolverInterfaces.size() == 0) {
54 std::cout << "ERROR: Zero MD instances on rank " << rank << ". Maybe you forgot to adjust number-md-simulations in couette.xml?" << std::endl;
55 exit(EXIT_FAILURE);
56 }
57
58 const tarch::la::Vector<dim,double> mdDomainSize(mdSolverInterfaces[0]->getGlobalMDDomainSize());
59 const tarch::la::Vector<dim,double> mdDomainOffset(mdSolverInterfaces[0]->getGlobalMDDomainOffset());
60
61 // determine globally unique IDs for each macroscopic cell service. Assumptions:
62 // - the global parallel topology is subdivided into equally sized blocks of ranks
63 // - on each block of ranks, the SAME number of MD simulations is executed. This yields that the variable _topologyOffset together with the local ID of an MD simulation yields a unique global ID
64 // - an MD simulation executes on one full block of ranks
65 _macroscopicCellServices = new coupling::services::MacroscopicCellService<dim>* [_totalNumberMDSimulations];
66 if (_macroscopicCellServices==NULL){std::cout << "ERROR coupling::services::MultiMDCellService::MultiMDCellService(...): _macroscopicCellServices==NULL!" << std::endl; exit(EXIT_FAILURE);}
67
68
69 // allocate all macroscopic cell services for macro-only-solver BEFORE topology offset
70 // -> we have localNumberMDSimulations that run per block of ranks on intNumberProcesses
71 // -> this yields localNumberMDSimulations*topologyOffset/intNumberProcesses MD simulations before the actual block of ranks
72 for (unsigned int i = 0; i < _localNumberMDSimulations*_topologyOffset/_intNumberProcesses; i++){
73 _macroscopicCellServices[i] = new coupling::services::MacroscopicCellServiceMacroOnly<dim>(
74 i, macroscopicSolverInterface, numberProcesses, rank, mdDomainSize, mdDomainOffset,
75 parallelTopologyConfiguration, macroscopicCellConfiguration, (i/_localNumberMDSimulations)*_intNumberProcesses
76 );
77 if (_macroscopicCellServices[i]==NULL){
78 std::cout << "ERROR coupling::services::MultiMDCellService::MultiMDCellService(...): _macroscopicCellServices[" << i << "]==NULL!" << std::endl; exit(EXIT_FAILURE);
79 }
80 }
81
82 // allocate all macroscopic cell services for macro- and micro-interactions
83 for (unsigned int i = _localNumberMDSimulations*_topologyOffset/_intNumberProcesses; i<_localNumberMDSimulations*(_topologyOffset+_intNumberProcesses)/_intNumberProcesses; i++){
84 _macroscopicCellServices[i] = new coupling::services::MacroscopicCellServiceImpl<LinkedCell,dim>(
85 i, mdSolverInterfaces[i-_localNumberMDSimulations*_topologyOffset/_intNumberProcesses], macroscopicSolverInterface, numberProcesses, rank, particleInsertionConfiguration,
86 momentumInsertionConfiguration, boundaryForceConfiguration, transferStrategyConfiguration, parallelTopologyConfiguration, thermostatConfiguration,
87 numberMDTimestepsPerCouplingCycle, macroscopicCellConfiguration, filterPipelineConfiguration, multiMDService, _topologyOffset, _tws
88 );
89 if (_macroscopicCellServices[i]==NULL){
90 std::cout << "ERROR coupling::services::MultiMDCellService::MultiMDCellService(...): _macroscopicCellServices[" << i << "]==NULL!" << std::endl; exit(EXIT_FAILURE);
91 }
92 }
93
94 // allocate all macroscopic cell services for macro-only-solver AFTER topology offset
95 for (unsigned int i = _localNumberMDSimulations*(_topologyOffset+_intNumberProcesses)/_intNumberProcesses; i < _totalNumberMDSimulations; i++){
96 _macroscopicCellServices[i] = new coupling::services::MacroscopicCellServiceMacroOnly<dim>(
97 i, macroscopicSolverInterface, numberProcesses, rank, mdDomainSize, mdDomainOffset,
98 parallelTopologyConfiguration, macroscopicCellConfiguration, (i/_localNumberMDSimulations)*_intNumberProcesses
99 );
100 if (_macroscopicCellServices[i]==NULL){
101 std::cout << "ERROR coupling::services::MultiMDCellService::MultiMDCellService(...): _macroscopicCellServices[" << i << "]==NULL!" << std::endl; exit(EXIT_FAILURE);
102 }
103 }
104 }

138 double sendFromMD2Macro(
139 const std::vector<coupling::datastructures::MacroscopicCell<dim>* > &macroscopicCellsFromMacroscopicSolver,
140 const unsigned int * const globalCellIndicesFromMacroscopicSolver
141 ){
142 double res = 0;
143 auto size = macroscopicCellsFromMacroscopicSolver.size(); // we assume globalCellIndicesFromMacroscopicSolver to be of identical size
144
145 /*
146 * If this is first coupling step, we must allocate space for the macroscopic cells we filter and determine averages with.
147 */
148 if(_macroscopicCells.empty()) {
149 //Allocate & init _macroscopicCells
150 for(unsigned int c = _macroscopicCells.size(); c < size; c++) _macroscopicCells.push_back(new coupling::datastructures::MacroscopicCell<dim>());
151 }
152
153 /*
154 * If this is the first coupling step, copy cells to be in format compatible with filtering system.
155 * Can be optimized if indices don't change dynamically.
156 */
157 if(_cellIndices.empty()) {
158 for(unsigned int i = 0; i < size; i++) _cellIndices.push_back(globalCellIndicesFromMacroscopicSolver[i]);
159 }
160
161 /*
162 * If this is the first coupling step, we must init the post multi instance filter pipeline operating on averaged cell data.
163 * The ENABLE_POST_MULTI_INSTANCE_FILTERING flag is used for debugging purposes and shall be removed later.
164 * If you wish to not use post multi-instance filtering in deployment, you can simply leave the corresponding XML-Subtag empty.
165 */
166 #ifdef ENABLE_POST_MULTI_INSTANCE_FILTERING
167 if(_postMultiInstanceFilterPipeline == nullptr) {
168 //Init filter pipeline
169 _postMultiInstanceFilterPipeline = new coupling::FilterPipeline<dim>(_macroscopicCells, _cellIndices, &(_macroscopicCellServices[0]->getIndexConversion()), _macroscopicSolverInterface, _multiMDService, coupling::Scope::postMultiInstance, _filterPipelineConfiguration.c_str());
170 }
171 #endif
172
173 // reset macroscopic data (only those should be used by macroscopic solver anyway) in cells
174 for (unsigned int i = 0; i < size; i++){
175 _macroscopicCells[i]->setMacroscopicMass(0.0); _macroscopicCells[i]->setMacroscopicMomentum(tarch::la::Vector<dim,double>(0.0));
176 }
177
178 // receive data from each MD simulation and accumulate information in cells
179 for (unsigned int l = 0; l < _totalNumberMDSimulations; l++){
180 //std::cout << "Rank " << _macroscopicCellServices[l]->getIndexConversion().getThisRank() << ": Send from MD to Macro for Simulation no. " << l << std::endl;
181 res += _macroscopicCellServices[l]->sendFromMD2Macro(macroscopicCellsFromMacroscopicSolver,globalCellIndicesFromMacroscopicSolver);
182 for (unsigned int i = 0; i < size; i++){
183 _macroscopicCells[i]->addMacroscopicMass(macroscopicCellsFromMacroscopicSolver[i]->getMacroscopicMass());
184 _macroscopicCells[i]->addMacroscopicMomentum(macroscopicCellsFromMacroscopicSolver[i]->getMacroscopicMomentum());
185 }
186 }
187
188 // average data
189 for (unsigned int i = 0; i < size; i++) {
190 _macroscopicCells[i]->setMacroscopicMass(_macroscopicCells[i]->getMacroscopicMass()/_totalNumberMDSimulations);
191 _macroscopicCells[i]->setMacroscopicMomentum(1.0/_totalNumberMDSimulations * _macroscopicCells[i]->getMacroscopicMomentum());
192 }
193
194 // apply post multi instance FilterPipeline on cell data
195 #ifdef ENABLE_POST_MULTI_INSTANCE_FILTERING
196
197 #ifdef DEBUG_FILTER_PIPELINE
198 std::cout << "FP: Now applying post-multi-instance filter pipeline" << std::endl;
199 #endif
200
201 (*_postMultiInstanceFilterPipeline)();
202 #endif
203
204
205 // store data in macroscopicCellsFromMacroscopicSolver
206 for (unsigned int i = 0; i < size; i++){
207 macroscopicCellsFromMacroscopicSolver[i]->setMacroscopicMass(_macroscopicCells[i]->getMacroscopicMass());
208 macroscopicCellsFromMacroscopicSolver[i]->setMacroscopicMomentum(_macroscopicCells[i]->getMacroscopicMomentum());
209 }
210
211 return res;
212 }

216 unsigned int computeTopologyOffset(tarch::la::Vector<dim,unsigned int> numberProcesses,unsigned int rank) const {
217 // determine topology offset of this rank
218 const unsigned int intNumberProcesses = computeScalarNumberProcesses(numberProcesses);
219 const unsigned int topologyOffset = (rank/intNumberProcesses)*intNumberProcesses;
220 return topologyOffset;
221 }
222
223 unsigned int computeScalarNumberProcesses(tarch::la::Vector<dim,unsigned int> numberProcesses) const {
224 unsigned int np = numberProcesses[0];
225 for (unsigned int d = 1; d < dim; d++){ np = np*numberProcesses[d]; }
226 return np;
227 }
