28 MultiMDCellService(std::vector<coupling::interface::MDSolverInterface<LinkedCell, dim> *> mdSolverInterfaces, // MD solver interfaces for each MD simulation
29 // (which uses this rank)
30 coupling::interface::MacroscopicSolverInterface<dim> *macroscopicSolverInterface, tarch::la::Vector<dim, unsigned int> numberProcesses,
31 unsigned int rank, unsigned int totalNumberMDSimulations,
32 const coupling::configurations::ParticleInsertionConfiguration &particleInsertionConfiguration,
33 const coupling::configurations::MomentumInsertionConfiguration &momentumInsertionConfiguration,
34 const coupling::configurations::BoundaryForceConfiguration<dim> &boundaryForceConfiguration,
35 const coupling::configurations::TransferStrategyConfiguration<dim> &transferStrategyConfiguration,
36 const coupling::configurations::ParallelTopologyConfiguration &parallelTopologyConfiguration,
37 const coupling::configurations::ThermostatConfiguration &thermostatConfiguration, unsigned int numberMDTimestepsPerCouplingCycle,
38 const coupling::configurations::MacroscopicCellConfiguration<dim> &macroscopicCellConfiguration,
39 const char *filterPipelineConfiguration, // location of the .xml file containing
40 // the static filter pipeline config
41 const tarch::utils::MultiMDService<dim> &multiMDService, int tws = 0)
42 : _localNumberMDSimulations((unsigned int)mdSolverInterfaces.size()), _totalNumberMDSimulations(totalNumberMDSimulations), _macroscopicCellServices(NULL),
43 _topologyOffset(computeTopologyOffset(numberProcesses, rank)), _tws(tws), _intNumberProcesses(computeScalarNumberProcesses(numberProcesses)),
44 _macroscopicSolverInterface(macroscopicSolverInterface), _filterPipelineConfiguration(filterPipelineConfiguration), _multiMDService(multiMDService),
45 _rank(rank), _postMultiInstanceFilterPipeline(nullptr) {
46
47 // If we allow for zero MD instances on ranks, this initializion process
48 // would segfault...
49 if (mdSolverInterfaces.size() == 0) {
50 std::cout << "ERROR: Zero MD instances on rank " << rank
51 << ". Maybe you forgot to adjust number-md-simulations in "
52 "couette.xml?"
53 << std::endl;
54 exit(EXIT_FAILURE);
55 }
56
57 const tarch::la::Vector<dim, double> mdDomainSize(mdSolverInterfaces[0]->getGlobalMDDomainSize());
58 const tarch::la::Vector<dim, double> mdDomainOffset(mdSolverInterfaces[0]->getGlobalMDDomainOffset());
59
60 // determine globally unique IDs for each macroscopic cell service.
61 // Assumptions:
62 // - the global parallel topology is subdivided into equally sized blocks of
63 // ranks
64 // - on each block of ranks, the SAME number of MD simulations is executed.
65 // This yields that the variable _topologyOffset together with the local ID
66 // of an MD simulation yields a unique global ID
67 // - an MD simulation executes on one full block of ranks
68 _macroscopicCellServices = new coupling::services::MacroscopicCellService<dim> *[_totalNumberMDSimulations];
69 if (_macroscopicCellServices == NULL) {
70 std::cout << "ERROR "
71 "coupling::services::MultiMDCellService::MultiMDCellService("
72 "...): _macroscopicCellServices==NULL!"
73 << std::endl;
74 exit(EXIT_FAILURE);
75 }
76
77 // allocate all macroscopic cell services for macro-only-solver BEFORE
78 // topology offset
79 // -> we have localNumberMDSimulations that run per block of ranks on
80 // intNumberProcesses
81 // -> this yields
82 // localNumberMDSimulations*topologyOffset/intNumberProcesses MD
83 // simulations before the actual block of ranks
84 for (unsigned int i = 0; i < _localNumberMDSimulations * _topologyOffset / _intNumberProcesses; i++) {
85 _macroscopicCellServices[i] = new coupling::services::MacroscopicCellServiceMacroOnly<dim>(
86 i, macroscopicSolverInterface, numberProcesses, rank, mdDomainSize, mdDomainOffset, parallelTopologyConfiguration, macroscopicCellConfiguration,
87 (i / _localNumberMDSimulations) * _intNumberProcesses);
88 if (_macroscopicCellServices[i] == NULL) {
89 std::cout << "ERROR "
90 "coupling::services::MultiMDCellService::"
91 "MultiMDCellService(...): _macroscopicCellServices["
92 << i << "]==NULL!" << std::endl;
93 exit(EXIT_FAILURE);
94 }
95 }
96
97 // allocate all macroscopic cell services for macro- and micro-interactions
98 for (unsigned int i = _localNumberMDSimulations * _topologyOffset / _intNumberProcesses;
99 i < _localNumberMDSimulations * (_topologyOffset + _intNumberProcesses) / _intNumberProcesses; i++) {
100 _macroscopicCellServices[i] = new coupling::services::MacroscopicCellServiceImpl<LinkedCell, dim>(
101 i, mdSolverInterfaces[i - _localNumberMDSimulations * _topologyOffset / _intNumberProcesses], macroscopicSolverInterface, numberProcesses, rank,
102 particleInsertionConfiguration, momentumInsertionConfiguration, boundaryForceConfiguration, transferStrategyConfiguration,
103 parallelTopologyConfiguration, thermostatConfiguration, numberMDTimestepsPerCouplingCycle, macroscopicCellConfiguration, filterPipelineConfiguration,
104 multiMDService, _topologyOffset, _tws);
105 if (_macroscopicCellServices[i] == NULL) {
106 std::cout << "ERROR "
107 "coupling::services::MultiMDCellService::"
108 "MultiMDCellService(...): _macroscopicCellServices["
109 << i << "]==NULL!" << std::endl;
110 exit(EXIT_FAILURE);
111 }
112 }
113
114 // allocate all macroscopic cell services for macro-only-solver AFTER
115 // topology offset
116 for (unsigned int i = _localNumberMDSimulations * (_topologyOffset + _intNumberProcesses) / _intNumberProcesses; i < _totalNumberMDSimulations; i++) {
117 _macroscopicCellServices[i] = new coupling::services::MacroscopicCellServiceMacroOnly<dim>(
118 i, macroscopicSolverInterface, numberProcesses, rank, mdDomainSize, mdDomainOffset, parallelTopologyConfiguration, macroscopicCellConfiguration,
119 (i / _localNumberMDSimulations) * _intNumberProcesses);
120 if (_macroscopicCellServices[i] == NULL) {
121 std::cout << "ERROR "
122 "coupling::services::MultiMDCellService::"
123 "MultiMDCellService(...): _macroscopicCellServices["
124 << i << "]==NULL!" << std::endl;
125 exit(EXIT_FAILURE);
126 }
127 }
128 }

180 double sendFromMD2Macro(const std::vector<coupling::datastructures::MacroscopicCell<dim> *> &macroscopicCellsFromMacroscopicSolver,
181 const unsigned int *const globalCellIndicesFromMacroscopicSolver) {
182 double res = 0;
183 auto size = macroscopicCellsFromMacroscopicSolver.size(); // we assume globalCellIndicesFromMacroscopicSolver
184 // to be of identical size
185
186 /*
187 * If this is first coupling step, we must allocate space for the
188 * macroscopic cells we filter and determine averages with.
189 */
190 if (_macroscopicCells.empty()) {
191 // Allocate & init _macroscopicCells
192 for (unsigned int c = _macroscopicCells.size(); c < size; c++)
193 _macroscopicCells.push_back(new coupling::datastructures::MacroscopicCell<dim>());
194 }
195
196 /*
197 * If this is the first coupling step, copy cells to be in format compatible
198 * with filtering system. Can be optimized if indices don't change
199 * dynamically.
200 */
201 if (_cellIndices.empty()) {
202 for (unsigned int i = 0; i < size; i++)
203 _cellIndices.push_back(globalCellIndicesFromMacroscopicSolver[i]);
204 }
205
206/*
207 * If this is the first coupling step, we must init the post multi instance
208 * filter pipeline operating on averaged cell data. The
209 * ENABLE_POST_MULTI_INSTANCE_FILTERING flag is used for debugging purposes and
210 * shall be removed later. If you wish to not use post multi-instance filtering
211 * in deployment, you can simply leave the corresponding XML-Subtag empty.
212 */
213#ifdef ENABLE_POST_MULTI_INSTANCE_FILTERING
214 if (_postMultiInstanceFilterPipeline == nullptr) {
215 // Init filter pipeline
216 _postMultiInstanceFilterPipeline = new coupling::filtering::FilterPipeline<dim>(_macroscopicCells, coupling::filtering::Scope::postMultiInstance,
217 _multiMDService, _filterPipelineConfiguration.c_str());
218 }
219#endif
220
221 // reset macroscopic data (only those should be used by macroscopic solver
222 // anyway) in cells
223 for (unsigned int i = 0; i < size; i++) {
224 _macroscopicCells[i]->setMacroscopicMass(0.0);
225 _macroscopicCells[i]->setMacroscopicMomentum(tarch::la::Vector<dim, double>(0.0));
226 }
227
228 // receive data from each MD simulation and accumulate information in cells
229 for (unsigned int l = 0; l < _totalNumberMDSimulations; l++) {
230 // std::cout << "Rank " <<
231 // _macroscopicCellServices[l]->getIndexConversion().getThisRank() << ":
232 // Send from MD to Macro for Simulation no. " << l << std::endl;
233 res += _macroscopicCellServices[l]->sendFromMD2Macro(macroscopicCellsFromMacroscopicSolver, globalCellIndicesFromMacroscopicSolver);
234 for (unsigned int i = 0; i < size; i++) {
235 _macroscopicCells[i]->addMacroscopicMass(macroscopicCellsFromMacroscopicSolver[i]->getMacroscopicMass());
236 _macroscopicCells[i]->addMacroscopicMomentum(macroscopicCellsFromMacroscopicSolver[i]->getMacroscopicMomentum());
237 }
238 }
239
240 // average data
241 for (unsigned int i = 0; i < size; i++) {
242 _macroscopicCells[i]->setMacroscopicMass(_macroscopicCells[i]->getMacroscopicMass() / _totalNumberMDSimulations);
243 _macroscopicCells[i]->setMacroscopicMomentum(1.0 / _totalNumberMDSimulations * _macroscopicCells[i]->getMacroscopicMomentum());
244 }
245
246// apply post multi instance FilterPipeline on cell data
247#ifdef ENABLE_POST_MULTI_INSTANCE_FILTERING
248
249#ifdef DEBUG_FILTER_PIPELINE
250 std::cout << "FP: Now applying post-multi-instance filter pipeline" << std::endl;
251#endif
252
253 (*_postMultiInstanceFilterPipeline)();
254#endif
255
256 // store data in macroscopicCellsFromMacroscopicSolver
257 for (unsigned int i = 0; i < size; i++) {
258 macroscopicCellsFromMacroscopicSolver[i]->setMacroscopicMass(_macroscopicCells[i]->getMacroscopicMass());
259 macroscopicCellsFromMacroscopicSolver[i]->setMacroscopicMomentum(_macroscopicCells[i]->getMacroscopicMomentum());
260 }
261
262 return res;
263 }

265 // Must be called after construction of MultiMDCellService, but before the
266 // first coupling step.
267 void constructFilterPipelines() {
268 for (unsigned int md = 0; md < _totalNumberMDSimulations; md++) {
269 // get macroscopic cell service of instance
270 auto &mcs = _macroscopicCellServices[md];
271
272 // only Impl instances of MCS contain filtering
273 if (dynamic_cast<coupling::services::MacroscopicCellServiceImpl<LinkedCell, dim> *>(mcs) == nullptr)
274 continue;
275
276 if (mcs->getFilterPipeline() == nullptr) {
277 mcs->initFiltering();
278 }
279 }

283 unsigned int computeTopologyOffset(tarch::la::Vector<dim, unsigned int> numberProcesses, unsigned int rank) const {
284 // determine topology offset of this rank
285 const unsigned int intNumberProcesses = computeScalarNumberProcesses(numberProcesses);
286 const unsigned int topologyOffset = (rank / intNumberProcesses) * intNumberProcesses;
287 return topologyOffset;
288 }
289
290 unsigned int computeScalarNumberProcesses(tarch::la::Vector<dim, unsigned int> numberProcesses) const {
291 unsigned int np = numberProcesses[0];
292 for (unsigned int d = 1; d < dim; d++) {
293 np = np * numberProcesses[d];
294 }
295 return np;
296 }
