222 double sendFromMD2Macro(const std::vector<coupling::datastructures::MacroscopicCell<dim>*>& macroscopicCellsFromMacroscopicSolver,
223 const unsigned int* const globalCellIndicesFromMacroscopicSolver) {
224 double res = 0;
225 const unsigned int size = (unsigned int)macroscopicCellsFromMacroscopicSolver.size();
226
227 /*
228 * If this is first coupling step, we must allocate space for the
229 * macroscopic cells we filter and determine averages with.
230 */
231 if (_macroscopicCells.empty()) {
232 // Allocate & init _macroscopicCells
233 for (unsigned int c = _macroscopicCells.size(); c < size; c++)
234 _macroscopicCells.push_back(new coupling::datastructures::MacroscopicCell<dim>());
235 }
236
237 /*
238 * If this is the first coupling step, copy cells to be in format compatible
239 * with filtering system. Can be optimized if indices don't change
240 * dynamically.
241 */
242 if (_cellIndices.empty()) {
243 for (unsigned int i = 0; i < size; i++)
244 _cellIndices.push_back(globalCellIndicesFromMacroscopicSolver[i]);
245 }
246
247 /*
248 * If this is the first coupling step, we must init the post multi instance
249 * filter pipeline operating on averaged cell data. If you wish to not use post multi-instance filtering
250 * in deployment, you can simply leave the corresponding XML-Subtag empty.
251 */
252 if (_postMultiInstanceFilterPipeline == nullptr) {
253 // Init filter pipeline
254 _postMultiInstanceFilterPipeline = new coupling::filtering::FilterPipeline<dim>(_macroscopicCells, coupling::filtering::Scope::postMultiInstance,
255 _multiMDService, _filterPipelineConfiguration.c_str());
256 }
257
258 // reset macroscopic data (only those should be used by macroscopic solver
259 // anyway) in duplicate
260 for (unsigned int i = 0; i < size; i++) {
261 _macroscopicCells[i]->setMacroscopicMass(0.0);
262 _macroscopicCells[i]->setMacroscopicMomentum(tarch::la::Vector<dim, double>(0.0));
263 }
264
265 // receive data from each MD simulation and accumulate information in
266 // duplicate
267 unsigned int totalNumberEquilibratedMDSimulations = 0;
268 for (unsigned int l = 0; l < _totalNumberMDSimulations; l++) {
269 // std::cout << "Rank " <<
270 //_macroscopicCellServices[l]->getIndexConversion().getThisRank() << ":
271 // Send from MD to Macro for Simulation no. " << l << std::endl;
272 if (_macroscopicCellServices[l] != nullptr && _warmupPhase[l] == 0) {
273 res += _macroscopicCellServices[l]->sendFromMD2Macro(macroscopicCellsFromMacroscopicSolver, globalCellIndicesFromMacroscopicSolver);
274 for (unsigned int i = 0; i < size; i++) {
275 _macroscopicCells[i]->addMacroscopicMass(macroscopicCellsFromMacroscopicSolver[i]->getMacroscopicMass());
276 _macroscopicCells[i]->addMacroscopicMomentum(macroscopicCellsFromMacroscopicSolver[i]->getMacroscopicMomentum());
277 }
278 totalNumberEquilibratedMDSimulations += 1;
279 }
280 }
281
282 // average data
283 for (unsigned int i = 0; i < size; i++) {
284 _macroscopicCells[i]->setMacroscopicMass(_macroscopicCells[i]->getMacroscopicMass() / (double)totalNumberEquilibratedMDSimulations);
285 _macroscopicCells[i]->setMacroscopicMomentum(1.0 / (double)totalNumberEquilibratedMDSimulations * _macroscopicCells[i]->getMacroscopicMomentum());
286 }
287
288 // apply post multi instance FilterPipeline on cell data
289
290#ifdef DEBUG_FILTER_PIPELINE
291 std::cout << "FP: Now applying post-multi-instance filter pipeline" << std::endl;
292#endif
293
294 (*_postMultiInstanceFilterPipeline)();
295
296 // store data in macroscopicCellsFromMacroscopicSolver
297 for (unsigned int i = 0; i < size; i++) {
298 macroscopicCellsFromMacroscopicSolver[i]->setMacroscopicMass(_macroscopicCells[i]->getMacroscopicMass());
299 macroscopicCellsFromMacroscopicSolver[i]->setMacroscopicMomentum(_macroscopicCells[i]->getMacroscopicMomentum());
300 }
301
302 return res;
303 }

335 void removeSimulationBlock() {
336
337 unsigned int newLocalNumberMDSimulations = _localNumberMDSimulations - 1;
338 unsigned int newTotalNumberMDSimulations = _totalNumberMDSimulations - _multiMDService.getNumberLocalComms();
339 unsigned int newBlockOffset = newLocalNumberMDSimulations * _topologyOffset / _intNumberProcesses;
340
341 auto** newMacroscopicCellServices = new MacroscopicCellService<dim>*[newTotalNumberMDSimulations];
342
343 for (unsigned int i = 0; i < _multiMDService.getNumberLocalComms(); ++i) {
344 for (unsigned int j = 0; j < newLocalNumberMDSimulations; ++j) {
345 unsigned int index = i * _localNumberMDSimulations + j;
346 unsigned int newIndex = i * newLocalNumberMDSimulations + j;
347
348 newMacroscopicCellServices[newIndex] = _macroscopicCellServices[index];
349 if (newIndex < newBlockOffset || newIndex >= newBlockOffset + newLocalNumberMDSimulations) {
350 // Need to update topology offset in indexconversion->paralleltopology
351 newMacroscopicCellServices[newIndex]->updateIndexConversion((newIndex / newLocalNumberMDSimulations) * _intNumberProcesses);
352 }
353 }
354
355 auto pos = _warmupPhase.begin() + newLocalNumberMDSimulations * i - i;
356 _warmupPhase.erase(pos);
357 }
358
359 // Update local variables
360 _localNumberMDSimulations = newLocalNumberMDSimulations;
361 _totalNumberMDSimulations = newTotalNumberMDSimulations;
362 _blockOffset = newBlockOffset;
363
364 delete[] _macroscopicCellServices;
365 _macroscopicCellServices = newMacroscopicCellServices;

379 void addSimulationBlock() {
380
381 unsigned int newLocalNumberMDSimulations = _localNumberMDSimulations + 1;
382 unsigned int newTotalNumberMDSimulations = _totalNumberMDSimulations + _multiMDService.getNumberLocalComms();
383 unsigned int newBlockOffset = newLocalNumberMDSimulations * _topologyOffset / _intNumberProcesses;
384
385 auto** newMacroscopicCellServices = new MacroscopicCellService<dim>*[newTotalNumberMDSimulations];
386 for (unsigned int i = 0; i < _multiMDService.getNumberLocalComms(); ++i) {
387 for (unsigned int j = 0; j < _localNumberMDSimulations; ++j) {
388 unsigned int index = i * _localNumberMDSimulations + j;
389 unsigned int newIndex = i * newLocalNumberMDSimulations + j;
390 newMacroscopicCellServices[newIndex] = _macroscopicCellServices[index];
391 if (newIndex < newBlockOffset || newIndex >= newBlockOffset + newLocalNumberMDSimulations) {
392 // Need to update topologoy offset in
393 // indexconversion->paralleltopology
394 if (newMacroscopicCellServices[newIndex] != nullptr) {
395 newMacroscopicCellServices[newIndex]->updateIndexConversion((newIndex / newLocalNumberMDSimulations) * _intNumberProcesses);
396 }
397 }
398 }
399 newMacroscopicCellServices[(i + 1) * newLocalNumberMDSimulations - 1] = nullptr;
400
401 auto pos = _warmupPhase.begin() + ((i + 1) * newLocalNumberMDSimulations - 1);
402 _warmupPhase.insert(pos, 0);
403 }
404
405 // Update local variables
406 _localNumberMDSimulations = newLocalNumberMDSimulations;
407 _totalNumberMDSimulations = newTotalNumberMDSimulations;
408 _blockOffset = newBlockOffset;
409 delete[] _macroscopicCellServices;
410 _macroscopicCellServices = newMacroscopicCellServices;
411 }

417 unsigned int addMDSimulation(coupling::InstanceHandling<LinkedCell, dim>& instanceHandling,
418 coupling::interface::MacroscopicSolverInterface<dim>* macroscopicSolverInterface, const unsigned int& slot) {
419
420 if (slot >= _totalNumberMDSimulations) {
421 std::cout << "ERROR coupling::services::MultiMDCellService::addMDSimulation(): "
422 "Invalid slot "
423 << slot << "!" << std::endl;
424 std::exit(EXIT_FAILURE);
425 }
426 if (_macroscopicCellServices[slot] != nullptr) {
427 std::cout << "ERROR! "
428 "coupling::services::MultiMDCellService::addMDSimulation(): "
429 "Simulation at "
430 << slot << " already exists!" << std::endl;
431 std::exit(EXIT_FAILURE);
432 }
433
434 std::stringstream filestem;
435 filestem << "restart_checkpoint_" << _multiMDService.getGlobalRank() / computeScalarNumberProcesses();
436 instanceHandling.writeCheckpoint(filestem.str().c_str(), 0);
437
438 if (slot < _blockOffset || slot >= _blockOffset + _localNumberMDSimulations) {
439
440 _macroscopicCellServices[slot] = new coupling::services::MacroscopicCellServiceMacroOnly<dim>(
441 slot, macroscopicSolverInterface, _mdConfiguration.getMPIConfiguration().getNumberOfProcesses(), _multiMDService.getGlobalRank(),
442 _mdConfiguration.getDomainConfiguration().getGlobalDomainSize(), _mdConfiguration.getDomainConfiguration().getGlobalDomainOffset(),
443 _mamicoConfiguration.getParallelTopologyConfiguration(), _mamicoConfiguration.getMacroscopicCellConfiguration(),
444 (slot / _localNumberMDSimulations) * _intNumberProcesses);
445 } else {
446
447 unsigned localIndex = slot - _blockOffset;
448 auto* mdSolverInterface = instanceHandling.addMDSimulation(slot, localIndex);
449
450 _macroscopicCellServices[slot] = new coupling::services::MacroscopicCellServiceImpl<LinkedCell, dim>(
451 slot, mdSolverInterface, macroscopicSolverInterface, _mdConfiguration.getMPIConfiguration().getNumberOfProcesses(), _multiMDService.getGlobalRank(),
452 _mamicoConfiguration.getParticleInsertionConfiguration(), _mamicoConfiguration.getMomentumInsertionConfiguration(),
453 _mamicoConfiguration.getBoundaryForceConfiguration(), _mamicoConfiguration.getTransferStrategyConfiguration(),
454 _mamicoConfiguration.getParallelTopologyConfiguration(), _mamicoConfiguration.getThermostatConfiguration(),
455 _mdConfiguration.getSimulationConfiguration().getNumberOfTimesteps(), _mamicoConfiguration.getMacroscopicCellConfiguration(),
456 _filterPipelineConfiguration.c_str(), _multiMDService, _topologyOffset, _tws);
457 instanceHandling.getSimpleMD()[localIndex]->setMacroscopicCellService((_macroscopicCellServices[slot]));
458 _macroscopicCellServices[slot]->perturbateVelocity();
459 if (_macroscopicCellServices[slot]->getFilterPipeline() == nullptr) {
460 _macroscopicCellServices[slot]->initFiltering();
461 }
462 }
463
464 _warmupPhase[slot] = 10;
465
466 return slot;
467 }

528 coupling::IndexConversion<dim>* initIndexConversion(tarch::la::Vector<dim, double> macroscopicCellSize, tarch::la::Vector<dim, unsigned int> numberProcesses,
529 unsigned int rank, tarch::la::Vector<dim, double> globalMDDomainSize,
530 tarch::la::Vector<dim, double> globalMDDomainOffset,
531 coupling::paralleltopology::ParallelTopologyType parallelTopologyType,
532 unsigned int topologyOffset) const {
533
534 tarch::la::Vector<dim, unsigned int> globalNumberMacroscopicCells(0);
535 for (unsigned int d = 0; d < dim; d++) {
536 globalNumberMacroscopicCells[d] = (unsigned int)floor(globalMDDomainSize[d] / macroscopicCellSize[d] + 0.5);
537 if (fabs(globalNumberMacroscopicCells[d] * macroscopicCellSize[d] - globalMDDomainSize[d]) > 1e-13) {
538 std::cout << "coupling::services::MultiMDCellService::initIndexConversi"
539 "on(): Deviation of domain size > 1e-13!"
540 << std::endl;
541 }
542 }
543 coupling::IndexConversion<dim>* ic = new coupling::IndexConversion<dim>(globalNumberMacroscopicCells, numberProcesses, rank, globalMDDomainSize,
544 globalMDDomainOffset, parallelTopologyType, topologyOffset);
545 if (ic == NULL) {
546 std::cout << "coupling::services::MultiMDCellService::initIndexConversion"
547 "(): ic==NULL!"
548 << std::endl;
549 exit(EXIT_FAILURE);
550 }
551 return ic;
552 }
