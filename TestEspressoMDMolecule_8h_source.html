49 /* compareParticleList() compares the list of particles in each cell, one directly read from Espresso Simulation and other from the list sent to the
50 * coupling tool (EspressoMDMolecule) */
51
52 void compareParticleLists(int c, int& flag) {
53 int j = 0;
54 int counter = 0;
55
56 // Define a molecule iterator and iterate over all the molecules in that particular cell
57 ParticleList* list = local_cells.cell[c];
58 coupling::interface::EspressoMDMoleculeIterator it(*list);
59 Particle* part;
60 part = list->part;
61 int number = list->n;
62
63 // I define two lists, listEspresso and listEspressoCouplingInterface and add the respective data to them
64 std::vector<tarch::la::Vector<3, double>> listEspresso;
65 std::vector<tarch::la::Vector<3, double>> listEspressoCouplingInterface;
66 tarch::la::Vector<3, double> velocity(0.0);
67 tarch::la::Vector<3, double> position(0.0);
68
69 for (it.begin(); it.continueIteration(); it.next()) {
70 velocity = it.get().getVelocity();
71 position = it.get().getPosition();
72
73 listEspressoCouplingInterface.push_back(velocity);
74 listEspressoCouplingInterface.push_back(position);
75
76 for (int i = 0; i < 3; i++) {
77 velocity[i] = part[j].m.v[i];
78 position[i] = part[j].r.p[i];
79 }
80 listEspresso.push_back(velocity);
81 listEspresso.push_back(position);
82
83 j++;
84 counter++;
85 }
86
87 if (listEspresso.size() != (unsigned int)2 * number) {
88 std::cout << "ERROR: particle list has wrong size!" << std::endl;
89 flag = 0;
90 }
91 if (listEspresso != listEspressoCouplingInterface) {
92 std::cout << "ERROR: the particle positions and velocites obtianed by the Coupling interfaces and Espresso simulation do not match " << std::endl;
93 flag = 0;
94 }
95 }
