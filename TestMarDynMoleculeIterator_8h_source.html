37 void testIterator() {
38 unsigned int errors = 0;
39 // get particle container from simulation
40 LinkedCells *cells = (LinkedCells *)this->_marDyn->getMolecules();
41
42 ParticleCell pc;
43 for (int i = 0; i < 2744; i++) {
44 pc = cells->getCell(i);
45 MarDynCell cell = MarDynCell(&pc, cells->getCutoff());
46
47 // set up molecule iterator for testing
48 MarDynMoleculeIterator mdmi(cell);
49
50 // number of molecules in current particle cell
51 unsigned int moleculeCount = pc.getMoleculeCount();
52
53 // determine number of molecules through interface
54 unsigned int moleculesInInterface = 0;
55 for (mdmi.begin(); mdmi.continueIteration(); mdmi.next()) {
56 moleculesInInterface++;
57
58 // std::cout << "pos: " << mdmi.getConst().getPosition() << std::endl;
59 // std::cout << "vel: " << mdmi.getConst().getVelocity() << std::endl;
60 // std::cout << "------" << std::endl;
61 }
62
63 if (moleculeCount != 0)
64 std::cout << "Test: " << moleculeCount
65 << " =?= " << moleculesInInterface << std::endl;
66
67 if (moleculeCount != moleculesInInterface) {
68 errors++;
69 std::cout << "ERROR: number of molecules in current cell " << i
70 << "should be " << moleculeCount
71 << ". The interface instead has: " << moleculesInInterface
72 << std::endl;
73 }
74 }
75
76 if (errors == 0)
77 std::cout << "MarDynMoleculeIterator test was successful!" << std::endl;
78 else
79 std::cout << "MarDynMoleculeIterator test was not successful! (Errors: "
80 << errors << ")" << std::endl;
81 }
82};
