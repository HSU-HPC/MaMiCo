21 virtual void run(){
22 // set a molecule into the very middle of the existing molecules
23 coupling::datastructures::Molecule<dim> molecule;
24 const tarch::la::Vector<dim,double> thisPosition(3.0);
25 molecule.setPosition(thisPosition);
26
27 // initialise all interfaces and simulation parts
28 if (dim==2){
29 TestLammps<dim>::loadLammpsTestConfiguration("inputpositionsonly2D.xyz",4);
30 } else {
31 TestLammps<dim>::loadLammpsTestConfiguration("inputpositionsonly3D.xyz",8);
32 }
33 TestLammps<dim>::loadMacroscopicSolverConfiguration();
34 TestLammps<dim>::loadMamicoTestConfiguration();
35
36 // run one time step to sort molecules into linked lists
37 LAMMPS_NS::LAMMPS* lammps = TestLammps<dim>::_lammps;
38 lammps->input->one("run 1");
39
40 // get MD solver interface (required for sorting)
41 LAMMPS_NS::MamicoLammpsMDSolverInterface<dim>* mdSolverInterface = (LAMMPS_NS::MamicoLammpsMDSolverInterface<dim>*)
42 coupling::interface::MamicoInterfaceProvider<LAMMPS_NS::MamicoCell,dim>::getInstance().getMDSolverInterface();
43 if (mdSolverInterface==NULL){ std::cout << "ERROR TestLammpsGhost: could not cast MD Solver interface!" << std::endl; exit(EXIT_FAILURE); }
44 const coupling::IndexConversion<dim>& indexConversion = coupling::interface::
45 MamicoInterfaceProvider<LAMMPS_NS::MamicoCell,dim>::getInstance().getMacroscopicCellService()->getIndexConversion();
46
47 // compute LJ parameters (copy-paste from calculateForceAndEnergy)
48 const double sigma6 = pow(mdSolverInterface->getMoleculeSigma(),6.0);
49 const double epsilon = mdSolverInterface->getMoleculeEpsilon();
50 const double cutOffRadius = 2.5;
51 const double cutOffRadiusSquared = cutOffRadius*cutOffRadius;
52 const double sigma6OverCutoff6 = sigma6/(cutOffRadiusSquared*cutOffRadiusSquared*cutOffRadiusSquared);
53 const double cutOffEnergy = 4.0 * epsilon * sigma6OverCutoff6 * (sigma6OverCutoff6 - 1.0);
54 // set tolerance in measurements
55 const double tolerance = 1.0e-8;
56
57 // compute LJ potential energy and force reference values
58 const double rij2 = tarch::la::dot(tarch::la::Vector<dim,double>(4.0)-thisPosition,tarch::la::Vector<dim,double>(4.0)-thisPosition);
59 const double rij6 = rij2*rij2*rij2;
60 const double potEnergyRef = pow(2.0,dim)*0.5*(4.0*epsilon * (sigma6/rij6) * ((sigma6/rij6) - 1.0) - cutOffEnergy);
61 const tarch::la::Vector<dim,double> forceRef(0.0);
62
63 // compute force/energy onto this molecule -> only do this on rank 0, since this is the rank containing the cell with the molecule at 3.0x3.0x3.0
64 if (indexConversion.getThisRank()==0){
65 mdSolverInterface->calculateForceAndEnergy(molecule);
66
67 tarch::la::Vector<dim,double> force = molecule.getForce();
68 double potEnergy = molecule.getPotentialEnergy();
69
70 if (fabs(potEnergy-potEnergyRef)>tolerance){
71 std::cout << "ERROR TestLammpsCalculateForceEnergy: potential Energies do not match1! Potential energy ref.=" << potEnergyRef << ", potential energy=" << potEnergy << std::endl; exit(EXIT_FAILURE);
72 }
73 if (fabs(tarch::la::norm2(force-forceRef)>tolerance)){
74 std::cout << "ERROR TestLammpsCalculcateForceEnergy: forces do not match1! Force ref.=" << forceRef << ", force=" << force << std::endl; exit(EXIT_FAILURE);
75 }
76
77 // add molecule to simulation
78 std::cout << "Add molecule to MD simulation..." << std::endl;
79 mdSolverInterface->addMoleculeToMDSimulation(molecule);
80 }
81
82 // synchronise molecules over all processes
83 mdSolverInterface->synchronizeMoleculesAfterMassModification();
84
85 // search for this molecule and try to extract potential energy from it; this energy should again return the same value
86 if (indexConversion.getThisRank()==0){
87 std::cout << "Molecule synchronised; do next test..." << std::endl;
88 const int nlocal = lammps->atom->nlocal;
89 int n = -1;
90 for (int i = 0; i < nlocal; i++){
91 bool found = true; for (int d = 0; d < dim; d++){ found = found && (thisPosition[d]==lammps->atom->x[i][d]); }
92 if (found){ n=i; break;}
93 }
94 if (n<0){ std::cout << "ERROR TestLammpsCalculateForceEnergy: Could not find molecule after adding it!" << std::endl; exit(EXIT_FAILURE); }
95 else {
96 std::cout << "Molecule added at local position " << n << std::endl;
97 }
98
99 // construct a lammps molecule
100 LAMMPS_NS::MamicoLammpsMolecule<dim> lammpsMolecule(lammps->atom->x,lammps->atom->v,lammps->atom->f,n,cutOffRadius);
101 double potEnergy = lammpsMolecule.getPotentialEnergy();
102
103 if (fabs(potEnergy-potEnergyRef)>tolerance){
104 std::cout << "ERROR TestLammpsCalculateForceEnergy: potential Energies do not match2! Potential energy ref.=" << potEnergyRef << ", potential energy=" << potEnergy << std::endl; exit(EXIT_FAILURE);
105 }
106 }
107 }

109};
