58 double stdDeviation = std::sqrt(dim * _kB * _temperature / _molecularMass);
59
60 tarch::la::Vector<dim, double> randomNumbers(0.0);
61
62 coupling::interface::MoleculeIterator<LinkedCell, dim> *molecule =
63 _mdSolverInterface->getMoleculeIterator(cell);
64 molecule->begin();
65 while (molecule->continueIteration()) {
66 coupling::interface::Molecule<dim> &wrapper(molecule->get());
67 randomNumbers[0] = tarch::utils::RandomNumberService::getInstance()
68 .getGaussianRandomNumber();
69 for (unsigned int d = 1; d < dim; ++d) {
70 randomNumbers[d] = tarch::utils::RandomNumberService::getInstance()
71 .getGaussianRandomNumber();
72 }
73
74 tarch::la::Vector<dim, double> mVelocity = wrapper.getVelocity();
75 if (dim == 3) {
76 mVelocity[0] =
77 _velocity[0] +
78 stdDeviation * (randomNumbers[0] * std::sin(randomNumbers[1]) *
79 std::cos(randomNumbers[2]));
80 mVelocity[1] =
81 _velocity[1] +
82 stdDeviation * (randomNumbers[0] * std::sin(randomNumbers[1]) *
83 std::sin(randomNumbers[2]));
84 mVelocity[2] =
85 _velocity[2] +
86 stdDeviation * (randomNumbers[0] * std::cos(randomNumbers[1]));
87 } else if (dim == 1) {
88 mVelocity = _velocity + stdDeviation * randomNumbers;
89 } else if (dim == 2) {
90 mVelocity[0] =
91 _velocity[0] +
92 stdDeviation * (randomNumbers[0] * std::cos(randomNumbers[1]));
93 mVelocity[1] =
94 _velocity[1] +
95 stdDeviation * (randomNumbers[0] * std::sin(randomNumbers[1]));
96 }
97 wrapper.setVelocity(mVelocity);
98
99 molecule->next();
100 }
101 delete molecule;
102 }
