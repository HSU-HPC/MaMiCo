88 coupling::IndexConversion<dim>* initIndexConversion(tarch::la::Vector<dim, double> macroscopicCellSize, tarch::la::Vector<dim, unsigned int> numberProcesses,
89 unsigned int rank, tarch::la::Vector<dim, double> globalMDDomainSize,
90 tarch::la::Vector<dim, double> globalMDDomainOffset,
91 coupling::paralleltopology::ParallelTopologyType parallelTopologyType,
92 unsigned int topologyOffset) const {
93 tarch::la::Vector<dim, unsigned int> globalNumberMacroscopicCells(0);
94 for (unsigned int d = 0; d < dim; d++) {
95 globalNumberMacroscopicCells[d] = (unsigned int)floor(globalMDDomainSize[d] / macroscopicCellSize[d] + 0.5);
96 if (fabs(globalNumberMacroscopicCells[d] * macroscopicCellSize[d] - globalMDDomainSize[d]) > 1e-13) {
97 std::cout << "coupling::services::MacroscopicCellServiceMacroOnly::"
98 "initIndexConversion(): Deviation of domain size > 1e-13!"
99 << std::endl;
100 }
101 }
102 coupling::IndexConversion<dim>* ic = new coupling::IndexConversion<dim>(globalNumberMacroscopicCells, numberProcesses, rank, globalMDDomainSize,
103 globalMDDomainOffset, parallelTopologyType, topologyOffset);
104 if (ic == NULL) {
105 std::cout << "coupling::services::MacroscopicCellServiceImpl::"
106 "initIndexConversion(): ic==NULL!"
107 << std::endl;
108 exit(EXIT_FAILURE);
109 }
110 return ic;
111 }
