132 // switch on coupling -> add mamico fix with ID=2; we further check correctness of number of cells
133 // -> afterwards, the MDSolverInterface of mamico will be initialised and provided via MamicoInterfaceProvider.
134 virtual void switchOnCoupling() {
135 std::stringstream ss;
136 unsigned int localNumberCells = 1;
137 const tarch::la::Vector<MDSIMULATIONFACTORY_DIMENSION,unsigned int> numberProcs = _configuration.getMPIConfiguration().getNumberOfProcesses();
138 const tarch::la::Vector<MDSIMULATIONFACTORY_DIMENSION,double> domainSize = _configuration.getDomainConfiguration().getGlobalDomainSize();
139 const tarch::la::Vector<MDSIMULATIONFACTORY_DIMENSION,double> meshsize = _mamicoConfiguration.getMacroscopicCellConfiguration().getMacroscopicCellSize();
140 // determine max. number of cells required on each process
141 for (unsigned int d = 0; d < MDSIMULATIONFACTORY_DIMENSION; d++){
142 const unsigned int cells = (unsigned int) ceil(domainSize[d]/meshsize[d]/numberProcs[d]);
143 if (!tarch::la::equals(cells*numberProcs[d]*meshsize[d],domainSize[d],_tolerance)){
144 std::cout << "ERROR switchOnCoupling(): cells and domainsize/meshsize/numberProcs do not match!" << std::endl; exit(EXIT_FAILURE);
145 }
146 // take cells+2 as we also need to allocate ghost layer
147 localNumberCells = localNumberCells * (cells+2);
148 }
149
150 ss << "fix 2 all mamico " << localNumberCells << " 123456 " << _configuration.getDomainConfiguration().getCutoffRadius();
151 _lmp->input->one(ss.str().c_str());
152 }

189 private:
190 void initSingleSimulation(MPI_Comm comm,unsigned int localMDSimulation){
191 int rank; MPI_Comm_rank(comm,&rank);
192 std::stringstream ss;
193 std::string inputFile;
194
195 // use logging separately for each LAMMPS instance
196 ss.str(""); ss << "log log.lammps" << localMDSimulation;
197 _lmp->input->one(ss.str().c_str());
198 ss.str(""); ss << "log log.lammps" << localMDSimulation << " append";
199 _lmp->input->one(ss.str().c_str());
200
201 // define lj atomic input
202
203 _lmp->input->one("units lj");
204 _lmp->input->one("atom_style atomic");
205 // define number of processes
206 ss.str("");
207 tarch::la::Vector<3,unsigned int> procs(1);
208 for (int d = 0; d < MDSIMULATIONFACTORY_DIMENSION; d++) { procs[d] = _configuration.getMPIConfiguration().getNumberOfProcesses()[d]; }
209 ss << "processors " << procs[0] << " " << procs[1] << " " << procs[2] << " map xyz";
210 _lmp->input->one(ss.str().c_str());
211
212 // define 2D if necessary
213 if (MDSIMULATIONFACTORY_DIMENSION==2){ _lmp->input->one("dimension 2"); }
214 else { _lmp->input->one("dimension 3");}
215
216 // incorporate boundary conditions: currently, only reflecting and periodic boundaries are supported; to be extended to 2D
217 ss.str("");
218 ss << "boundary";
219 if (MDSIMULATIONFACTORY_DIMENSION==3){
220 const tarch::la::Vector<MD_LINKED_CELL_NEIGHBOURS,simplemd::BoundaryType> bc = _configuration.getDomainConfiguration().getBoundary();
221 int indices[3] = {12,10,4};
222 for (int i = 0; i < MDSIMULATIONFACTORY_DIMENSION; i++){
223 if (bc[indices[i]]==simplemd::PERIODIC_BOUNDARY){ ss << " p";}
224 else if (bc[indices[i]]==simplemd::REFLECTING_BOUNDARY){ ss << " f";}
225 else { std::cout << "ERROR LammpsMDSimulation::initSingleSimulation(): only reflecting and periodic boundaries supported!" << std::endl; exit(EXIT_FAILURE);}
226 }
227 } else { std::cout << "ERROR LammpsMDSimulation::initSingleSimulation(): only 3D supported at the moment!" << std::endl; }
228 _lmp->input->one(ss.str().c_str());
229
230 // create atoms based on input configuration; currently, only creation of molecules on a lattice is supported -------------------
231 tarch::la::Vector<3,double> domainOffset(0.0); for (int d=0; d<MDSIMULATIONFACTORY_DIMENSION; d++){ domainOffset[d] = _configuration.getDomainConfiguration().getGlobalDomainOffset()[d];}
232 tarch::la::Vector<3,double> domainSize (1.0); for (int d=0; d<MDSIMULATIONFACTORY_DIMENSION; d++){ domainSize[d] = _configuration.getDomainConfiguration().getGlobalDomainSize()[d];}
233 double particleDensity=1.0;
234 for (int d=0; d<MDSIMULATIONFACTORY_DIMENSION; d++){
235 particleDensity = particleDensity*_configuration.getDomainConfiguration().getMoleculesPerDirection()[d]/domainSize[d];
236 }
237 ss.str("");
238 ss << "lattice sc " << particleDensity << " origin 0.5 0.5 0.5";
239 _lmp->input->one(ss.str().c_str());
240
241
242 // define simulation region
243 ss.str("");
244 ss << "region box block " << domainOffset[0] << " " << (domainOffset[0]+domainSize[0]) << " " << domainOffset[1] << " " << (domainOffset[1]+domainSize[1]) << " ";
245 ss << domainOffset[2] << " " << (domainOffset[2]+domainSize[2]) << " units box";
246 _lmp->input->one(ss.str().c_str());
247
248
249 _lmp->input->one("create_box 1 box");
250
251 _lmp->input->one("create_atoms 1 box");
252 // set mass and correct temperature
253 ss.str(""); ss << "mass 1 " << _configuration.getMoleculeConfiguration().getMass();
254 _lmp->input->one(ss.str().c_str());
255 // initialise velocities on molecules; for parallel simulations, we do this according to global MD simulation number (we add 1 as 0 is not a valid seed)
256 ss.str(""); ss << "velocity all create " << _configuration.getMoleculeConfiguration().getTemperature() << " " << localMDSimulation+1 << " loop geom";
257 _lmp->input->one(ss.str().c_str());
258
259 // incorporate reflecting boundaries, pt.2 (this needs to be done after the box was created); to be extended to 2D
260 if (MDSIMULATIONFACTORY_DIMENSION==3){
261 const tarch::la::Vector<MD_LINKED_CELL_NEIGHBOURS,simplemd::BoundaryType> bc = _configuration.getDomainConfiguration().getBoundary();
262 int indices[3] = {12,10,4}; std::string faces[3] = {"x", "y", "z"};
263 for (int i = 0; i < MDSIMULATIONFACTORY_DIMENSION; i++){
264 if (bc[indices[i]]==simplemd::REFLECTING_BOUNDARY){
265 ss.str(""); ss << "fix reflection" << faces[i] << " all wall/reflect " << faces[i] << "lo EDGE " << faces[i] << "hi EDGE units box";
266 _lmp->input->one(ss.str().c_str());
267 }
268 }
269 }
270 // --------------------------------------------------------------------------------------------------------------------------------------------
271
272 // define molecule and LJ parameters
273 ss.str(""); ss << "pair_style lj/cut " << _configuration.getDomainConfiguration().getCutoffRadius(); _lmp->input->one(ss.str().c_str());
274 _lmp->input->one("pair_modify shift yes");
275 ss.str(""); ss << "pair_coeff 1 1 " << _configuration.getMoleculeConfiguration().getEpsilon() << " ";
276 ss << _configuration.getMoleculeConfiguration().getSigma() << " " << _configuration.getDomainConfiguration().getCutoffRadius();
277 _lmp->input->one(ss.str().c_str());
278
279 // set time step
280 ss.str(""); ss << "timestep " << _configuration.getSimulationConfiguration().getDt(); _lmp->input->one(ss.str().c_str());
281
282 // set some verlet-specific values; currently hard-coded
283 _lmp->input->one("neighbor 0.5 bin");
284 _lmp->input->one("neigh_modify delay 0 every 20 check no");
285
286 // write output; though this is not vtk format, we use this config for that purpose
287 if (_configuration.getVTKConfiguration().getWriteEveryTimestep() != 0){
288 ss.str(""); ss << "dump myDump all custom " << _configuration.getVTKConfiguration().getWriteEveryTimestep() << " ";
289 ss << _configuration.getVTKConfiguration().getFilename() << "_" << localMDSimulation << "_" << "*.dat id type x y z vx vy vz";
290 _lmp->input->one(ss.str().c_str());
291 }
292 // write checkpointing
293 if (_configuration.getCheckpointConfiguration().getWriteEveryTimestep() != 0){
294 ss.str(""); ss << "restart " << _configuration.getCheckpointConfiguration().getWriteEveryTimestep() << " ";
295 ss << _configuration.getCheckpointConfiguration().getFilename() << "_" << localMDSimulation << "_" << "1 " << _configuration.getCheckpointConfiguration().getFilename() << "_" << localMDSimulation << "_" << "2";
296 _lmp->input->one(ss.str().c_str());
297 }
298 // set nve ensemble
299 _lmp->input->one("fix 1 all nve");
300 }
301
302};
303#endif

391 void initSingleSimulation(MPI_Comm comm,unsigned int localMDSimulation){
392 int rank; MPI_Comm_rank(comm,&rank);
393 std::stringstream ss;
394 std::string inputFile;
395 // particle insertion is currently only supported for single-centered LJ particles
396 if (_mamicoConfiguration.getParticleInsertionConfiguration().getParticleInsertionType()!=coupling::configurations::ParticleInsertionConfiguration::NO_INSERTION){
397 std::cout << "ERROR coupling::LammpsDPDSImulation::initSingleSimulation(): Particle insertion currently not supported!" << std::endl; exit(EXIT_FAILURE);
398 }
399
400 // set logs for each LAMMPS instance separately
401 ss.str(""); ss << "log log.lammps" << localMDSimulation;
402 _lmp->input->one(ss.str().c_str());
403 ss.str(""); ss << "log log.lammps" << localMDSimulation << " append";
404 _lmp->input->one(ss.str().c_str());
405 _lmp->input->one("variable ndim equal 3");
406 ss.str(""); ss << "variable kB equal " << _configuration.getDomainConfiguration().getKB();
407 _lmp->input->one(ss.str().c_str());
408 ss.str(""); ss << "variable kBT equal " << _configuration.getMoleculeConfiguration().getTemperature()*_configuration.getDomainConfiguration().getKB();
409 _lmp->input->one(ss.str().c_str());
410 ss.str(""); ss << "variable T equal " << _configuration.getMoleculeConfiguration().getTemperature();
411 _lmp->input->one(ss.str().c_str());
412 ss.str(""); ss << "variable rc equal " << _configuration.getDomainConfiguration().getCutoffRadius();
413 _lmp->input->one(ss.str().c_str());
414 _lmp->input->one("units lj");
415 _lmp->input->one("dimension ${ndim}");
416
417 // set up boundary treatment, Pt1: for "periodic", we just set the "p"-option and "f" otherwise;
418 // for the latter, we further have to introduce the respective walls, see boundary treatment, Pt2, further down
419 ss.str(""); ss << "boundary";
420 if (MDSIMULATIONFACTORY_DIMENSION==3){
421 const tarch::la::Vector<MD_LINKED_CELL_NEIGHBOURS,simplemd::BoundaryType> bc = _configuration.getDomainConfiguration().getBoundary();
422 int indices[3] = {12,10,4};
423 for (int i = 0; i < MDSIMULATIONFACTORY_DIMENSION; i++){
424 if (bc[indices[i]]==simplemd::PERIODIC_BOUNDARY){ ss << " p";}
425 else if (bc[indices[i]]==simplemd::REFLECTING_BOUNDARY){ ss << " f";}
426 else { std::cout << "ERROR LammpsDPDSimulation::initSingleSimulation(): only reflecting and periodic boundaries supported!" << std::endl; exit(EXIT_FAILURE);}
427 }
428 } else { std::cout << "ERROR LammpsDPDSimulation::initSingleSimulation(): only 3D supported at the moment!" << std::endl; }
429 _lmp->input->one(ss.str().c_str());
430
431 _lmp->input->one("atom_style atomic");
432 // define number of processes with xyz-topology
433 ss.str(""); ss << "processors " << _configuration.getMPIConfiguration().getNumberOfProcesses()[0] << " " << _configuration.getMPIConfiguration().getNumberOfProcesses()[1] << " ";
434 ss << _configuration.getMPIConfiguration().getNumberOfProcesses()[2] << " map xyz";
435 _lmp->input->one(ss.str().c_str());
436 ss.str(""); ss << "variable seed1 equal " << localMDSimulation+rank+1;
437 _lmp->input->one(ss.str().c_str()); // currently fixed random seed
438 ss.str(""); ss << "variable seed2 equal " << localMDSimulation+rank+10;
439 _lmp->input->one(ss.str().c_str()); // new seed for velocity generation; currently deduced from _randSeed+10
440 _lmp->input->one("print \"seed1: ${seed1}\"");
441 _lmp->input->one("print \"seed2: ${seed2}\"");
442 _lmp->input->one("pair_style dpd/general ${kBT} ${rc} ${seed1}"); // DPD scheme with forcing by Xin
443
444 const tarch::la::Vector<MDSIMULATIONFACTORY_DIMENSION,double> domainSize(_configuration.getDomainConfiguration().getGlobalDomainSize());
445 const tarch::la::Vector<MDSIMULATIONFACTORY_DIMENSION,double> domainOffset(_configuration.getDomainConfiguration().getGlobalDomainOffset());
