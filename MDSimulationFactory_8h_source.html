185 // switch on coupling -> add mamico fix with ID=2; we further check correctness of number of cells
186 // -> afterwards, the MDSolverInterface of mamico will be initialised and provided via MamicoInterfaceProvider.
187 virtual void switchOnCoupling() {
188 std::stringstream ss;
189 unsigned int localNumberCells = 1;
190 const tarch::la::Vector<MDSIMULATIONFACTORY_DIMENSION,unsigned int> numberProcs = _configuration.getMPIConfiguration().getNumberOfProcesses();
191 const tarch::la::Vector<MDSIMULATIONFACTORY_DIMENSION,double> domainSize = _configuration.getDomainConfiguration().getGlobalDomainSize();
192 const tarch::la::Vector<MDSIMULATIONFACTORY_DIMENSION,double> meshsize = _mamicoConfiguration.getMacroscopicCellConfiguration().getMacroscopicCellSize();
193 // determine max. number of cells required on each process
194 for (unsigned int d = 0; d < MDSIMULATIONFACTORY_DIMENSION; d++){
195 const unsigned int cells = (unsigned int) ceil(domainSize[d]/meshsize[d]/numberProcs[d]);
196 if (!tarch::la::equals(cells*numberProcs[d]*meshsize[d],domainSize[d],_tolerance)){
197 std::cout << "ERROR switchOnCoupling(): cells and domainsize/meshsize/numberProcs do not match!" << std::endl; exit(EXIT_FAILURE);
198 }
199 // take cells+2 as we also need to allocate ghost layer
200 localNumberCells = localNumberCells * (cells+2);
201 }
202
203 ss << "fix 2 all mamico " << localNumberCells << " 123456 " << _configuration.getDomainConfiguration().getCutoffRadius();
204 _lmp->input->one(ss.str().c_str());
205 }

242 private:
243 void initSingleSimulation(MPI_Comm comm,unsigned int localMDSimulation){
244 int rank; MPI_Comm_rank(comm,&rank);
245 std::stringstream ss;
246 std::string inputFile;
247
248 // use logging separately for each LAMMPS instance
249 ss.str(""); ss << "log log.lammps" << localMDSimulation;
250 _lmp->input->one(ss.str().c_str());
251 ss.str(""); ss << "log log.lammps" << localMDSimulation << " append";
252 _lmp->input->one(ss.str().c_str());
253
254 // define lj atomic input
255
256 _lmp->input->one("units lj");
257 _lmp->input->one("atom_style atomic");
258 // define number of processes
259 ss.str("");
260 tarch::la::Vector<3,unsigned int> procs(1);
261 for (int d = 0; d < MDSIMULATIONFACTORY_DIMENSION; d++) { procs[d] = _configuration.getMPIConfiguration().getNumberOfProcesses()[d]; }
262 ss << "processors " << procs[0] << " " << procs[1] << " " << procs[2] << " map xyz";
263 _lmp->input->one(ss.str().c_str());
264
265 // define 2D if necessary
266 if (MDSIMULATIONFACTORY_DIMENSION==2){ _lmp->input->one("dimension 2"); }
267 else { _lmp->input->one("dimension 3");}
268
269 // incorporate boundary conditions: currently, only reflecting and periodic boundaries are supported; to be extended to 2D
270 ss.str("");
271 ss << "boundary";
272 if (MDSIMULATIONFACTORY_DIMENSION==3){
273 const tarch::la::Vector<MD_LINKED_CELL_NEIGHBOURS,simplemd::BoundaryType> bc = _configuration.getDomainConfiguration().getBoundary();
274 int indices[3] = {12,10,4};
275 for (int i = 0; i < MDSIMULATIONFACTORY_DIMENSION; i++){
276 if (bc[indices[i]]==simplemd::PERIODIC_BOUNDARY){ ss << " p";}
277 else if (bc[indices[i]]==simplemd::REFLECTING_BOUNDARY){ ss << " f";}
278 else { std::cout << "ERROR LammpsMDSimulation::initSingleSimulation(): only reflecting and periodic boundaries supported!" << std::endl; exit(EXIT_FAILURE);}
279 }
280 } else { std::cout << "ERROR LammpsMDSimulation::initSingleSimulation(): only 3D supported at the moment!" << std::endl; }
281 _lmp->input->one(ss.str().c_str());
282
283 // create atoms based on input configuration; currently, only creation of molecules on a lattice is supported -------------------
284 tarch::la::Vector<3,double> domainOffset(0.0); for (int d=0; d<MDSIMULATIONFACTORY_DIMENSION; d++){ domainOffset[d] = _configuration.getDomainConfiguration().getGlobalDomainOffset()[d];}
285 tarch::la::Vector<3,double> domainSize (1.0); for (int d=0; d<MDSIMULATIONFACTORY_DIMENSION; d++){ domainSize[d] = _configuration.getDomainConfiguration().getGlobalDomainSize()[d];}
286 if (!_configuration.getDomainConfiguration().initFromCheckpoint()
287 && !_configuration.getDomainConfiguration().initFromSequentialCheckpoint() ) {
288 tarch::la::Vector<3,unsigned int> moleculesPerDirection(0); for(int d=0; d<MDSIMULATIONFACTORY_DIMENSION; d++){ moleculesPerDirection[d] = _configuration.getDomainConfiguration().getMoleculesPerDirection()[d];}
289 double particleDensity=1.0;
290 for (int d=0; d<MDSIMULATIONFACTORY_DIMENSION; d++){
291 particleDensity = particleDensity*_configuration.getDomainConfiguration().getMoleculesPerDirection()[d]/domainSize[d];
292 }
293 ss.str("");
294 ss << "lattice sc " << particleDensity << " origin 0.5 0.5 0.5";
295 _lmp->input->one(ss.str().c_str());
296 }
297
298
299 // define simulation region
300 ss.str("");
301 ss << "region box block " << domainOffset[0] << " " << (domainOffset[0]+domainSize[0]) << " " << domainOffset[1] << " " << (domainOffset[1]+domainSize[1]) << " ";
302 ss << domainOffset[2] << " " << (domainOffset[2]+domainSize[2]) << " units box";
303 _lmp->input->one(ss.str().c_str());
304
305
306
307 if(_configuration.getDomainConfiguration().initFromCheckpoint() ||
308 _configuration.getDomainConfiguration().initFromSequentialCheckpoint() ) {
309 std::stringstream command;
310 command << "read_restart " << _configuration.getDomainConfiguration().getCheckpointFilestem() << ".restart";
311 _lmp->input->one(command.str().c_str());
312 } else {
313 _lmp->input->one("create_box 1 box");
314 _lmp->input->one("create_atoms 1 box");
315 // set mass and correct temperature
316 ss.str(""); ss << "mass 1 " << _configuration.getMoleculeConfiguration().getMass();
317 _lmp->input->one(ss.str().c_str());
318 // initialise velocities on molecules; for parallel simulations, we do this according to global MD simulation number (we add 1 as 0 is not a valid seed)
319 ss.str(""); ss << "velocity all create " << _configuration.getMoleculeConfiguration().getTemperature() << " " << localMDSimulation+1 << " loop geom";
320 _lmp->input->one(ss.str().c_str());
321 }
322
323 // incorporate reflecting boundaries, pt.2 (this needs to be done after the box was created); to be extended to 2D
324 if (MDSIMULATIONFACTORY_DIMENSION==3){
325 const tarch::la::Vector<MD_LINKED_CELL_NEIGHBOURS,simplemd::BoundaryType> bc = _configuration.getDomainConfiguration().getBoundary();
326 int indices[3] = {12,10,4}; std::string faces[3] = {"x", "y", "z"};
327 for (int i = 0; i < MDSIMULATIONFACTORY_DIMENSION; i++){
328 if (bc[indices[i]]==simplemd::REFLECTING_BOUNDARY){
329 ss.str(""); ss << "fix reflection" << faces[i] << " all wall/reflect " << faces[i] << "lo EDGE " << faces[i] << "hi EDGE units box";
330 _lmp->input->one(ss.str().c_str());
331 }
332 }
333 }
334 // --------------------------------------------------------------------------------------------------------------------------------------------
335
336 // define molecule and LJ parameters
337 ss.str(""); ss << "pair_style lj/cut " << _configuration.getDomainConfiguration().getCutoffRadius(); _lmp->input->one(ss.str().c_str());
338 _lmp->input->one("pair_modify shift yes");
339 ss.str(""); ss << "pair_coeff 1 1 " << _configuration.getMoleculeConfiguration().getEpsilon() << " ";
340 ss << _configuration.getMoleculeConfiguration().getSigma() << " " << _configuration.getDomainConfiguration().getCutoffRadius();
341 _lmp->input->one(ss.str().c_str());
342
343 // set time step
344 ss.str(""); ss << "timestep " << _configuration.getSimulationConfiguration().getDt(); _lmp->input->one(ss.str().c_str());
345
346 // set some verlet-specific values; currently hard-coded
347 _lmp->input->one("neighbor 0.5 bin");
348 _lmp->input->one("neigh_modify delay 0 every 1 check no");
349
350 // write output; though this is not vtk format, we use this config for that purpose
351 if (_configuration.getVTKConfiguration().getWriteEveryTimestep() != 0){
352 ss.str(""); ss << "dump myDump all custom " << _configuration.getVTKConfiguration().getWriteEveryTimestep() << " ";
353 ss << _configuration.getVTKConfiguration().getFilename() << "_" << localMDSimulation << "_" << "*.dat id type x y z vx vy vz";
354 _lmp->input->one(ss.str().c_str());
355 }
356 // write checkpointing
357 if (_configuration.getCheckpointConfiguration().getWriteEveryTimestep() != 0){
358 ss.str(""); ss << "restart " << _configuration.getCheckpointConfiguration().getWriteEveryTimestep() << " ";
359 ss << _configuration.getCheckpointConfiguration().getFilename() << "_" << localMDSimulation << "_" << "1.restart " << _configuration.getCheckpointConfiguration().getFilename() << "_" << localMDSimulation << "_" << "2.restart";
360 _lmp->input->one(ss.str().c_str());
361 }
362 // set nve ensemble
363 _lmp->input->one("fix 1 all nve");
364 }
365
366};
367#endif

455 void initSingleSimulation(MPI_Comm comm,unsigned int localMDSimulation){
456 int rank; MPI_Comm_rank(comm,&rank);
457 std::stringstream ss;
458 std::string inputFile;
459 // particle insertion is currently only supported for single-centered LJ particles
460 if (_mamicoConfiguration.getParticleInsertionConfiguration().getParticleInsertionType()!=coupling::configurations::ParticleInsertionConfiguration::NO_INSERTION){
461 std::cout << "ERROR coupling::LammpsDPDSImulation::initSingleSimulation(): Particle insertion currently not supported!" << std::endl; exit(EXIT_FAILURE);
462 }
463
464 // set logs for each LAMMPS instance separately
465 ss.str(""); ss << "log log.lammps" << localMDSimulation;
466 _lmp->input->one(ss.str().c_str());
467 ss.str(""); ss << "log log.lammps" << localMDSimulation << " append";
468 _lmp->input->one(ss.str().c_str());
469 _lmp->input->one("variable ndim equal 3");
470 ss.str(""); ss << "variable kB equal " << _configuration.getDomainConfiguration().getKB();
471 _lmp->input->one(ss.str().c_str());
472 ss.str(""); ss << "variable kBT equal " << _configuration.getMoleculeConfiguration().getTemperature()*_configuration.getDomainConfiguration().getKB();
473 _lmp->input->one(ss.str().c_str());
474 ss.str(""); ss << "variable T equal " << _configuration.getMoleculeConfiguration().getTemperature();
475 _lmp->input->one(ss.str().c_str());
476 ss.str(""); ss << "variable rc equal " << _configuration.getDomainConfiguration().getCutoffRadius();
477 _lmp->input->one(ss.str().c_str());
478 _lmp->input->one("units lj");
479 _lmp->input->one("dimension ${ndim}");
480
481 // set up boundary treatment, Pt1: for "periodic", we just set the "p"-option and "f" otherwise;
482 // for the latter, we further have to introduce the respective walls, see boundary treatment, Pt2, further down
483 ss.str(""); ss << "boundary";
484 if (MDSIMULATIONFACTORY_DIMENSION==3){
485 const tarch::la::Vector<MD_LINKED_CELL_NEIGHBOURS,simplemd::BoundaryType> bc = _configuration.getDomainConfiguration().getBoundary();
486 int indices[3] = {12,10,4};
487 for (int i = 0; i < MDSIMULATIONFACTORY_DIMENSION; i++){
488 if (bc[indices[i]]==simplemd::PERIODIC_BOUNDARY){ ss << " p";}
489 else if (bc[indices[i]]==simplemd::REFLECTING_BOUNDARY){ ss << " f";}
490 else { std::cout << "ERROR LammpsDPDSimulation::initSingleSimulation(): only reflecting and periodic boundaries supported!" << std::endl; exit(EXIT_FAILURE);}
491 }
492 } else { std::cout << "ERROR LammpsDPDSimulation::initSingleSimulation(): only 3D supported at the moment!" << std::endl; }
493 _lmp->input->one(ss.str().c_str());
