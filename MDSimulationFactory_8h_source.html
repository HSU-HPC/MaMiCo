129 simulateOneCouplingTimestep(t);

133 void simulateOneCouplingTimestep(const unsigned int& t) {
134 // if coupling is switched off, perform "normal" MD timestep
135 if (!_couplingSwitchedOn) {
136 simulateOneTimestep(t);
137 return;
138 }
139 if (_parallelTopologyService->isIdle()) {
140 return;
141 }
142
143 _boundaryTreatment->putBoundaryParticlesToInnerCellsAndFillBoundaryCells(_localBoundary, *_parallelTopologyService);
144
145 // call to synchronise data in cells; needs to be at this point of the
146 // coupling algorithm as the particles need to be placed inside the correct
147 // sampling volumes (hence: after communication with neighbours and molecule
148 // updates); do it BEFORE quantities are manipulated as we can then also do
149 // some pre-processing here.
150 _macroscopicCellService->processInnerMacroscopicCellAfterMDTimestep();
151 // ------------ coupling step: distribute mass ---------------------
152 _macroscopicCellService->distributeMass(t);
153 // for isothermal simulations: apply thermostat
154 _macroscopicCellService->applyTemperatureToMolecules(t);
155
156 // ---------- from here: go on with usual MD algorithm
157 // ------------------------------
158
159 // compute forces. After this step, each molecule has received all force
160 // contributions from its neighbors.
161 _linkedCellService->iterateCellPairs(*_lennardJonesForce, false);
162
163 // distribute momentum -> some methods require modification of force terms,
164 // therefore we call it AFTER the force computation and before everything else
165 _macroscopicCellService->distributeMomentum(t);
166 // apply boundary forces
167 _macroscopicCellService->applyBoundaryForce(t);
168 // evaluate statistics
169 evaluateStatistics(t);
170
171 _boundaryTreatment->emptyGhostBoundaryCells();
172
173 // plot VTK output
174 if ((_configuration.getVTKConfiguration().getWriteEveryTimestep() != 0) && (t % _configuration.getVTKConfiguration().getWriteEveryTimestep() == 0)) {
175 _vtkMoleculeWriter->setTimestep(t);
176 _moleculeService->iterateMolecules(*_vtkMoleculeWriter, false);
177 }
178
179// plot ADIOS2 output
180#if BUILD_WITH_ADIOS2
181 if ((_configuration.getAdios2Configuration().getWriteEveryTimestep() != 0) && (t % _configuration.getAdios2Configuration().getWriteEveryTimestep() == 0)) {
182 _Adios2Writer->setTimestep(t);
183 _moleculeService->iterateMolecules(*_Adios2Writer, false);
184 }
185#endif
186
187 // write checkpoint
188 if ((_configuration.getCheckpointConfiguration().getWriteEveryTimestep() != 0) &&
189 (t % _configuration.getCheckpointConfiguration().getWriteEveryTimestep() == 0)) {
190 _moleculeService->writeCheckPoint(*_parallelTopologyService, _configuration.getCheckpointConfiguration().getFilename(), t);
191 }
192 // reorganise memory if needed
193 if ((_configuration.getSimulationConfiguration().getReorganiseMemoryEveryTimestep() != 0) &&
194 (t % _configuration.getSimulationConfiguration().getReorganiseMemoryEveryTimestep() == 0)) {
195 _moleculeService->reorganiseMemory(*_parallelTopologyService, *_linkedCellService);
196 }
197 // plot also macroscopic cell information
198 _macroscopicCellService->plotEveryMicroscopicTimestep(t);
199
200 _linkedCellService->iterateCells(*_emptyLinkedListsMapping, false);
201
202 // time integration. After this step, the velocities and the positions of the
203 // molecules have been updated.
204 _moleculeService->iterateMolecules(*_timeIntegrator, false);
205
206 // sort molecules into linked cells
207 _moleculeService->iterateMolecules(*_updateLinkedCellListsMapping, false);
208
209 if (_parallelTopologyService->getProcessCoordinates() == tarch::la::Vector<MD_DIM, unsigned int>(0)) {
210 // if(t%50==0) std::cout <<"Finish MD timestep " << t << "..." << std::endl;
211 }
212 }
213
214 virtual void sortMoleculesIntoCells() {
215 // nop required, since the linked cells are very tightly linked to mamico
216 }
217
218 virtual void setMacroscopicCellService(coupling::services::MacroscopicCellService<MDSIMULATIONFACTORY_DIMENSION>* macroscopicCellService) {
219 _macroscopicCellService = macroscopicCellService;
220 // set the cell service also in singleton of mamico interface provider ->
221 // typically not required in coupling, but makes the simulation state more
222 // consistent compared to using LAMMPS
223 coupling::interface::MamicoInterfaceProvider<simplemd::LinkedCell, MDSIMULATIONFACTORY_DIMENSION>::getInstance().setMacroscopicCellService(
224 macroscopicCellService);
225 }

228
229 virtual void init(const tarch::utils::MultiMDService<MDSIMULATIONFACTORY_DIMENSION>& multiMDService, unsigned int localMDSimulation) {
230 initServices(multiMDService, localMDSimulation);
231 }
232
233 virtual void shutdown() { shutdownServices(); }
234
235 virtual void writeCheckpoint(const std::string& filestem, const unsigned int& t) {
236 getMoleculeService().writeCheckPoint(getParallelTopologyService(), filestem, t);
237 }

240 // called from factory
241 simplemd::BoundaryTreatment& getBoundaryTreatment() { return *_boundaryTreatment; }
242 simplemd::services::ParallelTopologyService& getParallelTopologyService() { return *_parallelTopologyService; }
243 simplemd::services::MoleculeService& getMoleculeService() { return *_moleculeService; }
244 simplemd::services::LinkedCellService& getLinkedCellService() { return *_linkedCellService; }
245 const simplemd::services::MolecularPropertiesService& getMolecularPropertiesService() { return *_molecularPropertiesService; }

288 // switch on coupling -> add mamico fix with ID=2; we further check
289 // correctness of number of cells
290 // -> afterwards, the MDSolverInterface of mamico will be initialised and
291 // provided via MamicoInterfaceProvider.
292 virtual void switchOnCoupling() {
293 std::stringstream ss;
294 unsigned int localNumberCells = 1;
295 const tarch::la::Vector<MDSIMULATIONFACTORY_DIMENSION, unsigned int> numberProcs = _configuration.getMPIConfiguration().getNumberOfProcesses();
296 const tarch::la::Vector<MDSIMULATIONFACTORY_DIMENSION, double> domainSize = _configuration.getDomainConfiguration().getGlobalDomainSize();
297 const tarch::la::Vector<MDSIMULATIONFACTORY_DIMENSION, double> meshsize = _mamicoConfiguration.getMacroscopicCellConfiguration().getMacroscopicCellSize();
298 // determine max. number of cells required on each process
299 for (unsigned int d = 0; d < MDSIMULATIONFACTORY_DIMENSION; d++) {
300 const unsigned int cells = (unsigned int)ceil(domainSize[d] / meshsize[d] / numberProcs[d]);
301 if (!tarch::la::equals(cells * numberProcs[d] * meshsize[d], domainSize[d], _tolerance)) {
302 std::cout << "ERROR switchOnCoupling(): cells and "
303 "domainsize/meshsize/numberProcs do not match!"
304 << std::endl;
305 exit(EXIT_FAILURE);
306 }
307 // take cells+2 as we also need to allocate ghost layer
308 localNumberCells = localNumberCells * (cells + 2);
309 }
310
311 ss << "fix 2 all mamico " << localNumberCells << " 123456 " << _configuration.getDomainConfiguration().getCutoffRadius();
312 _lmp->input->one(ss.str().c_str());
313 }

361private:
362 void initSingleSimulation(MPI_Comm comm, unsigned int localMDSimulation) {
363 int rank;
364 MPI_Comm_rank(comm, &rank);
365 std::stringstream ss;
366 std::string inputFile;
367
368 // use logging separately for each LAMMPS instance
369 ss.str("");
370 ss << "log log.lammps" << localMDSimulation;
371 _lmp->input->one(ss.str().c_str());
372 ss.str("");
373 ss << "log log.lammps" << localMDSimulation << " append";
374 _lmp->input->one(ss.str().c_str());
375
376 // define lj atomic input
377
378 _lmp->input->one("units lj");
379 _lmp->input->one("atom_style atomic");
380 // define number of processes
381 ss.str("");
382 tarch::la::Vector<3, unsigned int> procs(1);
383 for (int d = 0; d < MDSIMULATIONFACTORY_DIMENSION; d++) {
384 procs[d] = _configuration.getMPIConfiguration().getNumberOfProcesses()[d];
385 }
386 ss << "processors " << procs[0] << " " << procs[1] << " " << procs[2] << " map xyz";
387 _lmp->input->one(ss.str().c_str());
388
389 // define 2D if necessary
390 if (MDSIMULATIONFACTORY_DIMENSION == 2) {
391 _lmp->input->one("dimension 2");
392 } else {
393 _lmp->input->one("dimension 3");
394 }
395
396 // incorporate boundary conditions: currently, only reflecting and periodic
397 // boundaries are supported; to be extended to 2D
398 ss.str("");
399 ss << "boundary";
400 if (MDSIMULATIONFACTORY_DIMENSION == 3) {
401 const tarch::la::Vector<MD_LINKED_CELL_NEIGHBOURS, simplemd::BoundaryType> bc = _configuration.getDomainConfiguration().getBoundary();
402 int indices[3] = {12, 10, 4};
403 for (int i = 0; i < MDSIMULATIONFACTORY_DIMENSION; i++) {
404 if (bc[indices[i]] == simplemd::PERIODIC_BOUNDARY) {
405 ss << " p";
406 } else if (bc[indices[i]] == simplemd::REFLECTING_BOUNDARY) {
407 ss << " f";
408 } else {
409 std::cout << "ERROR LammpsMDSimulation::initSingleSimulation(): only "
410 "reflecting and periodic boundaries supported!"
411 << std::endl;
412 exit(EXIT_FAILURE);
413 }
414 }
415 } else {
416 std::cout << "ERROR LammpsMDSimulation::initSingleSimulation(): only 3D "
417 "supported at the moment!"
418 << std::endl;
419 }
420 _lmp->input->one(ss.str().c_str());
421
422 // create atoms based on input configuration; currently, only creation of
423 // molecules on a lattice is supported -------------------
424 tarch::la::Vector<3, double> domainOffset(0.0);
425 for (int d = 0; d < MDSIMULATIONFACTORY_DIMENSION; d++) {
426 domainOffset[d] = _configuration.getDomainConfiguration().getGlobalDomainOffset()[d];
427 }
428 tarch::la::Vector<3, double> domainSize(1.0);
429 for (int d = 0; d < MDSIMULATIONFACTORY_DIMENSION; d++) {
430 domainSize[d] = _configuration.getDomainConfiguration().getGlobalDomainSize()[d];
431 }
432 if (!_configuration.getDomainConfiguration().initFromCheckpoint() && !_configuration.getDomainConfiguration().initFromSequentialCheckpoint()) {
433 tarch::la::Vector<3, unsigned int> moleculesPerDirection(0);
434 for (int d = 0; d < MDSIMULATIONFACTORY_DIMENSION; d++) {
435 moleculesPerDirection[d] = _configuration.getDomainConfiguration().getMoleculesPerDirection()[d];
436 }
437 double particleDensity = 1.0;
438 for (int d = 0; d < MDSIMULATIONFACTORY_DIMENSION; d++) {
439 particleDensity = particleDensity * _configuration.getDomainConfiguration().getMoleculesPerDirection()[d] / domainSize[d];
440 }
441 }
442 // --------------------------------------------------------------------------------------------------------------------------------------------
443
444 // define molecule and LJ parameters
445 ss.str("");
446 ss << "pair_style lj/cut " << _configuration.getDomainConfiguration().getCutoffRadius();
447 _lmp->input->one(ss.str().c_str());
448 _lmp->input->one("pair_modify shift yes");
449 ss.str("");
450 ss << "pair_coeff 1 1 " << _configuration.getMoleculeConfiguration().getEpsilon() << " ";
451 ss << _configuration.getMoleculeConfiguration().getSigma() << " " << _configuration.getDomainConfiguration().getCutoffRadius();
452 _lmp->input->one(ss.str().c_str());
453
454 // set time step
455 ss.str("");
456 ss << "timestep " << _configuration.getSimulationConfiguration().getDt();
457 _lmp->input->one(ss.str().c_str());
458
459 // set some verlet-specific values; currently hard-coded
460 _lmp->input->one("neighbor 0.5 bin");
461 _lmp->input->one("neigh_modify delay 0 every 20 check no");
462
463 // write output; though this is not vtk format, we use this config for that purpose
464 if (_configuration.getVTKConfiguration().getWriteEveryTimestep() != 0) {
465 ss.str("");
466 ss << "dump myDump all custom " << _configuration.getVTKConfiguration().getWriteEveryTimestep() << " ";
467 ss << _configuration.getVTKConfiguration().getFilename() << "_" << localMDSimulation << "_"
468 << "*.dat id type x y z vx vy vz";
469 _lmp->input->one(ss.str().c_str());
470 }
471
472 // define simulation region
473 ss.str("");
474 ss << "region box block " << domainOffset[0] << " " << (domainOffset[0] + domainSize[0]) << " " << domainOffset[1] << " "
475 << (domainOffset[1] + domainSize[1]) << " ";
476 ss << domainOffset[2] << " " << (domainOffset[2] + domainSize[2]) << " units box";
477 _lmp->input->one(ss.str().c_str());
478
479 if (_configuration.getDomainConfiguration().initFromCheckpoint() || _configuration.getDomainConfiguration().initFromSequentialCheckpoint()) {
480 std::stringstream command;
481 command << "read_restart " << _configuration.getDomainConfiguration().getCheckpointFilestem() << ".restart";
482 _lmp->input->one(command.str().c_str());
483 } else {
484 _lmp->input->one("create_box 1 box");
485 _lmp->input->one("create_atoms 1 box");
486 // set mass and correct temperature
487 ss.str("");
488 ss << "mass 1 " << _configuration.getMoleculeConfiguration().getMass();
489 _lmp->input->one(ss.str().c_str());
490 // initialise velocities on molecules; for parallel simulations, we do
491 // this according to global MD simulation number (we add 1 as 0 is not a
492 // valid seed)
493 ss.str("");
494 ss << "velocity all create " << _configuration.getMoleculeConfiguration().getTemperature() << " " << localMDSimulation + 1 << " loop geom";
495 _lmp->input->one(ss.str().c_str());
496 }
497
498 // incorporate reflecting boundaries, pt.2 (this needs to be done after the
499 // box was created); to be extended to 2D
500 if (MDSIMULATIONFACTORY_DIMENSION == 3) {
501 const tarch::la::Vector<MD_LINKED_CELL_NEIGHBOURS, simplemd::BoundaryType> bc = _configuration.getDomainConfiguration().getBoundary();
502 int indices[3] = {12, 10, 4};
503 std::string faces[3] = {"x", "y", "z"};
504 for (int i = 0; i < MDSIMULATIONFACTORY_DIMENSION; i++) {
505 if (bc[indices[i]] == simplemd::REFLECTING_BOUNDARY) {
506 ss.str("");
507 ss << "fix reflection" << faces[i] << " all wall/reflect " << faces[i] << "lo EDGE " << faces[i] << "hi EDGE units box";
508 _lmp->input->one(ss.str().c_str());
509 }
510 }
511 }
512 // --------------------------------------------------------------------------------------------------------------------------------------------
513
514 // define molecule and LJ parameters
515 ss.str("");
516 ss << "pair_style lj/cut " << _configuration.getDomainConfiguration().getCutoffRadius();
517 _lmp->input->one(ss.str().c_str());
518 _lmp->input->one("pair_modify shift yes");
519 ss.str("");
520 ss << "pair_coeff 1 1 " << _configuration.getMoleculeConfiguration().getEpsilon() << " ";
521 ss << _configuration.getMoleculeConfiguration().getSigma() << " " << _configuration.getDomainConfiguration().getCutoffRadius();
522 _lmp->input->one(ss.str().c_str());
523
524 // set time step
525 ss.str("");
526 ss << "timestep " << _configuration.getSimulationConfiguration().getDt();
527 _lmp->input->one(ss.str().c_str());
528
529 // set some verlet-specific values; currently hard-coded
530 _lmp->input->one("neighbor 0.5 bin");
531 _lmp->input->one("neigh_modify delay 0 every 1 check no");
532
533 // write output; though this is not vtk format, we use this config for that
534 // purpose
535 if (_configuration.getVTKConfiguration().getWriteEveryTimestep() != 0) {
536 ss.str("");
537 ss << "dump myDump all custom " << _configuration.getVTKConfiguration().getWriteEveryTimestep() << " ";
538 ss << _configuration.getVTKConfiguration().getFilename() << "_" << localMDSimulation << "_"
539 << "*.dat id type x y z vx vy vz";
540 _lmp->input->one(ss.str().c_str());
541 }
542 // write checkpointing
543 if (_configuration.getCheckpointConfiguration().getWriteEveryTimestep() != 0) {
544 ss.str("");
545 ss << "restart " << _configuration.getCheckpointConfiguration().getWriteEveryTimestep() << " ";
546 ss << _configuration.getCheckpointConfiguration().getFilename() << "_" << localMDSimulation << "_"
547 << "1.restart " << _configuration.getCheckpointConfiguration().getFilename() << "_" << localMDSimulation << "_"
548 << "2.restart";
549 _lmp->input->one(ss.str().c_str());
550 }
551 // set nve ensemble
552 _lmp->input->one("fix 1 all nve");
553 }
554};
555#endif

590 const tarch::la::Vector<MDSIMULATIONFACTORY_DIMENSION, double> domainSize = _configuration.getDomainConfiguration().getGlobalDomainSize();
591 const tarch::la::Vector<MDSIMULATIONFACTORY_DIMENSION, double> meshsize = _mamicoConfiguration.getMacroscopicCellConfiguration().getMacroscopicCellSize();
592 // determine max. number of cells required on each process
593 for (unsigned int d = 0; d < MDSIMULATIONFACTORY_DIMENSION; d++) {
594 const unsigned int cells = (unsigned int)ceil(domainSize[d] / meshsize[d] / numberProcs[d]);
595 if (!tarch::la::equals(cells * numberProcs[d] * meshsize[d], domainSize[d], _tolerance)) {
596 std::cout << "ERROR switchOnCoupling(): cells and "
597 "domainsize/meshsize/numberProcs do not match!"
598 << std::endl;
599 exit(EXIT_FAILURE);
600 }
601 // take cells+2 as we also need to allocate ghost layer
602 localNumberCells = localNumberCells * (cells + 2);
603 }
604
605 ss << "fix 2 all mamico " << localNumberCells << " 123456 " << _configuration.getDomainConfiguration().getCutoffRadius();
606 _lmp->input->one(ss.str().c_str());
607 }

656 void initSingleSimulation(MPI_Comm comm, unsigned int localMDSimulation) {
657 int rank;
658 MPI_Comm_rank(comm, &rank);
659 std::stringstream ss;
660 std::string inputFile;
661 // particle insertion is currently only supported for single-centered LJ
662 // particles
663 if (_mamicoConfiguration.getParticleInsertionConfiguration().getParticleInsertionType() !=
664 coupling::configurations::ParticleInsertionConfiguration::NO_INSERTION) {
665 std::cout << "ERROR coupling::LammpsDPDSImulation::initSingleSimulation(): "
666 "Particle insertion currently not supported!"
667 << std::endl;
668 exit(EXIT_FAILURE);
669 }
670
671 // set logs for each LAMMPS instance separately
672 ss.str("");
673 ss << "log log.lammps" << localMDSimulation;
674 _lmp->input->one(ss.str().c_str());
675 ss.str("");
676 ss << "log log.lammps" << localMDSimulation << " append";
677 _lmp->input->one(ss.str().c_str());
678 _lmp->input->one("variable ndim equal 3");
679 ss.str("");
680 ss << "variable kB equal " << _configuration.getDomainConfiguration().getKB();
681 _lmp->input->one(ss.str().c_str());
682 ss.str("");
683 ss << "variable kBT equal " << _configuration.getMoleculeConfiguration().getTemperature() * _configuration.getDomainConfiguration().getKB();
684 _lmp->input->one(ss.str().c_str());
685 ss.str("");
686 ss << "variable T equal " << _configuration.getMoleculeConfiguration().getTemperature();
687 _lmp->input->one(ss.str().c_str());
688 ss.str("");
689 ss << "variable rc equal " << _configuration.getDomainConfiguration().getCutoffRadius();
690 _lmp->input->one(ss.str().c_str());
691 _lmp->input->one("units lj");
692 _lmp->input->one("dimension ${ndim}");
693
694 // set up boundary treatment, Pt1: for "periodic", we just set the
695 // "p"-option and "f" otherwise; for the latter, we further have to
696 // introduce the respective walls, see boundary treatment, Pt2, further down
697 ss.str("");
698 ss << "boundary";
699 if (MDSIMULATIONFACTORY_DIMENSION == 3) {
700 const tarch::la::Vector<MD_LINKED_CELL_NEIGHBOURS, simplemd::BoundaryType> bc = _configuration.getDomainConfiguration().getBoundary();
701 int indices[3] = {12, 10, 4};
702 for (int i = 0; i < MDSIMULATIONFACTORY_DIMENSION; i++) {
703 if (bc[indices[i]] == simplemd::PERIODIC_BOUNDARY) {
704 ss << " p";
705 } else if (bc[indices[i]] == simplemd::REFLECTING_BOUNDARY) {
706 ss << " f";
707 } else {
708 std::cout << "ERROR LammpsDPDSimulation::initSingleSimulation(): "
709 "only reflecting and periodic boundaries supported!"
710 << std::endl;
711 exit(EXIT_FAILURE);
712 }
713 }
714 } else {
715 std::cout << "ERROR LammpsDPDSimulation::initSingleSimulation(): only 3D "
716 "supported at the moment!"
717 << std::endl;
718 }
719 _lmp->input->one(ss.str().c_str());
720
721 _lmp->input->one("atom_style atomic");
722 // define number of processes with xyz-topology
723 ss.str("");
724 ss << "processors " << _configuration.getMPIConfiguration().getNumberOfProcesses()[0] << " "
725 << _configuration.getMPIConfiguration().getNumberOfProcesses()[1] << " ";
726 ss << _configuration.getMPIConfiguration().getNumberOfProcesses()[2] << " map xyz";
727 _lmp->input->one(ss.str().c_str());
728 ss.str("");
729 ss << "variable seed1 equal " << localMDSimulation + rank + 1;
730 _lmp->input->one(ss.str().c_str()); // currently fixed random seed
731 ss.str("");
732 ss << "variable seed2 equal " << localMDSimulation + rank + 10;
733 _lmp->input->one(ss.str().c_str()); // new seed for velocity generation;
734 // currently deduced from _randSeed+10
735 _lmp->input->one("print \"seed1: ${seed1}\"");
736 _lmp->input->one("print \"seed2: ${seed2}\"");
737 _lmp->input->one("pair_style dpd/general ${kBT} ${rc} ${seed1}"); // DPD scheme with
738 // forcing by Xin
739
740 const tarch::la::Vector<MDSIMULATIONFACTORY_DIMENSION, double> domainSize(_configuration.getDomainConfiguration().getGlobalDomainSize());
741 const tarch::la::Vector<MDSIMULATIONFACTORY_DIMENSION, double> domainOffset(_configuration.getDomainConfiguration().getGlobalDomainOffset());
742 ss.str("");
743 ss << "variable Lx equal " << domainSize[0];
744 _lmp->input->one(ss.str().c_str());
745 ss.str("");
746 ss << "variable Ly equal " << domainSize[1];
747 _lmp->input->one(ss.str().c_str());
748 ss.str("");
749 ss << "variable Lz equal " << domainSize[2];
750 _lmp->input->one(ss.str().c_str());
751 // The following command computes the average density based on
752 // molecules-per-direction. However, particles will be randomly distributed
753 // - they are not arranged on a grid
754 const tarch::la::Vector<MDSIMULATIONFACTORY_DIMENSION, unsigned int> molecules(_configuration.getDomainConfiguration().getMoleculesPerDirection());
755 unsigned int numberMolecules = 1;
756 for (unsigned int d = 0; d < MDSIMULATIONFACTORY_DIMENSION; d++) {
757 numberMolecules = numberMolecules * molecules[d];
758 }
759 ss.str("");
760 ss << "variable Npart equal " << numberMolecules;
761 _lmp->input->one(ss.str().c_str());
762 _lmp->input->one("variable Ntype equal 1");
763
764 ss.str("");
765 ss << "region region_dpd block " << domainOffset[0] << " " << domainOffset[0] + domainSize[0] << " " << domainOffset[1] << " "
766 << domainOffset[1] + domainSize[1] << " ";
767 ss << domainOffset[2] << " " << domainOffset[2] + domainSize[2] << " units box";
768 _lmp->input->one(ss.str().c_str());
769 _lmp->input->one("create_box ${Ntype} region_dpd");
770 _lmp->input->one("create_atoms 1 random ${Npart} 12345 NULL"); // new randomized
771 // initialisation
772
773 // boundary treatment, Pt2: introduce reflecting walls if required (fixes
774 // need to be defined after box definition)
775 if (MDSIMULATIONFACTORY_DIMENSION == 3) {
776 const tarch::la::Vector<MD_LINKED_CELL_NEIGHBOURS, simplemd::BoundaryType> bc = _configuration.getDomainConfiguration().getBoundary();
777 int indices[3] = {12, 10, 4};
778 if (bc[indices[0]] == simplemd::REFLECTING_BOUNDARY) {
779 _lmp->input->one("fix reflectionX all wall/reflect xlo EDGE xhi EDGE");
780 }
781 if (bc[indices[1]] == simplemd::REFLECTING_BOUNDARY) {
782 _lmp->input->one("fix reflectionY all wall/reflect ylo EDGE yhi EDGE");
783 }
784 if (bc[indices[2]] == simplemd::REFLECTING_BOUNDARY) {
785 _lmp->input->one("fix reflectionZ all wall/reflect zlo EDGE zhi EDGE");
786 }
787 }
788
789 ss.str("");
790 ss << "variable m equal " << _configuration.getMoleculeConfiguration().getMass();
791 _lmp->input->one(ss.str().c_str());
792
793 ss.str("");
794 ss << "variable alpha equal " << _configuration.getMoleculeConfiguration().getEpsilon(); // configurable via epsilon parameter in xml config
795 _lmp->input->one(ss.str().c_str());
796 ss.str("");
797 ss << "variable sigma equal " << _configuration.getMoleculeConfiguration().getSigma();
798 _lmp->input->one(ss.str().c_str());
799 _lmp->input->one("variable gamma equal ${sigma}^2/(2*${kBT})");
800
801 _lmp->input->one("variable power equal 0.25");
802
803 ss.str("");
804 ss << "variable Ntime equal " << _configuration.getSimulationConfiguration().getNumberOfTimesteps();
805 _lmp->input->one(ss.str().c_str());
806 ss.str("");
807 ss << "variable Nrepeat equal " << (int)(0.8 * _configuration.getSimulationConfiguration().getNumberOfTimesteps());
808 _lmp->input->one(ss.str().c_str());
809
810 _lmp->input->one("pair_coeff 1 1 ${alpha} ${gamma} ${power} ${rc}");
811
812 _lmp->input->one("neighbor 0.5 bin");
813 _lmp->input->one("neigh_modify delay 0 every 1 check no");
814
815 _lmp->input->one("communicate single vel yes");
816 _lmp->input->one("group extra_atoms id > ${Npart}"); // remove particles to
817 // reach correct
818 // density
819 _lmp->input->one("delete_atoms group extra_atoms");
820
821 ss.str("");
822 ss << "timestep " << _configuration.getSimulationConfiguration().getDt();
823 _lmp->input->one(ss.str().c_str());
824
825 _lmp->input->one("mass 1 ${m}");
826 _lmp->input->one("velocity all create ${kBT} ${seed2} loop geom");
827
828 _lmp->input->one("thermo 1");
829 _lmp->input->one("thermo_style custom step time ke temp press");
830 _lmp->input->one("thermo_modify flush yes");
831
832 // wall fix bottom
833 if (_mamicoConfiguration.getBoundaryForceConfiguration().getBoundary()[4]) {
834 if (_configuration.getDomainConfiguration().getBoundary()[4] != simplemd::REFLECTING_BOUNDARY) {
835 std::cout << "ERROR "
836 "coupling::MDSimulationFactory::LammpsDPDSimulation::"
837 "initSingleSimulation(): bottom boundary is not reflecting!"
838 << std::endl;
839 exit(EXIT_FAILURE);
840 }
841 ss.str("");
842 ss << "fix lower_wall all dpd/boundary/force ${rc} 100 dpd_plane_fc.dat "
843 "dpd_plane_fd1.dat dpd_plane_fd2.dat 2 0 "
844 << domainOffset[2] << " 0.0";
845 _lmp->input->one(ss.str().c_str());
846 }
847 // wall fix top
848 if (_mamicoConfiguration.getBoundaryForceConfiguration().getBoundary()[5]) {
849 if (_configuration.getDomainConfiguration().getBoundary()[21] != simplemd::REFLECTING_BOUNDARY) {
850 std::cout << "ERROR "
851 "coupling::MDSimulationFactory::LammpsDPDSimulation::"
852 "initSingleSimulation(): top boundary is not reflecting!"
853 << std::endl;
854 exit(EXIT_FAILURE);
855 }
856 ss.str("");
857 ss << "fix upper_wall all dpd/interface/force ${rc} 100 dpd_plane_fc.dat "
858 "dpd_plane_fd1.dat dpd_plane_fd2.dat 2 1 "
859 << domainOffset[2] + domainSize[2] << " 0.0";
860 _lmp->input->one(ss.str().c_str());
861 }
862
863 _lmp->input->one("fix 1 all nve");
864 }
