52 void handleCell(LinkedCell& cell, const unsigned int& cellIndex) {
53 // return, if we already deleted the respective particle
54 if (_particleCounter > _particle) {
55 return;
56 }
57
58 // otherwise: loop over particles
59 coupling::interface::MoleculeIterator<LinkedCell, dim>* it = _mdSolverInterface->getMoleculeIterator(cell);
60 it->begin();
61 while (it->continueIteration()) {
62 // if we reached the respective particle, delete it
63 if (_particleCounter == _particle) {
64 const coupling::interface::Molecule<dim>& myMolecule(it->getConst());
65
66 // save copy of particle
67 const tarch::la::Vector<dim, double> position = myMolecule.getPosition();
68 const tarch::la::Vector<dim, double> velocity = myMolecule.getVelocity();
69 const tarch::la::Vector<dim, double> force = myMolecule.getForce();
70 const double potentialEnergy = myMolecule.getPotentialEnergy();
71 _deletedMoleculeCopy = coupling::datastructures::Molecule<dim>(position, velocity, force, potentialEnergy);
72
73 // delete molecule from MD simulation
74 _mdSolverInterface->deleteMoleculeFromMDSimulation(myMolecule, cell);
75 _particleCounter++;
76 break;
77 }
78 _particleCounter++;
79
80 it->next();
81 }
82 delete it;
83 }
