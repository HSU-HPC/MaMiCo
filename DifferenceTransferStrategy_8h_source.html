67 virtual void processInnerMacroscopicCellAfterReceivingMacroscopicSolverData(coupling::datastructures::MacroscopicCellWithLinkedCells<LinkedCell, dim>& cell,
68 const unsigned int& index) {
69 // compute difference between macroscopic and microscopic mass and momentum
70 // values; this value is set in the microscopic data buffer of the
71 // macroscopic cell.
72 const double diffMass = cell.getMicroscopicMass() - cell.getMacroscopicMass();
73 const tarch::la::Vector<dim, double> diffMomentum = cell.getMicroscopicMomentum() - cell.getMacroscopicMomentum();
74 cell.setMicroscopicMass(diffMass);
75 cell.setMicroscopicMomentum(diffMomentum);
76 // reset macroscopic-buffers
77 cell.setMacroscopicMass(0.0);
78 cell.setMacroscopicMomentum(_zero);
79 }
80
85 virtual void processInnerMacroscopicCellBeforeSendingMDSolverData(coupling::datastructures::MacroscopicCellWithLinkedCells<LinkedCell, dim>& cell,
86 const unsigned int& index) {
87 // average quantities
88 const double mass = cell.getMacroscopicMass() / ((double)_numberMDsteps);
89 const tarch::la::Vector<dim, double> momentum = cell.getMacroscopicMomentum() * (1.0 / ((double)_numberMDsteps));
90 cell.setMacroscopicMass(mass);
91 cell.setMacroscopicMomentum(momentum);
92 }
93
98 virtual void processInnerMacroscopicCellAfterMDTimestep(coupling::datastructures::MacroscopicCellWithLinkedCells<LinkedCell, dim>& cell,
99 const unsigned int& index) {
100 cell.iterateConstCells(_massMapping);
101 const double mass = _massMapping.getMass();
102 cell.iterateConstCells(_momentumMapping);
103 const tarch::la::Vector<dim, double> momentum = _momentumMapping.getMomentum();
104 cell.addMacroscopicMass(mass);
105 cell.addMacroscopicMomentum(momentum);
106 }
107
108private:
110 const unsigned int _numberMDsteps;

115 coupling::cellmappings::ComputeMassMapping<LinkedCell, dim> _massMapping;
