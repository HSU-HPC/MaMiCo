71 virtual void processInnerMacroscopicCellAfterReceivingMacroscopicSolverData(
72 coupling::datastructures::MacroscopicCellWithLinkedCells<LinkedCell,dim> &cell, const unsigned int &index
73 ){
74 // compute difference between macroscopic and microscopic mass and momentum values; this value is
75 // set in the microscopic data buffer of the macroscopic cell.
76 const double diffMass = cell.getMicroscopicMass() - cell.getMacroscopicMass();
77 const tarch::la::Vector<dim,double> diffMomentum = cell.getMicroscopicMomentum() - cell.getMacroscopicMomentum();
78 cell.setMicroscopicMass(diffMass);
79 cell.setMicroscopicMomentum(diffMomentum);
80 // reset macroscopic-buffers
81 cell.setMacroscopicMass(0.0);
82 cell.setMacroscopicMomentum(_zero);
83 }

88 virtual void processInnerMacroscopicCellBeforeSendingMDSolverData(
89 coupling::datastructures::MacroscopicCellWithLinkedCells<LinkedCell,dim> &cell, const unsigned int &index
90 ){
91 // average quantities
92 const double mass = cell.getMacroscopicMass()/((double)_numberMDsteps);
93 const tarch::la::Vector<dim,double> momentum = cell.getMacroscopicMomentum()*(1.0/((double)_numberMDsteps));
94 cell.setMacroscopicMass(mass);
95 cell.setMacroscopicMomentum(momentum);
96 }
97
101 virtual void processInnerMacroscopicCellAfterMDTimestep(
102 coupling::datastructures::MacroscopicCellWithLinkedCells<LinkedCell,dim> &cell, const unsigned int &index
103 ){
104 cell.iterateConstCells(_massMapping);
105 const double mass = _massMapping.getMass();
106 cell.iterateConstCells(_momentumMapping);
107 const tarch::la::Vector<dim,double> momentum = _momentumMapping.getMomentum();
108 cell.addMacroscopicMass(mass);
109 cell.addMacroscopicMomentum(momentum);
110 }
111
112 private:

121};
