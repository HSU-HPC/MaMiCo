154 void advance(double dt) override {
155 if (skipRank()) {
156 return;
157 }
158 const int timesteps = floor(dt / _dt + 0.5);
159 if (fabs(timesteps * _dt - dt) / _dt > 1.0e-8) {
160 std::cout
161 << "ERROR LBCouetteSolver::advance(): time steps and dt do not match!"
162 << std::endl;
163 exit(EXIT_FAILURE);
164 }
165 for (int i = 0; i < timesteps; i++) {
166 plot();
167 collidestream();
168 communicate(); // exchange between neighbouring MPI subdomains
169 _counter++;
170 }

179 void setMDBoundaryValues(
180 std::vector<coupling::datastructures::MacroscopicCell<3> *> &recvBuffer,
181 const unsigned int *const recvIndices,
182 const coupling::IndexConversion<3> &indexConversion) override {
183 if (skipRank()) {
184 return;
185 }
186 // loop over all received cells
187 const unsigned int size = (unsigned int)recvBuffer.size();
188 for (unsigned int i = 0; i < size; i++) {
189 // determine cell index of this cell in LB domain
190 tarch::la::Vector<3, unsigned int> globalCellCoords =
191 indexConversion.getGlobalVectorCellIndex(recvIndices[i]);
192 globalCellCoords[0] =
193 (globalCellCoords[0] + _offset[0]) - _coords[0] * _avgDomainSizeX;
194 globalCellCoords[1] =
195 (globalCellCoords[1] + _offset[1]) - _coords[1] * _avgDomainSizeY;
196 globalCellCoords[2] =
197 (globalCellCoords[2] + _offset[2]) - _coords[2] * _avgDomainSizeZ;
198#if (COUPLING_MD_DEBUG == COUPLING_MD_YES)
199 std::cout << "Process coords: " << _coords
200 << ": GlobalCellCoords for index "
201 << indexConversion.getGlobalVectorCellIndex(recvIndices[i])
202 << ": " << globalCellCoords << std::endl;
203#endif
204 const int index =
205 get(globalCellCoords[0], globalCellCoords[1], globalCellCoords[2]);
206#if (COUPLING_MD_DEBUG == COUPLING_MD_YES)
207 if (_flag[index] != MD_BOUNDARY) {
208 std::cout << "ERROR LBCouetteSolver::setMDBoundaryValues(): Cell "
209 << index << " is no MD boundary cell!" << std::endl;
210 exit(EXIT_FAILURE);
211 }
212#endif
213 // set velocity value and pdfs in MD boundary cell (before streaming); the
214 // boundary velocities are interpolated between the neighbouring and this
215 // cell. This interpolation is valid for FLUID-MD_BOUNDARY neighbouring
216 // relations only. determine local velocity received from MaMiCo and
217 // convert it to LB units; store the velocity in _vel
218 tarch::la::Vector<3, double> localVel(
219 (1.0 / recvBuffer[i]->getMacroscopicMass()) * (_dt / _dx) *
220 recvBuffer[i]->getMacroscopicMomentum());
221 for (unsigned int d = 0; d < 3; d++) {
222 _vel[3 * index + d] = localVel[d];
223 }
224 // loop over all pdfs and set them according to interpolated moving-wall
225 // conditions
226 for (unsigned int q = 0; q < 19; q++) {
227 // index of neighbour cell; only if cell is located inside local domain
228 if (((int)globalCellCoords[0] + _C[q][0] > 0) &&
229 ((int)globalCellCoords[0] + _C[q][0] < _domainSizeX + 1) &&
230 ((int)globalCellCoords[1] + _C[q][1] > 0) &&
231 ((int)globalCellCoords[1] + _C[q][1] < _domainSizeY + 1) &&
232 ((int)globalCellCoords[2] + _C[q][2] > 0) &&
233 ((int)globalCellCoords[2] + _C[q][2] < _domainSizeZ + 1)) {
234 const int nbIndex = get((_C[q][0] + globalCellCoords[0]),
235 (_C[q][1] + globalCellCoords[1]),
236 (_C[q][2] + globalCellCoords[2]));
237 const tarch::la::Vector<3, double> interpolVel(
238 0.5 * (_vel[3 * index] + _vel[3 * nbIndex]),
239 0.5 * (_vel[3 * index + 1] + _vel[3 * nbIndex + 1]),
240 0.5 * (_vel[3 * index + 2] + _vel[3 * nbIndex + 2]));
241 _pdf1[19 * index + q] = _pdf1[19 * nbIndex + 18 - q] -
242 6.0 * _W[q] * _density[nbIndex] *
243 (_C[18 - q][0] * interpolVel[0] +
244 _C[18 - q][1] * interpolVel[1] +
245 _C[18 - q][2] * interpolVel[2]);
246 }
247 }
248 }

254 tarch::la::Vector<3, double>
255 getVelocity(tarch::la::Vector<3, double> pos) const override {
256 tarch::la::Vector<3, unsigned int> coords;
257 const tarch::la::Vector<3, double> domainOffset(
258 _coords[0] * _dx * _avgDomainSizeX, _coords[1] * _dx * _avgDomainSizeY,
259 _coords[2] * _dx * _avgDomainSizeZ);
260 // check pos-data for process locality (todo: put this in debug mode in
261 // future releases)
262 if ((pos[0] < domainOffset[0]) ||
263 (pos[0] > domainOffset[0] + _domainSizeX * _dx) ||
264 (pos[1] < domainOffset[1]) ||
265 (pos[1] > domainOffset[1] + _domainSizeY * _dx) ||
266 (pos[2] < domainOffset[2]) ||
267 (pos[2] > domainOffset[2] + _domainSizeZ * _dx)) {
268 std::cout << "ERROR LBCouetteSolver::getVelocity(): Position " << pos
269 << " out of range!" << std::endl;
270 exit(EXIT_FAILURE);
271 }
272 // compute index for respective cell (_dx+... for ghost cells); use coords
273 // to store local cell coordinates
274 for (unsigned int d = 0; d < 3; d++) {
275 coords[d] = (unsigned int)((_dx + pos[d] - domainOffset[d]) / _dx);
276 }
277 const int index = get(coords[0], coords[1], coords[2]);
278 tarch::la::Vector<3, double> vel(0.0);
279 // extract and scale velocity to "real"=MD units
280 for (int d = 0; d < 3; d++) {
281 vel[d] = _dx / _dt * _vel[3 * index + d];
282 }
283#if (COUPLING_MD_DEBUG == COUPLING_MD_YES)
284 std::cout << "Position " << pos << " corresponds to cell: " << coords
285 << "; vel=" << vel << std::endl;
286#endif
287 return vel;
288 }

329 void collidestream() {
330#pragma omp parallel for
331 for (int z = 1; z < _domainSizeZ + 1; z++) {
332 for (int y = 1; y < _domainSizeY + 1; y++) {
333 for (int x = 1; x < _domainSizeX + 1; x++) {
334 const int index = get(x, y, z);
335 if (_flag[index] == FLUID) {
336 stream(index);
337 collide(index, x, y, z);
338 }
339 }
340 }
341 }
342 // swap fields
343 double *swap = _pdf1;
344 _pdf1 = _pdf2;
345 _pdf2 = swap;
346 }

349 void stream(int index) {
350 const int pI = 19 * index;
351 for (int q = 0; q < 9; q++) {
352 const int nb = 19 * (_C[q][0] + _C[q][1] * _xO + _C[q][2] * _yO);
353 _pdf2[pI + q] = _pdf1[pI + q - nb];
354 _pdf2[pI + 18 - q] = _pdf1[pI + 18 - q + nb];
355 }
356 _pdf2[pI + 9] = _pdf1[pI + 9];
357 }
358
360 void collide(int index, int x, int y, int z) {
361 // index of start of cell-local pdfs in AoS
362 const int pI = 19 * index;
363 // compute and store density, velocity
364 double *vel = &_vel[3 * index];
365 computeDensityAndVelocity(vel, _density[index], &_pdf2[pI]);
366 // collide (BGK); always handle pdfs no. q and inv(q)=18-q in one step
367 const double u2 =
368 1.0 - 1.5 * (vel[0] * vel[0] + vel[1] * vel[1] + vel[2] * vel[2]);
369 // pdf 0,18
370 double cu = -vel[1] - vel[2];
371 int nb = -_xO - _yO;
372 double feq = _W[0] * _density[index] * (u2 + 3.0 * cu + 4.5 * cu * cu);
373 _pdf2[pI] -= _omega * (_pdf2[pI] - feq);
374 boundary(_pdf2, pI, x, y, z, 0, _flag[index + nb], pI + 19 * nb);
375 feq -= 6.0 * _W[0] * _density[index] * cu;
376 _pdf2[pI + 18] -= _omega * (_pdf2[pI + 18] - feq);
377 boundary(_pdf2, pI, x, y, z, 18, _flag[index - nb], pI - 19 * nb);
378 // pdf 1,17
379 cu = -vel[0] - vel[2];
380 nb = -1 - _yO;
381 feq = _W[1] * _density[index] * (u2 + 3.0 * cu + 4.5 * cu * cu);
382 _pdf2[pI + 1] -= _omega * (_pdf2[pI + 1] - feq);
383 boundary(_pdf2, pI, x, y, z, 1, _flag[index + nb], pI + 19 * nb);
384 feq -= 6.0 * _W[1] * _density[index] * cu;
385 _pdf2[pI + 17] -= _omega * (_pdf2[pI + 17] - feq);
386 boundary(_pdf2, pI, x, y, z, 17, _flag[index - nb], pI - 19 * nb);
387 // pdf 2,16
388 cu = -vel[2];
389 nb = -_yO;
390 feq = _W[2] * _density[index] * (u2 + 3.0 * cu + 4.5 * cu * cu);
391 _pdf2[pI + 2] -= _omega * (_pdf2[pI + 2] - feq);
392 boundary(_pdf2, pI, x, y, z, 2, _flag[index + nb], pI + 19 * nb);
393 feq -= 6.0 * _W[2] * _density[index] * cu;
394 _pdf2[pI + 16] -= _omega * (_pdf2[pI + 16] - feq);
395 boundary(_pdf2, pI, x, y, z, 16, _flag[index - nb], pI - 19 * nb);
396 // pdf 3,15
397 cu = vel[0] - vel[2];
398 nb = 1 - _yO;
399 feq = _W[3] * _density[index] * (u2 + 3.0 * cu + 4.5 * cu * cu);
400 _pdf2[pI + 3] -= _omega * (_pdf2[pI + 3] - feq);
401 boundary(_pdf2, pI, x, y, z, 3, _flag[index + nb], pI + 19 * nb);
402 feq -= 6.0 * _W[3] * _density[index] * cu;
403 _pdf2[pI + 15] -= _omega * (_pdf2[pI + 15] - feq);
404 boundary(_pdf2, pI, x, y, z, 15, _flag[index - nb], pI - 19 * nb);
405 // pdf 4,14
406 cu = vel[1] - vel[2];
407 nb = _xO - _yO;
408 feq = _W[4] * _density[index] * (u2 + 3.0 * cu + 4.5 * cu * cu);
409 _pdf2[pI + 4] -= _omega * (_pdf2[pI + 4] - feq);
410 boundary(_pdf2, pI, x, y, z, 4, _flag[index + nb], pI + 19 * nb);
411 feq -= 6.0 * _W[4] * _density[index] * cu;
412 _pdf2[pI + 14] -= _omega * (_pdf2[pI + 14] - feq);
413 boundary(_pdf2, pI, x, y, z, 14, _flag[index - nb], pI - 19 * nb);
414 // pdf 5,13
415 cu = -vel[0] - vel[1];
416 nb = -1 - _xO;
417 feq = _W[5] * _density[index] * (u2 + 3.0 * cu + 4.5 * cu * cu);
418 _pdf2[pI + 5] -= _omega * (_pdf2[pI + 5] - feq);
419 boundary(_pdf2, pI, x, y, z, 5, _flag[index + nb], pI + 19 * nb);
420 feq -= 6.0 * _W[5] * _density[index] * cu;
421 _pdf2[pI + 13] -= _omega * (_pdf2[pI + 13] - feq);
422 boundary(_pdf2, pI, x, y, z, 13, _flag[index - nb], pI - 19 * nb);
423 // pdf 6,12
424 cu = -vel[1];
425 nb = -_xO;
426 feq = _W[6] * _density[index] * (u2 + 3.0 * cu + 4.5 * cu * cu);
427 _pdf2[pI + 6] -= _omega * (_pdf2[pI + 6] - feq);
428 boundary(_pdf2, pI, x, y, z, 6, _flag[index + nb], pI + 19 * nb);
429 feq -= 6.0 * _W[6] * _density[index] * cu;
430 _pdf2[pI + 12] -= _omega * (_pdf2[pI + 12] - feq);
431 boundary(_pdf2, pI, x, y, z, 12, _flag[index - nb], pI - 19 * nb);
432 // pdf 7,11
433 cu = vel[0] - vel[1];
434 nb = 1 - _xO;
435 feq = _W[7] * _density[index] * (u2 + 3.0 * cu + 4.5 * cu * cu);
436 _pdf2[pI + 7] -= _omega * (_pdf2[pI + 7] - feq);
437 boundary(_pdf2, pI, x, y, z, 7, _flag[index + nb], pI + 19 * nb);
438 feq -= 6.0 * _W[7] * _density[index] * cu;
439 _pdf2[pI + 11] -= _omega * (_pdf2[pI + 11] - feq);
440 boundary(_pdf2, pI, x, y, z, 11, _flag[index - nb], pI - 19 * nb);
441 // pdf 8,10
442 cu = -vel[0];
443 nb = -1;
444 feq = _W[8] * _density[index] * (u2 + 3.0 * cu + 4.5 * cu * cu);
445 _pdf2[pI + 8] -= _omega * (_pdf2[pI + 8] - feq);
446 boundary(_pdf2, pI, x, y, z, 8, _flag[index + nb], pI + 19 * nb);
447 feq -= 6.0 * _W[8] * _density[index] * cu;
448 _pdf2[pI + 10] -= _omega * (_pdf2[pI + 10] - feq);
449 boundary(_pdf2, pI, x, y, z, 10, _flag[index - nb], pI - 19 * nb);
450 // pdf 9
451 _pdf2[pI + 9] -= _omega * (_pdf2[pI + 9] - _W[9] * _density[index] * u2);
452 }

463 void boundary(double *const pdf, int index, int x, int y, int z, int q,
464 const Flag &flag, int nbIndex) {
465 if (flag != FLUID) {
466 if (flag == NO_SLIP) {
467 // half-way bounce back
468 pdf[nbIndex + 18 - q] = pdf[index + q];
469 } else if (flag == MOVING_WALL) {
470 // half-way bounce back + moving wall acceleration (only x-direction for
471 // wall supported at the moment)
472 pdf[nbIndex + 18 - q] =
473 pdf[index + q] -
474 6.0 * _W[q] * _density[index / 19] *
475 (_C[q][0] * _wallVelocity[0] + _C[q][1] * _wallVelocity[1] +
476 _C[q][2] * _wallVelocity[2]);
477 } else if (flag == PERIODIC) {
478 // periodic treatment
479 int target[3] = {x, y, z};
480 if (target[0] + _C[q][0] == 0) {
481 target[0] = _domainSizeX + 1;
482 } else if (target[0] + _C[q][0] == _domainSizeX + 1) {
483 target[0] = 0;
484 }
485 if (target[1] + _C[q][1] == 0) {
486 target[1] = _domainSizeY + 1;
487 } else if (target[1] + _C[q][1] == _domainSizeY + 1) {
488 target[1] = 0;
489 }
490 if (target[2] + _C[q][2] == 0) {
491 target[2] = _domainSizeZ + 1;
492 } else if (target[2] + _C[q][2] == _domainSizeZ + 1) {
493 target[2] = 0;
494 }
495 const int periodicNb =
496 target[0] +
497 (_domainSizeX + 2) * (target[1] + (_domainSizeY + 2) * target[2]);
498 pdf[19 * periodicNb + q] = pdf[index + q];
499 }
500 }

507 void computeDensityAndVelocity(double *const vel, double &density,
508 const double *const pdf) {
509 vel[0] = -(pdf[1] + pdf[5] + pdf[8] + pdf[11] + pdf[15]);
510 density = pdf[3] + pdf[7] + pdf[10] + pdf[13] + pdf[17];
511 vel[1] = (pdf[4] + pdf[11] + pdf[12] + pdf[13] + pdf[18]) -
512 (pdf[0] + pdf[5] + pdf[6] + pdf[7] + pdf[14]);
513 vel[0] = density + vel[0];
514 density = density + pdf[0] + pdf[1] + pdf[2] + pdf[4] + pdf[5] + pdf[6] +
515 pdf[8] + pdf[9] + pdf[11] + pdf[12] + pdf[14] + pdf[15] +
516 pdf[16] + pdf[18];
517 vel[2] = (pdf[14] + pdf[15] + pdf[16] + pdf[17] + pdf[18]) -
518 (pdf[0] + pdf[1] + pdf[2] + pdf[3] + pdf[4]);
519 vel[0] = vel[0] / density;
520 vel[1] = vel[1] / density;
521 vel[2] = vel[2] / density;
522 }

538 void communicatePart(double *pdf, double *sendBuffer, double *recvBuffer,
539 NbFlag nbFlagTo, NbFlag nbFlagFrom,
540 tarch::la::Vector<3, int> startSend,
541 tarch::la::Vector<3, int> endSend,
542 tarch::la::Vector<3, int> startRecv,
543 tarch::la::Vector<3, int> endRecv) {
544#if (COUPLING_MD_PARALLEL == COUPLING_MD_YES)
545 // directions that point to LEFT/RIGHT,... -> same ordering as enums!
546 const int directions[6][5] = {{1, 5, 8, 11, 15}, {3, 7, 10, 13, 17},
547 {4, 11, 12, 13, 18}, {0, 5, 6, 7, 14},
548 {0, 1, 2, 3, 4}, {14, 15, 16, 17, 18}};
549 MPI_Request requests[2];
550 MPI_Status status[2];
551 tarch::la::Vector<2, int> plane;
552 tarch::la::Vector<2, int> domainSize;
553 // find out plane coordinates
554 if (nbFlagTo == LEFT || nbFlagTo == RIGHT) {
555 plane[0] = 1;
556 plane[1] = 2;
557 domainSize[0] = _domainSizeY;
558 domainSize[1] = _domainSizeZ;
559 } else if (nbFlagTo == FRONT || nbFlagTo == BACK) {
560 plane[0] = 0;
561 plane[1] = 2;
562 domainSize[0] = _domainSizeX;
563 domainSize[1] = _domainSizeZ;
564 } else if (nbFlagTo == TOP || nbFlagTo == BOTTOM) {
565 plane[0] = 0;
566 plane[1] = 1;
567 domainSize[0] = _domainSizeX;
568 domainSize[1] = _domainSizeY;
569 } else {

580 sendBuffer[q + 5 * getParBuf(coords[plane[0]], coords[plane[1]],
581 domainSize[0], domainSize[1])] =

583 19 * get(coords[0], coords[1], coords[2])];

590 MPI_DOUBLE, _parallelNeighbours[nbFlagFrom], 1000, MPI_COMM_WORLD,
591 &requests[0]);
592 MPI_Isend(sendBuffer, (domainSize[0] + 2) * (domainSize[1] + 2) * 5,
593 MPI_DOUBLE, _parallelNeighbours[nbFlagTo], 1000, MPI_COMM_WORLD,
594 &requests[1]);
595 MPI_Waitall(2, requests, status);
596 // write data back to pdf field
597 if (_parallelNeighbours[nbFlagFrom] != MPI_PROC_NULL) {
598 for (coords[2] = startRecv[2]; coords[2] < endRecv[2]; coords[2]++) {
599 for (coords[1] = startRecv[1]; coords[1] < endRecv[1]; coords[1]++) {
600 for (coords[0] = startRecv[0]; coords[0] < endRecv[0]; coords[0]++) {
601 for (int q = 0; q < 5; q++) {
602 if (_flag[get(coords[0], coords[1], coords[2])] ==
603 PARALLEL_BOUNDARY) {
604 pdf[directions[nbFlagTo][q] +
605 19 * get(coords[0], coords[1], coords[2])] =
606 recvBuffer[q + 5 * getParBuf(coords[plane[0]],
607 coords[plane[1]],
608 domainSize[0], domainSize[1])];
609 }
610 }
611 }
612 }
613 }
614 }
