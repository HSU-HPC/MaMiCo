139 void advance(double dt) override {
140 if (skipRank()) {
141 return;
142 }
143 const int timesteps = floor(dt / _dt + 0.5);
144 if (fabs(timesteps * _dt - dt) / _dt > 1.0e-8) {
145 std::cout << "ERROR LBCouetteSolver::advance(): time steps and dt do not match!" << std::endl;
146 exit(EXIT_FAILURE);
147 }
148 for (int i = 0; i < timesteps; i++) {
149 plot();
150 collidestream();
151 communicate(); // exchange between neighbouring MPI subdomains
152 _counter++;
153 }

162 void setMDBoundaryValues(std::vector<coupling::datastructures::MacroscopicCell<3> *> &recvBuffer, const unsigned int *const recvIndices,
163 const coupling::IndexConversion<3> &indexConversion) override {
164 if (skipRank()) {
165 return;
166 }
167 // loop over all received cells
168 const unsigned int size = (unsigned int)recvBuffer.size();
169 for (unsigned int i = 0; i < size; i++) {
170 // determine cell index of this cell in LB domain
171 tarch::la::Vector<3, unsigned int> globalCellCoords = indexConversion.getGlobalVectorCellIndex(recvIndices[i]);
172 globalCellCoords[0] = (globalCellCoords[0] + _offset[0]) - _coords[0] * _avgDomainSizeX;
173 globalCellCoords[1] = (globalCellCoords[1] + _offset[1]) - _coords[1] * _avgDomainSizeY;
174 globalCellCoords[2] = (globalCellCoords[2] + _offset[2]) - _coords[2] * _avgDomainSizeZ;
175#if (COUPLING_MD_DEBUG == COUPLING_MD_YES)
176 std::cout << "Process coords: " << _coords << ": GlobalCellCoords for index " << indexConversion.getGlobalVectorCellIndex(recvIndices[i]) << ": "
177 << globalCellCoords << std::endl;
178#endif
179 const int index = get(globalCellCoords[0], globalCellCoords[1], globalCellCoords[2]);
180#if (COUPLING_MD_DEBUG == COUPLING_MD_YES)
181 if (_flag[index] != MD_BOUNDARY) {
182 std::cout << "ERROR LBCouetteSolver::setMDBoundaryValues(): Cell " << index << " is no MD boundary cell!" << std::endl;
183 exit(EXIT_FAILURE);
184 }
185#endif
186 // set velocity value and pdfs in MD boundary cell (before streaming); the
187 // boundary velocities are interpolated between the neighbouring and this
188 // cell. This interpolation is valid for FLUID-MD_BOUNDARY neighbouring
189 // relations only. determine local velocity received from MaMiCo and
190 // convert it to LB units; store the velocity in _vel
191 tarch::la::Vector<3, double> localVel((1.0 / recvBuffer[i]->getMacroscopicMass()) * (_dt / _dx) * recvBuffer[i]->getMacroscopicMomentum());
192 for (unsigned int d = 0; d < 3; d++) {
193 _vel[3 * index + d] = localVel[d];
194 }
195 // loop over all pdfs and set them according to interpolated moving-wall
196 // conditions
197 for (unsigned int q = 0; q < 19; q++) {
198 // index of neighbour cell; only if cell is located inside local domain
199 if (((int)globalCellCoords[0] + _C[q][0] > 0) && ((int)globalCellCoords[0] + _C[q][0] < _domainSizeX + 1) &&
200 ((int)globalCellCoords[1] + _C[q][1] > 0) && ((int)globalCellCoords[1] + _C[q][1] < _domainSizeY + 1) &&
201 ((int)globalCellCoords[2] + _C[q][2] > 0) && ((int)globalCellCoords[2] + _C[q][2] < _domainSizeZ + 1)) {
202 const int nbIndex = get((_C[q][0] + globalCellCoords[0]), (_C[q][1] + globalCellCoords[1]), (_C[q][2] + globalCellCoords[2]));
203 const tarch::la::Vector<3, double> interpolVel(0.5 * (_vel[3 * index] + _vel[3 * nbIndex]), 0.5 * (_vel[3 * index + 1] + _vel[3 * nbIndex + 1]),
204 0.5 * (_vel[3 * index + 2] + _vel[3 * nbIndex + 2]));
205 _pdf1[19 * index + q] =
206 _pdf1[19 * nbIndex + 18 - q] -
207 6.0 * _W[q] * _density[nbIndex] * (_C[18 - q][0] * interpolVel[0] + _C[18 - q][1] * interpolVel[1] + _C[18 - q][2] * interpolVel[2]);
208 }
209 }
210 }

216 tarch::la::Vector<3, double> getVelocity(tarch::la::Vector<3, double> pos) const override {
217 tarch::la::Vector<3, unsigned int> coords;
218 const tarch::la::Vector<3, double> domainOffset(_coords[0] * _dx * _avgDomainSizeX, _coords[1] * _dx * _avgDomainSizeY, _coords[2] * _dx * _avgDomainSizeZ);
219 // check pos-data for process locality (todo: put this in debug mode in
220 // future releases)
221 if ((pos[0] < domainOffset[0]) || (pos[0] > domainOffset[0] + _domainSizeX * _dx) || (pos[1] < domainOffset[1]) ||
222 (pos[1] > domainOffset[1] + _domainSizeY * _dx) || (pos[2] < domainOffset[2]) || (pos[2] > domainOffset[2] + _domainSizeZ * _dx)) {
223 std::cout << "ERROR LBCouetteSolver::getVelocity(): Position " << pos << " out of range!" << std::endl;
224 exit(EXIT_FAILURE);
225 }
226 // compute index for respective cell (_dx+... for ghost cells); use coords
227 // to store local cell coordinates
228 for (unsigned int d = 0; d < 3; d++) {
229 coords[d] = (unsigned int)((_dx + pos[d] - domainOffset[d]) / _dx);
230 }
231 const int index = get(coords[0], coords[1], coords[2]);
232 tarch::la::Vector<3, double> vel(0.0);
233 // extract and scale velocity to "real"=MD units
234 for (int d = 0; d < 3; d++) {
235 vel[d] = _dx / _dt * _vel[3 * index + d];
236 }
237#if (COUPLING_MD_DEBUG == COUPLING_MD_YES)
238 std::cout << "Position " << pos << " corresponds to cell: " << coords << "; vel=" << vel << std::endl;
239#endif
240 return vel;
241 }

273#pragma omp parallel for
274 for (int z = 1; z < _domainSizeZ + 1; z++) {
275 for (int y = 1; y < _domainSizeY + 1; y++) {
276 for (int x = 1; x < _domainSizeX + 1; x++) {
277 const int index = get(x, y, z);
278 if (_flag[index] == FLUID) {
279 stream(index);
280 collide(index, x, y, z);

292 void stream(int index) {
293 const int pI = 19 * index;
294 for (int q = 0; q < 9; q++) {
295 const int nb = 19 * (_C[q][0] + _C[q][1] * _xO + _C[q][2] * _yO);
296 _pdf2[pI + q] = _pdf1[pI + q - nb];
297 _pdf2[pI + 18 - q] = _pdf1[pI + 18 - q + nb];
298 }
299 _pdf2[pI + 9] = _pdf1[pI + 9];
300 }

303 void collide(int index, int x, int y, int z) {
304 // index of start of cell-local pdfs in AoS
305 const int pI = 19 * index;
306 // compute and store density, velocity
307 double *vel = &_vel[3 * index];
308 computeDensityAndVelocity(vel, _density[index], &_pdf2[pI]);
309 // collide (BGK); always handle pdfs no. q and inv(q)=18-q in one step
310 const double u2 = 1.0 - 1.5 * (vel[0] * vel[0] + vel[1] * vel[1] + vel[2] * vel[2]);
311 // pdf 0,18
312 double cu = -vel[1] - vel[2];
313 int nb = -_xO - _yO;
314 double feq = _W[0] * _density[index] * (u2 + 3.0 * cu + 4.5 * cu * cu);
315 _pdf2[pI] -= _omega * (_pdf2[pI] - feq);
316 boundary(_pdf2, pI, x, y, z, 0, _flag[index + nb], pI + 19 * nb);
317 feq -= 6.0 * _W[0] * _density[index] * cu;
318 _pdf2[pI + 18] -= _omega * (_pdf2[pI + 18] - feq);
319 boundary(_pdf2, pI, x, y, z, 18, _flag[index - nb], pI - 19 * nb);
320 // pdf 1,17
321 cu = -vel[0] - vel[2];
322 nb = -1 - _yO;
323 feq = _W[1] * _density[index] * (u2 + 3.0 * cu + 4.5 * cu * cu);
324 _pdf2[pI + 1] -= _omega * (_pdf2[pI + 1] - feq);
325 boundary(_pdf2, pI, x, y, z, 1, _flag[index + nb], pI + 19 * nb);
326 feq -= 6.0 * _W[1] * _density[index] * cu;
327 _pdf2[pI + 17] -= _omega * (_pdf2[pI + 17] - feq);
328 boundary(_pdf2, pI, x, y, z, 17, _flag[index - nb], pI - 19 * nb);
329 // pdf 2,16
330 cu = -vel[2];
331 nb = -_yO;
332 feq = _W[2] * _density[index] * (u2 + 3.0 * cu + 4.5 * cu * cu);
333 _pdf2[pI + 2] -= _omega * (_pdf2[pI + 2] - feq);
334 boundary(_pdf2, pI, x, y, z, 2, _flag[index + nb], pI + 19 * nb);
335 feq -= 6.0 * _W[2] * _density[index] * cu;
336 _pdf2[pI + 16] -= _omega * (_pdf2[pI + 16] - feq);
337 boundary(_pdf2, pI, x, y, z, 16, _flag[index - nb], pI - 19 * nb);
338 // pdf 3,15
339 cu = vel[0] - vel[2];
340 nb = 1 - _yO;
341 feq = _W[3] * _density[index] * (u2 + 3.0 * cu + 4.5 * cu * cu);
342 _pdf2[pI + 3] -= _omega * (_pdf2[pI + 3] - feq);
343 boundary(_pdf2, pI, x, y, z, 3, _flag[index + nb], pI + 19 * nb);
344 feq -= 6.0 * _W[3] * _density[index] * cu;
345 _pdf2[pI + 15] -= _omega * (_pdf2[pI + 15] - feq);
346 boundary(_pdf2, pI, x, y, z, 15, _flag[index - nb], pI - 19 * nb);
347 // pdf 4,14
348 cu = vel[1] - vel[2];
349 nb = _xO - _yO;
350 feq = _W[4] * _density[index] * (u2 + 3.0 * cu + 4.5 * cu * cu);
351 _pdf2[pI + 4] -= _omega * (_pdf2[pI + 4] - feq);
352 boundary(_pdf2, pI, x, y, z, 4, _flag[index + nb], pI + 19 * nb);
353 feq -= 6.0 * _W[4] * _density[index] * cu;
354 _pdf2[pI + 14] -= _omega * (_pdf2[pI + 14] - feq);
355 boundary(_pdf2, pI, x, y, z, 14, _flag[index - nb], pI - 19 * nb);
356 // pdf 5,13
357 cu = -vel[0] - vel[1];
358 nb = -1 - _xO;
359 feq = _W[5] * _density[index] * (u2 + 3.0 * cu + 4.5 * cu * cu);
360 _pdf2[pI + 5] -= _omega * (_pdf2[pI + 5] - feq);
361 boundary(_pdf2, pI, x, y, z, 5, _flag[index + nb], pI + 19 * nb);
362 feq -= 6.0 * _W[5] * _density[index] * cu;
363 _pdf2[pI + 13] -= _omega * (_pdf2[pI + 13] - feq);
364 boundary(_pdf2, pI, x, y, z, 13, _flag[index - nb], pI - 19 * nb);
365 // pdf 6,12
366 cu = -vel[1];
367 nb = -_xO;
368 feq = _W[6] * _density[index] * (u2 + 3.0 * cu + 4.5 * cu * cu);
369 _pdf2[pI + 6] -= _omega * (_pdf2[pI + 6] - feq);
370 boundary(_pdf2, pI, x, y, z, 6, _flag[index + nb], pI + 19 * nb);
371 feq -= 6.0 * _W[6] * _density[index] * cu;
372 _pdf2[pI + 12] -= _omega * (_pdf2[pI + 12] - feq);
373 boundary(_pdf2, pI, x, y, z, 12, _flag[index - nb], pI - 19 * nb);
374 // pdf 7,11
375 cu = vel[0] - vel[1];
376 nb = 1 - _xO;
377 feq = _W[7] * _density[index] * (u2 + 3.0 * cu + 4.5 * cu * cu);
378 _pdf2[pI + 7] -= _omega * (_pdf2[pI + 7] - feq);
379 boundary(_pdf2, pI, x, y, z, 7, _flag[index + nb], pI + 19 * nb);
380 feq -= 6.0 * _W[7] * _density[index] * cu;
381 _pdf2[pI + 11] -= _omega * (_pdf2[pI + 11] - feq);
382 boundary(_pdf2, pI, x, y, z, 11, _flag[index - nb], pI - 19 * nb);
383 // pdf 8,10
384 cu = -vel[0];
385 nb = -1;
386 feq = _W[8] * _density[index] * (u2 + 3.0 * cu + 4.5 * cu * cu);
387 _pdf2[pI + 8] -= _omega * (_pdf2[pI + 8] - feq);
388 boundary(_pdf2, pI, x, y, z, 8, _flag[index + nb], pI + 19 * nb);
389 feq -= 6.0 * _W[8] * _density[index] * cu;
390 _pdf2[pI + 10] -= _omega * (_pdf2[pI + 10] - feq);
391 boundary(_pdf2, pI, x, y, z, 10, _flag[index - nb], pI - 19 * nb);
392 // pdf 9
393 _pdf2[pI + 9] -= _omega * (_pdf2[pI + 9] - _W[9] * _density[index] * u2);
394 }

405 void boundary(double *const pdf, int index, int x, int y, int z, int q, const Flag &flag, int nbIndex) {
406 if (flag != FLUID) {
407 if (flag == NO_SLIP) {
408 // half-way bounce back
409 pdf[nbIndex + 18 - q] = pdf[index + q];
410 } else if (flag == MOVING_WALL) {
411 // half-way bounce back + moving wall acceleration (only x-direction for
412 // wall supported at the moment)
413 pdf[nbIndex + 18 - q] =
414 pdf[index + q] - 6.0 * _W[q] * _density[index / 19] * (_C[q][0] * _wallVelocity[0] + _C[q][1] * _wallVelocity[1] + _C[q][2] * _wallVelocity[2]);
415 } else if (flag == PERIODIC) {
416 // periodic treatment
417 int target[3] = {x, y, z};
418 if (target[0] + _C[q][0] == 0) {
419 target[0] = _domainSizeX + 1;
420 } else if (target[0] + _C[q][0] == _domainSizeX + 1) {
421 target[0] = 0;
422 }
423 if (target[1] + _C[q][1] == 0) {
424 target[1] = _domainSizeY + 1;
425 } else if (target[1] + _C[q][1] == _domainSizeY + 1) {
426 target[1] = 0;
427 }
428 if (target[2] + _C[q][2] == 0) {
429 target[2] = _domainSizeZ + 1;
430 } else if (target[2] + _C[q][2] == _domainSizeZ + 1) {
431 target[2] = 0;
432 }
433 const int periodicNb = target[0] + (_domainSizeX + 2) * (target[1] + (_domainSizeY + 2) * target[2]);
434 pdf[19 * periodicNb + q] = pdf[index + q];
435 }
436 }

443 void computeDensityAndVelocity(double *const vel, double &density, const double *const pdf) {
444 vel[0] = -(pdf[1] + pdf[5] + pdf[8] + pdf[11] + pdf[15]);
445 density = pdf[3] + pdf[7] + pdf[10] + pdf[13] + pdf[17];
446 vel[1] = (pdf[4] + pdf[11] + pdf[12] + pdf[13] + pdf[18]) - (pdf[0] + pdf[5] + pdf[6] + pdf[7] + pdf[14]);
447 vel[0] = density + vel[0];
448 density = density + pdf[0] + pdf[1] + pdf[2] + pdf[4] + pdf[5] + pdf[6] + pdf[8] + pdf[9] + pdf[11] + pdf[12] + pdf[14] + pdf[15] + pdf[16] + pdf[18];
449 vel[2] = (pdf[14] + pdf[15] + pdf[16] + pdf[17] + pdf[18]) - (pdf[0] + pdf[1] + pdf[2] + pdf[3] + pdf[4]);
450 vel[0] = vel[0] / density;
451 vel[1] = vel[1] / density;
452 vel[2] = vel[2] / density;
453 }

469 void communicatePart(double *pdf, double *sendBuffer, double *recvBuffer, NbFlag nbFlagTo, NbFlag nbFlagFrom, tarch::la::Vector<3, int> startSend,
470 tarch::la::Vector<3, int> endSend, tarch::la::Vector<3, int> startRecv, tarch::la::Vector<3, int> endRecv) {
471#if (COUPLING_MD_PARALLEL == COUPLING_MD_YES)
472 // directions that point to LEFT/RIGHT,... -> same ordering as enums!
473 const int directions[6][5] = {{1, 5, 8, 11, 15}, {3, 7, 10, 13, 17}, {4, 11, 12, 13, 18}, {0, 5, 6, 7, 14}, {0, 1, 2, 3, 4}, {14, 15, 16, 17, 18}};
474 MPI_Request requests[2];
475 MPI_Status status[2];
476 tarch::la::Vector<2, int> plane;
477 tarch::la::Vector<2, int> domainSize;
478 // find out plane coordinates
479 if (nbFlagTo == LEFT || nbFlagTo == RIGHT) {
480 plane[0] = 1;
481 plane[1] = 2;
482 domainSize[0] = _domainSizeY;
483 domainSize[1] = _domainSizeZ;
484 } else if (nbFlagTo == FRONT || nbFlagTo == BACK) {
485 plane[0] = 0;
486 plane[1] = 2;
487 domainSize[0] = _domainSizeX;
488 domainSize[1] = _domainSizeZ;
489 } else if (nbFlagTo == TOP || nbFlagTo == BOTTOM) {
490 plane[0] = 0;
491 plane[1] = 1;
492 domainSize[0] = _domainSizeX;
493 domainSize[1] = _domainSizeY;
494 } else {
495 std::cout << "ERROR LBCouetteSolver::communicatePart: d >2 or d < 0!" << std::endl;
496 exit(EXIT_FAILURE);
497 }
498 // extract data and write to send buffer
499 tarch::la::Vector<3, int> coords(0);
500 for (coords[2] = startSend[2]; coords[2] < endSend[2]; coords[2]++) {
501 for (coords[1] = startSend[1]; coords[1] < endSend[1]; coords[1]++) {
502 for (coords[0] = startSend[0]; coords[0] < endSend[0]; coords[0]++) {
503 for (int q = 0; q < 5; q++) {
504 sendBuffer[q + 5 * getParBuf(coords[plane[0]], coords[plane[1]], domainSize[0], domainSize[1])] =
