60 void parseSubtag(tinyxml2::XMLElement *node) {
61 std::string value;
62 tarch::configuration::ParseConfiguration::readStringMandatory(value, node,
63 "type");
64 if (value == "additive-momentum-insertion") {
65 _insertionType = ADDITIVE_MOMENTUM_INSERTION;
66 } else if (value == "direct-velocity-insertion") {
67 _insertionType = DIRECT_VELOCITY_INSERTION;
68 } else if (value == "velocity-gradient-relaxation") {
69 _insertionType = VELOCITY_GRADIENT_RELAXATION;
70 } else if (value == "none") {
71 _insertionType = NO_INSERTION;
72 } else if (value == "velocity-gradient-relaxation-top-only") {
73 _insertionType = VELOCITY_GRADIENT_RELAXATION_TOPONLY;
74 } else if (value == "nie-velocity-imposition") {
75 _insertionType = NIE_VELOCITY_IMPOSITION;
76 } else {
77 std::cout << "ERROR coupling::MomentumInsertionConfiguration: Wrong "
78 "insertion type!"
79 << std::endl;
80 _isValid = false;
81 exit(EXIT_FAILURE);
82 }
83
84 if (_insertionType == VELOCITY_GRADIENT_RELAXATION) {
85 tarch::configuration::ParseConfiguration::readDoubleMandatory(
86 _velocityRelaxationFactor, node, "velocity-relaxation-factor");
87 if ((_velocityRelaxationFactor <= 0.0) ||
88 (_velocityRelaxationFactor > 1.0)) {
89 std::cout << "ERROR coupling::MomentumInsertionConfiguration: "
90 "velocity-relaxation-factor="
91 << _velocityRelaxationFactor << "!";
92 std::cout << " It must be in the range (0.0,1.0]!" << std::endl;
93 _isValid = false;
94 exit(EXIT_FAILURE);
95 }
96 } else if (_insertionType == VELOCITY_GRADIENT_RELAXATION_TOPONLY) {
97 tarch::configuration::ParseConfiguration::readDoubleMandatory(
98 _velocityRelaxationFactor, node, "velocity-relaxation-factor");
99 if ((_velocityRelaxationFactor <= 0.0) ||
100 (_velocityRelaxationFactor > 1.0)) {
101 std::cout << "ERROR coupling::MomentumInsertionConfiguration: "
102 "velocity-relaxation-factor="
103 << _velocityRelaxationFactor << "!";
104 std::cout << " It must be in the range (0.0,1.0]!" << std::endl;
105 _isValid = false;
106 exit(EXIT_FAILURE);
107 }
108 } else if (_insertionType == NIE_VELOCITY_IMPOSITION) {
109 int buf;
110 tarch::configuration::ParseConfiguration::readIntMandatory(
111 buf, node, "innermost-overlap-layer");
112 if (buf <= 0) {
113 std::cout << "ERROR coupling::MomentumInsertionConfiguration: "
114 "innermost-overlap-layer="
115 << buf << "!" << std::endl;
116 _isValid = false;
117 exit(EXIT_FAILURE);
118 }
119 _innerOverlap = (unsigned int)buf;
120 tarch::configuration::ParseConfiguration::readIntMandatory(
121 buf, node, "outermost-overlap-layer");
122 if (((unsigned int)buf > _innerOverlap) || (buf <= 0)) {
123 std::cout << "ERROR coupling::MomentumInsertionConfiguration: "
124 "outermost-overlap-layer="
125 << buf << "!" << std::endl;
126 _isValid = false;
127 exit(EXIT_FAILURE);
128 }
129 _outerOverlap = (unsigned int)buf;
130 }
131 }

169 template <class LinkedCell, unsigned int dim>
170 MomentumInsertion<LinkedCell, dim> *interpreteConfiguration(
171 coupling::interface::MDSolverInterface<LinkedCell, dim>
172 *const mdSolverInterface,
173 const coupling::IndexConversion<dim> &indexConversion,
174 const coupling::datastructures::MacroscopicCellWithLinkedCells<
175 LinkedCell, dim> *const macroscopicCells,
176 unsigned int numberMDTimestepsPerCouplingCycle) const {
177 if (_insertionType == ADDITIVE_MOMENTUM_INSERTION) {
178 return new coupling::AdditiveMomentumInsertion<LinkedCell, dim>(
179 mdSolverInterface, numberMDTimestepsPerCouplingCycle);
180 } else if (_insertionType == DIRECT_VELOCITY_INSERTION) {
181 return new coupling::SetGivenVelocity4MomentumInsertion<LinkedCell, dim>(
182 mdSolverInterface);
183 } else if (_insertionType == VELOCITY_GRADIENT_RELAXATION) {
184 return new coupling::VelocityGradientRelaxation<LinkedCell, dim>(
185 _velocityRelaxationFactor, mdSolverInterface, indexConversion,
186 macroscopicCells);
187 } else if (_insertionType == NO_INSERTION) {
188 return new coupling::NoMomentumInsertion<LinkedCell, dim>(
189 mdSolverInterface);
190 } else if (_insertionType == VELOCITY_GRADIENT_RELAXATION_TOPONLY) {
191 return new coupling::VelocityGradientRelaxationTopOnly<LinkedCell, dim>(
192 _velocityRelaxationFactor, mdSolverInterface, indexConversion,
193 macroscopicCells);
194 } else if (_insertionType == NIE_VELOCITY_IMPOSITION) {
195 return new coupling::NieVelocityImposition<LinkedCell, dim>(
196 mdSolverInterface, indexConversion, _outerOverlap, _innerOverlap);
197 }
198 return NULL;
199 }
200
201private:
202 MomentumInsertionType _insertionType;
203 double _velocityRelaxationFactor; // required by velocity relaxation schemes
204 unsigned int _innerOverlap; // innermost layer of overlap cells (required by
205 // nie velocity imposition)
206 unsigned int _outerOverlap; // outermost layer of overlap cells (required by
207 // nie velocity imposition)
208 bool _isValid;
209};
