45 void parseSubtag( tinyxml2::XMLElement* node ){
46 std::string value;
47 tarch::configuration::ParseConfiguration::readStringMandatory(value,node,"type");
48 if (value=="additive-momentum-insertion"){
49 _insertionType = ADDITIVE_MOMENTUM_INSERTION;
50 } else if (value=="direct-velocity-insertion"){
51 _insertionType = DIRECT_VELOCITY_INSERTION;
52 } else if (value=="velocity-gradient-relaxation"){
53 _insertionType = VELOCITY_GRADIENT_RELAXATION;
54 } else if (value=="none"){
55 _insertionType = NO_INSERTION;
56 } else if (value=="velocity-gradient-relaxation-top-only"){
57 _insertionType = VELOCITY_GRADIENT_RELAXATION_TOPONLY;
58 } else if (value=="nie-velocity-imposition"){
59 _insertionType = NIE_VELOCITY_IMPOSITION;
60 } else {
61 std::cout << "ERROR coupling::MomentumInsertionConfiguration: Wrong insertion type!" << std::endl;
62 _isValid = false;
63 exit(EXIT_FAILURE);
64 }
65
66 if (_insertionType==VELOCITY_GRADIENT_RELAXATION){
67 tarch::configuration::ParseConfiguration::readDoubleMandatory(_velocityRelaxationFactor,node,"velocity-relaxation-factor");
68 if ((_velocityRelaxationFactor<=0.0) || (_velocityRelaxationFactor > 1.0) ){
69 std::cout << "ERROR coupling::MomentumInsertionConfiguration: velocity-relaxation-factor=" << _velocityRelaxationFactor << "!";
70 std::cout << " It must be in the range (0.0,1.0]!" << std::endl;
71 _isValid = false;
72 exit(EXIT_FAILURE);
73 }
74 } else if (_insertionType==VELOCITY_GRADIENT_RELAXATION_TOPONLY){
75 tarch::configuration::ParseConfiguration::readDoubleMandatory(_velocityRelaxationFactor,node,"velocity-relaxation-factor");
76 if ((_velocityRelaxationFactor<=0.0) || (_velocityRelaxationFactor > 1.0) ){
77 std::cout << "ERROR coupling::MomentumInsertionConfiguration: velocity-relaxation-factor=" << _velocityRelaxationFactor << "!";
78 std::cout << " It must be in the range (0.0,1.0]!" << std::endl;
79 _isValid = false;
80 exit(EXIT_FAILURE);
81 }
82 } else if (_insertionType==NIE_VELOCITY_IMPOSITION){
83 int buf;
84 tarch::configuration::ParseConfiguration::readIntMandatory(buf,node,"innermost-overlap-layer");
85 if (buf <=0){std::cout << "ERROR coupling::MomentumInsertionConfiguration: innermost-overlap-layer=" << buf <<"!" << std::endl; _isValid=false; exit(EXIT_FAILURE);}
86 _innerOverlap = (unsigned int) buf;
87 tarch::configuration::ParseConfiguration::readIntMandatory(buf,node,"outermost-overlap-layer");
88 if ( ((unsigned int)buf > _innerOverlap) || (buf <= 0) ){std::cout << "ERROR coupling::MomentumInsertionConfiguration: outermost-overlap-layer=" << buf <<"!" << std::endl; _isValid=false; exit(EXIT_FAILURE);}
89 _outerOverlap = (unsigned int) buf;
90
91 }
92 }

116 template<class LinkedCell,unsigned int dim>
117 MomentumInsertion<LinkedCell,dim>* interpreteConfiguration(
118 coupling::interface::MDSolverInterface<LinkedCell,dim> * const mdSolverInterface,
119 const coupling::IndexConversion<dim>& indexConversion,
120 const coupling::datastructures::MacroscopicCellWithLinkedCells<LinkedCell,dim> * const macroscopicCells,
121 unsigned int numberMDTimestepsPerCouplingCycle
122 ) const {
123 if (_insertionType == ADDITIVE_MOMENTUM_INSERTION){
124 return new coupling::AdditiveMomentumInsertion<LinkedCell,dim>(mdSolverInterface,numberMDTimestepsPerCouplingCycle);
125 } else if (_insertionType == DIRECT_VELOCITY_INSERTION){
126 return new coupling::SetGivenVelocity4MomentumInsertion<LinkedCell,dim>(mdSolverInterface);
127 } else if (_insertionType == VELOCITY_GRADIENT_RELAXATION){
128 return new coupling::VelocityGradientRelaxation<LinkedCell,dim>(_velocityRelaxationFactor,mdSolverInterface,indexConversion,macroscopicCells);
129 } else if (_insertionType == NO_INSERTION){
130 return new coupling::NoMomentumInsertion<LinkedCell,dim>(mdSolverInterface);
131 } else if (_insertionType == VELOCITY_GRADIENT_RELAXATION_TOPONLY){
132 return new coupling::VelocityGradientRelaxationTopOnly<LinkedCell,dim>(_velocityRelaxationFactor,mdSolverInterface,indexConversion,macroscopicCells);
133 } else if (_insertionType == NIE_VELOCITY_IMPOSITION){
134 return new coupling::NieVelocityImposition<LinkedCell,dim>(mdSolverInterface,indexConversion,_outerOverlap,_innerOverlap);
135 }
136 return NULL;
137 }
138
139 private:
140 MomentumInsertionType _insertionType;
141 double _velocityRelaxationFactor; // required by velocity relaxation schemes
142 unsigned int _innerOverlap; // innermost layer of overlap cells (required by nie velocity imposition)
143 unsigned int _outerOverlap; // outermost layer of overlap cells (required by nie velocity imposition)
144 bool _isValid;
145};
