52 bool receiveMacroscopicQuantityFromMDSolver(tarch::la::Vector<3, unsigned int> globalCellIndex) {
53 bool recv = true;

66 bool sendMacroscopicQuantityToMDSolver(tarch::la::Vector<3, unsigned int> globalCellIndex) {

78 virtual std::vector<unsigned int> getRanks(tarch::la::Vector<3, unsigned int> globalCellIndex) {
79 std::vector<unsigned int> ranks;
80 // determine global index of cell in LB simulation
81 tarch::la::Vector<3, unsigned int> globalLBCellIndex(globalCellIndex + _offsetMDDomain);
82 // modify global LB cell index due to ghost layer
83 for (int d = 0; d < 3; d++) {
84#if (COUPLING_MD_DEBUG == COUPLING_MD_YES)
85 std::cout << "LB cell index for global cell index " << globalCellIndex << ": " << globalLBCellIndex << std::endl;
86#endif
87 if (globalLBCellIndex[d] > 0) {
88 globalLBCellIndex[d]--;
89 }
90 }
91 // loop over all neighbouring cells within a one-cell surrounding and detect
92 // the respective ranks. IMPROVE: This currently only allows for simulations
93 // with MD located inside the domain (no simulation across boundary)
94 for (int z = -1; z < 2; z++) {
95 for (int y = -1; y < 2; y++) {
96 for (int x = -1; x < 2; x++) {
97 // neighbour cell index
98 const tarch::la::Vector<3, unsigned int> nbIndex(globalLBCellIndex[0] + x, globalLBCellIndex[1] + y, globalLBCellIndex[2] + z);
99 // coordinates of process of neighbour celll
100 const tarch::la::Vector<3, unsigned int> processCoordinates((nbIndex[0] / _avgNumberLBCells[0]), (nbIndex[1] / _avgNumberLBCells[1]),
101 (nbIndex[2] / _avgNumberLBCells[2]));
102 // corresponding rank
103 const unsigned int rank = processCoordinates[0] + _numberProcesses[0] * (processCoordinates[1] + processCoordinates[2] * _numberProcesses[1]);
104
105 // if this rank is not part of the vector, push it back
106 bool found = false;
107 for (unsigned int i = 0; i < ranks.size(); i++) {
108 found = found || (rank == ranks[i]);
109 }
110 if (!found) {
111 ranks.push_back(rank);
112 }
113 }
114 }
115 }
116#if (COUPLING_MD_DEBUG == COUPLING_MD_YES)
117 std::cout << "Ranks for cell " << globalCellIndex << ":";
118 for (unsigned int i = 0; i < ranks.size(); i++) {
119 std::cout << " " << ranks[i];
120 }
121 std::cout << std::endl;
122#endif
123 return ranks;
124 }

136 virtual std::vector<unsigned int> getSourceRanks(tarch::la::Vector<3, unsigned int> globalCellIndex) {
137 // determine global index of cell in LB simulation
138 tarch::la::Vector<3, unsigned int> globalLBCellIndex(globalCellIndex + _offsetMDDomain);
139 // modify global LB cell index due to ghost layer
140 for (int d = 0; d < 3; d++) {
141#if (COUPLING_MD_DEBUG == COUPLING_MD_YES)
142 std::cout << "LB cell index for global cell index " << globalCellIndex << ": " << globalLBCellIndex << std::endl;
143#endif
144 if (globalLBCellIndex[d] > 0) {
145 globalLBCellIndex[d]--;
146 }
147 }
148 // determine process coordinates and respective rank
149 const tarch::la::Vector<3, unsigned int> processCoordinates(globalLBCellIndex[0] / _avgNumberLBCells[0], globalLBCellIndex[1] / _avgNumberLBCells[1],
150 globalLBCellIndex[2] / _avgNumberLBCells[2]);
151 const unsigned int rank = processCoordinates[0] + _numberProcesses[0] * (processCoordinates[1] + processCoordinates[2] * _numberProcesses[1]);
152 std::vector<unsigned int> ranks;
153 ranks.push_back(rank);
154#if (COUPLING_MD_DEBUG == COUPLING_MD_YES)
155 std::cout << "Source rank for cell " << globalCellIndex << ": " << ranks[0] << std::endl;
156#endif
157 return ranks;
158 }
