71 virtual std::vector<unsigned int> getRanks(tarch::la::Vector<3,unsigned int> globalCellIndex){
72 std::vector<unsigned int> ranks;
73 // determine global index of cell in LB simulation
74 tarch::la::Vector<3,unsigned int> globalLBCellIndex(globalCellIndex + _offsetMDDomain);
75 // modify global LB cell index due to ghost layer
76 for (int d = 0; d < 3; d++){
77 #if (COUPLING_MD_DEBUG==COUPLING_MD_YES)
78 std::cout << "LB cell index for global cell index " << globalCellIndex << ": " << globalLBCellIndex << std::endl;
79 #endif
80 if (globalLBCellIndex[d]>0){ globalLBCellIndex[d]--;}
81 }
82 // loop over all neighbouring cells within a one-cell surrounding and detect the respective ranks.
83 // IMPROVE: This currently only allows for simulations with MD located inside the domain
84 // (no simulation across boundary)
85 for (int z = -1; z < 2; z++){ for (int y = -1; y < 2; y++){ for (int x = -1; x < 2; x++){
86 // neighbour cell index
87 const tarch::la::Vector<3,unsigned int> nbIndex(globalLBCellIndex[0]+x,globalLBCellIndex[1]+y,globalLBCellIndex[2]+z);
88 // coordinates of process of neighbour celll
89 const tarch::la::Vector<3,unsigned int> processCoordinates((nbIndex[0]/_avgNumberLBCells[0]),(nbIndex[1]/_avgNumberLBCells[1]),(nbIndex[2]/_avgNumberLBCells[2]));
90 // corresponding rank
91 const unsigned int rank = processCoordinates[0] + _numberProcesses[0]*(processCoordinates[1]+processCoordinates[2]*_numberProcesses[1]);
92
93 // if this rank is not part of the vector, push it back
94 bool found=false;
95 for (unsigned int i = 0; i<ranks.size();i++){found= found||(rank==ranks[i]);}
96 if (!found){ ranks.push_back(rank); }
97 }}}
98 #if (COUPLING_MD_DEBUG==COUPLING_MD_YES)
99 std::cout << "Ranks for cell " << globalCellIndex << ":"; for (unsigned int i = 0; i < ranks.size(); i++){ std::cout << " " << ranks[i];} std::cout << std::endl;
100 #endif
101 return ranks;
102 }

122 const unsigned int rank = processCoordinates[0] + _numberProcesses[0]*(processCoordinates[1]+processCoordinates[2]*_numberProcesses[1]);

125 #if (COUPLING_MD_DEBUG==COUPLING_MD_YES)

127 #endif
