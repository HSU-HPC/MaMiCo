113 virtual void

116 tarch::la::Vector<dim, double> macroscopicMomentum(0.0);
117 for (unsigned int d = 0; d < dim; d++) {
118 macroscopicMomentum[d] = buffer[1 + d];
119 }
120 cell.setMacroscopicMomentum(macroscopicMomentum);
121 cell.setMacroscopicMass(buffer[0]);
122 }
