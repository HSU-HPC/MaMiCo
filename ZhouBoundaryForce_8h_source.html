106 void handleCell(LinkedCell& cell,const unsigned int &cellIndex){
107 coupling::interface::MoleculeIterator<LinkedCell,dim> *it = _mdSolverInterface->getMoleculeIterator(cell);
108 it->begin();
109 while(it->continueIteration()){
110 coupling::interface::Molecule<dim> &wrapper(it->get());
111 //std::cout << "apply force to Molecule " << wrapper.getPosition() << std::endl;
112
113 // extract position and force of each molecule
114 const tarch::la::Vector<dim,double> position(wrapper.getPosition());
115 tarch::la::Vector<dim,double> force(wrapper.getForce());
116 // add boundary force
117 force = force + getBoundaryForces(position);
118 wrapper.setForce(force);
119
120 it->next();
121 }
122 delete it;
123 }

129 double getPotentialEnergy(const tarch::la::Vector<dim,double>& position) const {
130 double energy(0);
131 const double distance = 2.5*_sigma;
132 for (unsigned int d = 0; d < dim; d++){
133 if (_boundary[2*d] && (position[d]-_domainLowerLeft[d]<distance)) { energy += getPotentialEnergy((position[d]-_domainLowerLeft[d])/_sigma);}
134 if (_boundary[2*d+1] && (_domainUpperRight[d]-position[d]<distance)){ energy += getPotentialEnergy((_domainUpperRight[d]-position[d])/_sigma);}
135 }
136 return energy;
137 }

144 tarch::la::Vector<dim,double> getBoundaryForces(const tarch::la::Vector<dim,double>& position) const {
145 tarch::la::Vector<dim,double> force(0.0);
146 const double distance = 2.5*_sigma;
147
148 for (unsigned int d = 0; d < dim; d++){
149 // for left/lower/back boundary: add force; for right/upper/front boundary: subtract force value
150 if (_boundary[2*d] && (position[d]-_domainLowerLeft[d]<distance)) { force[d] += _forceFactor*getScalarBoundaryForce( (position[d]-_domainLowerLeft[d])/_sigma); }
151 if (_boundary[2*d+1] && (_domainUpperRight[d]-position[d]<distance)){ force[d] -= _forceFactor*getScalarBoundaryForce((_domainUpperRight[d]-position[d])/_sigma); }
152 }
153 return force;
154 }

162 double getPotentialEnergy(double rw) const {
163 double dx(2.5 * _sigma / (_energyResolution-1));
164 int upper = (int)((2.5-rw)/2.5*(_energyResolution-1)-.5);
165 int lower = (int)((2.5-rw)/2.5*(_energyResolution-1)+.5);
166 return _energyTable[lower] * ((2.5 - upper * dx - rw) / dx)
167 + _energyTable[upper] * ((rw - 2.5 + lower * dx) / dx);
168 }
169
174 double getScalarBoundaryForce(double rw) const {
175 if (rw < 1.04){ return _p1 + _p2*exp(pow((rw+0.25),3.4))*cos(_p3*rw) ; }
176 else { return -1.0/(_q1+_q2*(2.5-rw)*(2.5-rw) + _q3/((2.5-rw)*(2.5-rw))); }
177 }

184 double getP1(const double& density, const double& temperature) const {
185 const double logrho = log(density);
186 const double T2 = temperature*temperature;
187 const double T3 = T2*temperature;
188
189 // an additional factor 10**-5 is in the expression 4.599/(...) -> private communication with Wenjing Zhou (typo in paper)
190 const double p1 = (-18.953 + 53.369*temperature - 1.253*T2 + 4.599/100000.0*T3
191 +59.871*logrho + 19.737*logrho*logrho)/
192 (1.0 + 2.592*temperature - 0.557*T2 + 0.049*T3
193 -13.912*logrho + 18.657*logrho*logrho );
194 return p1;
195 }

202 double getP2(const double& density,const double &temperature) const {
203 const double logrho = log(density);
204 const double T2 = temperature*temperature;
205 const double T3 = T2*temperature;
206
207 const double p2 = (-0.094 + 2.808*temperature - 0.019*T2 - 0.001*T3
208 +2.823*logrho + 2.071*logrho*logrho)/ (1.0 + 0.168*temperature - 0.013*T2
209 -4.323*logrho + 2.557*logrho*logrho - 2.155*logrho*logrho*logrho);
210 return p2;
211 }

218 double getP3(const double& density,const double &temperature) const {
219 const double T394 = pow(temperature,0.394);
220 const double rho17437 = pow(density,17.437);
221 const double p3 = 3.934 + 0.099*T394 - 0.097*rho17437 + 0.075*T394 * rho17437;
222 return p3;
223 }

229 double getQ1(const double& density) const {
230 const double rho2 = density*density;
231 const double rho3 = rho2*density;
232 const double rho4 = rho2*rho2;
233 return -30.471 + 113.879*density - 207.205*rho2 + 184.242*rho3 - 62.879*rho4;
234 }
