53 ZhouBoundaryForce(const double &density, const double &temperature,
54 const double &epsilon, const double &sigma,
55 const tarch::la::Vector<2 * dim, bool> &boundary,
56 const tarch::la::Vector<dim, double> &domainOffset,
57 const tarch::la::Vector<dim, double> &domainSize,
58 coupling::interface::MDSolverInterface<LinkedCell, dim>
59 *const mdSolverInterface)
60 : _boundary(boundary), _domainLowerLeft(domainOffset),
61 _domainUpperRight(domainSize + domainOffset),
62 _mdSolverInterface(mdSolverInterface), _p1(getP1(density, temperature)),
63 _p2(getP2(density, temperature)), _p3(getP3(density, temperature)),
64 _q1(getQ1(density)), _q2(getQ2(density)), _q3(getQ3(density)),
65 _forceFactor(epsilon / sigma), _sigma(sigma),
67 _energyResolution(1000), _energyTable(new double[_energyResolution]) {
68 // simple trapezoidal integration scheme
69 double force(0);
70 double energy(0);
71 double dx(2.5 * sigma / (_energyResolution - 1));
72 for (unsigned int i = 0; i < _energyResolution; i++) {
73 force = getScalarBoundaryForce(2.5 - i * dx);
74 _energyTable[i] = energy + force / 2 * dx;
75 energy += force * dx;
76 }

104 void handleCell(LinkedCell &cell, const unsigned int &cellIndex) {
105 coupling::interface::MoleculeIterator<LinkedCell, dim> *it =
106 _mdSolverInterface->getMoleculeIterator(cell);
107 it->begin();
108 while (it->continueIteration()) {
109 coupling::interface::Molecule<dim> &wrapper(it->get());
110 // std::cout << "apply force to Molecule " << wrapper.getPosition() <<
111 // std::endl;
112
113 // extract position and force of each molecule
114 const tarch::la::Vector<dim, double> position(wrapper.getPosition());
115 tarch::la::Vector<dim, double> force(wrapper.getForce());
116 // add boundary force
117 force = force + getBoundaryForces(position);
118 wrapper.setForce(force);
119
120 it->next();
121 }
122 delete it;
123 }

129 double
130 getPotentialEnergy(const tarch::la::Vector<dim, double> &position) const {
131 double energy(0);
132 const double distance = 2.5 * _sigma;
133 for (unsigned int d = 0; d < dim; d++) {
134 if (_boundary[2 * d] && (position[d] - _domainLowerLeft[d] < distance)) {
135 energy +=
136 getPotentialEnergy((position[d] - _domainLowerLeft[d]) / _sigma);
137 }
138 if (_boundary[2 * d + 1] &&
139 (_domainUpperRight[d] - position[d] < distance)) {
140 energy +=
141 getPotentialEnergy((_domainUpperRight[d] - position[d]) / _sigma);
142 }
143 }
144 return energy;
145 }

154 tarch::la::Vector<dim, double>
155 getBoundaryForces(const tarch::la::Vector<dim, double> &position) const {
156 tarch::la::Vector<dim, double> force(0.0);
157 const double distance = 2.5 * _sigma;
158
159 for (unsigned int d = 0; d < dim; d++) {
160 // for left/lower/back boundary: add force; for right/upper/front
161 // boundary: subtract force value
162 if (_boundary[2 * d] && (position[d] - _domainLowerLeft[d] < distance)) {
163 force[d] +=
164 _forceFactor * getScalarBoundaryForce(
165 (position[d] - _domainLowerLeft[d]) / _sigma);
166 }
167 if (_boundary[2 * d + 1] &&
168 (_domainUpperRight[d] - position[d] < distance)) {
169 force[d] -=
170 _forceFactor * getScalarBoundaryForce(
171 (_domainUpperRight[d] - position[d]) / _sigma);
172 }
173 }
174 return force;
175 }

177private:
182 double getPotentialEnergy(double rw) const {
183 double dx(2.5 * _sigma / (_energyResolution - 1));
184 int upper = (int)((2.5 - rw) / 2.5 * (_energyResolution - 1) - .5);
185 int lower = (int)((2.5 - rw) / 2.5 * (_energyResolution - 1) + .5);
186 return _energyTable[lower] * ((2.5 - upper * dx - rw) / dx) +
187 _energyTable[upper] * ((rw - 2.5 + lower * dx) / dx);
188 }
189
196 double getScalarBoundaryForce(double rw) const {
197 if (rw < 1.04) {
198 return _p1 + _p2 * exp(pow((rw + 0.25), 3.4)) * cos(_p3 * rw);
199 } else {
200 return -1.0 / (_q1 + _q2 * (2.5 - rw) * (2.5 - rw) +
201 _q3 / ((2.5 - rw) * (2.5 - rw)));
202 }
203 }

210 double getP1(const double &density, const double &temperature) const {
211 const double logrho = log(density);
212 const double T2 = temperature * temperature;
213 const double T3 = T2 * temperature;
214
215 // an additional factor 10**-5 is in the expression 4.599/(...) -> private
216 // communication with Wenjing Zhou (typo in paper)
217 const double p1 =
218 (-18.953 + 53.369 * temperature - 1.253 * T2 + 4.599 / 100000.0 * T3 +
219 59.871 * logrho + 19.737 * logrho * logrho) /
220 (1.0 + 2.592 * temperature - 0.557 * T2 + 0.049 * T3 - 13.912 * logrho +
221 18.657 * logrho * logrho);
222 return p1;
223 }
224
230 double getP2(const double &density, const double &temperature) const {
231 const double logrho = log(density);
232 const double T2 = temperature * temperature;
233 const double T3 = T2 * temperature;
234
235 const double p2 =
236 (-0.094 + 2.808 * temperature - 0.019 * T2 - 0.001 * T3 +
237 2.823 * logrho + 2.071 * logrho * logrho) /
238 (1.0 + 0.168 * temperature - 0.013 * T2 - 4.323 * logrho +
239 2.557 * logrho * logrho - 2.155 * logrho * logrho * logrho);
240 return p2;
241 }

248 double getP3(const double &density, const double &temperature) const {
249 const double T394 = pow(temperature, 0.394);
250 const double rho17437 = pow(density, 17.437);
251 const double p3 =
252 3.934 + 0.099 * T394 - 0.097 * rho17437 + 0.075 * T394 * rho17437;
253 return p3;
254 }

260 double getQ1(const double &density) const {
261 const double rho2 = density * density;
262 const double rho3 = rho2 * density;
263 const double rho4 = rho2 * rho2;
264 return -30.471 + 113.879 * density - 207.205 * rho2 + 184.242 * rho3 -
265 62.879 * rho4;
