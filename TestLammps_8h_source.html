69 virtual void run() {
70 double tmp = -1.0; // test variable
71 tarch::la::Vector<dim, double> tmpVec(0.0); // test vector
72 bool valid = true;
73 std::cout << "Load lammps test configuration..." << std::endl;
74 if (dim == 2) {
75 loadLammpsTestConfiguration("inputpositionsonly2D.xyz", 4);
76 } else {
77 loadLammpsTestConfiguration("inputpositionsonly3D.xyz", 8);
78 }
79 std::cout << "Load void macroscopic solver configuration..." << std::endl;
80 loadMacroscopicSolverConfiguration();
81 std::cout
82 << "Load MaMiCo configuration and init macroscopic cell service..."
83 << std::endl;
84 loadMamicoTestConfiguration();
85 // test molecule mass
86 tmp = coupling::interface::MamicoInterfaceProvider<LAMMPS_NS::MamicoCell,
87 dim>::getInstance()
88 .getMDSolverInterface()
89 ->getMoleculeMass();
90 std::cout << "Molecule mass: " << tmp << std::endl;
91 if (tmp != 2.0) {
92 std::cout << "Wrong mass value, mass=" << tmp << ", instead of 2.0!"
93 << std::endl;
94 exit(EXIT_FAILURE);
95 }
96 // test Boltzmann constant kB
97 tmp = coupling::interface::MamicoInterfaceProvider<LAMMPS_NS::MamicoCell,
98 dim>::getInstance()
99 .getMDSolverInterface()
100 ->getKB();
101 std::cout << "Boltzmann constant: " << tmp << std::endl;
102 if (tmp != 1.0) {
103 std::cout << "Wrong boltzmann constant, kB=" << tmp << ", should be 1.0!"
104 << std::endl;
105 exit(EXIT_FAILURE);
106 }
107 // TODO put tests for sigma+epsilon here
108 // test time step dt
109 tmp = coupling::interface::MamicoInterfaceProvider<LAMMPS_NS::MamicoCell,
110 dim>::getInstance()
111 .getMDSolverInterface()
112 ->getDt();
113 std::cout << "Time step: " << tmp << std::endl;
114 if (tmp != 1.0e-14) {
115 std::cout << "Wrong time step, time step=" << tmp
116 << ", should be 1.0e-14!" << std::endl;
117 exit(EXIT_FAILURE);
118 }
119 // test domain size
120 tmpVec = coupling::interface::MamicoInterfaceProvider<LAMMPS_NS::MamicoCell,
121 dim>::getInstance()
122 .getMDSolverInterface()
123 ->getGlobalMDDomainSize();
124 valid = (tmpVec[0] == 10.0);
125 for (int d = 1; d < dim; d++) {
126 valid = valid && (tmpVec[d] == 10.0);
127 }
128 if (!valid) {
129 std::cout << "Wrong domain size " << tmpVec
130 << ", should be 10.0 in each component!" << std::endl;
131 exit(EXIT_FAILURE);
132 }
133 // test domain offset
134 tmpVec = coupling::interface::MamicoInterfaceProvider<LAMMPS_NS::MamicoCell,
135 dim>::getInstance()
136 .getMDSolverInterface()
137 ->getGlobalMDDomainOffset();
138 valid = (tmpVec[0] == 1.0);
139 for (int d = 1; d < dim; d++) {
140 valid = valid && (tmpVec[d] == 1.0);
141 }
142 if (!valid) {
143 std::cout << "Wrong domain offset " << tmpVec
144 << ", should be 1.0 in each component!" << std::endl;
145 exit(EXIT_FAILURE);

157 void loadLammpsTestConfiguration(std::string inputmoleculeconfiguration,
158 int numberAtoms) {
159 int size = -1;
160 MPI_Comm_size(MPI_COMM_WORLD, &size);
161
162 if ((dim != 2) && (dim != 3)) {
163 std::cout << "ERROR TestLammps::loadLammpsTestConfiguration: only dim=2 "
164 "or dim=3 supported!"
165 << std::endl;
166 exit(EXIT_FAILURE);
167 }
168 // allocate lammps
169 if (_lammps != NULL) {
170 delete _lammps;
171 _lammps = NULL;
172 }
173 _lammps = new LAMMPS_NS::LAMMPS(_argc, _argv, MPI_COMM_WORLD);
174 if (_lammps == NULL) {
175 std::cout << "ERROR TestLammps::loadLammpsTestConfiguration: Could not "
176 "allocate _lammps!"
177 << std::endl;
178 exit(EXIT_FAILURE);
179 }
180 // read input -> this should also initialise the md solver interface
181 //_lammps->input->file();
182 _lammps->input->one("units lj");
183 _lammps->input->one("atom_style atomic");
184 _lammps->input->one("processors * * * map xyz");
185 if (dim == 2) {
186 _lammps->input->one("dimension 2");
187 }
188 if (dim == 3) {
189 _lammps->input->one("region box block 1.0 11.0 1.0 11.0 1.0 11.0");
190 } else {
191 _lammps->input->one("region box block 1.0 11.0 1.0 11.0 0.0 10.0");
192 }
193 _lammps->input->one("create_box 1 box");
194 if (dim == 2) {
195 std::stringstream ss;
196 ss << "create_atoms 1 random " << numberAtoms << " 123 box";
197 _lammps->input->one(ss.str().c_str());
198 ss.str("");
199 ss.clear();
200 ss << "read_dump " << inputmoleculeconfiguration
201 << " 0 x y replace yes box no format xyz";
202 _lammps->input->one(ss.str().c_str());
203 } else {
204 std::stringstream ss;
205 ss << "create_atoms 1 random " << numberAtoms << " 123 box";
206 _lammps->input->one(ss.str().c_str());
207 ss.str("");
208 ss.clear();
209 ss << "read_dump " << inputmoleculeconfiguration
210 << " 0 x y z replace yes box no format xyz";
211 _lammps->input->one(ss.str().c_str());
212 }
213 _lammps->input->one("mass 1 2.0");
214 _lammps->input->one("velocity all create 1.8 12345678 loop geom");
215 _lammps->input->one("pair_style lj/cut 2.5");
216 _lammps->input->one("pair_modify shift yes");
217 _lammps->input->one("pair_coeff 1 1 1.0 1.0 2.5");
218 // use a very small time step to yield nearly no position changes after 1 or
219 // 2 time steps
220 _lammps->input->one("timestep 1.0e-14");
221 _lammps->input->one("neighbor 0.5 bin");
222 _lammps->input->one("neigh_modify delay 0 every 20 check no");
223 //_lammps->input->one("dump myDump all custom 100 atom*.dat id type x y z vx
224 // vy vz"); _lammps->input->one("restart 100 checkpoint1 checkpoint2");
225 _lammps->input->one("fix 1 all nve");
226 // always use a minimal number of local mamico cells in fix mamico. This
227 // implies that we have to choose this depending on the number of procs
228 if (dim == 2) {
229 switch (size) {
230 case 1:
231 _lammps->input->one("fix 2 all mamico 36 123456 2.5");
232 break;
233 case 4:
234 _lammps->input->one("fix 2 all mamico 16 123456 2.5");
235 break;
236 case 16:
237 _lammps->input->one("fix 2 all mamico 9 123456 2.5");
238 break;
239 default:
240 std::cout << "ERROR TestLammps: only 1,4,16 procs supported in 2D!"
241 << std::endl;
242 exit(EXIT_FAILURE);
243 break;
244 }
245 } else if (dim == 3) {
246 switch (size) {
247 case 1:
248 _lammps->input->one("fix 2 all mamico 216 123456 2.5");
249 break;
250 case 8:
251 _lammps->input->one("fix 2 all mamico 64 123456 2.5");
252 break;
253 case 64:
254 _lammps->input->one("fix 2 all mamico 27 123456 2.5");
255 break;
256 default:
257 std::cout << "ERROR TestLammps: only 1,8,64 procs supported in 3D!"
258 << std::endl;
259 exit(EXIT_FAILURE);
260 break;
261 }
262 }
263 }

270 void loadMamicoTestConfiguration() {
271 std::string mamicoLammpsTestConfiguration = "";
272 if (dim == 2) {
273 mamicoLammpsTestConfiguration = "mamico_lammps_test_configuration2D.xml";
274 } else if (dim == 3) {
275 mamicoLammpsTestConfiguration = "mamico_lammps_test_configuration3D.xml";
276 }
277 tarch::la::Vector<dim, unsigned int> numberProcesses(0);
278 int rank = 0;
279 int size = 0;
280 MPI_Comm_size(MPI_COMM_WORLD, &size);
281 MPI_Comm_rank(MPI_COMM_WORLD, &rank);
282 if (dim == 2) {
283 switch (size) {
284 case 1:
285 for (int d = 0; d < dim; d++) {
286 numberProcesses[d] = 1;
287 }
288 break;
289 case 4:
290 for (int d = 0; d < dim; d++) {
291 numberProcesses[d] = 2;
292 }
293 break;
294 case 16:
295 for (int d = 0; d < dim; d++) {
296 numberProcesses[d] = 4;
297 }
298 break;
299 default:
300 std::cout << "ERROR TestLammps: only 1,4,16 procs supported in 2D!"
301 << std::endl;
302 exit(EXIT_FAILURE);
303 break;
304 }
305 } else if (dim == 3) {
306 switch (size) {
307 case 1:
308 for (int d = 0; d < dim; d++) {
309 numberProcesses[d] = 1;
310 }
311 break;
312 case 8:
313 for (int d = 0; d < dim; d++) {
314 numberProcesses[d] = 2;
315 }
316 break;
317 case 64:
318 for (int d = 0; d < dim; d++) {
319 numberProcesses[d] = 4;
320 }
321 break;
322 default:
323 std::cout << "ERROR TestLammps: only 1,8,64 procs supported in 3D!"
324 << std::endl;
325 exit(EXIT_FAILURE);
326 break;
327 }
328 } else {
329 std::cout << "ERROR TestLammps::loadMamicoTestConfiguration: only 2D and "
330 "3D supported!"
331 << std::endl;
332 exit(EXIT_FAILURE);
333 }
334 const unsigned int numberMDTimestepsPerCouplingCycle = 100000;
335 coupling::configurations::MaMiCoConfiguration<dim> config;
336 std::cout << "Parse config " << mamicoLammpsTestConfiguration << std::endl;
337 tarch::configuration::ParseConfiguration::parseConfiguration<
338 coupling::configurations::MaMiCoConfiguration<dim>>(
339 mamicoLammpsTestConfiguration, "mamico", config);
340 std::cout << "Init macroscopic cell service..." << std::endl;
341 if (_macroscopicCellService != NULL) {
342 delete _macroscopicCellService;
343 _macroscopicCellService = NULL;
344 }
345 _macroscopicCellService =
346 new coupling::services::MacroscopicCellServiceImpl<
347 LAMMPS_NS::MamicoCell, dim>(
348 0,
349 coupling::interface::MamicoInterfaceProvider<LAMMPS_NS::MamicoCell,
350 dim>::getInstance()
351 .getMDSolverInterface(),
352 coupling::interface::MamicoInterfaceProvider<LAMMPS_NS::MamicoCell,
353 dim>::getInstance()
354 .getMacroscopicSolverInterface(),
355 numberProcesses, rank, config.getParticleInsertionConfiguration(),
356 config.getMomentumInsertionConfiguration(),
357 config.getTransferStrategyConfiguration(),
358 config.getParallelTopologyConfiguration(),

360 config.getMacroscopicCellConfiguration());
361 if (_macroscopicCellService == NULL) {
362 std::cout << "ERROR TestLammps: _macroscopicCellService==NULL!"
363 << std::endl;
364 exit(EXIT_FAILURE);
365 }
366 std::cout << "Set macroscopic cell service in MamicoInterfaceProvider..."
367 << std::endl;
368 coupling::interface::MamicoInterfaceProvider<LAMMPS_NS::MamicoCell,
369 dim>::getInstance()
370 .setMacroscopicCellService(_macroscopicCellService);
371 }

394 void printMolecules(bool printGhostMolecules = false) {
395 int size;
396 MPI_Comm_size(MPI_COMM_WORLD, &size);
397 int rank;
398 MPI_Comm_rank(MPI_COMM_WORLD, &rank);
399 MPI_Barrier(MPI_COMM_WORLD);
400
401 int numberAtoms = _lammps->atom->nlocal;
402 if (printGhostMolecules)
403 numberAtoms += _lammps->atom->nghost;
404
405 if (rank == 0)
406 std::cout << "Print molecule information: " << std::endl;
407 for (int r = 0; r < size; r++) {
408 if (r == rank) {
409 for (int i = 0; i < numberAtoms; i++) {
410 std::cout << "Rank " << rank << ": molecule " << i << ": "
411 << _lammps->atom->x[i][0] << "," << _lammps->atom->x[i][1]
412 << "," << _lammps->atom->x[i][2] << std::endl;
413 }
414 }
415 MPI_Barrier(MPI_COMM_WORLD);
416 }
417 }
