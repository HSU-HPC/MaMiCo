97 void loadLammpsTestConfiguration(std::string inputmoleculeconfiguration, int numberAtoms){
98 int size=-1;
99 MPI_Comm_size(MPI_COMM_WORLD,&size);
100
101 if ( (dim != 2 ) && (dim != 3) ){std::cout << "ERROR TestLammps::loadLammpsTestConfiguration: only dim=2 or dim=3 supported!" << std::endl; exit(EXIT_FAILURE); }
102 // allocate lammps
103 if (_lammps!=NULL){delete _lammps; _lammps = NULL;}
104 _lammps = new LAMMPS_NS::LAMMPS(_argc,_argv,MPI_COMM_WORLD);
105 if (_lammps==NULL){ std::cout << "ERROR TestLammps::loadLammpsTestConfiguration: Could not allocate _lammps!" << std::endl; exit(EXIT_FAILURE); }
106 // read input -> this should also initialise the md solver interface
107 //_lammps->input->file();
108 _lammps->input->one("units lj");
109 _lammps->input->one("atom_style atomic");
110 _lammps->input->one("processors * * * map xyz");
111 if (dim==2){ _lammps->input->one("dimension 2"); }
112 if (dim==3){ _lammps->input->one("region box block 1.0 11.0 1.0 11.0 1.0 11.0"); }
113 else { _lammps->input->one("region box block 1.0 11.0 1.0 11.0 0.0 10.0"); }
114 _lammps->input->one("create_box 1 box");
115 if (dim==2){ std::stringstream ss;
116 ss << "create_atoms 1 random " << numberAtoms << " 123 box";
117 _lammps->input->one(ss.str().c_str());
118 ss.str(""); ss.clear(); ss << "read_dump " << inputmoleculeconfiguration << " 0 x y replace yes box no format xyz";
119 _lammps->input->one(ss.str().c_str()); }
120 else { std::stringstream ss;
121 ss << "create_atoms 1 random " << numberAtoms << " 123 box";
122 _lammps->input->one(ss.str().c_str());
123 ss.str(""); ss.clear(); ss << "read_dump " << inputmoleculeconfiguration << " 0 x y z replace yes box no format xyz";
124 _lammps->input->one(ss.str().c_str()); }
125 _lammps->input->one("mass 1 2.0");
126 _lammps->input->one("velocity all create 1.8 12345678 loop geom");
127 _lammps->input->one("pair_style lj/cut 2.5");
128 _lammps->input->one("pair_modify shift yes");
129 _lammps->input->one("pair_coeff 1 1 1.0 1.0 2.5");
130 // use a very small time step to yield nearly no position changes after 1 or 2 time steps
131 _lammps->input->one("timestep 1.0e-14");
132 _lammps->input->one("neighbor 0.5 bin");
133 _lammps->input->one("neigh_modify delay 0 every 20 check no");
134 //_lammps->input->one("dump myDump all custom 100 atom*.dat id type x y z vx vy vz");
135 //_lammps->input->one("restart 100 checkpoint1 checkpoint2");
136 _lammps->input->one("fix 1 all nve");
137 // always use a minimal number of local mamico cells in fix mamico. This implies that we have to choose this depending on the number of procs
138 if (dim==2){
139 switch(size){
140 case 1:
141 _lammps->input->one("fix 2 all mamico 36 123456 2.5"); break;
142 case 4:
143 _lammps->input->one("fix 2 all mamico 16 123456 2.5"); break;
144 case 16:
145 _lammps->input->one("fix 2 all mamico 9 123456 2.5"); break;
146 default:
147 std::cout << "ERROR TestLammps: only 1,4,16 procs supported in 2D!" << std::endl; exit(EXIT_FAILURE); break;
148 }
149 } else if (dim==3){
150 switch(size){
151 case 1:
152 _lammps->input->one("fix 2 all mamico 216 123456 2.5"); break;
153 case 8:
154 _lammps->input->one("fix 2 all mamico 64 123456 2.5"); break;
155 case 64:
156 _lammps->input->one("fix 2 all mamico 27 123456 2.5"); break;
157 default:
158 std::cout << "ERROR TestLammps: only 1,8,64 procs supported in 3D!" << std::endl; exit(EXIT_FAILURE); break;
159 }
160 }

167 void loadMamicoTestConfiguration(){
168 std::string mamicoLammpsTestConfiguration="";
169 if (dim==2) { mamicoLammpsTestConfiguration="mamico_lammps_test_configuration2D.xml";
170 } else if (dim==3){ mamicoLammpsTestConfiguration="mamico_lammps_test_configuration3D.xml";}
171 tarch::la::Vector<dim,unsigned int> numberProcesses(0);
172 int rank = 0;
173 int size = 0;
174 MPI_Comm_size(MPI_COMM_WORLD,&size);
175 MPI_Comm_rank(MPI_COMM_WORLD,&rank);
176 if (dim==2){
177 switch(size){
178 case 1:
179 for (int d = 0; d < dim; d++){ numberProcesses[d] = 1; }
180 break;
181 case 4:
182 for (int d = 0; d < dim; d++){ numberProcesses[d] = 2; }
183 break;
184 case 16:
185 for (int d = 0; d < dim; d++){ numberProcesses[d] = 4; }
186 break;
187 default:
188 std::cout << "ERROR TestLammps: only 1,4,16 procs supported in 2D!" << std::endl; exit(EXIT_FAILURE); break;
189 }
190 } else if (dim==3){
191 switch(size){
192 case 1:
193 for (int d = 0; d < dim; d++){ numberProcesses[d] = 1; }
194 break;
195 case 8:
196 for (int d = 0; d < dim; d++){ numberProcesses[d] = 2; }
197 break;
198 case 64:
199 for (int d = 0; d < dim; d++){ numberProcesses[d] = 4; }
200 break;
201 default:
202 std::cout << "ERROR TestLammps: only 1,8,64 procs supported in 3D!" << std::endl; exit(EXIT_FAILURE); break;
203 }
204 } else {
205 std::cout << "ERROR TestLammps::loadMamicoTestConfiguration: only 2D and 3D supported!" << std::endl; exit(EXIT_FAILURE);
206 }
207 const unsigned int numberMDTimestepsPerCouplingCycle = 100000;
208 coupling::configurations::MaMiCoConfiguration<dim> config;
209 std::cout << "Parse config " << mamicoLammpsTestConfiguration << std::endl;
210 tarch::configuration::ParseConfiguration::parseConfiguration<coupling::configurations::MaMiCoConfiguration<dim> > (mamicoLammpsTestConfiguration,"mamico", config);
211 std::cout << "Init macroscopic cell service..." << std::endl;
212 if (_macroscopicCellService != NULL){delete _macroscopicCellService; _macroscopicCellService = NULL; }
213 _macroscopicCellService = new coupling::services::MacroscopicCellServiceImpl<LAMMPS_NS::MamicoCell,dim>(
214 0,coupling::interface::MamicoInterfaceProvider<LAMMPS_NS::MamicoCell,dim>::getInstance().getMDSolverInterface(),
215 coupling::interface::MamicoInterfaceProvider<LAMMPS_NS::MamicoCell,dim>::getInstance().getMacroscopicSolverInterface(),
216 numberProcesses,
217 rank,
218 config.getParticleInsertionConfiguration(),
219 config.getMomentumInsertionConfiguration(),
220 config.getTransferStrategyConfiguration(),
221 config.getParallelTopologyConfiguration(),
222 numberMDTimestepsPerCouplingCycle,
223 config.getMacroscopicCellConfiguration()
224 );
225 if (_macroscopicCellService==NULL){std::cout << "ERROR TestLammps: _macroscopicCellService==NULL!" << std::endl; exit(EXIT_FAILURE); }
226 std::cout << "Set macroscopic cell service in MamicoInterfaceProvider..." << std::endl;
227 coupling::interface::MamicoInterfaceProvider<LAMMPS_NS::MamicoCell,dim>::getInstance().setMacroscopicCellService(_macroscopicCellService);
228 }

240 void printMolecules(bool printGhostMolecules=false){
241 int size; MPI_Comm_size(MPI_COMM_WORLD,&size);
242 int rank; MPI_Comm_rank(MPI_COMM_WORLD,&rank);
243 MPI_Barrier(MPI_COMM_WORLD);
244
245 int numberAtoms = _lammps->atom->nlocal;
246 if (printGhostMolecules) numberAtoms += _lammps->atom->nghost;
247
248 if (rank==0) std::cout << "Print molecule information: " << std::endl;
249 for (int r = 0; r < size; r++){
250 if (r==rank){
251 for (int i = 0; i < numberAtoms; i++){
252 std::cout << "Rank " << rank << ": molecule " << i << ": " << _lammps->atom->x[i][0] << "," << _lammps->atom->x[i][1] << "," << _lammps->atom->x[i][2] << std::endl;
253 }
254 }
255 MPI_Barrier(MPI_COMM_WORLD);
256 }
