template<class LinkedCell, unsigned int dim>
class coupling::cellmappings::ComputeAvgForceAndVelocity< LinkedCell, dim >

sums up all force/velocity vectors and counts molecules inside a linked cell. Afterwards, the average force/velocity contribution is computed.
