79 bool inner = true;
80 tarch::la::Vector<dim, unsigned int> globalIndexUnsigned{globalCellIndex.get()};

82 inner = inner && (globalIndexUnsigned[d] > _innermostLayer && globalIndexUnsigned[d] < 1 + I09::numberCellsInDomain[d] - _innermostLayer);
