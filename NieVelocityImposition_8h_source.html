83 const tarch::la::Vector<dim, unsigned int> globalNumberMacroscopicCells(_indexConversion.getGlobalNumberMacroscopicCells());
84 const tarch::la::Vector<dim, unsigned int> globalCellIndex(
85 _indexConversion.getGlobalVectorCellIndex(_indexConversion.convertLocalToGlobalCellIndex(currentLocalMacroscopicCellIndex)));
86 bool inner = true;
87 for (unsigned int d = 0; d < dim; d++)
88 inner = inner && (globalCellIndex[d] > _innermostLayer && globalCellIndex[d] < 1 + globalNumberMacroscopicCells[d] - _innermostLayer);
89 bool outer = false;
90 for (unsigned int d = 0; d < dim; d++)
91 outer = outer || (globalCellIndex[d] < _outermostLayer || globalCellIndex[d] > 1 + globalNumberMacroscopicCells[d] - _outermostLayer);
92 return !inner && !outer;
93 }
