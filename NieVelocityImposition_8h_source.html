96 bool isInsideImpositionLayer(
97 const unsigned int &currentLocalMacroscopicCellIndex) const {
98 const tarch::la::Vector<dim, unsigned int> globalNumberMacroscopicCells(
99 _indexConversion.getGlobalNumberMacroscopicCells());
100 const tarch::la::Vector<dim, unsigned int> globalCellIndex(
101 _indexConversion.getGlobalVectorCellIndex(
102 _indexConversion.convertLocalToGlobalCellIndex(
103 currentLocalMacroscopicCellIndex)));
104 bool inner = true;
105 for (unsigned int d = 0; d < dim; d++)
106 inner =
107 inner && (globalCellIndex[d] > _innermostLayer &&
108 globalCellIndex[d] <
109 1 + globalNumberMacroscopicCells[d] - _innermostLayer);
110 bool outer = false;
111 for (unsigned int d = 0; d < dim; d++)
112 outer =
113 outer || (globalCellIndex[d] < _outermostLayer ||
114 globalCellIndex[d] >
115 1 + globalNumberMacroscopicCells[d] - _outermostLayer);
116 return !inner && !outer;
117 }
